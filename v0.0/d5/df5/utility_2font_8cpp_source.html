26ref_ptr<Font> Font::get(const std::string& f, GLuint size, GLuint dpi) {
27 if (!isFreetypeInitialized_) {
28 if (FT_Init_FreeType(&ftlib_)) { throw Font::Error("FT_Init_FreeType failed."); }
29 isFreetypeInitialized_ = GL_TRUE;
30 }
31 // unique font identifier
32 std::string fontKey = REGEN_STRING(f << size << "_" << dpi);
33 // check for cached font
34 auto result = fonts_.find(fontKey);
35 if (result != fonts_.end()) { return result->second; }
36
37 // create the font
38 ref_ptr<Font> font = ref_ptr<Font>::alloc(f, size, dpi);
39 fonts_[fontKey] = font;
40
41 return font;
42}

49Font::Font(const std::string &fontPath, GLuint size, GLuint dpi)
50 : fontPath_(fontPath),
51 size_(size),
52 dpi_(dpi),
53 lineHeight_(1.0f / .83f) {
54 // The object in which Freetype holds information on a given
55 // font is called a "face".
56 FT_Face face;
57
58 if (!boost::filesystem::exists(fontPath)) {
59 throw Error(REGEN_STRING(
60 "Unable to font file at '" << fontPath << "'."));
61 }
62
63 // This is where we load in the font information from the file.
64 // Of all the places where the code might die, this is the most likely,
65 // as FT_New_Face will die if the font file does not exist or is somehow broken.
66 if (FT_New_Face(ftlib_, fontPath.c_str(), 0, &face)) {
67 throw Error(REGEN_STRING(
68 "FT_New_Face failed. " <<
69 "There is probably a problem with the font file at " << fontPath
70 << "."));
71 }
72 // Freetype measures font size in terms of 1/64ths of pixels.
73 // Thus, to make a font h pixels high, we need to request a size of h*64.
74 // (h << 6 is just a prettier way of writing h*64)
75 FT_Set_Char_Size(face, size << 6, size << 6, dpi, dpi);
76
77 // find the bounding box that can hold any glyph of this font,
78 // we need to find this box because all glyph textures must have the same size
79 // for Texture2DArray.
80 int pixels_x = ::FT_MulFix((face->bbox.xMax - face->bbox.xMin), face->size->metrics.x_scale);
81 int pixels_y = ::FT_MulFix((face->bbox.yMax - face->bbox.yMin), face->size->metrics.y_scale);
82 auto textureWidth = (GLuint) (pixels_x / 64);
83 auto textureHeight = (GLuint) (pixels_y / 64);
84 // make sure texture dimensions are multiple of 2
85 if (textureWidth % 2 != 0) textureWidth += 1;
86 if (textureHeight % 2 != 0) textureHeight += 1;
87
88 // remembers some geometry infos about glyphs
89 faceData_ = new FaceData[NUMBER_OF_GLYPHS];
90
91 // create a array texture for the glyphs
92 arrayTexture_ = ref_ptr<Texture2DArray>::alloc(1);
93 arrayTexture_->set_format(GL_RED);
94 arrayTexture_->set_internalFormat(GL_R8);
95 arrayTexture_->set_pixelType(GL_UNSIGNED_BYTE);
96 arrayTexture_->set_rectangleSize(textureWidth, textureHeight);
97 arrayTexture_->set_depth(NUMBER_OF_GLYPHS);
98 arrayTexture_->begin(RenderState::get());
99 arrayTexture_->wrapping().push(GL_CLAMP_TO_BORDER);
100 arrayTexture_->filter().push(GL_LINEAR);
101 arrayTexture_->swizzle().push(GL_RED);
102 // GL expects 4byte aligned rows
103 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
104 arrayTexture_->texImage();
105 for (unsigned short i = 0; i < NUMBER_OF_GLYPHS; i++) {
106 initGlyph(face, i, textureWidth, textureHeight);
107 }
108 // reset to default
109 glPixelStorei(GL_UNPACK_ALIGNMENT, 4);
110 arrayTexture_->end(RenderState::get());
111
112 FT_Done_Face(face);
113}

124GLubyte *Font::invertPixmapWithAlpha(
125 const FT_Bitmap &bitmap, GLuint width, GLuint height) {
126 const unsigned int arraySize = width * height;
127 auto *inverse = new GLubyte[arraySize];
128 auto *inverse_ptr = inverse;
129
130 memset(inverse, 0x0, sizeof(GLubyte) * arraySize);
131
132 for (auto r = 0u; r < bitmap.rows; ++r) {
133 GLubyte *bitmap_ptr = &bitmap.buffer[bitmap.pitch * r];
134 for (auto p = 0u; p < bitmap.width; ++p) {
135 *inverse_ptr++ = *bitmap_ptr++;
136 }
137 inverse_ptr += (width - bitmap.width);
138 }
139
140 return inverse;
141}

143void Font::initGlyph(FT_Face face, GLushort ch, GLuint textureWidth, GLuint textureHeight) {
144 FaceData &glyphData = faceData_[ch];
145 FT_Glyph glyph;
146 GLubyte *inverted;
147
148 // Load the Glyph for our character.
149 if (FT_Load_Glyph(face, FT_Get_Char_Index(face, ch), FT_LOAD_DEFAULT)) {
150 throw Error("FT_Load_Glyph failed");
151 }
152 // Move the face's glyph into a Glyph object.
153 if (FT_Get_Glyph(face->glyph, &glyph)) {
154 throw Error("FT_Get_Glyph failed");
155 }
156 // Convert the glyph to a bitmap.
157 FT_Glyph_To_Bitmap(&glyph, ft_render_mode_normal, 0, 1);
158
159 auto bitmap_glyph = (FT_BitmapGlyph) glyph;
160 // This reference will make accessing the bitmap easier
161 FT_Bitmap &bitmap = bitmap_glyph->bitmap;
162
163 {
164 auto bitmapWith = (GLfloat) bitmap.width;
165 auto bitmapHeight = (GLfloat) bitmap.rows;
166 GLfloat sizeFactor = 1.0f / (GLfloat) size();
167
168 glyphData.uvX = bitmapWith / ((GLfloat) textureWidth);
169 glyphData.uvY = bitmapHeight / ((GLfloat) textureHeight);
170 glyphData.height = bitmapHeight * sizeFactor;
171 glyphData.width = bitmapWith * sizeFactor;
172 glyphData.advanceX = (face->glyph->advance.x >> 6) * sizeFactor;
173 glyphData.left = bitmap_glyph->left * sizeFactor;
174 glyphData.top = bitmap_glyph->top * sizeFactor;
175 }
176
177 {
178 inverted = invertPixmapWithAlpha(bitmap, textureWidth, textureHeight);
179 arrayTexture_->texSubImage((int) ch, inverted);
180 delete[]inverted;
181 }
182
183 FT_Done_Glyph(glyph);
184}
