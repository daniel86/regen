215void BoidsSimulation_CPU::animate(double dt) {
216 auto dt_f = static_cast<float>(dt) * 0.001f;
217 // advance boids simulation
218 simulateBoids(dt_f);
219 // update boids model transformation using the boids data
220 if (tf_.get()){
221 auto &tfInput = tf_->get();
222 auto tfData = tfInput->mapClientData<Mat4f>(ShaderData::READ | ShaderData::WRITE);
223 for (GLuint i = 0; i < tfInput->numInstances(); ++i) {
224 auto &d = boidData_[i];
225 // calculate boid matrix, also need to compute rotation from velocity, model z
226 // should point in the direction of velocity.
227 if (d.velocity.length() > 0.001f) {
228 // Convert the normalized direction vector to Euler angles
229 boidRotation_.setEuler(
230 atan2(-d.direction.x, d.direction.z) + baseOrientation_,
231 asin(-d.direction.y),
232 0.0f);
233 tfData.w[i] = boidRotation_.calculateMatrix();
234 tfData.w[i].scale(boidsScale_);
235 tfData.w[i].translate(d.position);
236 } else {
237 tfData.w[i] = tfData.r[i];
238 }
239 }
240 }
241 else if (position_.get()) {
242 auto positionData = position_->mapClientData<Vec3f>(ShaderData::WRITE);
243 // TODO: rather use memcpy
244 for (GLuint i = 0; i < position_->numInstances(); ++i) {
245 positionData.w[i] = boidData_[i].position;
246 }
247 }
248 // recompute neighborhood relationships of all boids
249 auto maxDistance = visualRange_->getVertex(0);
250 auto maxNeighbours = maxNumNeighbors_->getVertex(0);
251 for (size_t i = 0; i < boidData_.size(); i++) {
252 auto &b1 = boidData_[i];
253 unsigned int numNeighbors = 0;
254 for (size_t j = i + 1; j < boidData_.size(); j++) {
255 auto &b2 = boidData_[j];
256 auto distance = (b1.position - b2.position).length();
257 if (distance < maxDistance.r) {
258 b1.neighbors.push_back(&b2);
259 b2.neighbors.push_back(&b1);
260 numNeighbors++;
261 if (numNeighbors >= maxNeighbours.r) {
262 break;
263 }
264 }
265 }
266 }
267}

276void BoidsSimulation_CPU::simulateBoid(BoidData &boid, float dt) {
277 boid.force = Vec3f::zero();
278 // a boid is lost if it is outside the bounds
279 bool isBoidLost = !bounds_.contains(boid.position);
280 if (boid.neighbors.empty()) {
281 // a boid without neighbors is lost
282 isBoidLost = true;
283 } else {
284 // simulate the boid using the three rules of boids
285 avgPosition_ = Vec3f::zero();
286 avgVelocity_ = Vec3f::zero();
287 separation_ = Vec3f::zero();
288 for (auto neighbor: boid.neighbors) {
289 avgPosition_ += neighbor->position;
290 avgVelocity_ += neighbor->velocity;
291 boidDirection_ = boid.position - neighbor->position;
292 float distance = boidDirection_.length();
293 if (distance < 0.001f) {
294 boidDirection_ = Vec3f::random();
295 // avoid "jumping" up
296 //boidDirection_.y = -abs(boidDirection_.y);
297 boidDirection_.normalize();
298 distance = 0.0f;
299 } else {
300 boidDirection_ /= distance * distance;
301 }
302 if (distance < avoidanceDistance_->getVertex(0).r) {
303 separation_ += boidDirection_;
304 }
305 }
306 avgPosition_ /= static_cast<float>(boid.neighbors.size());
307 avgVelocity_ /= static_cast<float>(boid.neighbors.size());
308 boid.force +=
309 separation_ * separationWeight_->getVertex(0).r +
310 (avgVelocity_ - boid.velocity) * alignmentWeight_->getVertex(0).r +
311 (avgPosition_ - boid.position) * coherenceWeight_->getVertex(0).r;
312 }
313
314 // put some restrictions on the boid's velocity.
315 // a boid that cannot avoid collisions is considered lost.
316 isBoidLost = !avoidCollisions(boid, dt) || isBoidLost;
317 auto isInDanger = !dangers_.empty() && !avoidDanger(boid);
318 if (!isInDanger) {
319 // note: ignore attractors if in danger
320 attract(boid);
321 }
322 // drift towards home if lost
323 if (isBoidLost) { homesickness(boid); }
324
325 boid.position += boid.velocity * dt;
326 boid.velocity += boid.force * dt;
327 limitVelocity(boid);
328
329 boid.direction = boid.velocity;
330 boid.direction.normalize();
331}

333void BoidsSimulation_CPU::limitVelocity(BoidData &boid) {
334 // limit translation speed
335 auto maxSpeed = maxBoidSpeed_->getVertex(0).r;
336 if (boid.velocity.length() > maxSpeed) {
337 boid.velocity.normalize();
338 boid.velocity *= maxSpeed;
339 }
340
341 // limit angular speed
342 auto maxAngularSpeed = maxAngularSpeed_->getVertex(0).r;
343 boidDirection_ = boid.velocity;
344 boidDirection_.normalize();
345 auto angle = acos(boidDirection_.dot(boid.direction));
346 if (angle > maxAngularSpeed) {
347 auto axis = boidDirection_.cross(boid.direction);
348 axis.normalize();
349 boidRotation_.setAxisAngle(axis, maxAngularSpeed);
350 auto newDirection = boidRotation_.rotate(boid.direction);
351 boid.velocity = newDirection * boid.velocity.length();
352 }
353}

355void BoidsSimulation_CPU::homesickness(BoidData &boid) {
356 // a boid seems to have lost track, and wants to go home!
357 // first find closest home point...
358 const Vec3f *closestHomePoint = &Vec3f::zero();
359 float minDistance;
360 if (!homePoints_.empty()) {
361 minDistance = std::numeric_limits<float>::max();
362 for (auto &home: homePoints_) {
363 auto distance = (boid.position - home).length();
364 if (distance < minDistance) {
365 minDistance = distance;
366 closestHomePoint = &home;
367 }
368 }
369 } else {
370 minDistance = (boid.position - *closestHomePoint).length();
371 }
372 // second steer towards the closest home point ...
373 boidDirection_ = *closestHomePoint - boid.position;
374 if (minDistance < 0.001f) {
375 boidDirection_ = Vec3f::random();
376 } else {
377 boidDirection_ /= minDistance;
378 }
379 auto repulsionFactor = repulsionFactor_->getVertex(0).r * separationWeight_->getVertex(0).r;
380 boid.force += boidDirection_ * repulsionFactor * 0.1;
381}

398bool BoidsSimulation_CPU::avoidDanger(BoidData &boid) {
399 auto maxDistance = visualRange_->getVertex(0).r;
400 bool isInDanger = false;
401 for (auto &danger: dangers_) {
402 auto dir = Vec3f::zero();
403 if (danger.pos.get()) {
404 dir = danger.pos.get()->getVertex(0).r;
405 }
406 if (danger.tf.get()) {
407 dir += danger.tf.get()->getVertex(0).r.position();
408 }
409 dir -= boid.position;
410 auto distance = dir.length();
411 if (distance < maxDistance) {
412 if (distance < 0.001f) {
413 dir = Vec3f::random();
414 } else {
415 dir /= distance;
416 }
417 auto repulsionFactor = repulsionFactor_->getVertex(0).r * separationWeight_->getVertex(0).r;
418 boid.force += dir * repulsionFactor;
419 isInDanger = true;
420 }
421 }
422 return !isInDanger;
423}

425void BoidsSimulation_CPU::attract(BoidData &boid) {
426 auto maxDistance = visualRange_->getVertex(0).r * 100; // TODO: parameter
427 for (auto &attractor: attractors_) {
428 auto dir = Vec3f::zero();
429 if (attractor.pos.get()) {
430 dir = attractor.pos.get()->getVertex(0).r;
431 }
432 if (attractor.tf.get()) {
433 dir += attractor.tf.get()->getVertex(0).r.position();

444bool BoidsSimulation_CPU::avoidCollisions(BoidData &boid, float dt) {
445 auto repulsionFactor = repulsionFactor_->getVertex(0).r * separationWeight_->getVertex(0).r;
446 auto avoidDistance = avoidanceDistance_->getVertex(0).r;
447 auto avoidDistanceHalf = avoidDistance * 0.5f;
448 auto nextVelocity = boid.velocity + boid.force * dt;
449 nextVelocity.normalize();
450 auto lookAhead = boid.position + nextVelocity * lookAheadDistance_->getVertex(0).r;
451 bool isCollisionFree = true;
452
456 if (lookAhead.y < bounds_.min.y + avoidDistance) {
457 boid.force.y += repulsionFactor;
458 }
459 if (lookAhead.y > bounds_.max.y - avoidDistance) {
460 boid.force.y -= repulsionFactor;
461 }
462 if (lookAhead.x < bounds_.min.x + avoidDistance) {
463 boid.force.x += repulsionFactor;
464 }
465 if (lookAhead.x > bounds_.max.x - avoidDistance) {
466 boid.force.x -= repulsionFactor;
467 }
468 if (lookAhead.z < bounds_.min.z + avoidDistance) {
469 boid.force.z += repulsionFactor;
470 }
471 if (lookAhead.z > bounds_.max.z - avoidDistance) {
472 boid.force.z -= repulsionFactor;
473 }
474
478 if (heightMap_.get()) {
479 // boid position in height map space [0, 1]
480 auto boidCoord = computeUV(boid.position, mapCenter_, mapSize_);
481 auto currentY = heightMap_->sampleLinear<float>(boidCoord, heightMap_->textureData());
482 currentY *= heightMapFactor_;
483 currentY += mapCenter_.y + avoidDistanceHalf;
484 // sample the height map at the projected boid position
485 boidCoord = computeUV(lookAhead, mapCenter_, mapSize_);
486 auto nextY = heightMap_->sampleLinear<float>(boidCoord, heightMap_->textureData());
487 nextY *= heightMapFactor_;
488 nextY += mapCenter_.y + avoidDistanceHalf;
489
490 if (boid.position.y < currentY) {
491 // The boid is below the minimum height of the height map -> push boid up.
492 if (currentY < bounds_.max.y - avoidDistanceHalf) {
493 // only push up in case the height map y is below the maximum y of boids
494 boid.force.y += repulsionFactor;
495 }
496 isCollisionFree = false;
497 }
498 if (lookAhead.y < nextY) {
499 // Assuming the boid follows the lookAhead direction, it will eventually reach a point where
500 // it is below the surface of the height map -> push boid up.
501 if (nextY < bounds_.max.y - avoidDistanceHalf) {
502 // only push up in case the height map y is below the maximum y of boids
503 boid.force.y += repulsionFactor;
504 }
505 }
506 if (nextY > bounds_.max.y - avoidDistanceHalf) {
507 // Assuming the boid follows the lookAhead direction, it will eventually reach a point where
508 // it is above the max y of the boid bounds.
509 // We could sample normal map here, and compute reflection vector. But this might be a bit
510 // overkill for huge number of boids.
511 // An easy approach is to push the boid to where it came from:
512 boid.force.x -= nextVelocity.x * repulsionFactor;
513 boid.force.z -= nextVelocity.z * repulsionFactor;
514 }
515 }
516
517 return isCollisionFree;
518}
