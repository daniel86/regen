17static GLfloat sampleNoise(
18 const NoiseGenerator &noiseGen,
19 GLdouble x, GLdouble y, GLdouble z,
20 GLboolean isSeamless2D,
21 GLboolean isSeamless3D) {
22 GLfloat val;
23 if (isSeamless2D) {
24 GLdouble a = noiseGen.GetValue(x, y, z);
25 GLdouble b = noiseGen.GetValue(x + 1.0, y, z);
26 GLdouble c = noiseGen.GetValue(x, y + 1.0, z);
27 GLdouble d = noiseGen.GetValue(x + 1.0, y + 1.0, z);
28 val = math::mix(
29 math::mix(a, b, 1.0 - x),
30 math::mix(c, d, 1.0 - x), 1.0 - y);
31 } else if (isSeamless3D) {
32 GLdouble a0 = noiseGen.GetValue(x, y, z);
33 GLdouble b0 = noiseGen.GetValue(x + 1.0, y, z);
34 GLdouble c0 = noiseGen.GetValue(x, y + 1.0, z);
35 GLdouble d0 = noiseGen.GetValue(x + 1.0, y + 1.0, z);
36 GLdouble val0 = math::mix(
37 math::mix(a0, b0, 1.0 - x),
38 math::mix(c0, d0, 1.0 - x), 1.0 - y);
39
40 GLdouble a1 = noiseGen.GetValue(x, y, z + 1.0);
41 GLdouble b1 = noiseGen.GetValue(x + 1.0, y, z + 1.0);
42 GLdouble c1 = noiseGen.GetValue(x, y + 1.0, z + 1.0);
43 GLdouble d1 = noiseGen.GetValue(x + 1.0, y + 1.0, z + 1.0);
44 GLdouble val1 = math::mix(
45 math::mix(a1, b1, 1.0 - x),
46 math::mix(c1, d1, 1.0 - x), 1.0 - y);
47
48 val = (GLfloat) math::mix(val0, val1, 1.0 - z);
49 } else {
50 val = (GLfloat) noiseGen.GetValue(x, y, z);
51 }
52 // map roughly to [0,1] and clamp to that range
53 return math::clamp((val + 1.0f) * 0.5f, 0.0f, 1.0f);
54}

75void NoiseTexture2D::updateNoise() {
76 if (generator_.get() == nullptr) return;
77 const auto &gen = *generator_.get();
78
79 auto *data = new GLubyte[width() * height()];
80 GLubyte *dataPtr = data;
81 for (GLuint x = 0u; x < width(); ++x) {
82 for (GLuint y = 0u; y < height(); ++y) {
83 GLfloat val = sampleNoise(gen, x, y, 0.0, isSeamless_, GL_FALSE);
84 *dataPtr = (GLubyte) (val * 255.0f);
85 ++dataPtr;
86 }
87 }
88
89 begin(RenderState::get());
90 set_textureData(data);
91 texImage();
92 set_textureData(nullptr);
93 delete[]data;
94 end(RenderState::get());
95}

172ref_ptr<NoiseGenerator> NoiseGenerator::preset_clouds(GLint randomSeed) {
173 // Base of the cloud texture.
174 // The billowy noise produces the basic shape of soft, fluffy clouds.
175 auto cloudBase = ref_ptr<module::Billow>::alloc();
176 cloudBase->SetSeed(randomSeed);
177 cloudBase->SetFrequency(2.0);
178 cloudBase->SetPersistence(0.375);
179 cloudBase->SetLacunarity(2.12109375);
180 cloudBase->SetOctaveCount(4);
181 cloudBase->SetNoiseQuality(QUALITY_BEST);
182 auto baseGen = ref_ptr<NoiseGenerator>::alloc("base", cloudBase);
183
184 // Perturb the cloud texture for more realism.
185 auto finalClouds = ref_ptr<module::Turbulence>::alloc();
186 finalClouds->SetSeed(randomSeed + 1);
187 finalClouds->SetFrequency(16.0);
188 finalClouds->SetPower(1.0 / 64.0);
189 finalClouds->SetRoughness(2);
190 auto finalGen = ref_ptr<NoiseGenerator>::alloc("perturbed", finalClouds);
191 finalGen->addSource(baseGen);
192 return finalGen;
193}

195ref_ptr<NoiseGenerator> NoiseGenerator::preset_wood(GLint randomSeed) {
196 // Base wood texture. The base texture uses concentric cylinders aligned
197 // on the z axis, like a log.
198 auto baseWood = ref_ptr<module::Cylinders>::alloc();
199 baseWood->SetFrequency(16.0);
200 auto baseGen = ref_ptr<NoiseGenerator>::alloc("base", baseWood);
201
202 // Perlin noise to use for the wood grain.
203 auto woodGrainNoise = ref_ptr<module::Perlin>::alloc();
204 woodGrainNoise->SetSeed(randomSeed);
205 woodGrainNoise->SetFrequency(48.0);
206 woodGrainNoise->SetPersistence(0.5);
207 woodGrainNoise->SetLacunarity(2.20703125);
208 woodGrainNoise->SetOctaveCount(3);
209 woodGrainNoise->SetNoiseQuality(QUALITY_STD);
210 auto grainNoiseGen = ref_ptr<NoiseGenerator>::alloc("grain-noise", woodGrainNoise);
211
212 // Stretch the Perlin noise in the same direction as the center of the
213 // log. This produces a nice wood-grain texture.
214 auto scaledBaseWoodGrain = ref_ptr<module::ScalePoint>::alloc();
215 scaledBaseWoodGrain->SetYScale(0.25);
216 auto scaledGen = ref_ptr<NoiseGenerator>::alloc("scaled", scaledBaseWoodGrain);
217 scaledGen->addSource(grainNoiseGen);
218
219 // Scale the wood-grain values so that they may be added to the base wood
220 // texture.
221 auto woodGrain = ref_ptr<module::ScaleBias>::alloc();
222 woodGrain->SetScale(0.25);
223 woodGrain->SetBias(0.125);
224 auto grainGen = ref_ptr<NoiseGenerator>::alloc("grain", woodGrain);
225 grainGen->addSource(scaledGen);
226
227 // Add the wood grain texture to the base wood texture.
228 auto combinedGen = ref_ptr<NoiseGenerator>::alloc(
229 "combined", ref_ptr<module::Add>::alloc());
230 combinedGen->addSource(baseGen);
231 combinedGen->addSource(grainGen);
232
233 // Slightly perturb the wood texture for more realism.
234 auto perturbedWood = ref_ptr<module::Turbulence>::alloc();
235 perturbedWood->SetSeed(randomSeed + 1);
236 perturbedWood->SetFrequency(4.0);
237 perturbedWood->SetPower(1.0 / 256.0);
238 perturbedWood->SetRoughness(4);
239 auto perturbedGen = ref_ptr<NoiseGenerator>::alloc("perturbed", perturbedWood);
240 perturbedGen->addSource(combinedGen);
241
242 // Cut the wood texture a small distance from the center of the "log".
243 auto translatedWood = ref_ptr<module::TranslatePoint>::alloc();
244 translatedWood->SetZTranslation(1.48);
245 auto translatedGen = ref_ptr<NoiseGenerator>::alloc("translated", translatedWood);
246 translatedGen->addSource(perturbedGen);
247
248 // Cut the wood texture on an angle to produce a more interesting wood
249 // texture.
250 auto rotatedWood = ref_ptr<module::RotatePoint>::alloc();
251 rotatedWood->SetAngles(84.0, 0.0, 0.0);
252 auto rotatedGen = ref_ptr<NoiseGenerator>::alloc("rotated", rotatedWood);
253 rotatedGen->addSource(translatedGen);
254
255 // Finally, perturb the wood texture to produce the final texture.
256 auto finalWood = ref_ptr<module::Turbulence>::alloc();
257 finalWood->SetSeed(randomSeed + 2);
258 finalWood->SetFrequency(2.0);
259 finalWood->SetPower(1.0 / 64.0);
260 finalWood->SetRoughness(4);
261 auto finalGen = ref_ptr<NoiseGenerator>::alloc("final", finalWood);
262 finalGen->addSource(rotatedGen);
263
264 return finalGen;
265}

267ref_ptr<NoiseGenerator> NoiseGenerator::preset_granite(GLint randomSeed) {
268 // Primary granite texture. This generates the "roughness" of the texture
269 // when lit by a light source.
270 auto primaryGranite = ref_ptr<module::Billow>::alloc();
271 primaryGranite->SetSeed(randomSeed);
272 primaryGranite->SetFrequency(8.0);
273 primaryGranite->SetPersistence(0.625);
274 primaryGranite->SetLacunarity(2.18359375);
275 primaryGranite->SetOctaveCount(6);
276 primaryGranite->SetNoiseQuality(QUALITY_STD);
277 auto primaryGen = ref_ptr<NoiseGenerator>::alloc("primary", primaryGranite);
278
279 // Use Voronoi polygons to produce the small grains for the granite texture.
280 auto baseGrains = ref_ptr<module::Voronoi>::alloc();
281 baseGrains->SetSeed(randomSeed + 1);
282 baseGrains->SetFrequency(16.0);
283 baseGrains->EnableDistance(true);
284 auto baseGrainsGen = ref_ptr<NoiseGenerator>::alloc("base-grains", baseGrains);
285
286 // Scale the small grain values so that they may be added to the base
287 // granite texture. Voronoi polygons normally generate pits, so apply a
288 // negative scaling factor to produce bumps instead.
289 auto scaledGrains = ref_ptr<module::ScaleBias>::alloc();
290 scaledGrains->SetScale(-0.5);
291 scaledGrains->SetBias(0.0);
292 auto grainGen = ref_ptr<NoiseGenerator>::alloc("grains", scaledGrains);
293 grainGen->addSource(baseGrainsGen);
294
295 // Combine the primary granite texture with the small grain texture.
296 auto combinedGen = ref_ptr<NoiseGenerator>::alloc(
297 "combined", ref_ptr<module::Add>::alloc());
298 combinedGen->addSource(primaryGen);
299 combinedGen->addSource(grainGen);
300
301 // Finally, perturb the granite texture to add realism.
302 auto finalGranite = ref_ptr<module::Turbulence>::alloc();
303 finalGranite->SetSeed(randomSeed + 2);
304 finalGranite->SetFrequency(4.0);
305 finalGranite->SetPower(1.0 / 8.0);
306 finalGranite->SetRoughness(6);
307 auto finalGen = ref_ptr<NoiseGenerator>::alloc("final", finalGranite);
308 finalGen->addSource(combinedGen);
309 return finalGen;
310}
