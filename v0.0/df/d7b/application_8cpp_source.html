133void Application::updateMousePosition() {
134 auto mousePosition = mousePosition_->getVertex(0);
135 auto viewport = windowViewport_->getVertex(0);
136 // mouse position in range [0,1] within viewport
137 mouseTexco_->setVertex(0, Vec2f(
138 mousePosition.r.x / (GLfloat) viewport.r.x,
139 1.0 - mousePosition.r.y / (GLfloat) viewport.r.y));

142void Application::mouseMove(const Vec2i &pos) {
143 boost::posix_time::ptime time(
144 boost::posix_time::microsec_clock::local_time());
145 GLint dx = pos.x - mousePosition_->getVertex(0).r.x;
146 GLint dy = pos.y - mousePosition_->getVertex(0).r.y;
147 mousePosition_->setVertex(0, Vec2f(pos.x, pos.y));
148 updateMousePosition();
149
150 ref_ptr<MouseMotionEvent> event = ref_ptr<MouseMotionEvent>::alloc();
151 event->dt = ((GLdouble) (time - lastMotionTime_).total_microseconds()) / 1000.0;
152 event->dx = dx;
153 event->dy = dy;
154 queueEmit(MOUSE_MOTION_EVENT, event);
155 lastMotionTime_ = time;
156}

195 //glewExperimental=GL_TRUE;
196 GLenum err = glewInit();
197 if (GLEW_OK != err) {
198 std::cerr << "Error: " << glewGetErrorString(err) << std::endl;
199 exit(1);
200 }
201 // glewInit may calls glGetString(GL_EXTENSIONS),
202 // then invalid enum error is generated.
203 GL_ERROR_LOG();

205 REGEN_DEBUG("VENDOR: " << glGetString(GL_VENDOR));
206 REGEN_DEBUG("RENDERER: " << glGetString(GL_RENDERER));
207 REGEN_DEBUG("VERSION: " << glGetString(GL_VERSION));
208
209 // check for required and optional extensions
210 for (auto & it : requiredExt_) {
211 if (!glewIsSupported(it.c_str())) {
212 REGEN_ERROR(it << " unsupported.");
213 exit(0);
214 } else {REGEN_DEBUG(it << " supported."); }
215 }
216 for (auto & it : optionalExt_) {
217 if (!glewIsSupported(it.c_str())) {REGEN_DEBUG(it << " unsupported."); }
218 else {REGEN_DEBUG(it << " supported."); }
219 }

333void Application::updateTime() {
334 if (!isTimeInitialized_) {
335 setTime();
336 }
337 boost::posix_time::ptime t(boost::posix_time::microsec_clock::local_time());
338 auto dt = (t - lastTime_).total_milliseconds();
339 timeDelta_->setVertex(0, (double)dt);
340 timeSeconds_->setVertex(0, t.time_of_day().total_microseconds()/1e+6);
341 lastTime_ = t;
342 worldTime_.p_time += boost::posix_time::milliseconds(static_cast<long>(dt * worldTime_.scale));
343}

378void Application::withGLContext(std::function<void()> f) {
379 auto anim = ref_ptr<FunctionCallWithGL>::alloc(f);
380 glCalls_.emplace_back(anim);
381 anim->connect(Animation::ANIMATION_STOPPED, ref_ptr<LambdaEventHandler>::alloc(
382 [this](EventObject *emitter, EventData *data) {
383 for (auto & it : glCalls_) {
384 if (it.get() == emitter) {

386 break;
387 }
388 }
389 }));
390 anim->startAnimation();
