47void UBO::computePaddedSize() {
48 requiredSize_ = 0;
49 for (auto &uboInput : uboInputs_) {
50 auto &uniform = uboInput.input;
51 // note we need to compute the "aligned" offset for std140 layout
52 GLuint numElements = uniform->numArrayElements() * uniform->numInstances();
53 GLuint baseSize = uniform->inputSize() / numElements;
54
55 // Compute the alignment based on the type
56 GLuint baseAlignment = baseSize;
57 GLuint alignmentCount = 1;
58 if (baseSize == 12) { // vec3
59 baseAlignment = 16;
60 }
61 else if (baseSize == 48) { // mat3
62 baseAlignment = 16;
63 alignmentCount = 3;
64 }
65 else if (baseSize == 64) { // mat4
66 baseAlignment = 16;
67 alignmentCount = 4;
68 }
69 else if (numElements > 1) {
70 baseAlignment = 16;
71 }
72
73 // Align the offset to the required alignment
74 if (requiredSize_ % baseAlignment != 0) {
75 requiredSize_ += baseAlignment - (requiredSize_ % baseAlignment);
76 }
77 uboInput.offset = requiredSize_;
78 if (numElements > 1) {
79 requiredSize_ += baseAlignment * alignmentCount * numElements;
80 } else {
81 requiredSize_ += uniform->inputSize();
82 }
83 }

86void UBO::updateAlignedData(UBO_Input &uboInput) {
87 // the GL specification states that the stride between array elements must be
88 // rounded up to 16 bytes.
89 // this is quite ugly to do element-wise memcpy below, hence non 16-byte aligned
90 // arrays should be avoided in general.
91 auto &in = uboInput.input;
92 auto numElements = in->numArrayElements() * in->numInstances();
93 if (numElements == 1) {
94 return;
95 }
96 auto elementSizeUnaligned = in->valsPerElement() * in->dataTypeBytes();
97 if (elementSizeUnaligned % 16 == 0) {
98 return;
99 }
100 auto elementSizeAligned = elementSizeUnaligned + (16 - elementSizeUnaligned % 16);
101 auto dataSizeAligned = elementSizeAligned * numElements;
102 if (dataSizeAligned != uboInput.alignedSize) {
103 delete[] uboInput.alignedData;
104 uboInput.alignedSize = dataSizeAligned;
105 uboInput.alignedData = new byte[uboInput.alignedSize];
106 }
107 auto clientData = in->mapClientDataRaw(ShaderData::READ);
108 auto *src = clientData.r;
109 auto *dst = uboInput.alignedData;
110 for (unsigned int i = 0; i < numElements; ++i) {
111 memcpy(dst, src, elementSizeUnaligned);
112 src += elementSizeUnaligned;
113 dst += elementSizeAligned;
114 }
115}

117void UBO::update(bool forceUpdate) {

134

136 if (bufferData) {

138 if (forceUpdate || uboInput.input->stamp() != uboInput.lastStamp) {
139 if (!uboInput.input->hasClientData()) {
140 continue;
141 }
142 // copy the data to the buffer.
143 updateAlignedData(uboInput);
144 if (uboInput.alignedData) {
145 memcpy(static_cast<char*>(bufferData) + uboInput.offset,
146 uboInput.alignedData, uboInput.alignedSize);
147 } else {
148 auto mapped = uboInput.input->mapClientDataRaw(ShaderData::READ);
149 memcpy(static_cast<char*>(bufferData) + uboInput.offset,
150 mapped.r,
151 uboInput.input->inputSize());
152 }
153 uboInput.lastStamp = uboInput.input->stamp();
154 }
155 }
156 glUnmapBuffer(GL_UNIFORM_BUFFER);
157 } else {

159 }
160
161 glBindBuffer(GL_UNIFORM_BUFFER, 0);
