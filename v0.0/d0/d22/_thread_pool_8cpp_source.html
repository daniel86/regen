39void ThreadPool::shutdown() {
40 for (Worker *t: workerThreads_) {
41 t->hasTerminateRequest_ = true;
42 }
43 workCV_.notify_all();
44#ifdef KNOWROB_THREADING_DETACH_ON_EXIT
45 auto start = std::chrono::steady_clock::now();
46#endif
47 for (Worker *t: workerThreads_) {
48#ifdef KNOWROB_THREADING_DETACH_ON_EXIT
49 while (!t->isTerminated_) {
50 auto now = std::chrono::steady_clock::now();
51 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - start).count();
52 if (elapsed >= MAX_WORKER_TERMINATE_TIME_MS) {
53 REGEN_WARN("Worker thread does not seem to exit, it will be detached!");
54 // Note: the most reliable way I found to forcefully terminate a thread without causing
55 // a SIGABRT exit of KnowRob is detaching the worker threads that do not want to exit.
56 t->thread_.detach();
57 break;
58 }
59 std::this_thread::sleep_for(std::chrono::milliseconds(100));
60 }
61#endif
62 // note: worker destructor joins the thread
63 delete t;
64 }
65 workerThreads_.clear();
66}

68void ThreadPool::pushWork(const std::shared_ptr<ThreadPool::Runner> &goal, ExceptionHandler exceptionHandler) {
69 {
70 std::lock_guard<std::mutex> scoped_lock(workMutex_);
71 goal->setExceptionHandler(std::move(exceptionHandler));
72 workQueue_.push(goal);
73
74 uint32_t numAliveThreads = workerThreads_.size() - numFinishedThreads_;
75 uint32_t numAvailableThreads = numAliveThreads - numActiveWorker_;
76 if (numAvailableThreads == 0) {
77 // add another thread if max num not reached yet
78 if (workerThreads_.size() < (maxNumThreads_ + numFinishedThreads_)) {
79 workerThreads_.push_back(new Worker(this));
80 }
81 }
82 }
83 // wake up a worker if any is sleeping
84 workCV_.notify_one();
85}

111void ThreadPool::Worker::run() {
112 REGEN_DEBUG("Worker started.");
113 // let the pool do some thread specific initialization
114 if (!threadPool_->initializeWorker()) {
115 REGEN_ERROR("Worker initialization failed.");
116 isTerminated_ = true;
117 threadPool_->numFinishedThreads_ += 1;
118 return;
119 }
120 threadPool_->numActiveWorker_ += 1;
121
122 // loop until the application exits
123 while (!hasTerminateRequest_) {
124 // wait for a claim
125 if (threadPool_->numQueuedWork() == 0) {
126 std::unique_lock<std::mutex> lk(threadPool_->workMutex_);
127 threadPool_->numActiveWorker_ -= 1;
128 if (!hasTerminateRequest_) {
129 threadPool_->workCV_.wait(lk, [this] {
130 return hasTerminateRequest_ || !threadPool_->workQueue_.empty();
131 });
132 }
133 threadPool_->numActiveWorker_ += 1;
134 }
135 if (hasTerminateRequest_) {
136 break;
137 }
138
139 // pop work from queue
140 auto goal = threadPool_->popWork();
141 // do the work
142 if (goal) {
143 goal->runInternal();
144 }
145 }
146
147 isTerminated_ = true;
148 // tell the thread pool that a worker thread exited
149 if (threadPool_->finalizeWorker_) {
150 threadPool_->finalizeWorker_();
151 }
152 // note: counter indicates that there are finished threads in workerThreads_ list.
153 threadPool_->numFinishedThreads_ += 1;
154 threadPool_->numActiveWorker_ -= 1;
155 REGEN_DEBUG("Worker terminated.");
156}
