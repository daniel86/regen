53void Filter::setInput(const ref_ptr<Texture> &input) {
54 set_input(input);
55
56 //GLuint inputSize = min(input_->width(), input_->height());
57 //GLuint bufferSize = scaleFactor_*inputSize;
58 GLuint bufferW = scaleFactor_ * input_->width();
59 GLuint bufferH = scaleFactor_ * input_->height();
60
61 GLuint inputDepth = 1;
62 if (dynamic_cast<Texture3D *>(input_.get())) {
63 inputDepth = (dynamic_cast<Texture3D *>(input_.get()))->depth();
64 }
65
66 // create the render target. As this is the first filter with this target
67 // the attachment point is GL_COLOR_ATTACHMENT0 and only a single texture
68 // is added to the fbo.
69 out_ = ref_ptr<Output>::alloc();
70 out_->fbo_ = ref_ptr<FBO>::alloc(bufferW, bufferH, inputDepth);
71 out_->tex0_ = createTexture();
72
73 // call drawBuffer( GL_COLOR_ATTACHMENT0 )
74 outputAttachment_ = GL_COLOR_ATTACHMENT0;
75 if (drawBufferState_.get()) {
76 disjoinStates(drawBufferState_);
77 }
78 drawBufferState_ = ref_ptr<DrawBufferState>::alloc(out_->fbo_);
79 drawBufferState_->colorBuffers.buffers_.push_back(outputAttachment_);
80 joinStatesFront(drawBufferState_);
81}

83void Filter::setInput(
84 const ref_ptr<Output> &lastOutput, GLenum lastAttachment) {
85 // take last output as input
86 if (lastAttachment == GL_COLOR_ATTACHMENT0) {
87 input_ = lastOutput->tex0_;
88 } else {
89 input_ = lastOutput->tex1_;
90 }
91 set_input(input_);
92
93 out_ = lastOutput;
94 // make sure two textures created for ping pong rendering
95 if (!out_->tex1_.get()) {
96 out_->tex1_ = createTexture();
97 }
98
99 // call drawBuffer( outputAttachment_ )
100 outputAttachment_ = GL_COLOR_ATTACHMENT0 + (GL_COLOR_ATTACHMENT1 - lastAttachment);
101 if (drawBufferState_.get()) {
102 disjoinStates(drawBufferState_);
103 }
104 drawBufferState_ = ref_ptr<DrawBufferState>::alloc(out_->fbo_);
105 drawBufferState_->colorBuffers.buffers_.push_back(outputAttachment_);
106 joinStatesFront(drawBufferState_);
107}

146 if (dynamic_cast<TextureCube *>(input_.get())) {
147 shaderDefine("RENDER_LAYER", "6");

152 shaderDefine("RENDER_TARGET", "2D_ARRAY");

178void FilterSequence::addFilter(const ref_ptr<Filter> &f) {
179 if (filterSequence_.empty()) {
180 f->set_bindInput(bindInput_);
181 f->set_format(format_);
182 f->set_internalFormat(internalFormat_);
183 f->set_pixelType(pixelType_);
184 // no filter was added before, create a new framebuffer
185 f->setInput(input_);
186 } else {
187 ref_ptr<Filter> lastFilter = *filterSequence_.rbegin();
188 // another filter was added before
189 if (math::isApprox(f->scaleFactor(), 1.0)) {
190 // filter does not apply scaling. We gonna ping pong render
191 // to the framebuffer provided by previous filter.
192 f->setInput(lastFilter->output(), lastFilter->outputAttachment());
193 } else {
194 // create a new framebuffer for this filter
195 GLenum last = lastFilter->outputAttachment();
196 if (last == GL_COLOR_ATTACHMENT0) {
197 f->setInput(lastFilter->output()->tex0_);
198 } else {
199 f->setInput(lastFilter->output()->tex1_);
200 }
201 }
202 }
203
204 filterSequence_.push_back(f);
205}

216void FilterSequence::resize() {
217 //GLuint size = min(input_->width(), input_->height());
218 GLuint width = input_->width();
219 GLuint height = input_->height();
220 if (width == lastWidth_ && height == lastHeight_) return;
221
222 FBO *last = nullptr;
223 for (auto & it : filterSequence_) {
224 auto *f = (Filter *) it.get();
225 FBO *fbo = f->output()->fbo_.get();
226
227 if (last != fbo) {
228 width *= f->scaleFactor();
229 height *= f->scaleFactor();
230 fbo->resize(width, height, fbo->depth());
231 }
232
233 last = fbo;
234 }
235
236 lastWidth_ = width;
237 lastHeight_ = height;
238}

240void FilterSequence::enable(RenderState *rs) {
241 State::enable(rs);
242
243 if (filterSequence_.empty()) { return; }
244 // auto-resize if input texture size changed
245 resize();
246
247 if (clearFirstFilter_) {
248 Filter *firstFilter = (Filter *) (*filterSequence_.begin()).get();
249 rs->drawFrameBuffer().push(firstFilter->output()->fbo_->id());
250 rs->clearColor().push(clearColor_);
251 glClear(GL_COLOR_BUFFER_BIT);
252 rs->clearColor().pop();
253 rs->drawFrameBuffer().pop();
254 }
255 for (auto & it : filterSequence_) {
256 auto *f = (Filter *) it.get();
257
258 FBO *fbo = f->output()->fbo_.get();
259 rs->drawFrameBuffer().push(fbo->id());
260 rs->viewport().push(fbo->glViewport());
261 viewport_->setVertex(0, fbo->viewport()->getVertex(0).r);
262 inverseViewport_->setVertex(0, fbo->inverseViewport()->getVertex(0).r);
263
264 f->enable(rs);
265 f->disable(rs);
266
267 rs->viewport().pop();
268 rs->drawFrameBuffer().pop();
269 }
