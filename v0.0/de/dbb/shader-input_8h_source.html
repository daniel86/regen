571 template<class ValueType, class BaseType, GLenum TypeValue>
572 class ShaderInputTyped : public ShaderInput {
573 public:
579 ShaderInputTyped(
580 const std::string &name,
581 GLuint numArrayElements,
582 GLboolean normalize)
583 : ShaderInput(name,
584 TypeValue,
585 sizeof(BaseType),
586 sizeof(ValueType) / sizeof(BaseType),
587 numArrayElements, normalize) {}
588
592 virtual std::istream &operator<<(std::istream &in) {
593 ValueType value;
594 in >> value;
595 setUniformData(value);
596 return in;
597 }
598
602 void setUniformData(const ValueType &data) { setUniformUntyped((const byte *) &data); }
603
607 auto uniformData() { return getVertex(0); }
608
614 void setVertex(GLuint i, const ValueType &val) {
615 auto mapped = mapClientData<ValueType>(ShaderData::WRITE | ShaderData::INDEX);
616 mapped.w[i] = val;
617 }
618
623 ShaderVertex_ro<ValueType> getVertex(GLuint i) const {
624 return mapClientVertex<ValueType>(ShaderData::READ, i);
625 }
626
632 void setVertexClamped(GLuint i, const ValueType &val) { setVertex(numElements_ > i ? i : 0, val); }
633
639 auto getVertexClamped(GLuint i) const { return getVertex(numElements_ > i ? i : 0); }
640
644 virtual std::ostream &operator>>(std::ostream &out) const {
645 auto x = getVertex(0);
646 auto &v = x.r;
647 return out << v;
648 }
649
653 void write(std::ostream &out) const override {
654 auto x = getVertex(0);
655 auto &v = x.r;
656 out << v;
657 //out << getVertex(0).v;
658 }
659 };
