92float AnimalController::getHeight(const Vec2f &pos) {
93 float height = floorHeight_;
94 if (heightMap_.get()) {
95 auto mapCenter = (heightMapBounds_.max + heightMapBounds_.min) * 0.5f;
96 auto mapSize = heightMapBounds_.max - heightMapBounds_.min;
97 // compute UV for height map sampling
98 auto uv = pos - mapCenter;
99 uv /= mapSize * 0.5f;
100 uv += Vec2f(0.5f);
101#ifdef SMOOTH_HEIGHT
102 auto texelSize = Vec2f(1.0f / heightMap_->width(), 1.0f / heightMap_->height());
103 auto regionTS = texelSize*8.0f;
104 auto mapValue = heightMap_->sampleAverage<float>(uv, regionTS, heightMap_->textureData());
105#else
106 auto mapValue = heightMap_->sampleLinear(uv, heightMap_->textureData(), 1);
107#endif
108 mapValue *= heightMapFactor_;
109 // increase by small bias to avoid intersection with the floor
110 mapValue += 0.02f;
111 height += mapValue;
112 }
113 return height;
114}

137void AnimalController::activateMovement() {
138 // pick a random point in the territory
139 Vec2f target = (territoryBounds_.max - territoryBounds_.min) * Vec2f(
140 static_cast<float>(rand()) / static_cast<float>(RAND_MAX),
141 static_cast<float>(rand()) / static_cast<float>(RAND_MAX)
142 ) + territoryBounds_.min;
143 // pick a random orientation vector
144 auto orientation = Vec3f(
145 currentDir_.x,
146 static_cast<float>((static_cast<double>(rand()) / RAND_MAX - 0.5)*M_PI),
147 currentDir_.z
148 );
149
150 bezierPath_.p0 = Vec2f(currentPos_.x, currentPos_.z);
151 bezierPath_.p3 = target;
152 // compute control points using Euler angles
153 float directDistance = (bezierPath_.p0 - bezierPath_.p3).length() * 0.75f;
154 // add base orientation
155 float angle_rad1 = currentDir_.y + baseOrientation_;
156 float angle_rad2 = orientation.y + baseOrientation_;
157 bezierPath_.p1 = bezierPath_.p0 + Vec2f(cos(angle_rad1), sin(angle_rad1)) * directDistance;
158 bezierPath_.p2 = bezierPath_.p3 + Vec2f(cos(angle_rad2), sin(angle_rad2)) * directDistance;
159 // make sure control points are within the map bounds
160 intersectWithBounds(bezierPath_.p1, bezierPath_.p0, territoryBounds_);
161 intersectWithBounds(bezierPath_.p2, bezierPath_.p3, territoryBounds_);
162 float bezierLength = bezierPath_.length1();
163
164 // compute dt based on distance and speed
165 float dt = bezierLength / (behavior_==BEHAVIOR_RUN ? runSpeed_ : walkSpeed_);
166 // set the target position
167 setTarget(
168 Vec3f(target.x, getHeight(target), target.y),
169 orientation,
170 dt);
171 activateRandom();
172}
