60static Vec3f computeSphereTangent(const Vec3f &v) {
61 Vec3f vAbs = Vec3f(abs(v.x), abs(v.y), abs(v.z));
62 Vec3f v_;
63 if (1.0f - v.z < std::numeric_limits<float>::epsilon()) {
64 // there is a singularity at the back pole
65 v_ = Vec3f(1.0, 0.0, 0.0);
66 } else if (vAbs.x < vAbs.y && vAbs.x < vAbs.z) {
67 v_ = Vec3f(0.0, -v.z, v.y);
68 } else if (vAbs.y < vAbs.x && vAbs.y < vAbs.z) {
69 v_ = Vec3f(-v.z, 0, v.x);
70 } else {
71 v_ = Vec3f(-v.y, v.x, 0);
72 }
73 v_.normalize();
74 return v.cross(v_);
75}

77void pushVertex(
78 GLuint vertexIndex,
79 GLdouble u,
80 GLdouble v,
81 const Sphere::Config &cfg,
82 ShaderData_rw<Vec3f> &pos,
83 ShaderData_rw<Vec3f> &nor,
84 ShaderData_rw<Vec4f> &tan,
85 ShaderData_rw<Vec2f> &texco) {
86 GLdouble r = std::sin(M_PI * v);
87 pos.w[vertexIndex] = Vec3f(
88 static_cast<float>(r * std::cos(2.0 * M_PI * u)),
89 static_cast<float>(r * std::sin(2.0 * M_PI * u)),
90 static_cast<float>(std::cos(M_PI * v))
91 );
92 if (cfg.isNormalRequired) {
93 nor.w[vertexIndex] = pos.w[vertexIndex];
94 }
95 if (cfg.isTangentRequired) {
96 Vec3f t = computeSphereTangent(pos.w[vertexIndex]);
97 tan.w[vertexIndex] = Vec4f(t.x, t.y, t.z, 1.0);
98 }
99 if (texco.w) {
100 texco.w[vertexIndex] = Vec2f(
101 static_cast<float>(u) * cfg.texcoScale.x,
102 static_cast<float>(v) * cfg.texcoScale.y);
103 }
104 pos.w[vertexIndex] *= cfg.posScale * 0.5;
105}

107void Sphere::generateLODLevel(const Config &cfg,
108 GLuint lodLevel,
109 GLuint vertexOffset,
110 GLuint indexOffset) {
111 // map client data for writing
112 auto indices = indices_->mapClientData<GLuint>(ShaderData::WRITE);
113 auto v_pos = pos_->mapClientData<Vec3f>(ShaderData::WRITE);
114 auto v_nor = (cfg.isNormalRequired ?
115 nor_->mapClientData<Vec3f>(ShaderData::WRITE) :
116 ShaderData_rw<Vec3f>::nullData());
117 auto v_tan = (cfg.isTangentRequired ?
118 tan_->mapClientData<Vec4f>(ShaderData::WRITE) :
119 ShaderData_rw<Vec4f>::nullData());
120 auto v_texco = (texco_.get() ?
121 texco_->mapClientData<Vec2f>(ShaderData::WRITE) :
122 ShaderData_rw<Vec2f>::nullData());
123
124 GLdouble stepSizeInv = 1.0 / (GLdouble) lodLevel;
125 GLuint vertexIndex = vertexOffset, faceIndex = indexOffset / 6;
126
127 for (GLuint i = 0; i < lodLevel; i++) {
128 for (GLuint j = 0; j < lodLevel; j++) {
129 GLdouble u0 = (GLdouble) i * stepSizeInv;
130 GLdouble u1 = (GLdouble) (i + 1) * stepSizeInv;
131 GLdouble v0 = (GLdouble) j * stepSizeInv;
132 GLdouble v1 = (GLdouble) (j + 1) * stepSizeInv;
133
134 if (cfg.isHalfSphere && u0 < 0.5) continue;
135
136 // create two triangles for each quad
137 GLuint index = (faceIndex++) * 6;
138 indices.w[index + 0] = vertexIndex + 0;
139 indices.w[index + 1] = vertexIndex + 1;
140 indices.w[index + 2] = vertexIndex + 2;
141 indices.w[index + 3] = vertexIndex + 2;
142 indices.w[index + 4] = vertexIndex + 1;
143 indices.w[index + 5] = vertexIndex + 3;
144
145 // they are made of 4 vertices
146 pushVertex(vertexIndex++, u1, v1, cfg, v_pos, v_nor, v_tan, v_texco);
147 pushVertex(vertexIndex++, u1, v0, cfg, v_pos, v_nor, v_tan, v_texco);
148 pushVertex(vertexIndex++, u0, v1, cfg, v_pos, v_nor, v_tan, v_texco);
149 pushVertex(vertexIndex++, u0, v0, cfg, v_pos, v_nor, v_tan, v_texco);
150 }
151 }

154void Sphere::updateAttributes(const Config &cfg) {
155 std::vector<GLuint> LODs;
156 GLuint vertexOffset = 0;
157 GLuint indexOffset = 0;
158 for (auto &lod: cfg.levelOfDetails) {
159 GLuint lodLevel = 4 + lod * lod;
160 LODs.push_back(lodLevel);
161 GLuint numFaces;
162 if (cfg.isHalfSphere) {
163 numFaces = lodLevel * lodLevel;
164 } else {
165 numFaces = 2 * lodLevel * lodLevel;
166 }
167 auto &x = meshLODs_.emplace_back();
168 x.numVertices = numFaces * 2;
169 x.numIndices = numFaces * 3;
170 x.vertexOffset = vertexOffset;
171 x.indexOffset = indexOffset;
172 vertexOffset += x.numVertices;
173 indexOffset += x.numIndices;
174 }
175 GLuint numVertices = vertexOffset;
176 GLuint numIndices = indexOffset;
177
178 // allocate attributes
179 pos_->setVertexData(numVertices);
180 if (cfg.isNormalRequired) {
181 nor_->setVertexData(numVertices);
182 }
183 if (cfg.isTangentRequired) {
184 tan_->setVertexData(numVertices);
185 }
186 TexcoMode texcoMode = cfg.texcoMode;
187 if (cfg.isTangentRequired && cfg.texcoMode == TEXCO_MODE_NONE) {
188 texcoMode = TEXCO_MODE_UV;
189 }
190 if (texcoMode == TEXCO_MODE_UV) {
191 texco_ = ref_ptr<ShaderInput2f>::alloc("texco0");
192 texco_->setVertexData(numVertices);
193 }
194 indices_->setVertexData(numIndices);
195
196 for (auto i = 0u; i < LODs.size(); ++i) {
197 generateLODLevel(cfg,
198 LODs[i],
199 meshLODs_[i].vertexOffset,
200 meshLODs_[i].indexOffset);
201 }
202
203 begin(ShaderInputContainer::INTERLEAVED);
204 auto indexRef = setIndices(indices_, numVertices);
205 setInput(pos_);
206 if (cfg.isNormalRequired) {
207 setInput(nor_);
208 }
209 if (cfg.isTangentRequired) {
210 setInput(tan_);
211 }
212 if (texcoMode != TEXCO_MODE_NONE) {
213 setInput(texco_);
214 }
215 end();
216
217 for (auto &x: meshLODs_) {
218 // add the index buffer offset (in number of bytes)
219 x.indexOffset = indexRef->address() + x.indexOffset * sizeof(GLuint);
220 }
221 activateLOD(0);
222 minPosition_ = -cfg.posScale * 0.5f;
223 maxPosition_ = cfg.posScale * 0.5f;
224}

242void SphereSprite::updateAttributes(const Config &cfg) {
243 ref_ptr<ShaderInput1f> radiusIn = ref_ptr<ShaderInput1f>::alloc("sphereRadius");
244 radiusIn->setVertexData(cfg.sphereCount);
245 auto mappedRadius = radiusIn->mapClientData<float>(ShaderData::WRITE);
246
247 ref_ptr<ShaderInput3f> positionIn = ref_ptr<ShaderInput3f>::alloc(ATTRIBUTE_NAME_POS);
248 positionIn->setVertexData(cfg.sphereCount);
249 auto mappedPosition = positionIn->mapClientData<Vec3f>(ShaderData::WRITE);
250
251 minPosition_ = Vec3f(999999.0f);
252 maxPosition_ = Vec3f(-999999.0f);
253 Vec3f v;
254 for (GLuint i = 0; i < cfg.sphereCount; ++i) {
255 mappedRadius.w[i] = cfg.radius[i];
256 mappedPosition.w[i] = cfg.position[i];
257
258 v = cfg.position[i] - Vec3f(cfg.radius[i]);
259 minPosition_.setMin(v);
260 v = cfg.position[i] + Vec3f(cfg.radius[i]);
261 maxPosition_.setMax(v);
262 }
263
264 mappedRadius.unmap();
265 mappedPosition.unmap();
266
267 begin(ShaderInputContainer::INTERLEAVED);
268 setInput(radiusIn);
269 setInput(positionIn);
270 end();
271}
