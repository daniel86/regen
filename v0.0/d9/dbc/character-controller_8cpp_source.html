163void CharacterController::applyStep(GLfloat dt, const Vec3f &offset) {
164 // Make a step. This is called each frame.
165
166 if (dt <= std::numeric_limits<float>::epsilon()) {
167 return;
168 }
169
170 if (!btController_.get()) {
171 if (!initializePhysics()) {
172 return;
173 }
174 }
175
176 // synchronize the ghost object's transform with the OpenGL transform
177 btTransform ghostTransform = btController_->getGhostObject()->getWorldTransform();
178 ghostTransform.getOpenGLMatrix((btScalar *) &matVal_);
179 // decrease the y position by the collision height, else
180 // the character would float above the ground
181 GLfloat heightAdjust = btCollisionHeight_ * 0.5f + btCollisionRadius_;
182 matVal_.x[13] -= heightAdjust;
183
184 btVector3 btVelocity = btController_->getLinearVelocity();
185 btVelocity.setX(0.0);
186 btVelocity.setZ(0.0);
187 GLboolean isMoving = isMoving_;
188
189 // Apply the platform velocity to the character controller
190 if (btPlatform_) {
191 btVector3 currentPlatformPos = btPlatform_->getWorldTransform().getOrigin();
192 btVector3 platformVelocity = currentPlatformPos - previousPlatformPos_;
193 if (platformVelocity.length() > 0.0001) {
194 // not sure about the magic number 16.8, might
195 // be related to some hacks in the kinematics controller :/
196 btVelocity += 16.8 * platformVelocity / dt;
197 isMoving = GL_TRUE;
198 }
199 previousPlatformPos_ = currentPlatformPos;
200 }
201
202 if (!isMoving_) {
203 if (btIsMoving_ || isMoving) {
204 btVelocity.setY(btController_->getLinearVelocity().getY());
205 btController_->setLinearVelocity(btVelocity);
206 btIsMoving_ = GL_FALSE;
207 }
208 } else {
209 // Apply the step to the character controller
210 btVelocity += btVector3(offset.x, offset.y, offset.z)/dt;
211 btVelocity.setY(btController_->getLinearVelocity().getY());
212 btController_->setLinearVelocity(btVelocity);
213 btIsMoving_ = GL_TRUE;
214 }
