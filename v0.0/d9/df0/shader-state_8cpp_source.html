33void ShaderState::loadStage(
34 const std::map<std::string, std::string> &shaderConfig,
35 const std::string &effectName,
36 std::map<GLenum, std::string> &code,
37 GLenum stage) {
38 std::string stageName = glenum::glslStageName(stage);
39 std::string effectKey = REGEN_STRING(effectName << "." << glenum::glslStagePrefix(stage));
40 std::string ignoreKey = REGEN_STRING("IGNORE_" << stageName);
41
42 auto it = shaderConfig.find(ignoreKey);
43 if (it != shaderConfig.end() && it->second == "TRUE") { return; }
44
45 code[stage] = Includer::get().include(effectKey);
46 // failed to include ?
47 if (code[stage].empty()) { code.erase(stage); }
48}

50GLboolean ShaderState::createShader(const StateConfig &cfg, const std::string &shaderKey) {
51 std::map<GLenum, std::string> unprocessedCode;
52 for (GLint i = 0; i < glenum::glslStageCount(); ++i) {
53 loadStage(cfg.defines_, shaderKey, unprocessedCode, glenum::glslStages()[i]);
54 }
55 if (unprocessedCode.empty()) {
56 REGEN_ERROR("Failed to load shader with key '" << shaderKey << "'");
57 return GL_FALSE;
58 }
59 return createShader(cfg, unprocessedCode);
60}

74GLboolean ShaderState::createShader(const StateConfig &cfg, const std::map<GLenum, std::string> &unprocessedCode) {
75 const std::list<NamedShaderInput> specifiedInput = cfg.inputs_;
76 const std::map<std::string, ref_ptr<Texture> > &textures = cfg.textures_;
77 const std::map<std::string, std::string> &shaderConfig = cfg.defines_;
78 const std::map<std::string, std::string> &shaderFunctions = cfg.functions_;
79 std::map<GLenum, std::string> processedCode;
80
81 PreProcessorConfig preProcessCfg(cfg.version(),
82 unprocessedCode, shaderConfig,
83 shaderFunctions, specifiedInput);
84 Shader::preProcess(processedCode, preProcessCfg);
85 shader_ = ref_ptr<Shader>::alloc(processedCode);
86 // setup transform feedback attributes
87 shader_->setTransformFeedback(cfg.feedbackAttributes_, cfg.feedbackMode_, cfg.feedbackStage_);
88
89 if (!shader_->compile()) {
90 REGEN_ERROR("Shader failed to compiled.");
91 for (auto it = processedCode.begin(); it != processedCode.end(); ++it) {
92 REGEN_DEBUG("Shader code failed to compile:\n" << it->second);
93 }
94 return GL_FALSE;
95 }
96 if (!shader_->link()) {
97 REGEN_ERROR("Shader failed to link.");
98 for (auto it = processedCode.begin(); it != processedCode.end(); ++it) {
99 REGEN_DEBUG("Shader code failed to link:\n" << it->second);
100 }
101 } else {
102 set_isHidden(false);
103 }
104
105 shader_->setInputs(specifiedInput);
106 for (auto it = textures.begin(); it != textures.end(); ++it) {
107 shader_->setTexture(it->second, it->first);
108 }
109
110 return GL_TRUE;
111}
