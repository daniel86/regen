116 inline void setEuler(GLfloat fPitch, GLfloat fYaw, GLfloat fRoll) {
117 const GLfloat fSinPitch(sin(fPitch * 0.5f));
118 const GLfloat fCosPitch(cos(fPitch * 0.5f));
119 const GLfloat fSinYaw(sin(fYaw * 0.5f));
120 const GLfloat fCosYaw(cos(fYaw * 0.5f));
121 const GLfloat fSinRoll(sin(fRoll * 0.5f));
122 const GLfloat fCosRoll(cos(fRoll * 0.5f));
123 const GLfloat fCosPitchCosYaw(fCosPitch * fCosYaw);
124 const GLfloat fSinPitchSinYaw(fSinPitch * fSinYaw);
125 x = fSinRoll * fCosPitchCosYaw - fCosRoll * fSinPitchSinYaw;
126 y = fCosRoll * fSinPitch * fCosYaw + fSinRoll * fCosPitch * fSinYaw;
127 z = fCosRoll * fCosPitch * fSinYaw - fSinRoll * fSinPitch * fCosYaw;
128 w = fCosRoll * fCosPitchCosYaw + fSinRoll * fSinPitchSinYaw;
129 }

149 inline void setQuaternion(const Quaternion &normalized) {
150 x = normalized.x;
151 y = normalized.y;
152 z = normalized.z;
153
154 const GLfloat t = 1.0f - (x * x) - (y * y) - (z * z);
155 if (t < 0.0f) w = 0.0f;
156 else w = sqrt(t);
157 }

162 inline Mat4f calculateMatrix() const {
163 Mat4f mat4x4 = Mat4f::identity();
164 mat4x4.x[0] = -2.0f * (y * y + z * z) + 1.0f;
165 mat4x4.x[1] = 2.0f * (x * y - z * w);
166 mat4x4.x[2] = 2.0f * (x * z + y * w);
167 mat4x4.x[4] = 2.0f * (x * y + z * w);
168 mat4x4.x[5] = -2.0f * (x * x + z * z) + 1.0f;
169 mat4x4.x[6] = 2.0f * (y * z - x * w);
170 mat4x4.x[8] = 2.0f * (x * z - y * w);
171 mat4x4.x[9] = 2.0f * (y * z + x * w);
172 mat4x4.x[10] = -2.0f * (x * x + y * y) + 1.0f;
173 return mat4x4;
174 }

183 inline void interpolate(
184 const Quaternion &pStart,
185 const Quaternion &pEnd,
186 GLfloat pFactor) {
187 // calc cosine theta
188 GLfloat cosom =
189 pStart.x * pEnd.x +
190 pStart.y * pEnd.y +
191 pStart.z * pEnd.z +
192 pStart.w * pEnd.w;
193
194 // adjust signs (if necessary)
195 Quaternion end = pEnd;
196 if (cosom < 0.0f) {
197 cosom = -cosom;
198 end.x = -end.x; // Reverse all signs
199 end.y = -end.y;
200 end.z = -end.z;
201 end.w = -end.w;
202 }
203
204 // Calculate coefficients
205 GLfloat sclp, sclq;
206 if ((1.0f - cosom) > 0.0001f) // 0.0001 -> some epsillon
207 {
208 // Standard case (slerp)
209 float omega, sinom;
210 omega = acos(cosom); // extract theta from dot product's cos theta
211 sinom = sin(omega);
212 sclp = sin((1.0f - pFactor) * omega) / sinom;
213 sclq = sin(pFactor * omega) / sinom;
214 } else {
215 // Very close, do linear interp (because it's faster)
216 sclp = 1.0f - pFactor;
217 sclq = pFactor;
218 }
219
220 x = sclp * pStart.x + sclq * end.x;
221 y = sclp * pStart.y + sclq * end.y;
222 z = sclp * pStart.z + sclq * end.z;
223 w = sclp * pStart.w + sclq * end.w;
224 }

267 inline void normalize() {
268 const GLfloat mag = sqrt(x * x + y * y + z * z + w * w);
269 if (mag) {
270 const GLfloat invMag = 1.0f / mag;
271 x *= invMag;
272 y *= invMag;
273 z *= invMag;
274 w *= invMag;
275 }
276 }

291 inline Vec3f rotate(const Vec3f &v) {
292 Quaternion q2(0.0f, v.x, v.y, v.z);
293 Quaternion q = *this;
294 q.conjugate();
295 q = q * q2 * (*this);
296 return Vec3f(q.x, q.y, q.z);
297 }
