27void OBB::updateOBB() {
28 Vec3f offset = basePosition_;
29 if (modelOffset_.get()) {
30 offset += modelOffset_->getVertex(modelOffsetIndex_).r;
31 }
32 // compute axes of the OBB based on the model transformation
33 auto scaling = Vec3f::one();
34 if (transform_.get()) {
35 auto tf = transform_->get()->getVertex(transformIndex_);
36 boxAxes_[0] = (tf.r ^ Vec4f(Vec3f::right(), 0.0f)).xyz_();
37 boxAxes_[1] = (tf.r ^ Vec4f(Vec3f::up(), 0.0f)).xyz_();
38 boxAxes_[2] = (tf.r ^ Vec4f(Vec3f::front(), 0.0f)).xyz_();
39 // transform base position offset
40 offset = (tf.r ^ Vec4f(offset, 0.0f)).xyz_();
41 offset += tf.r.position();
42 } else {
43 boxAxes_[0] = Vec3f::right();
44 boxAxes_[1] = Vec3f::up();
45 boxAxes_[2] = Vec3f::front();
46 }
47
48 // compute vertices of the OBB
49 auto halfSize = (bounds_.max - bounds_.min) * scaling * 0.5f;
50 Vec3f x = boxAxes_[0] * halfSize.x;
51 Vec3f y = boxAxes_[1] * halfSize.y;
52 Vec3f z = boxAxes_[2] * halfSize.z;
53 vertices_[0] = offset - x - y - z;
54 vertices_[1] = offset - x - y + z;
55 vertices_[2] = offset - x + y - z;
56 vertices_[3] = offset - x + y + z;
57 vertices_[4] = offset + x - y - z;
58 vertices_[5] = offset + x - y + z;
59 vertices_[6] = offset + x + y - z;
60 vertices_[7] = offset + x + y + z;

63Vec3f OBB::closestPointOnSurface(const Vec3f &point) const {
64 auto halfSize = (bounds_.max - bounds_.min) * 0.5f;
65 const GLfloat *halfSizes = &halfSize.x;
66 const Vec3f *axes = boxAxes();
67 Vec3f obbCenter = getCenterPosition();
68 Vec3f d = point - obbCenter;
69 Vec3f closest = obbCenter;
70
71 for (int i = 0; i < 3; ++i) {
72 float distance = d.dot(axes[i]);
73 if (distance > halfSizes[i]) distance = halfSizes[i];
74 if (distance < -halfSizes[i]) distance = -halfSizes[i];
75 closest = closest + axes[i] * distance;
76 }
77
78 return closest;
79}
