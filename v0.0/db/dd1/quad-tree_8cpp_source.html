120bool QuadTree::insert(Node *node, Item *shape, bool allowSubdivision) { // NOLINT(no-recursion)
121 if (!node->intersects(shape->projection)) {
122 return false;
123 } else {
124 // only subdivide if the node is larger than the shape
125 auto nodeSize = node->bounds.max - node->bounds.min;
126 auto shapeSize = shape->projection.bounds().max - shape->projection.bounds().min;
127 bool isNodeLargeEnough = (nodeSize.x > 2.0*shapeSize.x && nodeSize.y > 2.0*shapeSize.y);
128
129 return insert1(node, shape, isNodeLargeEnough && allowSubdivision);
130 }

133bool QuadTree::insert1(Node *node, Item *newShape, bool allowSubdivision) { // NOLINT(no-recursion)
134 if (node->isLeaf()) {
135 // the node does not have child nodes (yet).
136 // the shape can be added to the node in three cases:
137 // 1. the node has still not exceeded the maximum number of shapes per node
138 // 2. the node has reached the minimum size and cannot be subdivided further
139 // 3. the node was just created by subdividing a parent node
140 if (!allowSubdivision ||
141 node->shapes.size() < QUAD_TREE_SUBDIVIDE_THRESHOLD ||
142 node->bounds.size() < minNodeSize_) {
143 node->shapes.push_back(newShape);
144 newShape->nodes.push_back(node);
145 return true;
146 } else {
147 // split the node into four children
148 subdivide(node);
149 bool inserted = false;
150 // remove the subdivided node from the list of nodes of the existing shapes
151 for (const auto &existingShape: node->shapes) {
152 existingShape->removeNode(node);
153 }
154 for (auto &child: node->children) {
155 // reinsert the existing shapes into the new children nodes
156 for (const auto &existingShape: node->shapes) {
157 insert(child, existingShape, true);
158 }
159 // also insert the new shape
160 inserted = insert(child, newShape, true) || inserted;
161 }
162 // subdivided node does not contain shapes anymore
163 node->shapes.clear();
164 return inserted;
165 }
166 } else {
167 // the node has child nodes, must insert into one of them
168 // note: a shape can be inserted into multiple nodes, so we allways need to check all children
169 // (at least on the next level)
170 bool inserted = false;
171 for (auto &child: node->children) {
172 inserted = insert(child, newShape, allowSubdivision) || inserted;
173 }
174 return inserted;
175 }

204void QuadTree::collapse(Node *node) {
205 if (!node->parent) {
206 return;
207 }
208 auto *parent = node->parent;
209 auto &firstShapes = parent->children[0]->shapes;
210 for (auto &child : parent->children) {
211 if (!child->isLeaf()) {
212 // at least one child is not a leaf -> no collapse
213 return;
214 }
215 }
216 //if (parent->children[0]->shapes.size() > QUAD_TREE_COLLAPSE_THRESHOLD) {
217 // no collapse
218 //return;
219 //}
220 for (int i = 1; i < 4; i++) {
221 if (parent->children[i]->shapes.size() != firstShapes.size()) {
222 // unequal number of shapes -> no collapse
223 return;
224 }
225 for (auto &shape: parent->children[i]->shapes) {
226 if (std::find(firstShapes.begin(), firstShapes.end(), shape) == firstShapes.end()) {
227 // not all children contain the same set of shapes -> no collapse
228 return;
229 }
230 }
231 }
232 // finally, collapse the node...
233 for (auto &shape: firstShapes) {
234 // remove the children nodes from the shape's list of nodes
235 for (int i = 0; i < 4; i++) {
236 auto it = std::find(shape->nodes.begin(), shape->nodes.end(), parent->children[i]);
237 shape->nodes.erase(it);
238 }
239 // add the parent node to the shape's list of nodes
240 shape->nodes.push_back(parent);
241 // add the shape to the parent node
242 parent->shapes.push_back(shape);
243 }
244 // free all children nodes
245 for (int i = 0; i < 4; i++) {
246 freeNode(parent->children[i]);
247 parent->children[i] = nullptr;
248 }
249 // continue collapsing the parent node
250 collapse(parent);
251}

253void QuadTree::subdivide(Node *node) {
254 // Subdivide the node into four children.
255 auto center = (node->bounds.min + node->bounds.max) * 0.5f;
256 Node *child;
257 // bottom-left
258 child = createNode(node->bounds.min, center);
259 child->parent = node;
260 node->children[0] = child;
261 // bottom-right
262 child = createNode(Vec2f(center.x, node->bounds.min.y), Vec2f(node->bounds.max.x, center.y));
263 child->parent = node;
264 node->children[1] = child;
265 // top-right
266 child = createNode(center, node->bounds.max);
267 child->parent = node;
268 node->children[2] = child;
269 // top-left
270 child = createNode(Vec2f(node->bounds.min.x, center.y), Vec2f(center.x, node->bounds.max.y));
271 child->parent = node;
272 node->children[3] = child;
273}

309std::pair<float, float> project(const Bounds<Vec2f> &b, const Vec2f &axis) {
310 std::array<float, 4> projections = {
311 b.min.dot(axis),
312 Vec2f(b.max.x, b.min.y).dot(axis),
313 Vec2f(b.min.x, b.max.y).dot(axis),
314 b.max.dot(axis)
315 };
316
317 auto [minIt, maxIt] = std::minmax_element(projections.begin(), projections.end());
318 return {*minIt, *maxIt};

321bool QuadTree::Node::intersects(const OrthogonalProjection &projection) const {
322 switch (projection.type) {
323 case OrthogonalProjection::Type::CIRCLE: {
324 const auto &radiusSqr = projection.points[1].x; // = radius * radius
325 const auto &center = projection.points[0];
326 // Calculate the squared distance from the circle's center to the AABB
327 float sqDist = 0.0f;
328 if (center.x < bounds.min.x) {
329 sqDist += (bounds.min.x - center.x) * (bounds.min.x - center.x);
330 } else if (center.x > bounds.max.x) {
331 sqDist += (center.x - bounds.max.x) * (center.x - bounds.max.x);
332 }
333 if (center.y < bounds.min.y) {
334 sqDist += (bounds.min.y - center.y) * (bounds.min.y - center.y);
335 } else if (center.y > bounds.max.y) {
336 sqDist += (center.y - bounds.max.y) * (center.y - bounds.max.y);
337 }
338 return sqDist < radiusSqr;
339 }
340 case OrthogonalProjection::Type::TRIANGLE:
341 case OrthogonalProjection::Type::RECTANGLE:
342 // Check for separation along the axes of the shape and the axis-aligned quad
343 for (const auto &axis: projection.axes) {
344 auto [minA, maxA] = project(bounds, axis.dir);
345 if (maxA < axis.min || axis.max < minA) {
346 return false;
347 }
348 }
349 return true;
350 }
351 return false;
352}

547void QuadTree::update(float dt) {
548 static auto maxFloat = Vec2f(std::numeric_limits<float>::lowest());
549 static auto minFloat = Vec2f(std::numeric_limits<float>::max());
550 bool hasChanged;
551
552 changedItems_.clear();
553 newBounds_.min = minFloat;
554 newBounds_.max = maxFloat;
555#ifdef QUAD_TREE_EVER_GROWING
556 if (root_ != nullptr) {
557 // never shrink the root node
558 newBounds_.extend(root_->bounds);
559 }
560#endif
561
562 // go through all items and update their geometry and transform, and the new bounds
563 for (const auto &it: items_) {
564 auto &item = it.second;
565 hasChanged = item->shape->updateGeometry();
566 hasChanged = item->shape->updateTransform(hasChanged) || hasChanged;
567 if (hasChanged) {
568 item->projection.update(*item->shape.get());
569 changedItems_.push_back(item);
570 }
571 newBounds_.extend(item->projection.bounds());
572 }
573 // do the same for any additional items added to the tree
574 for (const auto &item: newItems_) {
575 hasChanged = item->shape->updateGeometry();
576 hasChanged = item->shape->updateTransform(hasChanged) || hasChanged;
577 if (hasChanged) {
578 item->projection = OrthogonalProjection(*item->shape.get());
579 }
580 newBounds_.extend(item->projection.bounds());
581 }
582#ifdef QUAD_TREE_SQUARED
583 // make the bounds square
584 newBounds_.min.x = std::min(newBounds_.min.x, newBounds_.min.y);
585 newBounds_.min.y = newBounds_.min.x;
586 newBounds_.max.x = std::max(newBounds_.max.x, newBounds_.max.y);
587 newBounds_.max.y = newBounds_.max.x;
588#endif
589 // if bounds have changed, re-initialize the tree
590 auto reInit = (root_ == nullptr || newBounds_ != root_->bounds);
591 if (reInit) {
592 // free the root node and start all over
593 if(root_) freeNode(root_);
594 root_ = createNode(newBounds_.min, newBounds_.max);
595
596 for (auto &it: items_) {
597 auto &item = it.second;
598 item->nodes.clear();
599 insert1(root_, item, true);
600 }
601 }
602 // else remove/insert the changed items
603 else {
604 for (auto item: changedItems_) {
605 removeFromNodes(item);
606 insert1(root_, item, true);
607 }
608 }
609 // finally insert the new items
610 for (auto item: newItems_) {
611 if(insert1(root_, item, true)) {
612 items_[item->shape.get()] = item;
613 } else {
614 delete item;
615 REGEN_WARN("Failed to insert shape into quad tree. This should not happen!");
616 }
617 }
618 newItems_.clear();
619
620 // make the visibility computations
621 updateVisibility();
622}
