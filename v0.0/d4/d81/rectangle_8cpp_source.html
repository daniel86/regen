71void Rectangle::generateLODLevel(const Config &cfg,
72 const Tessellation &tessellation,
73 const Mat4f &rotMat,
74 GLuint vertexOffset,
75 GLuint indexOffset) {
76 // map client data for writing
77 auto indices = indices_->mapClientData<GLuint>(ShaderData::WRITE);
78 auto v_pos = pos_->mapClientData<Vec3f>(ShaderData::WRITE);
79 auto v_nor = (cfg.isNormalRequired ?
80 nor_->mapClientData<Vec3f>(ShaderData::WRITE) :
81 ShaderData_rw<Vec3f>::nullData());
82 auto v_tan = (cfg.isTangentRequired ?
83 tan_->mapClientData<Vec4f>(ShaderData::WRITE) :
84 ShaderData_rw<Vec4f>::nullData());
85 auto v_texco = (cfg.isTexcoRequired ?
86 texco_->mapClientData<Vec2f>(ShaderData::WRITE) :
87 ShaderData_rw<Vec2f>::nullData());
88
89 GLuint nextIndex = indexOffset;
90 for (auto &tessFace: tessellation.outputFaces) {
91 indices.w[nextIndex++] = vertexOffset + tessFace.v1;
92 indices.w[nextIndex++] = vertexOffset + tessFace.v2;
93 indices.w[nextIndex++] = vertexOffset + tessFace.v3;
94 }
95
96 GLuint triIndices[3];
97 Vec3f triVertices[3];
98 Vec2f triTexco[3];
99 Vec3f normal = rotMat.transformVector(Vec3f(0.0, -1.0, 0.0));
100 Vec3f startPos;
101 if (cfg.centerAtOrigin) {
102 startPos = Vec3f(-cfg.posScale.x * 0.5f, 0.0f, -cfg.posScale.z * 0.5f);
103 } else {
104 startPos = Vec3f(0.0f, 0.0f, 0.0f);
105 }
106
107 for (GLuint faceIndex = 0; faceIndex < tessellation.outputFaces.size(); ++faceIndex) {
108 auto &face = tessellation.outputFaces[faceIndex];
109 GLuint faceVertIndex = 0;
110
111 for (auto &tessIndex: {face.v1, face.v2, face.v3}) {
112 auto &vertex = tessellation.vertices[tessIndex];
113 auto vertexIndex = vertexOffset + tessIndex;
114 triIndices[faceVertIndex] = vertexIndex;
115
116 Vec3f pos = rotMat.transformVector(
117 cfg.posScale * vertex + startPos) + cfg.translation;
118 v_pos.w[vertexIndex] = pos;
119 minPosition_.setMin(pos);
120 maxPosition_.setMax(pos);
121 if (cfg.isNormalRequired) {
122 v_nor.w[vertexIndex] = normal;
123 }
124 if (cfg.isTexcoRequired) {
125 v_texco.w[vertexIndex] = cfg.texcoScale - (cfg.texcoScale * Vec2f(vertex.x, vertex.z));
126 }
127 if (cfg.isTangentRequired) {
128 triVertices[faceVertIndex] = v_pos.w[vertexIndex];
129 triTexco[faceVertIndex] = v_texco.w[vertexIndex];
130 }
131 faceVertIndex += 1;
132 }
133
134 if (cfg.isTangentRequired) {
135 Vec4f tangent = calculateTangent(triVertices, triTexco, normal);
136 for (GLuint i = 0; i < 3; ++i) {
137 v_tan.w[triIndices[i]] = tangent;
138 }
139 }
140 }

143void Rectangle::updateAttributes(Config cfg) {
144 std::vector<Tessellation> tessellations;
145 GLuint numVertices = 0;
146 GLuint numIndices = 0;
147 {
148 Tessellation baseTess;
149 baseTess.vertices.resize(4);
150 baseTess.vertices[0] = Vec3f(0.0, 0.0, 0.0);
151 baseTess.vertices[1] = Vec3f(1.0, 0.0, 0.0);
152 baseTess.vertices[2] = Vec3f(1.0, 0.0, 1.0);
153 baseTess.vertices[3] = Vec3f(0.0, 0.0, 1.0);
154 baseTess.inputFaces.resize(2);
155 baseTess.inputFaces[0] = TessellationFace(0, 1, 3);
156 baseTess.inputFaces[1] = TessellationFace(1, 2, 3);
157
158 for (GLuint lodLevel: cfg.levelOfDetails) {
159 auto &lodTess = tessellations.emplace_back();
160 lodTess.vertices = baseTess.vertices;
161 lodTess.inputFaces = baseTess.inputFaces;
162 tessellate(lodLevel, lodTess);
163
164 auto &x = meshLODs_.emplace_back();
165 x.numVertices = lodTess.vertices.size();
166 x.numIndices = lodTess.outputFaces.size() * 3;
167 x.vertexOffset = numVertices;
168 x.indexOffset = numIndices;
169 numVertices += lodTess.vertices.size();
170 numIndices += lodTess.outputFaces.size() * 3;
171 }
172 }
173 if (cfg.isTangentRequired) {
174 cfg.isNormalRequired = GL_TRUE;
175 cfg.isTexcoRequired = GL_TRUE;
176 }
177
178 // allocate attributes
179 pos_->setVertexData(numVertices);
180 if (cfg.isNormalRequired) {
181 nor_->setVertexData(numVertices);
182 }
183 if (cfg.isTexcoRequired) {
184 texco_->setVertexData(numVertices);
185 }
186 if (cfg.isTangentRequired) {
187 tan_->setVertexData(numVertices);
188 }
189 indices_->setVertexData(numIndices);
190
191 minPosition_ = Vec3f(0.0);
192 maxPosition_ = Vec3f(0.0);
193
194 Mat4f rotMat = Mat4f::rotationMatrix(cfg.rotation.x, cfg.rotation.y, cfg.rotation.z);
195 for (auto i = 0u; i < tessellations.size(); ++i) {
196 generateLODLevel(cfg,
197 tessellations[i],
198 rotMat,
199 meshLODs_[i].vertexOffset,
200 meshLODs_[i].indexOffset);
201 }
202
203 begin(ShaderInputContainer::INTERLEAVED);
204 auto indexRef = setIndices(indices_, numVertices);
205 setInput(pos_);
206 if (cfg.isNormalRequired)
207 setInput(nor_);
208 if (cfg.isTexcoRequired)
209 setInput(texco_);
210 if (cfg.isTangentRequired)
211 setInput(tan_);
212 end();
213
214 for (auto &x: meshLODs_) {
215 // add the index buffer offset (in number of bytes)
216 x.indexOffset = indexRef->address() + x.indexOffset * sizeof(GLuint);
217 }
218 activateLOD(0);
219}
