82void AnimationNode::calculateGlobalTransform() {
83 // concatenate all parent transforms to get the global transform for this node
84 globalTransform_ = localTransform_;
85 for (AnimationNode *p = parentNode_.get(); p != nullptr; p = p->parentNode_.get()) {
86 globalTransform_.multiplyr(p->localTransform_);

90void AnimationNode::updateTransforms(const std::vector<Mat4f> &transforms) {
91 std::vector<ref_ptr<AnimationNode> >::iterator it;
92 Stack<AnimationNode *> nodes;
93
94#if 1
95 // assume identity as local root node transform.
96 // not sure if this is safe.
97#else
98 globalTransform_ = channelIndex_!=-1 ?
99 transforms[channelIndex_] : localTransform_;
100 Mat4f rootInverse = globalTransform_.inverse();
101#endif
102 if (isBoneNode_) {
103 boneTransformationMatrix_ = offsetMatrix_;
104 }
105 for (it = nodeChilds_.begin(); it != nodeChilds_.end(); ++it) nodes.push(it->get());
106
107 while (!nodes.isEmpty()) {
108 AnimationNode *n = nodes.top();
109 nodes.pop();
110
111 // update node global transform
112 n->globalTransform_ = (n->channelIndex_ != -1) ?
113 transforms[n->channelIndex_] : n->localTransform_;
114 n->globalTransform_.multiplyr(n->parentNode_->globalTransform_);
115
116 if (n->isBoneNode_) {
117 // Bone matrices transform from mesh coordinates in bind pose
118 // to mesh coordinates in skinned pose
119#if 1
120 n->boneTransformationMatrix_ = (n->globalTransform_ * n->offsetMatrix_).transpose();
121#else
122 n->boneTransformationMatrix_ = (rootInverse*n->globalTransform_*n->offsetMatrix_).transpose();
123#endif
124 }
125
126 // continue for all children
127 for (it = n->nodeChilds_.begin(); it != n->nodeChilds_.end(); ++it) nodes.push(it->get());
128 }
129}

141// Look for present frame number.
142// Search from last position if time is after the last time, else from beginning
143template<class T>
144static void findFrameAfterTick(
145 GLdouble tick,
146 GLuint &frame,
147 const std::vector<T> &keys) {
148 while (frame < keys.size() - 1) {
149 if (tick - keys[++frame].time < 0.00001) {
150 --frame;
151 return;
152 }
153 }
154}

156template<class T>
157static void findFrameBeforeTick(
158 GLdouble &tick,
159 GLuint &frame,
160 const std::vector<T> &keys) {
161 if (keys.empty()) return;
162 for (frame = keys.size() - 1; frame > 0;) {
163 if (tick - keys[--frame].time > 0.000001) return;
164 }
165}

167template<class T>
168static inline GLdouble interpolationFactor(
169 const T &key, const T &nextKey,
170 GLdouble t, GLdouble duration) {
171 GLdouble timeDifference = nextKey.time - key.time;
172 if (timeDifference < 0.0) timeDifference += duration;
173 return ((t - key.time) / timeDifference);
174}

176template<class T>
177static inline GLboolean handleFrameLoop(
178 T &dst,
179 const GLuint &frame, const GLuint &lastFrame,
180 const NodeAnimation::Channel &channel,
181 const T &key, const T &first) {
182 if (frame >= lastFrame) {
183 return false;
184 }
185 switch (channel.postState) {
186 case NodeAnimation::BEHAVIOR_DEFAULT:
187 dst.value = first.value;
188 return true;
189 case NodeAnimation::BEHAVIOR_CONSTANT:
190 dst.value = key.value;
191 return true;
192 case NodeAnimation::BEHAVIOR_LINEAR:
193 case NodeAnimation::BEHAVIOR_REPEAT:
194 default:
195 return false;
196 }
197}

260void NodeAnimation::setAnimationIndexActive(
261 GLint animationIndex, const Vec2d &forcedTickRange) {
262 if (animationIndex_ == animationIndex) {
263 // already active
264 setTickRange(forcedTickRange);
265 return;
266 }
267
268 if (animationIndex_ > 0) {
269 animData_[animationIndex_]->active_ = false;
270 }
271 animationIndex_ = ANIM_INDEX(animationIndex);
272 if (animationIndex_ < 0) return;
273
274 NodeAnimation::Data &anim = *animData_[animationIndex_].get();
275 anim.active_ = true;
276 if (anim.transforms_.size() != anim.channels_->size()) {
277 anim.transforms_.resize(anim.channels_->size(), Mat4f::identity());
278 }
279 if (anim.startFramePosition_.size() != anim.channels_->size()) {
280 anim.startFramePosition_.resize(anim.channels_->size());
281 }
282
283 // set matching channel index
284 for (GLuint a = 0; a < anim.channels_->size(); a++) {
285 const std::string nodeName = anim.channels_->data()[a].nodeName_;
286 nameToNode_[nodeName]->set_channelIndex(a);
287 }
288
289 tickRange_.x = -1.0;
290 tickRange_.y = -1.0;
291 setTickRange(forcedTickRange);
292}

294void NodeAnimation::setTickRange(const Vec2d &forcedTickRange) {
295 if (animationIndex_ < 0) {
296 REGEN_WARN("can not set tick range without animation index set.");
297 return;
298 }
299 NodeAnimation::Data &anim = *animData_[animationIndex_].get();
300
301 // get first and last tick of animation
302 Vec2d tickRange;
303 if (forcedTickRange.x < 0.0f || forcedTickRange.y < 0.0f) {
304 tickRange.x = 0.0;
305 tickRange.y = anim.duration_;
306 } else {
307 tickRange = forcedTickRange;
308 }
309
310 if (tickRange.x == tickRange_.x &&
311 tickRange.y == tickRange_.y) {
312 // nothing changed
313 return;
314 }
315 tickRange_ = tickRange;
316
317 // set start frames
318 if (tickRange_.x < 0.00001) {
319 anim.startFramePosition_.assign(
320 anim.channels_->size(), Vec3ui(0u));
321 } else {
322 for (GLuint a = 0; a < anim.channels_->size(); a++) {
323 Channel &channel = anim.channels_->data()[a];
324 Vec3ui framePos(0u);
325 findFrameBeforeTick(tickRange_.x, framePos.x, *channel.positionKeys_.get());
326 findFrameBeforeTick(tickRange_.x, framePos.y, *channel.rotationKeys_.get());
327 findFrameBeforeTick(tickRange_.x, framePos.z, *channel.scalingKeys_.get());
328 anim.startFramePosition_[a] = framePos;
329 }
330 }
331
332 // initial last frame to start frame
333 anim.lastFramePosition_ = anim.startFramePosition_;
334
335 // set to start pos of the new tick range
336 anim.lastTime_ = 0.0;
337 anim.elapsedTime_ = 0.0;
338
339 // set start tick and duration in ticks
340 startTick_ = tickRange_.x;
341 duration_ = abs(tickRange_.y - tickRange_.x);
342}

379void NodeAnimation::animate(GLdouble milliSeconds) {
380 if (animationIndex_ < 0) return;
381
382 NodeAnimation::Data &anim = *animData_[animationIndex_].get();
383
384 anim.elapsedTime_ += milliSeconds;
385 if (duration_ <= 0.0) return;
386
387 GLdouble time = anim.elapsedTime_ * timeFactor_ / 1000.0;
388 // map into anim's duration
389 GLdouble timeInTicks = time * anim.ticksPerSecond_;
390 if (timeInTicks > duration_) {
391 // for repeating we could do...
392 //timeInTicks = fmod(timeInTicks, duration_);
393 stopNodeAnimation(anim);
394 return;
395 }
396 // Time runs backwards when start tick is higher then stop tick
397 if (tickRange_.x > tickRange_.y) {
398 timeInTicks = -timeInTicks;
399 }
400 timeInTicks += startTick_;
401
402 // update transformations
403 for (GLuint i = 0; i < anim.channels_->size(); i++) {
404 const Channel &channel = anim.channels_->data()[i];
405 Mat4f &m = anim.transforms_[i];
406
407 if (channel.rotationKeys_->empty()) {
408 m = Mat4f::identity();
409 } else if (channel.rotationKeys_->size() == 1) {
410 m = channel.rotationKeys_->data()[0].value.calculateMatrix();
411 } else {
412 m = nodeRotation(anim, channel, timeInTicks, i).calculateMatrix();
413 }
414
415 if (channel.scalingKeys_->empty()) {
416 } else if (channel.scalingKeys_->size() == 1) {
417 m.scale(channel.scalingKeys_->data()[0].value);
418 } else {
419 m.scale(nodeScaling(anim, channel, timeInTicks, i));
420 }
421
422 if (channel.positionKeys_->empty()) {
423 } else if (channel.positionKeys_->size() == 1) {
424 Vec3f &pos = channel.positionKeys_->data()[0].value;
425 m.x[3] = pos.x;
426 m.x[7] = pos.y;
427 m.x[11] = pos.z;
428 } else {
429 Vec3f pos = nodePosition(anim, channel, timeInTicks, i);
430 m.x[3] = pos.x;
431 m.x[7] = pos.y;
432 m.x[11] = pos.z;
433 }
434 }
435
436 anim.lastTime_ = timeInTicks;
437
438 rootNode_->updateTransforms(anim.transforms_);
439}

441Vec3f NodeAnimation::nodePosition(
442 NodeAnimation::Data &anim,
443 const Channel &channel,
444 GLdouble timeInTicks,
445 GLuint i) {
446 GLuint keyCount = channel.positionKeys_->size();
447 const std::vector<KeyFrame3f> &keys = *channel.positionKeys_.get();
448 KeyFrame3f pos;
449
450 pos.value = Vec3f(0);
451
452 // Look for present frame number.
453 GLuint lastFrame = anim.lastFramePosition_[i].x;
454 GLuint frame;
455 if (tickRange_.x > tickRange_.y) {
456 frame = (timeInTicks <= anim.lastTime_ ? lastFrame : anim.startFramePosition_[i].x);
457 findFrameBeforeTick(timeInTicks, frame, keys);
458 } else {
459 frame = (timeInTicks >= anim.lastTime_ ? lastFrame : anim.startFramePosition_[i].x);
460 findFrameAfterTick(timeInTicks, frame, keys);
461 }
462 anim.lastFramePosition_[i].x = frame;
463 // lookup nearest two keys
464 const KeyFrame3f &key = keys[frame];
465 // interpolate between this frame's value and next frame's value
466 if (!handleFrameLoop(pos, frame, lastFrame, channel, key, keys[0])) {
467 const KeyFrame3f &nextKey = keys[(frame + 1) % keyCount];
468 GLfloat fac = interpolationFactor(key, nextKey, timeInTicks, duration_);
469 if (fac <= 0) return key.value;
470 pos.value = key.value + (nextKey.value - key.value) * fac;
471 }
472
473 return pos.value;
474}

476Quaternion NodeAnimation::nodeRotation(
477 NodeAnimation::Data &anim,
478 const Channel &channel,
479 GLdouble timeInTicks,
480 GLuint i) {
481 KeyFrameQuaternion rot;
482 GLuint keyCount = channel.rotationKeys_->size();
483 const std::vector<KeyFrameQuaternion> &keys = *channel.rotationKeys_.get();
484
485 rot.value = Quaternion(1, 0, 0, 0);
486
487 // Look for present frame number.
488 GLuint lastFrame = anim.lastFramePosition_[i].y;
489 GLuint frame;
490 if (tickRange_.x > tickRange_.y) {
491 frame = (timeInTicks <= anim.lastTime_ ? lastFrame : anim.startFramePosition_[i].y);
492 findFrameBeforeTick(timeInTicks, frame, keys);
493 } else {
494 frame = (timeInTicks >= anim.lastTime_ ? lastFrame : anim.startFramePosition_[i].y);
495 findFrameAfterTick(timeInTicks, frame, keys);
496 }
497 anim.lastFramePosition_[i].y = frame;
498 // lookup nearest two keys
499 const KeyFrameQuaternion &key = keys[frame];
500 // interpolate between this frame's value and next frame's value
501 if (!handleFrameLoop(rot, frame, lastFrame, channel, key, keys[0])) {
502 const KeyFrameQuaternion &nextKey = keys[(frame + 1) % keyCount];
503 float fac = interpolationFactor(key, nextKey, timeInTicks, duration_);
504 if (fac <= 0) return key.value;
505#ifdef INTERPOLATE_QUATERNION_LINEAR
506 rot.value.interpolateLinear(key.value, nextKey.value, fac);
507#else
508 rot.value.interpolate(key.value, nextKey.value, fac);
509#endif
510 }
511
512 return rot.value;
513}

515Vec3f NodeAnimation::nodeScaling(
516 NodeAnimation::Data &anim,
517 const Channel &channel,
518 GLdouble timeInTicks,
519 GLuint i) {
520 const std::vector<KeyFrame3f> &keys = *channel.scalingKeys_.get();
521 KeyFrame3f scale;
522
523 // Look for present frame number.
524 GLuint lastFrame = anim.lastFramePosition_[i].z;
525 GLuint frame;
526 if (tickRange_.x > tickRange_.y) {
527 frame = (timeInTicks <= anim.lastTime_ ? lastFrame : anim.startFramePosition_[i].z);
528 findFrameBeforeTick(timeInTicks, frame, keys);
529 } else {
530 frame = (timeInTicks >= anim.lastTime_ ? lastFrame : anim.startFramePosition_[i].z);
531 findFrameAfterTick(timeInTicks, frame, keys);
532 }
533 anim.lastFramePosition_[i].z = frame;
534 // lookup nearest key
535 const KeyFrame3f &key = keys[frame];
536 // set current value
537 if (!handleFrameLoop(scale, frame, lastFrame, channel, key, keys[0])) {
538 scale.value = key.value;
539 }
540
541 return scale.value;
