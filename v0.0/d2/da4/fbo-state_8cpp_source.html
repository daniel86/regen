15FBOState::FBOState(const ref_ptr<FBO> &fbo)
16 : State(), fbo_(fbo) {
17#ifdef USE_FBO_UBO
18 // TODO: there are some nodes in the scene graph with special handling for viewport and inverseViewport
19 // uniforms. This includes BloomPass and FilterSequence. I think the problem is that FBOState appears
20 // as a parent, and then the shader is compiled for a uniform block.
21 // However, BloomPass and FilterSequence rather use gUniform* functions to set the uniforms, which
22 // pretty sure does not work with uniform blocks.
23 joinShaderInput(fbo->uniforms());
24#else
25 joinShaderInput(fbo->viewport());
26 joinShaderInput(fbo->inverseViewport());
27#endif
28 for (auto &attachment : fbo->colorTextures()) {
29 shaderDefine(REGEN_STRING("FBO_ATTACHMENT_" << attachment->name()), "TRUE");
30 }
31}

128ScreenState::ScreenState(

145void ScreenState::enable(RenderState *state) {
146 auto winViewport = windowViewport_->getVertex(0);
147 glViewport_.z = winViewport.r.x;
148 glViewport_.w = winViewport.r.y;
149 viewport_->setVertex(0, Vec2f(winViewport.r.x, winViewport.r.y));
150 inverseViewport_->setUniformData(
151 Vec2f(1.0f / (GLfloat) winViewport.r.x, 1.0f / (GLfloat) winViewport.r.y));
152 winViewport.unmap();
153
154 state->drawFrameBuffer().push(0);
155 FBO::screen().drawBuffer_.push(drawBuffer_);
156 state->viewport().push(glViewport_);
157 State::enable(state);
158}
