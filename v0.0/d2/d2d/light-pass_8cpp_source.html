16LightPass::LightPass(Light::Type type, const std::string &shaderKey)
17 : State(), lightType_(type), shaderKey_(shaderKey) {
18 switch (lightType_) {
19 case Light::DIRECTIONAL:
20 mesh_ = Rectangle::getUnitQuad();
21 break;
22 case Light::SPOT:
23 mesh_ = ConeClosed::getBaseCone();
24 joinStates(ref_ptr<CullFaceState>::alloc(GL_FRONT));
25 // enable depth clamping to avoid faces to be clipped away when out of depth range.
26 joinStates(ref_ptr<ToggleState>::alloc(RenderState::DEPTH_CLAMP, GL_TRUE));
27 break;
28 case Light::POINT:
29 mesh_ = Box::getUnitCube();
30 joinStates(ref_ptr<CullFaceState>::alloc(GL_FRONT));
31 // enable depth clamping to avoid faces to be clipped away when out of depth range.
32 joinStates(ref_ptr<ToggleState>::alloc(RenderState::DEPTH_CLAMP, GL_TRUE));
33 break;
34 }
35 shadowFiltering_ = SHADOW_FILTERING_NONE;
36 numShadowLayer_ = 1;
37
38 shader_ = ref_ptr<ShaderState>::alloc();
39 joinStates(shader_);
40 setShadowFiltering(SHADOW_FILTERING_NONE);
41}

60void LightPass::addLight(
61 const ref_ptr<Light> &l,
62 const ref_ptr<LightCamera> &lightCamera,
63 const ref_ptr<Texture> &shadowTexture,
64 const ref_ptr<Texture> &shadowColorTexture,
65 const std::list<ref_ptr<ShaderInput> > &inputs) {
66 LightPassLight light;
67 lights_.push_back(light);
68
69 auto it = lights_.end();
70 --it;
71 lightIterators_[l.get()] = it;
72
73 it->light = l;
74 it->camera = lightCamera;
75 it->shadow = shadowTexture;
76 it->shadowColor = shadowColorTexture;
77 it->inputs = inputs;
78 if (shadowTexture.get()) {
79 auto *tex3D = dynamic_cast<Texture3D *>(shadowTexture.get());
80 if (tex3D == nullptr) {
81 numShadowLayer_ = 1;
82 } else {
83 numShadowLayer_ = tex3D->depth();
84 }
85 shaderDefine("USE_SHADOW_MAP", "TRUE");
86 if (shadowColorTexture.get()) {
87 shaderDefine("USE_SHADOW_COLOR", "TRUE");
88 }
89 }
90
91 if (shader_->shader().get()) { addLightInput(*it); }
92}

106void LightPass::createShader(const StateConfig &cfg) {
107 StateConfigurer _cfg(cfg);
108 _cfg.addState(this);
109 _cfg.addState(mesh_.get());
110 bool hasInstancedInputs = false;
111 if (!lights_.empty()) {
112 // add first light to shader to set up shader defines and also instance count
113 auto &firstLight = lights_.front();
114 for (auto &in : firstLight.light->inputContainer()->inputs()) {
115 if (in.in_->isUniformBlock()) {
116 auto *block = dynamic_cast<UniformBlock *>(in.in_.get());
117 _cfg.addInput(in.in_->name(), in.in_);
118 for (auto &blockUniform : block->uniforms()) {
119 if (blockUniform.in_->numInstances()>0) { hasInstancedInputs = true; }
120 }
121 } else {
122 _cfg.addInput(in.in_->name(), in.in_);
123 if (in.in_->numInstances()>0) { hasInstancedInputs = true; }
124 }
125 }
126 }
127 if (hasInstancedInputs) {
128 _cfg.define("HAS_INSTANCES", "TRUE");
129 }
130 _cfg.define("NUM_SHADOW_LAYER", REGEN_STRING(numShadowLayer_));
131 shader_->createShader(_cfg.cfg(), shaderKey_);
132 mesh_->updateVAO(RenderState::get(), _cfg.cfg(), shader_->shader());
133
134 for (auto it = lights_.begin(); it != lights_.end(); ++it) { addLightInput(*it); }
135 shadowMapLoc_ = shader_->shader()->uniformLocation("shadowTexture");
136 shadowColorLoc_ = shader_->shader()->uniformLocation("shadowColorTexture");
137}

139void LightPass::addLightInput(LightPassLight &light) {
140 // clear list of uniform loactions
141 light.inputLocations.clear();
142 // add user specified uniforms
143 for (auto jt = light.inputs.begin(); jt != light.inputs.end(); ++jt) {
144 ref_ptr<ShaderInput> &in = *jt;
145 addInputLocation(light, in, in->name());
146 }
147
148 // add shadow uniforms
149 if (light.camera.get()) {
150 addInputLocation(light, light.camera->lightCamera()->far(), "lightFar");
151 addInputLocation(light, light.camera->lightCamera()->near(), "lightNear");
152 addInputLocation(light, light.camera->lightMatrix(), "lightMatrix");
153 }
154 if (light.shadow.get()) {
155 addInputLocation(light, light.shadow->sizeInverse(), "shadowInverseSize");
156 addInputLocation(light, light.shadow->size(), "shadowSize");
157 }
158
159 // add light UBO, and special light type uniforms
160 addInputLocation(light, light.light->lightUBO(), "Light");
161 if (lightType_ == Light::SPOT) {
162 addInputLocation(light, light.light->coneMatrix(), "modelMatrix");
163 }
164}

175void LightPass::enable(RenderState *rs) {
176 State::enable(rs);
177 auto smChannel = rs->reserveTextureChannel();
178 auto smColorChannel = rs->reserveTextureChannel();
179
180 for (auto &l : lights_) {
181 ref_ptr<Texture> shadowTex;
182 // activate shadow map if specified
183 if (l.shadow.get()) {
184 l.shadow->begin(rs, smChannel);
185 glUniform1i(shadowMapLoc_, smChannel);
186 }
187 if (l.shadowColor.get()) {
188 l.shadowColor->begin(rs, smColorChannel);
189 glUniform1i(shadowColorLoc_, smColorChannel);
190 }
191 // enable light pass uniforms
192 for (auto jt = l.inputLocations.begin(); jt != l.inputLocations.end(); ++jt) {
193 if (lights_.size()>1 || jt->uploadStamp != jt->input->stamp()) {
194 jt->input->enableUniform(jt->location);
195 jt->uploadStamp = jt->input->stamp();
196 }
197 }
198
199 mesh_->enable(rs);
200 mesh_->disable(rs);
201
202 if (l.shadow.get()) { l.shadow->end(rs, smChannel); }
203 if (l.shadowColor.get()) { l.shadowColor->end(rs, smColorChannel); }
204 }
205
206 rs->releaseTextureChannel();
207 rs->releaseTextureChannel();
208}
