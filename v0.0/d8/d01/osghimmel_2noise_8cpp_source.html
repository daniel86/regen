176void Noise::generatePermutationMap(unsigned char *dest) const
177{
178 const unsigned int size2 = m_size * m_size;
179
180 if(size2 < 1)
181 return;
182
183 for(unsigned int s = 0; s < m_size; ++s)
184 for(unsigned int t = 0; t < m_size; ++t)
185 {
186 const unsigned int o = 4 * (t * m_size + s);
187 regen::Vec4f g(grad3h(s + m_xoff, t + m_yoff));
188 dest[o + 0] = static_cast<unsigned char>(g.x + 1);
189 dest[o + 1] = static_cast<unsigned char>(g.y + 1);
190 dest[o + 2] = static_cast<unsigned char>(g.z + 1);
191 dest[o + 3] = static_cast<unsigned char>(g.w);
192 }
193}

226const float Noise::noise2(
227 const float s
228, const float t) const
229{
230 //const float o10 = 1.0 / static_cast<float>(m_size);
231 //const float o05 = 0.5 / static_cast<float>(m_size);
232
233 const float is = floor(s);
234 const float it = floor(t);
235
236 const regen::Vec2f f(_frac(s), _frac(t));
237
238 // range [-1;+1]
239
240 const float aa = dot(grad2(is + 0, it + 0), (f ));
241 const float ba = dot(grad2(is + 1, it + 0), (f - regen::Vec2f(1.f, 0.f)));
242 const float ab = dot(grad2(is + 0, it + 1), (f - regen::Vec2f(0.f, 1.f)));
243 const float bb = dot(grad2(is + 1, it + 1), (f - regen::Vec2f(1.f, 1.f)));
244
245 const regen::Vec2f i = mix(regen::Vec2f(aa, ab), regen::Vec2f(ba, bb), fade(f.x));
246
247 return mix(i.x, i.y, fade(f.y));
248}

251const float Noise::noise2(
252 float s
253, float t
254, const unsigned int r) const
255{
256 s += m_xoff;
257 t += m_yoff;
258
259 s *= 1 << r;
260 t *= 1 << r;
261
262 const float is = floor(s);
263 const float it = floor(t);
264
265 const regen::Vec2f f(_frac(s), _frac(t));
266
267 // range [-1;+1]
268
269 const float aa = dot(grad2(is + 0, it + 0, r), (f ));
270 const float ba = dot(grad2(is + 1, it + 0, r), (f - regen::Vec2f(1.f, 0.f)));
271 const float ab = dot(grad2(is + 0, it + 1, r), (f - regen::Vec2f(0.f, 1.f)));
272 const float bb = dot(grad2(is + 1, it + 1, r), (f - regen::Vec2f(1.f, 1.f)));
273
274 const regen::Vec2f i = mix(regen::Vec2f(aa, ab), regen::Vec2f(ba, bb), fade(f.x));
275
276 return mix(i.x, i.y, fade(f.y));
277}
