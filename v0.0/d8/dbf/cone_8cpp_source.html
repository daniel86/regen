34void ConeOpened::generateLODLevel(const Config &cfg,
35 GLuint lodLevel,
36 GLuint vertexOffset,
37 GLuint indexOffset) {
38 // map client data for writing
39 auto v_pos = pos_->mapClientData<Vec3f>(ShaderData::WRITE);
40 auto v_nor = (cfg.isNormalRequired ?
41 nor_->mapClientData<Vec3f>(ShaderData::WRITE) :
42 ShaderData_rw<Vec3f>::nullData());
43
44 GLfloat phi = acos(cfg.cosAngle);
45 GLfloat radius = tan(phi) * cfg.height;
46 GLfloat angle = 0.0f;
47 GLfloat angleStep = 2.0f * M_PI / (GLfloat) lodLevel;
48 GLuint i = vertexOffset;
49
50 v_pos.w[i] = Vec3f(0.0f);
51 if (cfg.isNormalRequired) {
52 v_nor.w[i] = Vec3f(0.0f, -1.0f, 0.0f);
53 }
54
55 for (; i < lodLevel + 1; ++i) {
56 angle += angleStep;
57 GLfloat s = sin(angle) * radius;
58 GLfloat c = cos(angle) * radius;
59 Vec3f pos(c, s, cfg.height);
60 v_pos.w[i + 1] = pos;
61 minPosition_.setMin(pos);
62 maxPosition_.setMax(pos);
63 if (cfg.isNormalRequired) {
64 Vec3f n(c, 0.0, s);
65 n.normalize();
66 v_nor.w[i + 1] = n;
67 }
68 }
69}

142static void loadConeData(
143 Vec3f *pos, Vec3f *nor,
144 Vec3f &min,
145 Vec3f &max,
146 GLboolean useBase,
147 GLuint subdivisions,
148 GLfloat radius,
149 GLfloat height) {
150 GLfloat angle = 0.0f;
151 GLfloat angleStep = 2.0f * M_PI / (GLfloat) subdivisions;
152 GLint i = 0;
153
154 // apex
155 pos[i] = Vec3f(0.0f);
156 if (nor) nor[i] = Vec3f(0.0f, 0.0f, -1.0f);
157 ++i;
158 // base center
159 if (useBase) {
160 pos[i] = Vec3f(0.0f, 0.0f, height);
161 min.setMin(pos[i]);
162 max.setMax(pos[i]);
163 if (nor) nor[i] = Vec3f(0.0f, 0.0f, 1.0f);
164 ++i;
165 }
166
167 GLint numVertices = subdivisions + i;
168 for (; i < numVertices; ++i) {
169 angle += angleStep;
170 GLfloat s = sin(angle) * radius;
171 GLfloat c = cos(angle) * radius;
172 pos[i] = Vec3f(c, s, height);
173 min.setMin(pos[i]);
174 max.setMax(pos[i]);
175 if (nor) {
176 nor[i] = Vec3f(c, s, 0.0);
177 nor[i].normalize();
178 }
179 }

182void ConeClosed::generateLODLevel(
183 const Config &cfg,
184 GLuint lodLevel,
185 GLuint vertexOffset,
186 GLuint indexOffset) {
187 // map client data for writing
188 auto indices = indices_->mapClientData<GLuint>(ShaderData::WRITE);
189 auto v_pos = pos_->mapClientData<Vec3f>(ShaderData::WRITE);
190 auto v_nor = (cfg.isNormalRequired ?
191 nor_->mapClientData<Vec3f>(ShaderData::WRITE) :
192 ShaderData_rw<Vec3f>::nullData());
193
194 // create cone vertex data
195 loadConeData(
196 v_pos.w+vertexOffset,
197 (v_nor.w ? v_nor.w+vertexOffset : v_nor.w),
198 minPosition_, maxPosition_,
199 cfg.isBaseRequired, lodLevel,
200 cfg.radius, cfg.height);
201
202 // create cone index data
203 const GLuint apexIndex = vertexOffset;
204 const GLuint baseCenterIndex = vertexOffset + 1;
205 GLuint faceIndex = indexOffset;
206 GLint vIndex = vertexOffset + cfg.isBaseRequired ? 2 : 1;
207 // cone
208 for (GLuint i = 0; i < lodLevel; ++i) {
209 indices.w[faceIndex++] = apexIndex;
210 indices.w[faceIndex++] = (i + 1 == lodLevel ? vIndex : vIndex + i + 1);
211 indices.w[faceIndex++] = vIndex + i;
212 }
213 // base
214 if (cfg.isBaseRequired) {
215 for (GLuint i = 0; i < lodLevel; ++i) {
216 indices.w[faceIndex++] = baseCenterIndex;
217 indices.w[faceIndex++] = vIndex + i;
218 indices.w[faceIndex++] = (i + 1 == lodLevel ? vIndex : vIndex + i + 1);
219 }
220 }
221}

223void ConeClosed::updateAttributes(const Config &cfg) {
224 std::vector<GLuint> LODs;
225 GLuint vertexOffset = 0;
226 GLuint indexOffset = 0;
227 for (auto &lod: cfg.levelOfDetails) {
228 GLuint lodLevel = 4u * pow(lod, 2);
229 LODs.push_back(lodLevel);
230 auto &x = meshLODs_.emplace_back();
231 x.numVertices = lodLevel + 1;
232 if (cfg.isBaseRequired) x.numVertices += 1;
233 x.vertexOffset = vertexOffset;
234 vertexOffset += x.numVertices;
235
236 GLuint numFaces = lodLevel;
237 if (cfg.isBaseRequired) { numFaces *= 2; }
238 x.numIndices = numFaces * 3;
239 x.indexOffset = indexOffset;
240 indexOffset += x.numIndices;
241 }
242 GLuint numVertices = vertexOffset;
243 GLuint numIndices = indexOffset;
244
245 minPosition_ = Vec3f(0.0);
246 maxPosition_ = Vec3f(0.0);
247 pos_->setVertexData(numVertices);
248 if (cfg.isNormalRequired) {
249 nor_->setVertexData(numVertices);
250 }
251 indices_->setVertexData(numIndices);
252
253 for (auto i = 0u; i < LODs.size(); ++i) {
254 generateLODLevel(cfg,
255 LODs[i],
256 meshLODs_[i].vertexOffset,
257 meshLODs_[i].indexOffset);
258 }
259
260 begin(ShaderInputContainer::INTERLEAVED);
261 auto indexRef = setIndices(indices_, numVertices);
262 setInput(pos_);
263 if (cfg.isNormalRequired)
264 setInput(nor_);
265 end();
266
267 for (auto &x: meshLODs_) {
268 // add the index buffer offset (in number of bytes)
269 x.indexOffset = indexRef->address() + x.indexOffset * sizeof(GLuint);
270 }
271 activateLOD(0);
