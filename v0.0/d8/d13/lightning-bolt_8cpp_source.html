194static Vec3f getPerpendicular1(const Vec3f &v) {
195 if (std::abs(v.x) < std::abs(v.y) && std::abs(v.x) < std::abs(v.z)) {
196 return v.cross(Vec3f::right());
197 } else if (std::abs(v.y) < std::abs(v.z)) {
198 return v.cross(Vec3f::up());
199 } else {
200 return v.cross(Vec3f::front());
201 }
202}

204static Vec3f getPerpendicular(const Vec3f &v) {
205 // Compute a vector perpendicular to v
206 auto w = getPerpendicular1(v);
207 w.normalize();
208 // Generate a random angle
209 auto phi = static_cast<float>(math::random() * 2.0 * M_PI);
210 // Rotate the perpendicular vector around v by phi
211 return w * std::cos(phi) + v.cross(w) * std::sin(phi);
212}

214static Vec3f branch(
215 const Vec3f &start,
216 const Vec3f &dir,
217 const Vec3f &segmentEnd,
218 float offsetAmount,
219 float branchLength) {
220 Vec3f v = segmentEnd;
221 v += getPerpendicular(dir) * (offsetAmount * (math::random() * 2.0f - 1.0f));
222 v -= start;
223 v.normalize();
224 // pull the end point of the branch closer to origin by factor lengthScale
225 return start + v * dir.length() * branchLength;
226}

234void LightningBolt::updateSegmentData() {
235 // update CPU segment data.
236 // we use two std::vectors in the process which are swapped for each subdivision.
237 // internally the vector should keep the memory allocated, so there should be no
238 // reallocation of memory after the first iteration.
239 // the memory in the vector is further aligned with the GL buffer such that we can
240 // directly copy the data to the buffer.
241 // TODO: could rather generate a 2D bolt oriented wrt the camera.
242 auto source = source_->getVertex(0);
243 auto target = target_->getVertex(0);
244 segments_[1].clear();
245 segments_[0].clear();
246 segments_[0].emplace_back(source.r, target.r);
247 segmentIndex_ = 0;
248 unsigned int nextIndex = 1;
249 float offsetAmount = jitterOffset_->getVertex(0).r;
250 float heightRange = source.r.y - target.r.y;
251 auto branchProbability = branchProbability_->getVertex(0);
252 auto branchOffset = branchOffset_->getVertex(0);
253 auto branchDarkening = branchDarkening_->getVertex(0);
254
255 int numMainBranchVertices = static_cast<int>(pow(2, maxSubDivisions_));
256 int numSubBranchVertices = numMainBranchVertices;
257 int numRemainingVertices = numMainBranchVertices * static_cast<int>(maxBranches_ - 1);
258
259 // update the lifetime
260 auto lifetime_cfg = lifetimeConfig_->getVertex(0);
261 u_lifetime_ = lifetime_cfg.r.x + lifetime_cfg.r.y * (math::random() * 2.0f - 1.0f);
262 u_lifetimeBegin_ = u_lifetime_;
263
264 Vec3f midPoint, direction;
265 for (unsigned int i = 0u; i < maxSubDivisions_; ++i) {
266 numSubBranchVertices /= 2;
267
268 for (auto &segment: segments_[segmentIndex_]) {
269 midPoint = (segment.start.pos + segment.end.pos) * 0.5f;
270 direction = segment.end.pos - segment.start.pos;
271 direction.normalize();
272 midPoint += getPerpendicular(direction) * (offsetAmount * (math::random() * 2.0f - 1.0f));
273 segments_[nextIndex].emplace_back(segment.start.pos, midPoint, segment.start.brightness);
274 segments_[nextIndex].emplace_back(midPoint, segment.end.pos, segment.end.brightness);
275
276 direction = midPoint - segment.start.pos;
277 // with some probability create a new branch.
278 if (numRemainingVertices > numSubBranchVertices &&
279 math::random() < branchProbability.r *
280 // decrease the probability for sub-branches
281 std::min(1.0f, segment.start.brightness + 0.25f) *
282 // decrease the probability further away from the target
283 std::min(1.0f, (1.0f - (midPoint.y - target.r.y) / heightRange) + 0.25f)) {
284 segments_[nextIndex].emplace_back(
285 midPoint,
286 branch(midPoint, direction, segment.end.pos,
287 offsetAmount * branchOffset.r,
288 branchLength_->getVertex(0).r),
289 // decrease brightness for sub-branches
290 segment.start.brightness * branchDarkening.r);
291 numRemainingVertices -= numSubBranchVertices;
292 }
293 }
294 segments_[segmentIndex_].clear();
295 nextIndex = segmentIndex_;
296 segmentIndex_ = 1 - segmentIndex_;
297 offsetAmount *= 0.5f;
298 }
299}
