108 template<typename StateType>
109 StateType* findStateWithType() {
110 auto queue = std::queue<StateNode*>();
111 queue.push(this);
112
113 while (!queue.empty()) {
114 auto node = queue.front();
115 queue.pop();
116
117 auto *thisState = dynamic_cast<StateType *>(node->state_.get());
118 if (thisState) {
119 return thisState;
120 }
121 for (auto &joined : node->state_->joined()) {
122 auto *joinedState = dynamic_cast<StateType *>(joined.get());
123 if (joinedState) {
124 return joinedState;
125 }
126 }
127
128 for (auto &child: node->childs_) {
129 queue.push(child.get());
130 }
131 }
132
133 return nullptr;
134 }
