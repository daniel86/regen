87void Demuxer::set_file(const std::string &file) {
88 // (re)open file
89 if (formatCtx_) {
90 avformat_close_input(&formatCtx_);
91 formatCtx_ = nullptr;
92 }
93 if (avformat_open_input(&formatCtx_, file.c_str(), nullptr, nullptr) != 0) {
94 throw Error(REGEN_STRING("Couldn't open file " << file));
95 }
96
97 // Retrieve stream information
98 if (avformat_find_stream_info(formatCtx_, nullptr) < 0) {
99 throw Error("Couldn't find stream information");
100 }
101
102 // Find the first video/audio stream
103 videoStreamIndex_ = -1;
104 audioStreamIndex_ = -1;
105 for (GLuint i = 0u; i < formatCtx_->nb_streams; ++i) {
106 auto stream = formatCtx_->streams[i];
107 auto codec_type = stream->codecpar->codec_type;
108 if (videoStreamIndex_ == -1 && codec_type == AVMEDIA_TYPE_VIDEO) { videoStreamIndex_ = i; }
109 else if (audioStreamIndex_ == -1 && codec_type == AVMEDIA_TYPE_AUDIO) { audioStreamIndex_ = i; }
110 }
111 if (videoStreamIndex_ != -1) {
112 videoStream_ = ref_ptr<VideoStream>::alloc(
113 formatCtx_->streams[videoStreamIndex_], videoStreamIndex_, 100);
114 }
115 if (audioStreamIndex_ != -1) {
116 audioStream_ = ref_ptr<AudioSource>::alloc(
117 formatCtx_->streams[audioStreamIndex_], audioStreamIndex_, -1);
118 }
119}

160void Demuxer::seekTo(GLdouble p) {
161 if (!formatCtx_) { return; }
162 p = std::max(0.0, std::min(1.0, p));
163 seek_.isRequired = GL_TRUE;
164 seek_.flags &= ~AVSEEK_FLAG_BYTE;
165 seek_.rel = 0;
166 seek_.pos = p * formatCtx_->duration;
167 if (formatCtx_->start_time != AV_NOPTS_VALUE) {
168 seek_.pos += formatCtx_->start_time;
169 }
170}

172GLboolean Demuxer::decode() {
173 AVPacket packet;
174
175 SeekPosition seek = seek_;
176 seek_.isRequired = GL_FALSE;
177 if (seek.isRequired) {
178 int64_t seek_min = seek.rel > 0 ? seek.pos - seek.rel + 2 : INT64_MIN;
179 int64_t seek_max = seek.rel < 0 ? seek.pos - seek.rel - 2 : INT64_MAX;
180
181 int ret = avformat_seek_file(formatCtx_,
182 -1, seek_min, seek.pos, seek_max, seek.flags);
183 if (ret < 0) {
184 REGEN_ERROR("seeking failed");
185 } else {
186 clearQueue();
187 seeked_ = GL_TRUE;
188 }
189 } else if (pauseFlag_) {
190 return GL_TRUE;
191 }
192
193 if (av_read_frame(formatCtx_, &packet) < 0) {
194 // end of stream reached
195 if (repeatStream_) {
196 seekTo(0.0);
197 av_packet_unref(&packet);
198 return GL_FALSE;
199 } else {
200 stop();
201 return GL_TRUE;
202 }
203 }
204
205 // Is this a packet from the video stream?
206 if (packet.stream_index == videoStreamIndex_) {
207 videoStream_->decode(&packet);
208 } else if (packet.stream_index == audioStreamIndex_) {
209 audioStream_->decode(&packet);
210 } else {
211 // Free the packet that was allocated by av_read_frame
212 av_packet_unref(&packet);
213 }
214
215 return GL_FALSE;
216}
