192void TextureState::set_texture(const ref_ptr<Texture> &tex) {
193 texture_ = tex;
194 samplerType_ = tex->samplerType();
195 if (tex.get()) {
196 set_name(REGEN_STRING("Texture" << tex->id()));
197 shaderDefine(REGEN_TEX_NAME("TEX_SAMPLER_TYPE"), tex->samplerType());
198 shaderDefine(REGEN_TEX_NAME("TEX_DIM"), REGEN_STRING(tex->numComponents()));
199 shaderDefine(REGEN_TEX_NAME("TEX_TEXEL_X"), REGEN_STRING(1.0 / tex->width()));
200 shaderDefine(REGEN_TEX_NAME("TEX_TEXEL_Y"), REGEN_STRING(1.0 / tex->height()));
201 shaderDefine(REGEN_TEX_NAME("TEX_WIDTH"), REGEN_STRING(tex->width()));
202 shaderDefine(REGEN_TEX_NAME("TEX_HEIGHT"), REGEN_STRING(tex->height()));
203 }
204}

365void TextureState::enable(RenderState *rs) {
366 lastTexChannel_ = texture_->channel();

369 }
370 State::enable(rs);
