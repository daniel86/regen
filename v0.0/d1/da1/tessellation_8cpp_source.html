36 std::vector<TriangleFace> tessellate(GLuint lod, std::vector<TriangleFace> &in) {
37 GLuint inFaces = in.size();
38 // Each triangle is divided in 4 smaller triangles
39 // for each LoD level.
40 GLuint outFaces = pow(4.0, lod) * inFaces;
41 std::vector<TriangleFace> out(outFaces);
42 GLuint counter = inFaces;
43 TriangleVertex pa, pb, pc;
44 GLuint i, j;
45
46 GLuint lastIndex = 0;
47 for (i = 0; i < inFaces; ++i) {
48 out[i] = in[i];
49
50 auto *vertices = (TriangleVertex *) &in[i];
51 for (j = 0; j < 3; ++j) {
52 if (lastIndex < vertices[j].i) {
53 lastIndex = vertices[j].i;
54 }
55 }
56 }
57
58 for (j = 0; j < lod; ++j) {
59 const GLuint lastCount = counter;
60 for (i = 0; i < lastCount; ++i) {
61 pa.p = (out[i].v1.p + out[i].v2.p) * 0.5;
62 pb.p = (out[i].v2.p + out[i].v3.p) * 0.5;
63 pc.p = (out[i].v3.p + out[i].v1.p) * 0.5;
64 pa.i = lastIndex + 1;
65 pb.i = lastIndex + 2;
66 pc.i = lastIndex + 3;
67 lastIndex += 3;
68
69 out[counter++] = TriangleFace(out[i].v1, pa, pc);
70 out[counter++] = TriangleFace(pa, out[i].v2, pb);
71 out[counter++] = TriangleFace(pb, out[i].v3, pc);
72 out[i] = TriangleFace(pa, pb, pc);
73 }
74 }
75
76 return out;
77 }

79 void tessellate(GLuint lod, Tessellation &tess) {
80 GLuint inFaces = tess.inputFaces.size();
81 // Each triangle is divided in 4 smaller triangles
82 // for each LoD level.
83 GLuint outFaces = pow(4.0, lod) * inFaces;
84 tess.outputFaces.resize(outFaces);
85 GLuint counter = inFaces;
86 GLuint pa_i, pb_i, pc_i;
87 GLuint i, j;
88
89 std::unordered_map<Edge, GLuint> edgeMidpointMap;
90 GLuint lastIndex = tess.vertices.size() - 1;
91
92 for (i = 0; i < inFaces; ++i) {
93 tess.outputFaces[i] = tess.inputFaces[i];
94 }
95
96 for (j = 0; j < lod; ++j) {
97 const GLuint lastCount = counter;
98 for (i = 0; i < lastCount; ++i) {
99 auto &face = tess.outputFaces[i];
100 auto addMidpoint = [&](GLuint v1, GLuint v2) {
101 Edge edge = {v1, v2};
102 if (edgeMidpointMap.find(edge) == edgeMidpointMap.end()) {
103 tess.vertices.push_back((tess.vertices[v1] + tess.vertices[v2]) * 0.5);
104 edgeMidpointMap[edge] = ++lastIndex;
105 }
106 return edgeMidpointMap[edge];
107 };
108
109 pa_i = addMidpoint(face.v1, face.v2);
110 pb_i = addMidpoint(face.v2, face.v3);
111 pc_i = addMidpoint(face.v3, face.v1);
112
113 tess.outputFaces[counter++] = TessellationFace(face.v1, pa_i, pc_i);
114 tess.outputFaces[counter++] = TessellationFace(pa_i, face.v2, pb_i);
115 tess.outputFaces[counter++] = TessellationFace(pb_i, face.v3, pc_i);
116 tess.outputFaces[i] = TessellationFace(pa_i, pb_i, pc_i);
117 }
118 }
119 }
