643 inline Mat4f inverse() const {
644 // Compute the reciprocal determinant
645 float det = determinant();
646 if (det == 0.0f) {
647 // Matrix not invertible.
648 return identity();
649 }
650
651 float invdet = 1.0f / det;
652
653 Mat4f res;
654 res.x[0] = invdet * (
655 x[5] * (x[10] * x[15] - x[11] * x[14]) +
656 x[6] * (x[11] * x[13] - x[9] * x[15]) +
657 x[7] * (x[9] * x[14] - x[10] * x[13]));
658
659 res.x[1] = -invdet * (
660 x[1] * (x[10] * x[15] - x[11] * x[14]) +
661 x[2] * (x[11] * x[13] - x[9] * x[15]) +
662 x[3] * (x[9] * x[14] - x[10] * x[13]));
663
664 res.x[2] = invdet * (
665 x[1] * (x[6] * x[15] - x[7] * x[14]) +
666 x[2] * (x[7] * x[13] - x[5] * x[15]) +
667 x[3] * (x[5] * x[14] - x[6] * x[13]));
668
669 res.x[3] = -invdet * (
670 x[1] * (x[6] * x[11] - x[7] * x[10]) +
671 x[2] * (x[7] * x[9] - x[5] * x[11]) +
672 x[3] * (x[5] * x[10] - x[6] * x[9]));
673
674 res.x[4] = -invdet * (
675 x[4] * (x[10] * x[15] - x[11] * x[14]) +
676 x[6] * (x[11] * x[12] - x[8] * x[15]) +
677 x[7] * (x[8] * x[14] - x[10] * x[12]));
678
679 res.x[5] = invdet * (
680 x[0] * (x[10] * x[15] - x[11] * x[14]) +
681 x[2] * (x[11] * x[12] - x[8] * x[15]) +
682 x[3] * (x[8] * x[14] - x[10] * x[12]));

684 res.x[6] = -invdet * (
685 x[0] * (x[6] * x[15] - x[7] * x[14]) +
686 x[2] * (x[7] * x[12] - x[4] * x[15]) +
687 x[3] * (x[4] * x[14] - x[6] * x[12]));
688
689 res.x[7] = invdet * (
690 x[0] * (x[6] * x[11] - x[7] * x[10]) +
691 x[2] * (x[7] * x[8] - x[4] * x[11]) +
692 x[3] * (x[4] * x[10] - x[6] * x[8]));

694 res.x[8] = invdet * (
695 x[4] * (x[9] * x[15] - x[11] * x[13]) +
696 x[5] * (x[11] * x[12] - x[8] * x[15]) +
697 x[7] * (x[8] * x[13] - x[9] * x[12]));

699 res.x[9] = -invdet * (
700 x[0] * (x[9] * x[15] - x[11] * x[13]) +
701 x[1] * (x[11] * x[12] - x[8] * x[15]) +
702 x[3] * (x[8] * x[13] - x[9] * x[12]));

705 x[0] * (x[5] * x[15] - x[7] * x[13]) +
706 x[1] * (x[7] * x[12] - x[4] * x[15]) +
707 x[3] * (x[4] * x[13] - x[5] * x[12]));

709 res.x[11] = -invdet * (

711 x[1] * (x[7] * x[8] - x[4] * x[11]) +
712 x[3] * (x[4] * x[9] - x[5] * x[8]));
713
714 res.x[12] = -invdet * (
715 x[4] * (x[9] * x[14] - x[10] * x[13]) +
716 x[5] * (x[10] * x[12] - x[8] * x[14]) +
717 x[6] * (x[8] * x[13] - x[9] * x[12]));

720 x[0] * (x[9] * x[14] - x[10] * x[13]) +
721 x[1] * (x[10] * x[12] - x[8] * x[14]) +
722 x[2] * (x[8] * x[13] - x[9] * x[12]));
