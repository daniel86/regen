38void Particles::begin(ShaderInputContainer::DataLayout layout) {
39 HasInput::begin(layout);
40
41 GLuint numParticles = inputContainer()->numVertices();
42
43 // Initialize the random number generator and distribution
44 std::random_device rd;
45 std::mt19937 gen(rd());
46 std::uniform_int_distribution<> dis(0, std::numeric_limits<int>::max());
47
48 // get a random seed for each particle
49 ref_ptr<ShaderInput1ui> randomSeed_ = ref_ptr<ShaderInput1ui>::alloc("randomSeed");
50 randomSeed_->setVertexData(numParticles, nullptr);
51 for (GLuint i = 0u; i < numParticles; ++i) {
52 randomSeed_->setVertex(i, dis(gen));
53 }
54 setInput(randomSeed_);
55
56 // initially set lifetime to zero so that particles
57 // get emitted in the first step
58 ref_ptr<ShaderInput1f> lifetimeInput_ = ref_ptr<ShaderInput1f>::alloc("lifetime");
59 lifetimeInput_->setVertexData(numParticles, nullptr);
60 for (GLuint i = 0u; i < numParticles; ++i) {
61 lifetimeInput_->setVertex(i, -1.0);
62 }
63 setInput(lifetimeInput_);
64}

66VBOReference Particles::end() {
67 ShaderInputList particleInputs = inputContainer()->uploadInputs();
68
69 particleRef_ = HasInput::end();
70 feedbackRef_ = feedbackBuffer_->alloc(particleRef_->allocatedSize());
71 bufferRange_.size_ = particleRef_->allocatedSize();
72
73 // Create shader defines.
74 GLuint counter = 0;
75 for (auto it = particleInputs.begin(); it != particleInputs.end(); ++it) {
76 if (!it->in_->isVertexAttribute()) continue;
77 shaderDefine(
78 REGEN_STRING("PARTICLE_ATTRIBUTE" << counter << "_TYPE"),
79 glenum::glslDataType(it->in_->dataType(), it->in_->valsPerElement()));
80 shaderDefine(
81 REGEN_STRING("PARTICLE_ATTRIBUTE" << counter << "_NAME"),
82 it->in_->name());
83 if (advanceModes_.find(it->in_->name()) != advanceModes_.end()) {
84 configureAdvancing(it->in_, counter, advanceModes_[it->in_->name()]);
85 }
86 counter += 1;
87 REGEN_DEBUG("Particle attribute '" << it->in_->name() << "' added.");
88 }
89 shaderDefine("NUM_PARTICLE_ATTRIBUTES", REGEN_STRING(counter));
90 createUpdateShader(particleInputs);
91
92 for (auto &particleInput: particleInputs) {
93 const ref_ptr<ShaderInput> in = particleInput.in_;
94 if (!in->isVertexAttribute()) continue;
95 GLint loc = updateState_->shader()->attributeLocation(particleInput.in_->name());
96 if (loc == -1) continue;
97 particleAttributes_.emplace_back(in, loc);
98 }
99 // start with zero emitted particles
100 //inputContainer_->set_numVertices(0);
101
102 return particleRef_;
103}

114void Particles::configureAdvancing(
115 const ref_ptr<ShaderInput> &in,
116 GLuint counter,
117 AdvanceMode mode) {
118 std::string advanceImportKey;
119 std::string advanceFunction;
120 GLfloat advanceFactor = 1.0f;
121 switch (mode) {
122 case ADVANCE_MODE_CUSTOM: {
123 auto needle = advanceFunctions_.find(in->name());
124 if (needle == advanceFunctions_.end()) {
125 REGEN_WARN("No custom advance function found for '" << in->name() << "'.");
126 return;
127 }
128 advanceImportKey = needle->second;
129 // get string after the last dot, convention is that this is the function name
130 advanceFunction = advanceImportKey.substr(advanceImportKey.find_last_of('.') + 1);
131 // get the factor for this attribute
132 auto factorNeedle = advanceFactors_.find(in->name());
133 if (factorNeedle != advanceFactors_.end()) {
134 advanceFactor = factorNeedle->second;
135 }
136 break;
137 }
138 case ADVANCE_MODE_SRC:
139 advanceImportKey = "regen.states.blending.src";
140 advanceFunction = "blend_src";
141 break;
142 case ADVANCE_MODE_MULTIPLY:
143 advanceImportKey = "regen.states.blending.multiply";
144 advanceFunction = "blend_multiply";
145 break;
146 case ADVANCE_MODE_ADD:
147 advanceImportKey = "regen.states.blending.add";
148 advanceFunction = "blend_add";
149 break;
150 case ADVANCE_MODE_SMOOTH_ADD:
151 advanceImportKey = "regen.states.blending.smooth_add";
152 advanceFunction = "blend_smooth_add";
153 break;
154 case ADVANCE_MODE_SUBTRACT:
155 advanceImportKey = "regen.states.blending.sub";
156 advanceFunction = "blend_sub";
157 break;
158 case ADVANCE_MODE_REVERSE_SUBTRACT:
159 advanceImportKey = "regen.states.blending.reverse_sub";
160 advanceFunction = "blend_reverse_sub";
161 break;
162 case ADVANCE_MODE_DIFFERENCE:
163 advanceImportKey = "regen.states.blending.difference";
164 advanceFunction = "blend_difference";
165 break;
166 case ADVANCE_MODE_LIGHTEN:
167 advanceImportKey = "regen.states.blending.lighten";
168 advanceFunction = "blend_lighten";
169 break;
170 case ADVANCE_MODE_DARKEN:
171 advanceImportKey = "regen.states.blending.darken";
172 advanceFunction = "blend_darken";
173 break;
174 case ADVANCE_MODE_MIX:
175 advanceImportKey = "regen.states.blending.mix";
176 advanceFunction = "blend_mix";
177 break;
178 }
179 // create a new shader input for the advance function
180 shaderDefine(
181 REGEN_STRING("PARTICLE_ATTRIBUTE" << counter << "_ADVANCE_FUNCTION"),
182 advanceFunction);
183 shaderDefine(
184 REGEN_STRING("PARTICLE_ATTRIBUTE" << counter << "_ADVANCE_KEY"),
185 advanceImportKey);
186 shaderDefine(
187 REGEN_STRING("PARTICLE_ATTRIBUTE" << counter << "_ADVANCE_FACTOR"),
188 REGEN_STRING(advanceFactor));
189
190 // configure the ramp texture
191 auto rampNeedle = ramps_.find(in->name());
192 if (rampNeedle != ramps_.end()) {
193 auto &rampTexture = rampNeedle->second.texture;
194 auto &rampMode = rampNeedle->second.mode;
195 std::string rampImportKey, rampFunction;
196 switch (rampMode) {
197 case RAMP_MODE_CUSTOM: {
198 auto needle = rampFunctions_.find(in->name());
199 if (needle == rampFunctions_.end()) {
200 REGEN_WARN("No custom ramp function found for '" << in->name() << "'.");
201 return;
202 }
203 rampImportKey = needle->second;
204 rampFunction = advanceImportKey.substr(rampImportKey.find_last_of('.') + 1);
205 break;
206 }
207 case RAMP_MODE_TIME:
208 rampImportKey = "regen.particles.ramp.time";
209 rampFunction = "ramp_time";
210 break;
211 case RAMP_MODE_LIFETIME:
212 rampImportKey = "regen.particles.ramp.lifetime";
213 rampFunction = "ramp_lifetime";
214 break;
215 case RAMP_MODE_EMITTER_DISTANCE:
216 rampImportKey = "regen.particles.ramp.emitter_distance";
217 rampFunction = "ramp_emitter_distance";
218 break;
219 case RAMP_MODE_CAMERA_DISTANCE:
220 rampImportKey = "regen.particles.ramp.camera_distance";
221 rampFunction = "ramp_camera_distance";
222 break;
223 case RAMP_MODE_VELOCITY:
224 rampImportKey = "regen.particles.ramp.velocity";
225 rampFunction = "ramp_velocity";
226 break;
227 }
228 shaderDefine(
229 REGEN_STRING("PARTICLE_ATTRIBUTE" << counter << "_RAMP_KEY"),
230 rampImportKey);
231 shaderDefine(
232 REGEN_STRING("PARTICLE_ATTRIBUTE" << counter << "_RAMP_FUNCTION"),
233 rampFunction);
234 // finally create a texture state for the ramp texture
235 auto rampTextureState = ref_ptr<TextureState>::alloc();
236 rampTextureState->set_texture(rampTexture);
237 rampTextureState->set_name(REGEN_STRING(in->name() << "Ramp"));
238 joinStates(rampTextureState);
239 shaderDefine(
240 REGEN_STRING("PARTICLE_ATTRIBUTE" << counter << "_RAMP_TEXEL_SIZE"),
241 REGEN_STRING(1.0 / rampTexture->width()));
242 }

245void Particles::createUpdateShader(const ShaderInputList &inputs) {
246 StateConfigurer shaderConfigurer;
247 shaderConfigurer.addState(animationState_.get());
248 shaderConfigurer.addState(this);
249
250 StateConfig &shaderCfg = shaderConfigurer.cfg();
251 shaderCfg.feedbackAttributes_.clear();
252 for (const auto &input: inputs) {
253 if (!input.in_->isVertexAttribute()) continue;
254 shaderCfg.feedbackAttributes_.push_back(input.in_->name());
255 }
256 shaderCfg.feedbackMode_ = GL_INTERLEAVED_ATTRIBS;
257 shaderCfg.feedbackStage_ = GL_VERTEX_SHADER;
258 updateState_->createShader(shaderCfg, updateShaderKey_);
259 shaderCfg.feedbackAttributes_.clear();
260}

262void Particles::glAnimate(RenderState *rs, GLdouble dt) {
263 rs->toggles().push(RenderState::RASTARIZER_DISCARD, GL_TRUE);
264 updateState_->enable(rs);
265
266 rs->vao().push(particleVAO_.id());
267 glBindBuffer(GL_ARRAY_BUFFER, particleRef_->bufferID());
268 for (auto &particleAttribute: particleAttributes_) {
269 particleAttribute.input->enableAttribute(particleAttribute.location);
270 }
271
272 bufferRange_.buffer_ = feedbackRef_->bufferID();
273 bufferRange_.offset_ = feedbackRef_->address();
274 rs->feedbackBufferRange().push(0, bufferRange_);
275 rs->beginTransformFeedback(GL_POINTS);
276 // bind the atomic counter for computing the bounding box to fixed location 0
277 //boundingBoxCounter_->bindBufferBase(0);
278
279 /*
280 // only emit a limited number of particles per frame
281 if (inputContainer_->numVertices() < numParticles_) {
282 GLuint nextNumParticles = inputContainer_->numVertices() + maxEmits_;
283 if (nextNumParticles > numParticles_) {
284 inputContainer_->set_numVertices(numParticles_);
285 } else {
286 inputContainer_->set_numVertices(nextNumParticles);
287 }
288 }
289 */
290 glDrawArrays(primitive_, 0, inputContainer_->numVertices());
291
292 rs->endTransformFeedback();
293 rs->feedbackBufferRange().pop(0);
294
295 rs->vao().pop();
296 updateState_->disable(rs);
297 rs->toggles().pop(RenderState::RASTARIZER_DISCARD);
298
299 // Update particle attribute layout.
300 GLuint currOffset = bufferRange_.offset_;
301 for (auto &particleAttribute: particleAttributes_) {
302 particleAttribute.input->set_buffer(bufferRange_.buffer_, feedbackRef_);
303 particleAttribute.input->set_offset(currOffset);
304 currOffset += particleAttribute.input->elementSize();
305 }
306 // And update the VAO so that next drawing uses last feedback result.
307 std::set<Mesh *> particleMeshes;
308 getMeshViews(particleMeshes);
309 for (auto particleMesh: particleMeshes) { particleMesh->updateVAO(rs); }
310
311 // Ping-Pong VBO references so that next feedback goes to other buffer.
312 VBOReference buf = particleRef_;
313 particleRef_ = feedbackRef_;
314 feedbackRef_ = buf;
315
316 // Read atomic counter to get the bounding box of the particles
317 //auto bounds = boundingBoxCounter_->updateBounds();
318 //set_bounds(bounds.min, bounds.max);
319
320 GL_ERROR_LOG();
321}
