36 static ref_ptr<Texture> getTexture(
37 SceneParser *parser, SceneInputNode &input,
38 const string &idKey = "id",
39 const string &bufferKey = "fbo",
40 const string &attachmentKey = "attachment") {
41 ref_ptr<Texture> tex;
42 // Find the texture resource
43 if (input.hasAttribute(idKey)) {
44 tex = parser->getResources()->getTexture(parser, input.getValue(idKey));
45 } else if (input.hasAttribute(bufferKey)) {
46 ref_ptr<FBO> fbo = parser->getResources()->getFBO(parser, input.getValue(bufferKey));
47 if (fbo.get() == nullptr) {
48 REGEN_WARN("Unable to find FBO '" << input.getValue(bufferKey) <<
49 "' for " << input.getDescription() << ".");
50 return tex;
51 }
52 const auto val = input.getValue<string>(attachmentKey, "0");
53 if (val == "depth") {
54 tex = fbo->depthTexture();
55 } else {
56 vector<ref_ptr<Texture> > &textures = fbo->colorTextures();
57
58 unsigned int attachment;
59 stringstream ss(val);
60 ss >> attachment;
61
62 if (attachment < textures.size()) {
63 tex = textures[attachment];
64 } else {
65 REGEN_WARN("Invalid attachment '" << val <<
66 "' for " << input.getDescription() << ".");
67 }
68 }
69 }
70 return tex;
71 }
