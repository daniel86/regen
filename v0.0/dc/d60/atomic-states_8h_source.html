348 class ClearColorState : public ServerSideState {
349 public:
353 struct Data {
355 Vec4f clearColor;
357 DrawBuffers colorBuffers;
358 };
360 std::list<Data> data;
361
363 explicit ClearColorState(const ref_ptr<FBO> &fbo)
364 : ServerSideState(), fbo_(fbo) {}
365
366 // override
367 void enable(RenderState *rs) override {
368 for (std::list<Data>::iterator it = data.begin(); it != data.end(); ++it) {
369 if (!rs->drawFrameBuffer().isLocked()) {
370 fbo_->drawBuffers().push(it->colorBuffers);
371 rs->clearColor().push(it->clearColor);
372 glClear(GL_COLOR_BUFFER_BIT);
373 rs->clearColor().pop();
374 fbo_->drawBuffers().pop();
375 }
376 }
377 }
378
379 protected:
380 ref_ptr<FBO> fbo_;
381 };
