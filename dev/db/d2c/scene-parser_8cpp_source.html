113void SceneParser::loadShapes() {
114 auto root = getRoot();
115 auto dummy = ref_ptr<State>::alloc();
116
117 // load shapes that are globally defined
118 auto shapeNodes = root->getChildren("shape");
119 for (auto &shapeNode: shapeNodes) {
120 processState(dummy, "shape", shapeNode);
121 }
122 // load meshes and associated shapes
123 auto meshNodes = root->getChildren("mesh");
124 for (auto &meshNode: meshNodes) {
125 auto meshID = meshNode->getName();
126 auto meshVec = resources_->getMesh(this, meshID);
127 if (!meshVec.get()) {
128 REGEN_WARN("Could not load mesh with id '" << meshID << "'.");
129 }
130 }
131}

205void SceneParser::processNode(
206 const ref_ptr<StateNode> &parent,
207 const string &nodeName,
208 const string &nodeCategory) {
209 ref_ptr<NodeProcessor> processor = getNodeProcessor(nodeCategory);
210 if (processor.get() == nullptr) {
211 REGEN_WARN("No Processor registered for node category '" << nodeCategory << "'.");
212 return;
213 }
214 ref_ptr<SceneInputNode> input = getRoot()->getFirstChild(nodeCategory, nodeName);
215 if (input.get() == nullptr) {
216 REGEN_WARN("No input for node category '" <<
217 nodeCategory << "' and node name '" << nodeName << "'.");
218 return;
219 }
220 processor->processInput(this, *input.get(), parent);
221}

248vector<AnimRange> SceneParser::getAnimationRanges(const std::string &assetID) {
249 ref_ptr<SceneInputNode> root = getRoot();
250 ref_ptr<SceneInputNode> importer = root->getFirstChild("asset", assetID);
251 if (importer.get() == nullptr) {
252 REGEN_WARN("No asset with id '" << assetID << "' known.");
253 return {};
254 } else {
255 const list<ref_ptr<SceneInputNode> > &childs = importer->getChildren();
256
257 GLuint animRangeCount = 0u;
258 for (auto it = childs.begin(); it != childs.end(); ++it) {
259 if ((*it)->getCategory() == "anim-range") animRangeCount += 1;
260 }
261
262 vector<AnimRange> out(animRangeCount);
263 animRangeCount = 0u;
264
265 for (auto it = childs.begin(); it != childs.end(); ++it) {
266 ref_ptr<SceneInputNode> n = *it;
267 if (n->getCategory() != "anim-range") continue;
268 out[animRangeCount].name = n->getValue("name");
269 out[animRangeCount].range = n->getValue<Vec2d>("range", Vec2d(0.0));
270 out[animRangeCount].channelName = n->getValue("channel");
271 out[animRangeCount].channelIndex = n->getValue<GLuint>("index", 0u);
272 animRangeCount += 1u;
273 }
274
275 return out;
276 }
