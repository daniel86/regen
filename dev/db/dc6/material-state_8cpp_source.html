98void Material::set_texture(const ref_ptr<TextureState> &texState, TextureState::MapTo mapTo) {
99 texState->set_mapping(TextureState::MAPPING_TEXCO);
100 texState->set_mapTo(TextureState::MAP_TO_CUSTOM);
101 switch (mapTo) {
102 case TextureState::MAP_TO_HEIGHT:
103 // add the offset to the original position
104 texState->set_blendMode(BLEND_MODE_ADD);
105 // map height map values from [0,1] to [0,maxOffset]
106 texState->set_blendFactor(maxOffset_);
107 if (heightMapMode_ == HEIGHT_MAP_VERTEX) {
108 // only modify vertex position in case no height mapping via UV transfer is used
109 texState->set_mapTo(mapTo);
110 }
111 break;
112 case TextureState::MAP_TO_DISPLACEMENT:
113 // add the offset to the original position
114 texState->set_blendMode(BLEND_MODE_ADD);
115 // map height map values from [0,1] to [0,maxOffset]
116 texState->set_blendFactor(maxOffset_);
117 break;
118 case TextureState::MAP_TO_COLOR:
119 case TextureState::MAP_TO_DIFFUSE:
120 texState->set_blendMode(colorBlendMode_);
121 texState->set_blendFactor(colorBlendFactor_);
122 texState->set_mapTo(mapTo);
123 break;
124 default:
125 texState->set_blendMode(BLEND_MODE_SRC);
126 texState->set_blendFactor(1.0f);
127 texState->set_mapTo(mapTo);
128 // if the material has a height map which is mapped to UV transfer, then every
129 // texture should use the same UV transfer
130 if (textures_.find(TextureState::MAP_TO_HEIGHT) != textures_.end()) {
131 if (heightMapMode_ == HEIGHT_MAP_RELIEF) {
132 texState->set_texcoTransfer(TextureState::TRANSFER_TEXCO_RELIEF);
133 } else if (heightMapMode_ == HEIGHT_MAP_PARALLAX) {
134 texState->set_texcoTransfer(TextureState::TRANSFER_TEXCO_PARALLAX);
135 } else if (heightMapMode_ == HEIGHT_MAP_PARALLAX_OCCLUSION) {
136 texState->set_texcoTransfer(TextureState::TRANSFER_TEXCO_PARALLAX_OCC);
137 }
138 }
139 break;
140 }
141 joinStates(texState);
142}

144bool Material::getMapTo(std::string_view fileName, TextureState::MapTo &mapTo) {
145 if (fileName.find("diffuse") != std::string::npos) {
146 mapTo = TextureState::MAP_TO_DIFFUSE;
147 } else if (fileName.find("color") != std::string::npos) {
148 mapTo = TextureState::MAP_TO_COLOR;
149 } else if (fileName.find("ambient") != std::string::npos) {
150 mapTo = TextureState::MAP_TO_AMBIENT;
151 } else if (fileName.find("specular") != std::string::npos) {
152 mapTo = TextureState::MAP_TO_SPECULAR;
153 } else if (fileName.find("emission") != std::string::npos) {
154 mapTo = TextureState::MAP_TO_EMISSION;
155 } else if (fileName.find("normal") != std::string::npos) {
156 mapTo = TextureState::MAP_TO_NORMAL;
157 } else if (fileName.find("height") != std::string::npos) {
158 mapTo = TextureState::MAP_TO_HEIGHT;
159 } else if (fileName.find("displacement") != std::string::npos) {
160 mapTo = TextureState::MAP_TO_DISPLACEMENT;
161 } else if (fileName.find("displace") != std::string::npos) {
162 mapTo = TextureState::MAP_TO_DISPLACEMENT;
163 } else if (fileName.find("occlusion") != std::string::npos) {
164 mapTo = TextureState::MAP_TO_LIGHT;
165 } else {
166 return false;
167 }
168 return true;
169
170}

176bool Material::set_textures(std::string_view materialName, std::string_view variant) {
177 // find the base path with the textures
178 auto basePath0 = REGEN_STRING("res/textures/materials/" << materialName << "/" << variant);
179 basePath0 = resourcePath(basePath0);
180 if (!boost::filesystem::exists(basePath0)) {
181 basePath0 = REGEN_STRING("res/textures/" << materialName << "/" << variant);
182 basePath0 = resourcePath(basePath0);
183 }
184 if (!boost::filesystem::exists(basePath0)) {
185 return false;
186 }
187
188 // iterate over the texture files in the directory
189 for (auto &entry: boost::filesystem::directory_iterator(basePath0)) {
190 auto &filePath = entry.path().string();
191 auto fileName = entry.path().filename().string();
192 TextureState::MapTo mapTo = TextureState::MAP_TO_CUSTOM;
193 if (!getMapTo(fileName, mapTo)) {
194 continue;
195 }
196 auto tex = textures::load(
197 filePath,
198 mipmapFlag_,
199 forcedInternalFormat_,
200 forcedFormat_,
201 forcedSize_);
202 if (tex.get() != nullptr) {
203 auto texName = REGEN_STRING("materialTexture" << textures_.size());
204 auto texState = ref_ptr<TextureState>::alloc(tex, texName);
205 textures_[mapTo].push_back(texState);
206
207 if (wrapping_.has_value()) {
208 tex->begin(RenderState::get(), 0);
209 tex->wrapping().push(GL_CLAMP_TO_EDGE);
210 tex->end(RenderState::get());
211 }
212 }
213 }
214
215 for (auto &pair: textures_) {
216 for (auto &tex: pair.second) {
217 set_texture(tex, pair.first);
218 }
219 }
220 return true;
221}

336 std::ostream &operator<<(std::ostream &out, const Material::HeightMapMode &mode) {
337 switch (mode) {
338 case Material::HEIGHT_MAP_VERTEX:
339 return out << "VERTEX";
340 case Material::HEIGHT_MAP_RELIEF:
341 return out << "RELIEF";
342 case Material::HEIGHT_MAP_PARALLAX:
343 return out << "PARALLAX";
344 case Material::HEIGHT_MAP_PARALLAX_OCCLUSION:
345 return out << "PARALLAX_OCCLUSION";
346 }
347 return out;
348 }
349
350 std::istream &operator>>(std::istream &in, Material::HeightMapMode &mode) {
351 std::string val;
352 in >> val;
353 boost::to_lower(val);
354 if (val == "vertex") mode = Material::HEIGHT_MAP_VERTEX;
355 else if (val == "relief") mode = Material::HEIGHT_MAP_RELIEF;
356 else if (val == "parallax") mode = Material::HEIGHT_MAP_PARALLAX;
357 else if (val == "parallax_occlusion") mode = Material::HEIGHT_MAP_PARALLAX_OCCLUSION;
358 else {
359 REGEN_WARN("Unknown Height Map Mode '" << val <<
360 "'. Using default VERTEX Mapping.");
361 mode = Material::HEIGHT_MAP_VERTEX;
362 }
363 return in;
364 }

367ref_ptr<Material> Material::load(LoadingContext &ctx, scene::SceneInputNode &input) {
368 ref_ptr<Material> mat = ref_ptr<Material>::alloc();
369
370 if (input.hasAttribute("max-offset")) {
371 mat->set_maxOffset(input.getValue<GLfloat>("max-offset", 0.1f));
372 }
373 if (input.hasAttribute("height-map-mode")) {
374 mat->set_heightMapMode(input.getValue<Material::HeightMapMode>("height-map-mode",
375 Material::HEIGHT_MAP_VERTEX));
376 }
377 if (input.hasAttribute("color-blend-mode")) {
378 mat->set_colorBlendMode(input.getValue<BlendMode>("color-blend-mode", BLEND_MODE_SRC));
379 }
380 if (input.hasAttribute("color-blend-factor")) {
381 mat->set_colorBlendFactor(input.getValue<GLfloat>("color-blending-factor", 1.0f));
382 }
383
384
385 if (input.hasAttribute("asset")) {
386 ref_ptr<AssetImporter> assetLoader =
387 ctx.scene()->getResource<AssetImporter>(input.getValue("asset"));
388 if (assetLoader.get() == nullptr) {
389 REGEN_WARN("Skipping unknown Asset for '" << input.getDescription() << "'.");
390 } else {
391 const std::vector<ref_ptr<Material> > materials = assetLoader->materials();
392 auto materialIndex = input.getValue<GLuint>("asset-index", 0u);
393 if (materialIndex >= materials.size()) {
394 REGEN_WARN("Invalid Material index '" << materialIndex <<
395 "' for Asset '" << input.getValue("asset") << "'.");
396 } else {
397 mat = materials[materialIndex];
398 }
399 }
400 } else if (input.hasAttribute("preset")) {
401 std::string presetVal(input.getValue("preset"));
402 auto variant = input.getValue<Material::Variant>("variant", 0);
403 if (presetVal == "jade") mat->set_jade(variant);
404 else if (presetVal == "ruby") mat->set_ruby(variant);
405 else if (presetVal == "chrome") mat->set_chrome(variant);
406 else if (presetVal == "gold") mat->set_gold(variant);
407 else if (presetVal == "copper") mat->set_copper(variant);
408 else if (presetVal == "silver") mat->set_silver(variant);
409 else if (presetVal == "pewter") mat->set_pewter(variant);
410 else if (presetVal == "iron") mat->set_iron(variant);
411 else if (presetVal == "steel") mat->set_steel(variant);
412 else if (presetVal == "metal") mat->set_metal(variant);
413 else if (presetVal == "leather") mat->set_leather(variant);
414 else if (presetVal == "stone") mat->set_stone(variant);
415 else if (presetVal == "wood") mat->set_wood(variant);
416 else if (presetVal == "marble") mat->set_marble(variant);
417 else {
418 mat->set_textures(presetVal, variant);
419 }
420 }
421 if (input.hasAttribute("ambient"))
422 mat->ambient()->setVertex(0,
423 input.getValue<Vec3f>("ambient", Vec3f(0.0f)));
424 if (input.hasAttribute("diffuse"))
425 mat->diffuse()->setVertex(0,
426 input.getValue<Vec3f>("diffuse", Vec3f(1.0f)));
427 if (input.hasAttribute("specular"))
428 mat->specular()->setVertex(0,
429 input.getValue<Vec3f>("specular", Vec3f(0.0f)));
430 if (input.hasAttribute("shininess"))
431 mat->shininess()->setVertex(0,
432 input.getValue<GLfloat>("shininess", 1.0f));
433 if (input.hasAttribute("emission"))
434 mat->set_emission(input.getValue<Vec3f>("emission", Vec3f(0.0f)));
435 if (input.hasAttribute("textures")) {
436 mat->set_textures(
437 input.getValue("textures"),
438 input.getValue<Material::Variant>("variant", 0));
439 }
440
441 mat->alpha()->setVertex(0,
442 input.getValue<GLfloat>("alpha", 1.0f));
443 mat->refractionIndex()->setVertex(0,
444 input.getValue<GLfloat>("refractionIndex", 0.95f));
445 mat->set_fillMode(glenum::fillMode(
446 input.getValue<std::string>("fill-mode", "FILL")));
447 if (input.getValue<bool>("two-sided", false)) {
448 // this conflicts with shadow mapping front face culling.
449 mat->set_twoSided(true);
450 }
451
452 return mat;
453}
