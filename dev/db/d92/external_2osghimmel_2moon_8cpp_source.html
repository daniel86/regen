402void Moon::opticalLibrations(
403 const t_julianDay t
404, t_longf &l /* librations in longitude */
405, t_longf &b /* librations in latitude */)
406{
407 // (AA.51.1)
408
409 const t_longf Dr = _rad(Earth::longitudeNutation(t));
410
411 const t_longf F = _rad(meanLatitude(t));
412 const t_longf O = _rad(meanOrbitLongitude(t));
413
414 const t_ecld ecl = position(t);
415 const t_longf lo = _rad(ecl.longitude);
416 const t_longf la = _rad(ecl.latitude);
417
418 static const t_longf I = _rad(1.54242);
419
420 const t_longf cos_la = cos(la);
421 const t_longf sin_la = sin(la);
422 const t_longf cos_I = cos(I);
423 const t_longf sin_I = sin(I);
424
425 const t_longf W = _rev(lo - Dr - O);
426 const t_longf sin_W = sin(W);
427
428 const t_longf A = _rev(atan2(sin_W * cos_la * cos_I - sin_la * sin_I, cos(W) * cos_la));
429
430 l = _deg(A - F);
431 b = _deg(asin(-sin_W * cos_la * sin_I - sin_la * cos_I));
432}

435const t_longf Moon::parallacticAngle(
436 const t_aTime &aTime
437, const t_longf latitude
438, const t_longf longitude)
439{
440 // (AA.13.1)
441
442 const t_julianDay t(jd(aTime));
443
444 const t_longf la = _rad(latitude);
445 const t_longf lo = _rad(longitude);
446
447 const t_equd pos = apparentPosition(t);
448 const t_longf ra = _rad(pos.right_ascension);
449 const t_longf de = _rad(pos.declination);
450
451 const t_longf s = _rad(siderealTime(aTime));
452
453 // (AA.p88) - local hour angle
454
455 const t_longf H = s + lo - ra;
456
457 const t_longf cos_la = cos(la);
458 const t_longf P = atan2(sin(H) * cos_la, sin(la) * cos(de) - sin(de) * cos_la * cos(H));
459
460 return _deg(P);
461}

464const t_longf Moon::positionAngleOfAxis(const t_julianDay t)
465{
466 // (AA.p344)
467
468 const t_equd pos = apparentPosition(t);
469
470 const t_longf a = _rad(pos.right_ascension);
471 const t_longf e = _rad(Earth::meanObliquity(t));
472
473 const t_longf Dr = _rad(Earth::longitudeNutation(t));
474 const t_longf O = _rad(meanOrbitLongitude(t));
475
476 const t_longf V = O + Dr;
477
478 static const t_longf I = _rad(1.54242);
479 const t_longf sin_I = sin(I);
480
481 const t_longf X = sin_I * sin(V);
482 const t_longf Y = sin_I * cos(V) * cos(e) - cos(I) * sin(e);
483
484 // optical libration in latitude
485
486 const t_ecld ecl = position(t);
487
488 const t_longf lo = _rad(ecl.longitude);
489 const t_longf la = _rad(ecl.latitude);
490
491 const t_longf W = _rev(lo - Dr - O);
492 const t_longf b = asin(-sin(W) * cos(la) * sin_I - sin(la) * cos(I));
493
494 // final angle
495
496 const t_longf w = _rev(atan2(X, Y));
497 const t_longf P = asin(sqrt(X * X + Y * Y) * cos(a - w) / cos(b));
498
499 return _deg(P);
500}
