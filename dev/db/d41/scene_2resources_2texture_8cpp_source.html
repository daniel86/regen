53Vec3i TextureResource::getSize(
54 const ref_ptr<ShaderInput2i> &viewport,
55 const std::string &sizeMode,
56 const Vec3f &size) {
57 if (sizeMode == "abs") {
58 return Vec3i(size.x, size.y, size.z);
59 } else if (sizeMode == "rel") {
60 auto v = viewport->getVertex(0);
61 return Vec3i(
62 (GLint) (size.x * v.r.x),
63 (GLint) (size.y * v.r.y), 1);
64 } else {
65 REGEN_WARN("Unknown size mode '" << sizeMode << "'.");
66 return Vec3i(size.x, size.y, size.z);
67 }
68}

73void TextureResource::configureTexture(
74 ref_ptr<Texture> &tex, SceneInputNode &input) {
75 if (!input.getValue("sampler-type").empty()) {
76 tex->set_samplerType(input.getValue("sampler-type"));
77 }
78 tex->begin(RenderState::get(), 0);
79 {
80 if (!input.getValue("wrapping").empty()) {
81 tex->wrapping().push(glenum::wrappingMode(
82 input.getValue<std::string>("wrapping", "CLAMP_TO_EDGE")));
83 }
84 if (!input.getValue("aniso").empty()) {
85 tex->aniso().push(input.getValue<GLfloat>("aniso", 2.0f));
86 }
87 if (!input.getValue("lod").empty()) {
88 tex->lod().push(input.getValue<Vec2f>("lod", Vec2f(1.0f)));
89 }
90 if (!input.getValue("swizzle-r").empty() ||
91 !input.getValue("swizzle-g").empty() ||
92 !input.getValue("swizzle-b").empty() ||
93 !input.getValue("swizzle-a").empty()) {
94 auto swizzleR = static_cast<int>(glenum::textureSwizzle(
95 input.getValue<std::string>("swizzle-r", "RED")));
96 auto swizzleG = static_cast<int>(glenum::textureSwizzle(
97 input.getValue<std::string>("swizzle-g", "GREEN")));
98 auto swizzleB = static_cast<int>(glenum::textureSwizzle(
99 input.getValue<std::string>("swizzle-b", "BLUE")));
100 auto swizzleA = static_cast<int>(glenum::textureSwizzle(
101 input.getValue<std::string>("swizzle-a", "ALPHA")));
102 tex->swizzle().push(Vec4i(swizzleR, swizzleG, swizzleB, swizzleA));
103 }
104 if (!input.getValue("compare-mode").empty()) {
105 auto function = static_cast<int>(glenum::compareFunction(
106 input.getValue<std::string>("compare-function", "LEQUAL")));
107 auto mode = static_cast<int>(glenum::compareMode(
108 input.getValue<std::string>("compare-mode", "NONE")));
109 tex->compare().push(TextureCompare(mode, function));
110 }
111 if (!input.getValue("max-level").empty()) {
112 tex->maxLevel().push(input.getValue<GLint>("max-level", 1000));
113 }
114
115 if (!input.getValue("min-filter").empty() &&
116 !input.getValue("mag-filter").empty()) {
117 auto min = static_cast<int>(glenum::filterMode(input.getValue("min-filter")));
118 auto mag = static_cast<int>(glenum::filterMode(input.getValue("mag-filter")));
119 tex->filter().push(TextureFilter(min, mag));
120 } else if (!input.getValue("min-filter").empty() ||
121 !input.getValue("mag-filter").empty()) {
122 REGEN_WARN("Minification and magnification filters must be specified both." <<
123 " One missing for '"
124 << input.getDescription()
125 << "'.");
126 }
127 }
128 tex->end(RenderState::get(), 0);
129 GL_ERROR_LOG();
130}

132static std::vector<GLubyte> readTextureData(SceneInputNode &input, GLenum format) {
133 std::vector<GLubyte> data;
134 auto numPixelComponents = glenum::pixelComponents(format);
135 // iterate over all "texel" children
136 for (auto &child: input.getChildren("texel")) {
137 if (!child->hasAttribute("v")) {
138 REGEN_WARN("No 'v' attribute found for texel child '" << child->getName() << "'.");
139 continue;
140 }
141 int texelWidth = child->getValue<int>("width", 1);
142 for (int i = 0; i < texelWidth; ++i) {
143 if (numPixelComponents == 1) {
144 data.push_back(child->getValue<GLuint>("v", 0u));
145 } else if (numPixelComponents == 2) {
146 auto v = child->getValue<Vec2ui>("v", Vec2ui(0u));
147 data.push_back(v.x);
148 data.push_back(v.y);
149 } else if (numPixelComponents == 3) {
150 auto v = child->getValue<Vec3ui>("v", Vec3ui(0u));
151 data.push_back(v.x);
152 data.push_back(v.y);
153 data.push_back(v.z);
154 } else if (numPixelComponents == 4) {
155 auto v = child->getValue<Vec4ui>("v", Vec4ui(0u));
156 data.push_back(v.x);
157 data.push_back(v.y);
158 data.push_back(v.z);
159 data.push_back(v.w);
160 }
161 }
162 }
163 return data;
164}

166ref_ptr<Texture> TextureResource::createResource(
167 SceneParser *parser, SceneInputNode &input) {
168 ref_ptr<Texture> tex;
169
170 if (input.hasAttribute("file")) {
171 auto mipmapFlag = GL_DONT_CARE;
172 auto forcedInternalFormat = glenum::textureInternalFormat(
173 input.getValue<std::string>("forced-internal-format", "NONE"));
174 auto forcedFormat = glenum::textureFormat(
175 input.getValue<std::string>("forced-format", "NONE"));
176 auto forcedSize =
177 input.getValue<Vec3ui>("forced-size", Vec3ui(0u));
178 auto keepData = input.getValue<bool>("keep-data", false);
179 const std::string filePath =
180 resourcePath(input.getValue("file"));
181
182 try {
183 if (input.getValue<bool>("is-cube", false)) {
184 tex = textures::loadCube(
185 filePath,
186 input.getValue<bool>("cube-flip-back", false),
187 mipmapFlag,
188 forcedInternalFormat,
189 forcedFormat,
190 forcedSize);
191 } else if (input.getValue<bool>("is-array", false)) {
192 tex = textures::loadArray(
193 filePath,
194 input.getValue<std::string>("name-pattern", ".*"),
195 mipmapFlag,
196 forcedInternalFormat,
197 forcedFormat,
198 forcedSize);
199 } else if (input.getValue<bool>("is-raw", false)) {
200 tex = textures::loadRAW(
201 filePath,
202 input.getValue<Vec3ui>("raw-size", Vec3ui(256u)),
203 input.getValue<GLuint>("raw-components", 3u),
204 input.getValue<GLuint>("raw-bytes", 4u));
205 } else {
206 tex = textures::load(
207 filePath,
208 mipmapFlag,
209 forcedInternalFormat,
210 forcedFormat,
211 forcedSize,
212 keepData);
213 }
214 }
215 catch (textures::Error &ie) {
216 REGEN_ERROR("Failed to load Texture at " << filePath << ".");
217 }
218 } else if (input.hasAttribute("video")) {
219 const std::string filePath = resourcePath(input.getValue("video"));
220 ref_ptr<VideoTexture> video = ref_ptr<VideoTexture>::alloc();
221 try {
222 video->set_file(filePath);
223 auto filename = filePath.substr(filePath.find_last_of('/') + 1);
224 video->setAnimationName(REGEN_STRING("Video" << filename));
225 video->demuxer()->set_repeat(
226 input.getValue<bool>("repeat", 1));
227 tex = video;
228 video->play();
229 video->startAnimation();
230 }
231 catch (VideoTexture::Error &ve) {
232 REGEN_ERROR("Failed to load Video at " << filePath << ".");
233 }
234 } else if (input.hasAttribute("noise")) {
235 const std::string noiseMode = input.getValue("noise");
236
237 auto sizeMode = input.getValue<std::string>("size-mode", "abs");
238 auto sizeRel = input.getValue<Vec3f>("size", Vec3f(256.0, 256.0, 1.0));
239 auto sizeAbs = getSize(parser->getViewport(), sizeMode, sizeRel);
240
241 auto randomSeed = input.getValue<GLint>("random-seed", rand());
242 auto isSeamless = input.getValue<bool>("is-seamless", false);
243
244 // TODO: allow configuration of noise parameters
245 if (noiseMode == "cloud") {
246 auto noise = ref_ptr<NoiseTexture2D>::alloc(sizeAbs.x, sizeAbs.y, isSeamless);
247 noise->setNoiseGenerator(NoiseGenerator::preset_clouds(randomSeed));
248 tex = noise;
249 } else if (noiseMode == "wood") {
250 auto noise = ref_ptr<NoiseTexture2D>::alloc(sizeAbs.x, sizeAbs.y, isSeamless);
251 noise->setNoiseGenerator(NoiseGenerator::preset_wood(randomSeed));
252 tex = noise;
253 } else if (noiseMode == "granite") {
254 auto noise = ref_ptr<NoiseTexture2D>::alloc(sizeAbs.x, sizeAbs.y, isSeamless);
255 noise->setNoiseGenerator(NoiseGenerator::preset_granite(randomSeed));
256 tex = noise;
257 } else if (noiseMode == "perlin-2d" || noiseMode == "perlin") {
258 auto noise = ref_ptr<NoiseTexture2D>::alloc(sizeAbs.x, sizeAbs.y, isSeamless);
259 noise->setNoiseGenerator(NoiseGenerator::preset_perlin(randomSeed));
260 tex = noise;
261 } else if (noiseMode == "perlin-3d") {
262 auto noise = ref_ptr<NoiseTexture3D>::alloc(sizeAbs.x, sizeAbs.y, sizeAbs.z, isSeamless);
263 noise->setNoiseGenerator(NoiseGenerator::preset_perlin(randomSeed));
264 tex = noise;
265 }
266 } else if (input.hasAttribute("ramp")) {
267 auto ramp = input.getValue("ramp");
268 if (ramp == "dark-white") {
269 tex = RampTexture::darkWhite();
270 } else if (ramp == "dark-white-skewed") {
271 tex = RampTexture::darkWhiteSkewed();
272 } else if (ramp == "normal") {
273 tex = RampTexture::normal();
274 } else if (ramp == "three-step") {
275 tex = RampTexture::threeStep();
276 } else if (ramp == "four-step") {
277 tex = RampTexture::fourStep();
278 } else if (ramp == "four-step-skewed") {
279 tex = RampTexture::fourStepSkewed();
280 } else if (ramp == "black-white-black") {
281 tex = RampTexture::blackWhiteBlack();
282 } else if (ramp == "stripes") {
283 tex = RampTexture::stripes();
284 } else if (ramp == "stripe") {
285 tex = RampTexture::stripe();
286 } else if (ramp == "rgb") {
287 tex = RampTexture::rgb();
288 } else if (ramp == "inline") {
289 auto format = glenum::textureFormat(
290 input.getValue<std::string>("format", "LUMINANCE"));
291 auto internalFormat = format;
292 if (input.hasAttribute("internal-format")) {
293 internalFormat = glenum::textureInternalFormat(
294 input.getValue<std::string>("internal-format", "LUMINANCE"));
295 }
296 auto data = readTextureData(input, format);
297 tex = ref_ptr<RampTexture>::alloc(format, internalFormat, data);
298 } else {
299 REGEN_WARN("Unknown ramp type '" << ramp << "'.");
300 }
301 } else if (input.hasAttribute("spectrum")) {
302 auto spectrum = input.getValue<Vec2d>("spectrum", Vec2d(0.0, 1.0));
303 auto numTexels = input.getValue<GLint>("num-texels", 256u);
304 tex = regen::textures::loadSpectrum(spectrum.x, spectrum.y, numTexels);
305 } else if (input.hasAttribute("type")) {
306 const std::string typeName = input.getValue("type");
307 if (typeName == "bloom") {
308 auto numMips = input.getValue<GLuint>("num-mips", 5u);
309 auto bloomTexture = ref_ptr<BloomTexture>::alloc(numMips);
310
311 auto inputFBO = parser->getResources()->getFBO(parser, input.getValue("input-fbo"));
312 if (inputFBO.get() == nullptr) {
313 REGEN_WARN("Unable to find FBO for '" << input.getDescription() << "'.");
314 } else {
315 auto resizer = ref_ptr<TextureResizer>::alloc(parser->application(),
316 bloomTexture, parser->getViewport(), 1.0, 1.0);
317 parser->addEventHandler(Application::RESIZE_EVENT, resizer);
318 tex = bloomTexture;
319 bloomTexture->resize(inputFBO->width(), inputFBO->height());
320 }
321 } else {
322 REGEN_WARN("Unknown texture type '" << typeName << "'.");
323 }
324 } else {
325 auto sizeMode = input.getValue<std::string>("size-mode", "abs");
326 auto sizeRel = input.getValue<Vec3f>("size", Vec3f(256.0, 256.0, 1.0));
327 Vec3i sizeAbs = getSize(parser->getViewport(), sizeMode, sizeRel);
328
329 auto texCount = input.getValue<GLuint>("count", 1);
330 auto pixelSize = input.getValue<GLuint>("pixel-size", 16);
331 auto pixelComponents = input.getValue<GLuint>("pixel-components", 4);
332 auto pixelType = glenum::pixelType(
333 input.getValue<std::string>("pixel-type", "UNSIGNED_BYTE"));
334 auto textureTarget = glenum::textureTarget(
335 input.getValue<std::string>("target", sizeAbs.z > 1 ? "TEXTURE_3D" : "TEXTURE_2D"));
336
337
338 tex = FBO::createTexture(
339 sizeAbs.x, sizeAbs.y, sizeAbs.z,
340 texCount,
341 textureTarget,
342 glenum::textureFormat(pixelComponents),
343 glenum::textureInternalFormat(pixelType, pixelComponents, pixelSize),
344 pixelType);
345
346 if (input.hasAttribute("size-mode") && sizeMode == "rel") {
347 auto resizer = ref_ptr<TextureResizer>::alloc(parser->application(),
348 tex, parser->getViewport(), sizeRel.x, sizeRel.y);
349 parser->addEventHandler(Application::RESIZE_EVENT, resizer);
350 }
351 }
352
353 if (tex.get() == nullptr) {
354 REGEN_WARN("Failed to create Texture for '" << input.getDescription() << ".");
355 return tex;
356 }
357 tex->set_name(input.getName());
358 configureTexture(tex, input);
359
360 return tex;
361}
