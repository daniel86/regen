87void SceneLoader::loadShapes() {
88 auto root = getRoot();
89 auto dummy = ref_ptr<State>::alloc();
90
91 // load shapes that are globally defined
92 auto shapeNodes = root->getChildren("shape");
93 for (auto &shapeNode: shapeNodes) {
94 processState({}, dummy, "shape", shapeNode);
95 }
96 // load meshes and associated shapes
97 auto meshNodes = root->getChildren("mesh");
98 for (auto &meshNode: meshNodes) {
99 auto meshID = meshNode->getName();
100 auto meshVec = resources_->getMesh(this, meshID);
101 if (!meshVec.get()) {
102 REGEN_WARN("Could not load mesh with id '" << meshID << "'.");
103 }
104 }
105}

275std::vector<AnimRange> SceneLoader::getAnimationRanges(const std::string &assetID) {
276 ref_ptr<SceneInputNode> root = getRoot();
277 ref_ptr<SceneInputNode> importer = root->getFirstChild("asset", assetID);
278 if (importer.get() == nullptr) {
279 REGEN_WARN("No asset with id '" << assetID << "' known.");
280 return {};
281 } else {
282 const std::list<ref_ptr<SceneInputNode> > &childs = importer->getChildren();
283
284 GLuint animRangeCount = 0u;
285 for (auto it = childs.begin(); it != childs.end(); ++it) {
286 if ((*it)->getCategory() == "anim-range") animRangeCount += 1;
287 }
288
289 std::vector<AnimRange> out(animRangeCount);
290 animRangeCount = 0u;
291
292 for (auto it = childs.begin(); it != childs.end(); ++it) {
293 ref_ptr<SceneInputNode> n = *it;
294 if (n->getCategory() != "anim-range") continue;
295 out[animRangeCount].name = n->getValue("name");
296 out[animRangeCount].range = n->getValue<Vec2d>("range", Vec2d(0.0));
297 out[animRangeCount].channelName = n->getValue("channel");
298 out[animRangeCount].channelIndex = n->getValue<GLuint>("index", 0u);
299 animRangeCount += 1u;
300 }
301
302 return out;
303 }
