87void TwoUnitsChanger::updateSmallestRange()
88{
89 m_smallestRange = 1.f;
90
91 if(m_unitsByTime.size() < 2)
92 return;
93
94 const t_unitsByTime::const_iterator ie(m_unitsByTime.end());
95
96 t_unitsByTime::const_iterator i(m_unitsByTime.begin());
97 t_unitsByTime::const_iterator i0(i), i1(i);
98
99 // Iterate over all time keys, and retrieve the smallest range.
100 for(; i != ie; ++i)
101 {
102 ++i1;
103
104 // The last key has to be compared with the first.
105 if(i1 == ie)
106 i1 = i0;
107
108 const float range = _abs(i1->first - i->first);
109
110 if(range < m_smallestRange)
111 m_smallestRange = range;
112 }
113 checkAndAdjustTransitionDuration();
114}

126void TwoUnitsChanger::updateBackUnits() const
127{
128 t_unitsByTime::const_iterator i(m_unitsByTime.begin());
129 const t_unitsByTime::const_iterator ie(m_unitsByTime.end());
130
131 for(; i != ie; ++i)
132 {
133 float t(i->first + m_transitionDuration * 0.5f);
134
135 if(t >= 1.f)
136 t -= 1.f;
137
138 m_backUnitsByTime[t] = i->second;
139 }
140}

143void TwoUnitsChanger::update(const float time) const
144{
145 if(time == m_lastTime && m_updated)
146 return;
147
148 if(!m_updated)
149 m_updated = true;
150
151 m_lastTime = time;
152
153 // If no or only one unit is assigned do nothing.
154 if(m_unitsByTime.size() < 2)
155 {
156 m_back = 0;
157 m_src = 0;
158 m_srcAlpha = 0.f;
159
160 return;
161 }
162
163 // Update back units if required.
164 if(m_unitsByTime.size() != m_backUnitsByTime.size())
165 updateBackUnits();
166
167 const t_unitsByTime::const_iterator ie(m_backUnitsByTime.end());
168 const t_unitsByTime::const_iterator ib(m_backUnitsByTime.begin());
169 t_unitsByTime::const_iterator il = ie; --il; // last
170
171 t_unitsByTime::const_iterator isrc(m_backUnitsByTime.upper_bound(time));
172 t_unitsByTime::const_iterator iback;
173
174 if(isrc == ie || isrc == ib)
175 {
176 // If time is after last or before first key-point, src
177 // is at first key-point, and back is at last key-point.
178 isrc = ib;
179 iback = il;
180 }
181 else
182 {
183 iback = isrc;
184 std::advance(iback, -1);
185 }
186
187 if(iback->second == isrc->second)
188 m_srcAlpha = 0.f;
189 else
190 {
191 const float tsrc(isrc->first);
192 const float t = tsrc < time ? time - 1.f : time;
193
194 if(m_transitionDuration != 0.f)
195 {
196 float a = 1.f - (tsrc - t) / m_transitionDuration;
197 a = std::min<float>(std::max<float>(a, 0.f), 1.f);
198
199 m_srcAlpha = interpolate<float>(a, m_interpolationMethod);
200 }
201 else
202 m_srcAlpha = 0.f;
203 }
204
205 // NOTE: This could also be optimized for 8 or less textures by not
206 // switching their texture unit and adapting the shader: "EightUnitsChanger" :P
207
208 m_back = iback->second;
209 m_src = isrc->second;
210}
