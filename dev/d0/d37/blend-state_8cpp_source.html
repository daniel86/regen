105 BlendState::BlendState(BlendMode blendMode) : ServerSideState() {
106 joinStates(ref_ptr<ToggleState>::alloc(RenderState::BLEND, GL_TRUE));
107
108 switch (blendMode) {
109 case BLEND_MODE_ALPHA:
110 case BLEND_MODE_FRONT_TO_BACK:
111 // c = c0*(1-c1.a) + c1*c1.a
112 setBlendEquation(GL_FUNC_ADD);
113 setBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
114 break;
115 case BLEND_MODE_DST_ALPHA:
116 case BLEND_MODE_BACK_TO_FRONT:
117 // c = c0*(1-c1.a) + c1*c1.a
118 setBlendEquation(GL_FUNC_ADD);
119 setBlendFunc(GL_ONE_MINUS_DST_ALPHA, GL_DST_ALPHA);
120 break;
121 case BLEND_MODE_MULTIPLY:
122 // c = c0*c1, a=a0*a1
123 setBlendEquation(GL_FUNC_ADD);
124 setBlendFunc(GL_DST_COLOR, GL_ZERO);
125 break;
126 case BLEND_MODE_ADD:
127 // c = c0+c1, a=a0+a1
128 setBlendEquation(GL_FUNC_ADD);
129 setBlendFunc(GL_ONE, GL_ONE);
130 break;
131 case BLEND_MODE_SMOOTH_ADD: // aka average
132 // c = 0.5*c0 + 0.5*c1
133 // a = a0 + a1
134 setBlendEquation(GL_FUNC_ADD);
135 setBlendFuncSeparate(
136 GL_CONSTANT_ALPHA, GL_CONSTANT_ALPHA,
137 GL_ONE, GL_ONE);
138 setBlendColor(Vec4f(0.5f));
139 break;
140 case BLEND_MODE_SUBTRACT:
141 // c = c0-c1, a=a0-a1
142 setBlendEquation(GL_FUNC_SUBTRACT);
143 setBlendFunc(GL_ONE, GL_ONE);
144 break;
145 case BLEND_MODE_REVERSE_SUBTRACT:
146 // c = c1-c0, a=c1-c0
147 setBlendEquation(GL_FUNC_REVERSE_SUBTRACT);
148 setBlendFunc(GL_ONE, GL_ONE);
149 break;
150 case BLEND_MODE_LIGHTEN:
151 // c = max(c0,c1), a = max(a0,a1)
152 setBlendEquation(GL_MAX);
153 setBlendFunc(GL_ONE, GL_ONE);
154 break;
155 case BLEND_MODE_DARKEN:
156 // c = min(c0,c1), a = min(a0,a1)
157 setBlendEquation(GL_MIN);
158 setBlendFunc(GL_ONE, GL_ONE);
159 break;
160 case BLEND_MODE_SCREEN:
161 // c = c0 - c1*(1-c0), a = a0 - a1*(1-a0)
162 setBlendEquation(GL_FUNC_SUBTRACT);
163 setBlendFunc(GL_ONE_MINUS_DST_COLOR, GL_ONE);
164 break;
165 case BLEND_MODE_SRC_ALPHA:
166 // c = c1*c1.a
167 setBlendEquation(GL_FUNC_ADD);
168 setBlendFunc(GL_SRC_ALPHA, GL_ZERO);
169 break;
170 case BLEND_MODE_SRC:
171 // c = c1
172 setBlendEquation(GL_FUNC_ADD);
173 setBlendFunc(GL_ONE, GL_ZERO);
174 break;
175 case BLEND_MODE_DIVIDE:
176 case BLEND_MODE_DIFFERENCE:
177 case BLEND_MODE_MIX:
178 case BLEND_MODE_OVERLAY:
179 case BLEND_MODE_HUE:
180 case BLEND_MODE_SATURATION:
181 case BLEND_MODE_VALUE:
182 case BLEND_MODE_COLOR:
183 case BLEND_MODE_DODGE:
184 case BLEND_MODE_BURN:
185 case BLEND_MODE_SOFT:
186 case BLEND_MODE_LINEAR:
187 break;
188 }
189 }
