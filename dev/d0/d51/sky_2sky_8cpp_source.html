162GLfloat Sky::computeHorizonExtinction(const Vec3f &position, const Vec3f &dir, float radius) {
163 GLfloat u = dir.dot(-position);
164 if (u < 0.0) {
165 return 1.0;
166 }
167 Vec3f near = position + dir * u;
168 if (near.length() < radius) {
169 return 0.0;
170 } else {
171 near.normalize();
172 Vec3f v2 = near * radius - position;
173 v2.normalize();
174 GLfloat diff = acos(v2.dot(dir));
175 return math::smoothstep(0.0, 1.0, pow(diff * 2.0, 3.0));
176 }
177}

179GLfloat Sky::computeEyeExtinction(const Vec3f &eyedir) {
180 static const float surfaceHeight = 0.99f;
181 static const Vec3f eyePosition(0.0, surfaceHeight, 0.0);
182 return computeHorizonExtinction(eyePosition, eyedir, surfaceHeight - 0.15f);

193void Sky::animate(GLdouble dt) {
194 if (worldTime_) {
195 // convert to time_t
196 time_t t = boost::posix_time::to_time_t(worldTime_->p_time);
197 // get UTC offset
198 struct tm *tm = gmtime(&t);
199 time_t utcOffset = t - mktime(tm);
200 // create timef object
201 osgHimmel::TimeF time_osg(t, utcOffset);
202 astro_->update(osgHimmel::t_aTime::fromTimeF(time_osg));
203 }
204
205 // Compute sun/moon directions
206 Vec3f moon = astro_->getMoonPosition(false);
207 Vec3f sun = astro_->getSunPosition(false);
208 sun_->direction()->setVertex(0, sun);
209 moon_->direction()->setVertex(0, moon);
210 // Compute sun/moon diffuse color
211 GLfloat sunExt = computeEyeExtinction(sun);
212 Vec3f sunColor = math::mix(dawnColor_, noonColor_, abs(sunExt));
213 sun_->diffuse()->setVertex(0, computeColor(
214 sunColor,
215 sunExt)
216 );
217 moon_->diffuse()->setVertex(0, computeColor(
218 sunColor * moonSunLightReflectance_,
219 computeEyeExtinction(moon))
220 );
221
222 const float fovHalf = camera()->fov()->getVertex(0).r * 0.5f * DEGREE_TO_RAD;
223 const float height = static_cast<float>(viewport()->getVertex(0).r.y);
224 q_->setVertex(0, sqrt(2.0f) * 2.0f * tan(fovHalf) / height);
225 R_->setVertex(0, astro().getEquToHorTransform());
226 // Update random number in cmn uniform
227 updateSeed();
228}
