62void Mesh::addShaderInput(const std::string &name, const ref_ptr<ShaderInput> &in) {
63 if (!meshShader_.get()) return;
64
65 if (in->isUniformBlock()) {
66 auto block = (UniformBlock *) (in.get());
67 for (auto &blockUniform: block->uniforms()) {
68 if (blockUniform.in_->numInstances() > 1) {
69 inputContainer_->set_numInstances(blockUniform.in_->numInstances());
70 hasInstances_ = GL_TRUE;
71 }
72 }
73 }
74 if (in->numInstances() > 1) {
75 inputContainer_->set_numInstances(in->numInstances());
76 hasInstances_ = GL_TRUE;
77 }
78
79 if (in->isVertexAttribute()) {
80 GLint loc = meshShader_->attributeLocation(name);
81 if (loc == -1) {
82 // not used in shader
83 return;
84 }
85 if (!in->bufferIterator().get()) {
86 // allocate VBO memory if not already allocated
87 inputContainer_->inputBuffer()->alloc(in);
88 }
89
90 auto needle = vaoLocations_.find(loc);
91 if (needle == vaoLocations_.end()) {
92 vaoAttributes_.emplace_back(in, loc);
93 auto it = vaoAttributes_.end();
94 --it;
95 vaoLocations_[loc] = it;
96 } else {
97 *needle->second = ShaderInputLocation(in, loc);
98 }
99 } else if (!in->isConstant()) {
100 if (meshShader_->hasUniform(name) &&
101 meshShader_->input(name).get() == in.get()) {
102 // shader handles uniform already.
103 return;
104 }
105 GLint loc = meshShader_->uniformLocation(name);
106 if (loc == -1) {
107 // not used in shader
108 return;
109 }
110 meshUniforms_[loc] = ShaderInputLocation(in, loc);
111 }
112}

114void Mesh::updateVAO(
115 RenderState *rs,
116 const StateConfig &cfg,
117 const ref_ptr<Shader> &meshShader) {
118 // remember the shader
119 meshShader_ = meshShader;
120 hasInstances_ = GL_FALSE;
121
122 // reset attribute list
123 vaoAttributes_.clear();
124 vaoLocations_.clear();
125 meshUniforms_.clear();
126 // and load from Config
127 for (auto it = cfg.inputs_.begin(); it != cfg.inputs_.end(); ++it) { addShaderInput(it->name_, it->in_); }
128 // Get input from mesh and joined states (might be handled by StateConfig allready)
129 ShaderInputList localInputs;
130 collectShaderInput(localInputs);
131 for (auto it = localInputs.begin(); it != localInputs.end(); ++it) {
132 addShaderInput(it->name_, it->in_);
133 }
134 // Add Textures
135 for (auto it = cfg.textures_.begin(); it != cfg.textures_.end(); ++it) { addShaderInput(it->first, it->second); }
136
137 updateVAO(rs);
138 updateDrawFunction();
139}

157void Mesh::updateVAO(RenderState *rs) {
158 GLuint lastArrayBuffer = 0;
159 vao_->resetGL();
160 rs->vao().push(vao_->id());
161 // Setup attributes
162 for (auto it = vaoAttributes_.begin(); it != vaoAttributes_.end(); ++it) {
163 const ref_ptr<ShaderInput> &in = it->input;
164 if (lastArrayBuffer != in->buffer()) {
165 lastArrayBuffer = in->buffer();
166 glBindBuffer(GL_ARRAY_BUFFER, lastArrayBuffer);
167 }
168 in->enableAttribute(it->location);
169 if (in->numInstances() > 1) hasInstances_ = GL_TRUE;
170 }
171 // bind the index buffer
172 if (inputContainer_->indexBuffer() > 0) {
173 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, inputContainer_->indexBuffer());
174 }
175 rs->vao().pop();
176}

178unsigned int Mesh::getLODLevel(float cameraDistance) {
179 auto normalizedDistance = std::min(cameraDistance / lodFar_, 0.9999f);
180 auto lodLevel = static_cast<float>(meshLODs_.size()) * normalizedDistance;
181 lodLevel = std::trunc(lodLevel);
182 return static_cast<unsigned int>(lodLevel);

189void Mesh::activateLOD(GLuint lodLevel) {
190 if (meshLODs_.size() <= lodLevel) {
191 REGEN_WARN("LOD level " << lodLevel << " not available num LODs: " << meshLODs_.size());
192 return;
193 }
194 MeshLOD &lod = meshLODs_[lodLevel];
195 lodLevel_ = lodLevel;
196 if (inputContainer_->indexBuffer() > 0) {
197 inputContainer_->set_numIndices(lod.numIndices);
198 inputContainer_->set_indexOffset(lod.indexOffset);
199 } else {
200 inputContainer_->set_numVertices(lod.numVertices);
201 inputContainer_->set_vertexOffset(lod.vertexOffset);
202 }
203}

209void Mesh::enable(RenderState *rs) {
210 State::enable(rs);
211
212 for (auto it = meshUniforms_.begin(); it != meshUniforms_.end(); ++it) {
213 ShaderInputLocation &x = it->second;
214 // For uniforms below the shader it is expected that
215 // they will be set multiple times during shader lifetime.
216 // So we upload uniform data each time.
217 x.input->enableUniform(x.location);
218 }
219
220 if (feedbackRange_.get()) {
221 feedbackCount_ = 0;
222 rs->feedbackBufferRange().push(0, *feedbackRange_.get());
223 rs->beginTransformFeedback(GL_POINTS);
224 }
225
226 rs->vao().push(vao_->id());
227 (inputContainer_.get()->*draw_)(primitive_);
228}

230void Mesh::disable(RenderState *rs) {
231 if (feedbackRange_.get()) {
232 rs->endTransformFeedback();
233 rs->feedbackBufferRange().pop(0);
234 }
235
236 rs->vao().pop();
237 State::disable(rs);
