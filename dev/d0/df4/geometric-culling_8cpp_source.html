14GeometricCulling::GeometricCulling(
15 const ref_ptr<Camera> &camera,
16 const ref_ptr<SpatialIndex> &spatialIndex,
17 std::string_view shapeName)
18 : StateNode(),
19 camera_(camera),
20 spatialIndex_(spatialIndex) {
21 shapeIndex_ = spatialIndex_->getIndexedShape(camera, shapeName);
22 if (shapeIndex_.get()) {
23 numInstances_ = shapeIndex_->shape()->numInstances();
24 mesh_ = shapeIndex_->shape()->mesh();
25 }
26 if (!mesh_.get()) {
27 numInstances_ = 1;
28 }
29 if (numInstances_ > 1) {
30 // create array with numInstances_ elements
31 instanceIDMap_ = ref_ptr<ShaderInput1ui>::alloc("instanceIDMap", numInstances_);
32 instanceIDMap_->setInstanceData(1, 1, nullptr);
33 auto instanceData = instanceIDMap_->mapClientData<GLuint>(ShaderData::WRITE);
34 for (GLuint i = 0; i < numInstances_; ++i) {
35 instanceData.w[i] = i;
36 }
37 instanceData.unmap();
38 state()->joinShaderInput(instanceIDMap_);
39 lodGroups_.resize(mesh_->numLODs());
40 }

43void GeometricCulling::updateMeshLOD() {
44 if (!mesh_.get() || mesh_->numLODs() <= 1) {
45 return;
46 }
47 auto &shape = shapeIndex_->shape();
48 // set LOD level based on distance
49 auto camPos = camera_->position()->getVertex(0);
50 auto distance = (shape->getCenterPosition() - camPos.r).length();
51 camPos.unmap();
52 mesh_->updateLOD(distance);
53 for (auto &part : shape->parts()) {
54 if (part->numLODs()>1) {
55 part->updateLOD(distance);
56 }
57 }

60void computeLODGroups_(
61 const ref_ptr<Mesh> &mesh,
62 const ref_ptr<BoundingShape> &shape,
63 const ref_ptr<Camera> &camera,
64 std::vector<std::vector<GLuint>> &lodGroups,
65 const unsigned int *mappedData,
66 int begin,
67 int end,
68 int increment) {
69 auto &transform = shape->transform();
70 auto &modelOffset = shape->modelOffset();
71 auto camPos = camera->position()->getVertex(0);
72
73 if (lodGroups.size() == 1) {
74 for (int i = begin; i != end; i += increment) {
75 lodGroups[0].push_back(mappedData[i]);
76 }
77 }
78 else if (transform.get() && modelOffset.get()) {
79 auto modelOffsetData = modelOffset->mapClientData<Vec3f>(ShaderData::READ);
80 auto tfData = transform->get()->mapClientData<Mat4f>(ShaderData::READ);
81 for (int i = begin; i != end; i += increment) {
82 auto lodLevel = mesh->getLODLevel((
83 tfData.r[mappedData[i]].position() +
84 modelOffsetData.r[mappedData[i]] - camPos.r).length());
85 lodGroups[lodLevel].push_back(mappedData[i]);
86 }
87 }
88 else if (modelOffset.get()) {
89 auto modelOffsetData = modelOffset->mapClientData<Vec3f>(ShaderData::READ);
90 for (int i = begin; i != end; i += increment) {
91 auto lodLevel = mesh->getLODLevel((
92 modelOffsetData.r[mappedData[i]] - camPos.r).length());
93 lodGroups[lodLevel].push_back(mappedData[i]);
94 }
95 }
96 else if (transform.get()) {
97 auto tfData = transform->get()->mapClientData<Mat4f>(ShaderData::READ);
98 for (int i = begin; i != end; i += increment) {
99 auto lodLevel = mesh->getLODLevel((
100 tfData.r[mappedData[i]].position() - camPos.r).length());
101 lodGroups[lodLevel].push_back(mappedData[i]);
102 }
103 }
104 else {
105 for (int i = begin; i != end; i += increment) {
106 lodGroups[0].push_back(mappedData[i]);
107 }
108 }
109}

111void GeometricCulling::computeLODGroups() {
112 for (auto &lodGroup : lodGroups_) {
113 lodGroup.clear();
114 }
115 auto visible_ids = shapeIndex_->mapInstanceIDs(ShaderData::READ);
116 auto numVisible = visible_ids.r[0];
117 if (numVisible == 0) { return; }
118
119 if (instanceSortMode_ == SortMode::BACK_TO_FRONT) {
120 computeLODGroups_(
121 mesh_,
122 shapeIndex_->shape(),
123 camera_,
124 lodGroups_,
125 visible_ids.r+1,
126 static_cast<int>(numVisible) - 1,
127 -1,
128 -1);
129 } else {
130 computeLODGroups_(
131 mesh_,
132 shapeIndex_->shape(),
133 camera_,
134 lodGroups_,
135 visible_ids.r+1,
136 0,
137 static_cast<int>(numVisible),
138 1);
139 }

156void GeometricCulling::traverseInstanced1(RenderState *rs) {
157 // Shape does not have LOD levels, thus shapeIndex_ instance array can be used directly
158 // TODO: Avoid the copy here and below. Exchanging the data pointers should be enough.
159 // But shader input does not have an interface for this yet. It manages two data slots.
160 // So the new operation would need to write-lock one slot, but then exchange the data pointers, plus reversing
161 // after traversal.
162 auto instance_ids = instanceIDMap_->mapClientData<unsigned int>(ShaderData::WRITE);
163 auto visible_ids = shapeIndex_->mapInstanceIDs(ShaderData::READ);
164 auto numVisible = visible_ids.r[0];
165 std::memcpy(instance_ids.w, visible_ids.r+1, numVisible * sizeof(unsigned int));
166 visible_ids.unmap();
167 instance_ids.unmap();
168 traverseInstanced_(rs, numVisible);
169}

179void GeometricCulling::traverse(RenderState *rs) {
180 if (!spatialIndex_->hasCamera(*camera_.get()) || !shapeIndex_.get()) {
181 updateMeshLOD();
182 StateNode::traverse(rs);
183 }
184 else if (numInstances_ <= 1) {
185 if (shapeIndex_->isVisible()) {
186 updateMeshLOD();
187 StateNode::traverse(rs);
188 }
189 } else if (!mesh_.get()) {
190 REGEN_WARN("No mesh set for shape " << shapeIndex_->shape()->name());
191 numInstances_ = 0;
192 }
193 else {
194 if (!shapeIndex_->hasVisibleInstances()) {
195 // no visible instances
196 return;
197 }
198
199 if (lodGroups_.size() < 2) {
200 traverseInstanced1(rs);
201 }
202 else {
203 // build LOD groups, then traverse each group
204 computeLODGroups();
205
206 for (unsigned int lodLevel=0; lodLevel<mesh_->numLODs(); ++lodLevel) {
207 auto &lodGroup = lodGroups_[lodLevel];
208 if (lodGroup.empty()) { continue; }
209 // set the LOD level
210 if (lodGroups_.size() > 1) {
211 mesh_->activateLOD(lodLevel);
212 }
213 // set number of visible instances
214 for (auto &part : shapeIndex_->shape()->parts()) {
215 if (mesh_->numLODs() == part->numLODs() && part->numLODs() > 1) {
216 part->activateLOD(lodLevel);
217 }
218 else if (part->numLODs()>1) {
219 // could be part has different number of LODs, need to compute an adjusted
220 // LOD level for each part
221 part->activateLOD(static_cast<unsigned int>(std::round(static_cast<float>(lodLevel) *
222 static_cast<float>(part->numLODs()) / static_cast<float>(mesh_->numLODs()))));
223 }
224 }
225 traverseInstanced2(rs, lodGroup.data(), lodGroup.size());
226 }
227 // reset LOD level
228 if (lodGroups_.size() > 1) {
229 mesh_->activateLOD(0);
230 }
231 }
232 }
