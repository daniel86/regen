31void Bones::setBones(const std::list<ref_ptr<AnimationNode> > &bones) {
32 GL_ERROR_LOG();
33 RenderState *rs = RenderState::get();
34 bones_ = bones;
35 shaderDefine("NUM_BONES", REGEN_STRING(bones_.size()));
36
37 // create and join bone matrix uniform
38 boneMatrices_ = ref_ptr<ShaderInputMat4>::alloc("boneMatrices", bones.size());
39 boneMatrices_->set_forceArray(GL_TRUE);
40 boneMatrices_->setUniformUntyped();
41
42#ifdef USE_BONE_TBO
43 bufferSize_ = sizeof(GLfloat) * 16 * bones_.size();
44 vboRef_ = inputContainer_->inputBuffer()->alloc(bufferSize_);
45
46 // attach vbo to texture
47 rs->textureBuffer().push(vboRef_->bufferID());
48 boneMatrixTex_ = ref_ptr<TextureBuffer>::alloc(GL_RGBA32F);
49 boneMatrixTex_->begin(rs);
50 boneMatrixTex_->attach(inputContainer_->inputBuffer(), vboRef_);
51 boneMatrixTex_->end(rs);
52 rs->textureBuffer().pop();
53
54 // and make the tbo available
55 if (texState_.get()) disjoinStates(texState_);
56 texState_ = ref_ptr<TextureState>::alloc(boneMatrixTex_, "boneMatrices");
57 texState_->set_mapping(TextureState::MAPPING_CUSTOM);
58 texState_->set_mapTo(TextureState::MAP_TO_CUSTOM);
59 joinStates(texState_);
60 shaderDefine("USE_BONE_TBO", "TRUE");
61#else
62 joinShaderInput(boneMatrices_);
63 shaderDefine("USE_BONE_TBO", "FALSE");
64#endif
65
66 GL_ERROR_LOG();
67
68 // initially calculate the bone matrices
69 glAnimate(rs, 0.0f);
70}

72void Bones::animate(GLdouble dt) {
73 if (bufferSize_ <= 0) return;
74 auto mapped = boneMatrices_->mapClientData<Mat4f>(ShaderData::WRITE);
75 auto *boneMatrixData_ = mapped.w;
76
77 unsigned int i = 0;
78 for (auto it = bones_.begin(); it != bones_.end(); ++it) {
79 // the bone matrix is actually calculated in the animation thread
80 // by NodeAnimation.
81 boneMatrixData_[i] = (*it)->boneTransformationMatrix();
82 i += 1;
83 }
84}
