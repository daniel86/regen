73void Stars::set_brightStarsFile(const std::string &brightStars) {
74 osgHimmel::BrightStars bs(brightStars.c_str());
75 const osgHimmel::BrightStars::s_BrightStar *stars = bs.stars();
76 if (bs.numStars() == 0) {
77 REGEN_WARN("Unable to load bright stars catalog at " << brightStars << ".");
78 return;
79 }
80
81 pos_->setVertexData(bs.numStars());
82 col_->setVertexData(bs.numStars());
83
84 for (unsigned int i = 0; i < bs.numStars(); ++i) {
85 osgHimmel::t_equf equ;
86 equ.right_ascension = _rightascd(stars[i].RA, 0, 0);
87 equ.declination = stars[i].DE;
88
89 pos_->setVertex(i, Vec4f(equ.toEuclidean(), static_cast<float>(i)));
90 col_->setVertex(i, Vec4f(
91 stars[i].sRGB_R,
92 stars[i].sRGB_G,
93 stars[i].sRGB_B,
94 stars[i].Vmag + 0.4f // the 0.4 accounts for magnitude decrease due to the earth's atmosphere
95 ));
96 }
97
98 meshState_->begin(ShaderInputContainer::INTERLEAVED);
99 meshState_->setInput(pos_);
100 meshState_->setInput(col_);
101 meshState_->end();
102}

104void Stars::updateNoiseTexture() {
105 const int noiseN = 256;
106
107 byte *noiseMap = new byte[noiseN];
108 osgHimmel::RandomMapGenerator::generate1(noiseN, 1, noiseMap);
109
110 noiseTex_ = ref_ptr<Texture1D>::alloc();
111 noiseTex_->set_rectangleSize(noiseN, 1);
112 noiseTex_->set_format(GL_RED);
113 noiseTex_->set_internalFormat(GL_R8);
114 noiseTex_->set_pixelType(GL_UNSIGNED_BYTE);
115 noiseTex_->set_textureData(noiseMap);
116 noiseTex_->begin(RenderState::get());
117 noiseTex_->texImage();
118 noiseTex_->filter().push(GL_LINEAR);
119 noiseTex_->wrapping().push(GL_REPEAT);
120 noiseTex_->end(RenderState::get());
121 noiseTex_->set_textureData(nullptr);
122
123 delete[]noiseMap;
124
125 noiseTexState_->set_texture(noiseTex_);
126 noiseTexState_->set_name("noiseTexture");
127}
