59 template<class T>
60 static void setInput(SceneInputNode &input, ShaderInput *shaderInput, unsigned int count) {
61 auto v_values = shaderInput->mapClientData<T>(ShaderData::WRITE | ShaderData::READ);
62 auto default_value = input.getValue<T>("value", T(0));
63 for (unsigned int i = 0; i < count; ++i) {
64 v_values.w[i] = default_value;
65 }
66 for (auto &child : input.getChildren()) {
67 if (child->getCategory() == "set") {
68 std::list<GLuint> indices = child->getIndexSequence(count);
69 auto blendMode = child->getValue<BlendMode>("blend-mode", BLEND_MODE_SRC);
70 ValueGenerator<T> generator(child.get(), indices.size(),
71 child->getValue<T>("value", T(1)));
72 for (auto it = indices.begin(); it != indices.end(); ++it) {
73 switch (blendMode) {
74 case BLEND_MODE_ADD:
75 v_values.w[*it] = v_values.r[*it] + v_values.w[*it] + generator.next();
76 break;
77 case BLEND_MODE_MULTIPLY:
78 v_values.w[*it] = v_values.r[*it] * v_values.w[*it] * generator.next();
79 break;
80 default:
81 v_values.w[*it] = generator.next();
82 break;
83 }
84 }
85 } else {
86 REGEN_WARN("No processor registered for '" << child->getDescription() << "'.");
87 }
88 }
89 }

91 static void setInput(SceneInputNode &input, ShaderInput *in, unsigned int count) {
92 if (input.getChildren().empty()) return;
93 switch (in->dataType()) {
94 case GL_FLOAT:
95 switch (in->valsPerElement()) {
96 case 1: setInput<float>(input, in, count); break;
97 case 2: setInput<Vec2f>(input, in, count); break;
98 case 3: setInput<Vec3f>(input, in, count); break;
99 case 4: setInput<Vec4f>(input, in, count); break;
100 }
101 break;
102 case GL_INT:
103 switch (in->valsPerElement()) {
104 case 1: setInput<int>(input, in, count); break;
105 case 2: setInput<Vec2i>(input, in, count); break;
106 case 3: setInput<Vec3i>(input, in, count); break;
107 case 4: setInput<Vec4i>(input, in, count); break;
108 }
109 break;
110 case GL_UNSIGNED_INT:
111 switch (in->valsPerElement()) {
112 case 1: setInput<unsigned int>(input, in, count); break;
113 case 2: setInput<Vec2ui>(input, in, count); break;
114 case 3: setInput<Vec3ui>(input, in, count); break;
115 case 4: setInput<Vec4ui>(input, in, count); break;
116 }
117 break;
118 default:
119 REGEN_WARN("No processor registered for '" << in->name() << "'.");
120 }
121 }

123 static int getNumInstances(const ref_ptr<Mesh> &mesh) {
124 int num = mesh->inputContainer()->numInstances();
125 std::stack<ref_ptr<State>> stack;
126 stack.emplace(mesh);
127 while (!stack.empty()) {
128 auto state = stack.top();
129 stack.pop();
130 for (auto &joined : state->joined()) {
131 stack.push(joined);
132 }
133 auto *hasInput = dynamic_cast<HasInput *>(state.get());
134 if (hasInput != nullptr) {
135 num = std::max(num, hasInput->inputContainer()->numInstances());
136 }
137 }
138 return num;
139 }

145 static ref_ptr<ShaderInput> createShaderInput(
146 SceneParser *parser,
147 SceneInputNode &input,
148 const ref_ptr<State> &state) {
149 ref_ptr<ShaderInput> in;
150
151 if (input.hasAttribute("state")) {
152 // take uniform from state
153 ref_ptr<State> s = parser->getState(input.getValue("state"));
154 if (s.get() == nullptr) {
155 s = parser->getResources()->getTransform(parser, input.getValue("state"));
156 }
157 if (s.get() == nullptr) {
158 parser->getResources()->loadResources(parser, input.getValue("state"));
159 s = parser->getState(input.getValue("state"));
160 if (s.get() == nullptr) {
161 REGEN_WARN("No State found for for '" << input.getDescription() << "'.");
162 return {};
163 }
164 }
165 auto in_opt = s->findShaderInput(input.getValue("component"));
166 if (!in_opt.has_value() || !in_opt.value().in.get()) {
167 REGEN_WARN("No ShaderInput found for for '" << input.getDescription() << "'.");
168 return {};
169 }
170 in = in_opt.value().in;
171 if (in->isVertexAttribute()) {
172 in->setVertexData(in->numVertices(), nullptr);
173 setInput(input, in.get(), in->numVertices());
174 } else {
175 in->setInstanceData(in->numInstances(), 1, nullptr);
176 setInput(input, in.get(), in->numInstances());
177 }
178 }
179 else if (input.hasAttribute("mesh")) {
180 auto meshVec = parser->getResources()->getMesh(parser, input.getValue("mesh"));
181 if (meshVec.get() == nullptr || meshVec->empty()) {
182 REGEN_WARN("No Mesh found for '" << input.getDescription() << "'.");
183 return {};
184 }
185 auto meshIndex = input.getValue<GLuint>("mesh-index", 0);
186 ref_ptr<Mesh> mesh = meshVec->at(0);
187 if (meshVec->size() > meshIndex) {
188 mesh = meshVec->at(meshIndex);
189 }
190 auto in_opt = mesh->findShaderInput(input.getValue("component"));
191 if (!in_opt.has_value() || !in_opt.value().in.get()) {
192 REGEN_WARN("No ShaderInput found for for '" << input.getDescription() << "'.");
193 return {};
194 }
195 in = in_opt.value().in;
196 auto numInstances = getNumInstances(mesh);
197 in->setInstanceData(numInstances, 1, nullptr);
198 setInput(input, in.get(), numInstances);
199 }
200 else if (input.hasAttribute("ubo")) {
201 auto ubo = parser->getResources()->getUBO(parser, input.getValue("ubo"));
202 if (ubo.get() == nullptr) {
203 REGEN_WARN("No UBO found for '" << input.getDescription() << "'.");
204 return {};
205 }
206 in = ref_ptr<UniformBlock>::alloc(input.getValue("name"), ubo);
207 }
208 else {
209 auto type = input.getValue<std::string>("type", "");
210 if (type == "time") {
211 auto scale = input.getValue<GLfloat>("scale", 1.0f);
212 auto timer = ref_ptr<TimerInput>::alloc(scale);
213 in = timer;
214 timer->startAnimation();
215 } else if (type == "int") {
216 in = createShaderInputTyped<ShaderInput1i, GLint>(parser, input, state, GLint(0));
217 } else if (type == "ivec2") {
218 in = createShaderInputTyped<ShaderInput2i, Vec2i>(parser, input, state, Vec2i(0));
219 } else if (type == "ivec3") {
220 in = createShaderInputTyped<ShaderInput3i, Vec3i>(parser, input, state, Vec3i(0));
221 } else if (type == "ivec4") {
222 in = createShaderInputTyped<ShaderInput4i, Vec4i>(parser, input, state, Vec4i(0));
223 } else if (type == "uint") {
224 in = createShaderInputTyped<ShaderInput1ui, GLuint>(parser, input, state, GLuint(0));
225 } else if (type == "uvec2") {
226 in = createShaderInputTyped<ShaderInput2ui, Vec2ui>(parser, input, state, Vec2ui(0));
227 } else if (type == "uvec3") {
228 in = createShaderInputTyped<ShaderInput3ui, Vec3ui>(parser, input, state, Vec3ui(0));
229 } else if (type == "uvec4") {
230 in = createShaderInputTyped<ShaderInput4ui, Vec4ui>(parser, input, state, Vec4ui(0));
231 } else if (type == "float") {
232 in = createShaderInputTyped<ShaderInput1f, GLfloat>(parser, input, state, GLfloat(0));
233 } else if (type == "vec2") {
234 in = createShaderInputTyped<ShaderInput2f, Vec2f>(parser, input, state, Vec2f(0));
235 } else if (type == "vec3") {
236 in = createShaderInputTyped<ShaderInput3f, Vec3f>(parser, input, state, Vec3f(0));
237 } else if (type == "vec4") {
238 in = createShaderInputTyped<ShaderInput4f, Vec4f>(parser, input, state, Vec4f(0));
239 } else if (type == "mat3") {
240 in = createShaderInputTyped<ShaderInputMat3, Mat3f>(parser, input, state, Mat3f::identity());
241 } else if (type == "mat4") {
242 in = createShaderInputTyped<ShaderInputMat4, Mat4f>(parser, input, state, Mat4f::identity());
243 } else {
244 REGEN_WARN("Unknown input type '" << type << "'.");
245 }
246 }
247
248 return in;
249 }

254 // Override
255 void processInput(
256 SceneParser *parser,
257 SceneInputNode &input,
258 const ref_ptr<State> &state) override {
259 ref_ptr<ShaderInput> in = createShaderInput(parser, input, state);
260 if (in.get() == nullptr) {
261 REGEN_WARN("Failed to create input for " << input.getDescription() << ".");
262 return;
263 }
264
265 ref_ptr<State> s = state;
266 while (!s->joined().empty()) {
267 s = *s->joined().rbegin();
268 }
269 auto *x = dynamic_cast<HasInput *>(s.get());
270
271 if (in->name() != input.getValue("name")) {
272 // TODO: there is a problem with renaming of inputs, as state configurer
273 // uses the name. We can avoid problems in shader generation by adding some macros here manually.
274 // but this case of inserting inputs with different names should be handled better IMO.
275 s->shaderDefine(REGEN_STRING("HAS_"<<input.getValue("name")), "TRUE");
276 }
277
278 if (input.getValue<bool>("join", true)) {
279 if (x == nullptr) {
280 ref_ptr<HasInputState> inputState = ref_ptr<HasInputState>::alloc();
281 inputState->setInput(in, input.getValue("name"));
282 state->joinStates(inputState);
283 } else {
284 x->setInput(in, input.getValue("name"));
285 }
286 }

289 template<class U, class T>
290 static ref_ptr<U> createShaderInputTyped(
291 SceneParser *parser,
292 SceneInputNode &input,
293 const ref_ptr<State> &state,
294 const T &defaultValue) {
295 if (!input.hasAttribute("name")) {
296 REGEN_WARN("No name specified for " << input.getDescription() << ".");
297 return ref_ptr<U>();
298 }
299 ref_ptr<U> v = ref_ptr<U>::alloc(input.getValue("name"));
300 v->set_isConstant(input.getValue<bool>("is-constant", false));
301
302 auto numInstances = input.getValue<GLuint>("num-instances", 1u);
303 auto numVertices = input.getValue<GLuint>("num-vertices", 1u);
304 bool isInstanced = input.getValue<bool>("is-instanced", false);
305 bool isAttribute = input.getValue<bool>("is-attribute", false);
306 GLuint count = 1;
307
308 if (isInstanced) {
309 v->setInstanceData(numInstances, 1, nullptr);
310 count = numInstances;
311 } else if (isAttribute) {
312 v->setVertexData(numVertices, nullptr);
313 count = numVertices;
314 } else {
315 v->setUniformData(input.getValue<T>("value", defaultValue));
316 }
317
318 // Handle Attribute values.
319 if (isInstanced || isAttribute) {
320 auto values = v->mapClientDataRaw(ShaderData::WRITE);
321 auto typedValues = (T*)values.w;
322 for (GLuint i = 0; i < count; i += 1) typedValues[i] = defaultValue;
323 values.unmap();
324 setInput(input, v.get(), count);
325 }
326
327 // Load
328 for (const auto& n : input.getChildren("animation")) {
329 ref_ptr<InputAnimation<U, T> > inputAnimation = ref_ptr<InputAnimation<U, T> >::alloc(v);
330 for (const auto &m : n->getChildren("key-frame")) {
331 inputAnimation->push_back(
332 m->getValue<T>("value", defaultValue),
333 m->getValue<GLdouble>("dt", 1.0)
334 );
335 }
336 state->attach(inputAnimation);
337 inputAnimation->startAnimation();
338 }
339
340 return v;
341 }
342 };
