264 textureCounter_ = 0;
265 // init toggle states
266 GLenum enabledToggles[] = {
267 GL_CULL_FACE, GL_DEPTH_TEST,
268 GL_TEXTURE_CUBE_MAP_SEAMLESS
269 };
270 for (GLint i = 0; i < TOGGLE_STATE_LAST; ++i) {
271 GLenum e = toggleToID((Toggle) i);
272 // avoid initial state set for unsupported states
273 if (e == GL_NONE) continue;
274
275 GLboolean enabled = GL_FALSE;
276 for (GLuint j = 0; j < sizeof(enabledToggles) / sizeof(GLenum); ++j) {
277 if (enabledToggles[j] == e) {
278 enabled = GL_TRUE;
279 break;
280 }
281 }
282 toggles_.push(i, enabled);
283 }
284 // init value states
285 cullFace_.push(GL_BACK);
286 depthMask_.push(GL_TRUE);
287 depthFunc_.push(GL_LEQUAL);
288 depthClear_.push(1.0);
289 depthRange_.push(DepthRange(0.0, 1.0));

291 blendFunc_.push(BlendFunction(GL_ONE, GL_ONE, GL_ZERO, GL_ZERO));
292 polygonMode_.push(GL_FILL);
293 polygonOffset_.push(Vec2f(0.0f));
294 pointSize_.push(1.0);
295 lineWidth_.push(1.0);
296 colorMask_.push(ColorMask(GL_TRUE));
297 clearColor_.push(ClearColor(0.0f));
298 logicOp_.push(GL_COPY);
299 frontFace_.push(GL_CCW);
300 pointFadeThreshold_.push(1.0);
301 pointSpriteOrigin_.push(GL_UPPER_LEFT);
302 activeTexture_.push(GL_TEXTURE0);
303 textureBuffer_.push(0);
304 GL_ERROR_LOG();
305}

308 switch (target) {
309 case GL_UNIFORM_BUFFER:
310 return 0;
311 case GL_ATOMIC_COUNTER_BUFFER:

320void RenderState::bindBufferBase(GLenum target, GLuint index, GLuint buffer) {
321 auto arrayIndex = bufferBaseIndex(target);
322 auto &bindings = bufferBaseBindings_[arrayIndex];
323 auto needle = bindings.find(index);
324 if (needle != bindings.end() && needle->second == buffer) {
325 // buffer is already bound
326 return;
327 }
328 bindings[index] = buffer;
329 Regen_BindBufferBase(target, index, buffer);
330}
