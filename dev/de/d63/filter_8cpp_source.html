55void Filter::setInput(const ref_ptr<Texture> &input) {
56 set_input(input);
57
58 //GLuint inputSize = min(input_->width(), input_->height());
59 //GLuint bufferSize = scaleFactor_*inputSize;
60 GLuint bufferW = scaleFactor_ * input_->width();
61 GLuint bufferH = scaleFactor_ * input_->height();
62
63 GLuint inputDepth = 1;
64 if (dynamic_cast<Texture3D *>(input_.get())) {
65 inputDepth = (dynamic_cast<Texture3D *>(input_.get()))->depth();
66 }
67
68 // create the render target. As this is the first filter with this target
69 // the attachment point is GL_COLOR_ATTACHMENT0 and only a single texture
70 // is added to the fbo.
71 out_ = ref_ptr<Output>::alloc();
72 out_->fbo_ = ref_ptr<FBO>::alloc(bufferW, bufferH, inputDepth);
73 out_->tex0_ = createTexture();
74
75 // call drawBuffer( GL_COLOR_ATTACHMENT0 )
76 outputAttachment_ = GL_COLOR_ATTACHMENT0;
77 if (drawBufferState_.get()) {
78 disjoinStates(drawBufferState_);
79 }
80 drawBufferState_ = ref_ptr<DrawBufferState>::alloc(out_->fbo_);
81 drawBufferState_->colorBuffers.buffers_.push_back(outputAttachment_);
82 joinStatesFront(drawBufferState_);
83}

85void Filter::setInput(
86 const ref_ptr<Output> &lastOutput, GLenum lastAttachment) {
87 // take last output as input
88 if (lastAttachment == GL_COLOR_ATTACHMENT0) {
89 input_ = lastOutput->tex0_;
90 } else {
91 input_ = lastOutput->tex1_;
92 }
93 set_input(input_);
94
95 out_ = lastOutput;
96 // make sure two textures created for ping pong rendering
97 if (!out_->tex1_.get()) {
98 out_->tex1_ = createTexture();
99 }
100
101 // call drawBuffer( outputAttachment_ )
102 outputAttachment_ = GL_COLOR_ATTACHMENT0 + (GL_COLOR_ATTACHMENT1 - lastAttachment);
103 if (drawBufferState_.get()) {
104 disjoinStates(drawBufferState_);
105 }
106 drawBufferState_ = ref_ptr<DrawBufferState>::alloc(out_->fbo_);
107 drawBufferState_->colorBuffers.buffers_.push_back(outputAttachment_);
108 joinStatesFront(drawBufferState_);
109}

115FilterSequence::FilterSequence(const ref_ptr<Texture> &input, GLboolean bindInput)
116 : State(),

118 clearFirstFilter_(GL_FALSE),
119 clearColor_(Vec4f(0.0f)),
120 lastWidth_(0u),
121 lastHeight_(0u),
122 bindInput_(bindInput) {
123 format_ = GL_NONE;
124 internalFormat_ = GL_NONE;
125 pixelType_ = GL_NONE;
126
127 viewport_ = ref_ptr<ShaderInput2f>::alloc("viewport");
128 viewport_->setUniformData(Vec2f(
129 (GLfloat) input->width(), (GLfloat) input->height()));
130
131 inverseViewport_ = ref_ptr<ShaderInput2f>::alloc("inverseViewport");
132 inverseViewport_->setUniformData(Vec2f(
133 1.0f / (GLfloat) input->width(), 1.0f / (GLfloat) input->height()));
134
135#ifdef USE_FBO_UBO
136 uniforms_ = ref_ptr<UniformBlock>::alloc("FBO");
137 uniforms_->addUniform(viewport_);
138 uniforms_->addUniform(inverseViewport_);
139 joinShaderInput(uniforms_);
140#else
141 joinShaderInput(viewport_);
142 joinShaderInput(inverseViewport_);
143#endif
144
145 ref_ptr<ShaderInput2f> inverseViewport;
146
147 // use layered geometry shader for 3d textures
148 if (dynamic_cast<TextureCube *>(input_.get())) {
149 shaderDefine("RENDER_LAYER", "6");
150 shaderDefine("RENDER_TARGET", "CUBE");
151 } else if (dynamic_cast<Texture3D *>(input_.get())) {
152 auto *tex3D = (Texture3D *) input_.get();
153 shaderDefine("RENDER_LAYER", REGEN_STRING(tex3D->depth()));
154 shaderDefine("RENDER_TARGET", "2D_ARRAY");
155 } else {
156 shaderDefine("RENDER_LAYER", "1");
157 shaderDefine("RENDER_TARGET", "2D");
158 }
159}

161ref_ptr<FilterSequence> FilterSequence::load(LoadingContext &ctx, scene::SceneInputNode &input) {
162 auto scene = ctx.scene();
163 if (!input.hasAttribute("texture") && !input.hasAttribute("fbo")) {
164 REGEN_WARN("Ignoring " << input.getDescription() << " without input texture.");
165 return {};
166 }
167
168 ref_ptr<Texture> tex;
169 if (input.hasAttribute("texture")) {
170 tex = scene->getResource<Texture>(input.getValue("texture"));
171 if (tex.get() == nullptr) {
172 REGEN_WARN("Unable to find Texture for " << input.getDescription() << ".");
173 return {};
174 }
175 } else {
176 ref_ptr<FBO> fbo = scene->getResource<FBO>(input.getValue("fbo"));
177 if (fbo.get() == nullptr) {
178 REGEN_WARN("Unable to find FBO for " << input.getDescription() << ".");
179 return {};
180 }
181 auto attachment = input.getValue<GLuint>("attachment", 0);
182 std::vector<ref_ptr<Texture> > textures = fbo->colorTextures();
183 if (attachment >= textures.size()) {
184 REGEN_WARN("FBO " << input.getValue("fbo") <<
185 " has less then " << (attachment + 1) << " attachments.");
186 return {};
187 }
188 tex = textures[attachment];
189 }
190
191 ref_ptr<FilterSequence> filterSeq = ref_ptr<FilterSequence>::alloc(
192 tex,
193 input.getValue<bool>("bind-input", true));
194 filterSeq->set_format(glenum::textureFormat(
195 input.getValue<std::string>("format", "NONE")));
196 filterSeq->set_internalFormat(glenum::textureInternalFormat(
197 input.getValue<std::string>("internal-format", "NONE")));
198 filterSeq->set_pixelType(glenum::pixelType(
199 input.getValue<std::string>("pixel-type", "NONE")));
200 if (input.hasAttribute("clear-color")) {
201 filterSeq->setClearColor(
202 input.getValue<Vec4f>("clear-color", Vec4f(0.0f)));
203 }
204
205 const std::list<ref_ptr<scene::SceneInputNode> > &childs = input.getChildren();
206 for (auto it = childs.begin(); it != childs.end(); ++it) {
207 ref_ptr<scene::SceneInputNode> n = *it;
208 if (!n->hasAttribute("shader")) {
209 REGEN_WARN("Ignoring filter '" << n->getDescription() << "' without shader.");
210 continue;
211 }
212 filterSeq->addFilter(ref_ptr<Filter>::alloc(
213 n->getValue("shader"),
214 n->getValue<GLfloat>("scale", 1.0f)));
215 }
216
217 StateConfigurer shaderConfigurer;
218 shaderConfigurer.addNode(ctx.parent().get());
219 shaderConfigurer.addState(filterSeq.get());
220 filterSeq->createShader(shaderConfigurer.cfg());
221
222 // Make the filter output available to the Texture resource provider.
223 //scene->putTexture(input.getName(), filterSeq->output());
224 scene->putResource<Texture>(input.getName(), filterSeq->output());
225
226 return filterSeq;
227}

248void FilterSequence::addFilter(const ref_ptr<Filter> &f) {
249 if (filterSequence_.empty()) {
250 f->set_bindInput(bindInput_);
251 f->set_format(format_);
252 f->set_internalFormat(internalFormat_);
253 f->set_pixelType(pixelType_);
254 // no filter was added before, create a new framebuffer
255 f->setInput(input_);
256 } else {
257 ref_ptr<Filter> lastFilter = *filterSequence_.rbegin();
258 // another filter was added before
259 if (math::isApprox(f->scaleFactor(), 1.0)) {
260 // filter does not apply scaling. We gonna ping pong render
261 // to the framebuffer provided by previous filter.
262 f->setInput(lastFilter->output(), lastFilter->outputAttachment());
263 } else {
264 // create a new framebuffer for this filter
265 GLenum last = lastFilter->outputAttachment();
266 if (last == GL_COLOR_ATTACHMENT0) {
267 f->setInput(lastFilter->output()->tex0_);
268 } else {
269 f->setInput(lastFilter->output()->tex1_);
270 }
271 }
272 }
273
274 filterSequence_.push_back(f);
275}

286void FilterSequence::resize() {
287 //GLuint size = min(input_->width(), input_->height());
288 GLuint width = input_->width();
289 GLuint height = input_->height();
290 if (width == lastWidth_ && height == lastHeight_) return;
291
292 FBO *last = nullptr;
293 for (auto &it: filterSequence_) {
294 auto *f = (Filter *) it.get();
295 FBO *fbo = f->output()->fbo_.get();
296
297 if (last != fbo) {
298 width *= f->scaleFactor();
299 height *= f->scaleFactor();
300 fbo->resize(width, height, fbo->depth());
301 }
302
303 last = fbo;
304 }
305
306 lastWidth_ = width;
307 lastHeight_ = height;
308}

310void FilterSequence::enable(RenderState *rs) {
311 State::enable(rs);
312
313 if (filterSequence_.empty()) { return; }
314 // auto-resize if input texture size changed
315 resize();
316
317 if (clearFirstFilter_) {
318 Filter *firstFilter = (Filter *) (*filterSequence_.begin()).get();
319 rs->drawFrameBuffer().push(firstFilter->output()->fbo_->id());
320 rs->clearColor().push(clearColor_);
321 glClear(GL_COLOR_BUFFER_BIT);
322 rs->clearColor().pop();
323 rs->drawFrameBuffer().pop();
324 }
325 for (auto &it: filterSequence_) {
326 auto *f = (Filter *) it.get();
327
328 FBO *fbo = f->output()->fbo_.get();
329 rs->drawFrameBuffer().push(fbo->id());
330 rs->viewport().push(fbo->glViewport());
331 viewport_->setVertex(0, fbo->viewport()->getVertex(0).r);
332 inverseViewport_->setVertex(0, fbo->inverseViewport()->getVertex(0).r);
333
334 f->enable(rs);
335 f->disable(rs);
336
337 rs->viewport().pop();
338 rs->drawFrameBuffer().pop();
339 }
