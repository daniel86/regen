40const t_julianDay jd(t_aTime aTime)
41{
42 assert(aTime.month > 0);
43
44 if(aTime.month < 3)
45 {
46 // "If the date is in January or February, it is concidered to
47 // be 13th or 14th month of the preceding year."
48
49 --aTime.year;
50 aTime.month += 12;
51 }
52
53 // from "A Physically-Based Night Sky Model" - 2001 - Wann Jensen et al.
54 // "Local time is GMT with a zone correction. Terrestrial Time (TT)
55 // is essentially the time kept by atomic clocks. As it is not corrected
56 // for the slowing of the Earthï¿½s rotation, it gains on GMT by about
57 // a second per year. The current difference ␁T is about 64 sec in 2000.
58 // It should be added to s in all equations for precise computation..."
59
60 unsigned int s = aTime.second;
61
62 const t_longf h(_day(aTime.hour, aTime.minute, s));
63
64 signed int b = 0;
65
66 const int i = aTime.year * 10000 + aTime.month * 100 + aTime.day;
67 if(i >= 15821015)
68 {
69 // gregorian calendar
70 const signed int a = _int(aTime.year * 0.01);
71 b = 2 - a + _int(a * 0.25);
72 }
73 else if(i > 15821004)
74 return 0.0; // The gregorian calender follows with 15. on the 4.
75 // julian oktober in 1582 - poor people who missed a birthday these days :P
76
77 return _int(365.25 * (aTime.year + 4716))
78 + _int(30.600001 * (aTime.month + 1)) + aTime.day + h + b - 1524.5;
79}

105const t_aTime makeTime(
106 t_julianDay julianDate
107, const short GMTOffset)
108{
109 assert(julianDate >= 0.0);
110
111 julianDate += 0.5;
112
113 const int z = _int(julianDate);
114 const t_longf f = _frac(julianDate);
115
116 long int a(z); // julian
117 if(z >= 2299161) // gregorian
118 {
119 const int g = _int((z - 1867216.25) / 36524.25);
120 a = z + 1 + g - _int(g / 4);
121 }
122
123 const int b = _int(a + 1524);
124 const int c = _int((b - 122.1) / 365.25);
125 const int d = _int(365.25 * c);
126 const int e = _int((b - d) / 30.600001);
127
128
129 const short day = static_cast<short>(b - d - _int(30.600001 * e));
130 const short month = static_cast<short>(e < 14 ? e - 1 : e - 13);
131 const short year = static_cast<short>(month > 2 ? c - 4716 : c - 4715);
132
133 const t_longf h = f * 24.0;
134 const t_longf m = _frac(h) * 60.0;
135 const t_longf s = _frac(m) * 60.0001;
136
137 const short hour = _short(h);
138 const short minute = _short(m);
139 const short second = _short(s);
140
141 return t_aTime(year, month, day, hour, minute, second, GMTOffset);
142}
