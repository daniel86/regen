83void AnimationManager::addAnimation(Animation *animation) {
84 // Don't add while removing
85 while (removeInProgress_) usleepRegen(1000);
86
87 addThreadID_ = boost::this_thread::get_id();
88 addInProgress_ = true;
89
90 if (animation->isGPUAnimation()) {
91 if (rootState_.get() != nullptr) {
92 animation->joinAnimationState(rootState_);
93 }
94 if (glInProgress_ && addThreadID_ == glThreadID_) {
95 // Called from glAnimate().
96 glChangedDuringLoop_ = true;
97 gpuAnimations_.insert(animation);
98 } else {
99 // Wait for the current loop to finish.
100 while (glInProgress_) usleepRegen(1000);
101 // save to remove from set
102 gpuAnimations_.insert(animation);
103 }
104 }
105
106 if (animation->isCPUAnimation()) {
107 if (animation->isSynchronized()) {
108 if (animInProgress_ && addThreadID_ == animationThreadID_) {
109 // Called from animate().
110 animChangedDuringLoop_ = true;
111 synchronizedAnimations_.emplace_back(animation);
112 } else {
113 // Wait for the current loop to finish.
114 while (animInProgress_) usleepRegen(1000);
115 // save to remove from set
116 synchronizedAnimations_.emplace_back(animation);
117 }
118 } else {
119 // start a new dedicated thread
120 boost::unique_lock<boost::mutex> lock(unsynchronizedMut_);
121 unsynchronizedAnimations_.emplace_back(animation);
122 unsynchronizedThreads_.emplace_back([this, animation]()
123 { runUnsynchronized(animation); });
124 }
125 }
126
127 addInProgress_ = false;
128}

130void AnimationManager::removeAnimation(Animation *animation) {
131 // Don't remove while adding
132 while (addInProgress_) usleepRegen(1000);
133
134 removeThreadID_ = boost::this_thread::get_id();
135 removeInProgress_ = true;
136
137 if (animation->isGPUAnimation()) {
138 if (glInProgress_ && removeThreadID_ == glThreadID_) {
139 // Called from glAnimate().
140 glChangedDuringLoop_ = true;
141 gpuAnimations_.erase(animation);
142 } else {
143 // Wait for the current loop to finish.
144 while (glInProgress_) usleepRegen(1000);
145 // save to remove from set
146 gpuAnimations_.erase(animation);
147 }
148 }
149
150 if (animation->isCPUAnimation()) {
151 if (animation->isSynchronized()) {
152 if (animInProgress_ && removeThreadID_ == animationThreadID_) {
153 // Called from animate().
154 animChangedDuringLoop_ = true;
155 } else {
156 // Wait for the current loop to finish.
157 while (animInProgress_) usleepRegen(1000);
158 // save to remove from set
159 }
160 // remove from list
161 auto it = synchronizedAnimations_.begin();
162 while (it != synchronizedAnimations_.end()) {
163 if (*it == animation) {
164 synchronizedAnimations_.erase(it);
165 break;
166 }
167 ++it;
168 }
169 }
170 else {
171 // remove from list
172 boost::unique_lock<boost::mutex> lock(unsynchronizedMut_);
173 animation->isRunning_ = false;
174 for (size_t i = 0; i < unsynchronizedAnimations_.size(); i++) {
175 if (unsynchronizedAnimations_[i] == animation) {
176 unsynchronizedThreads_[i].join();
177 unsynchronizedAnimations_.erase(unsynchronizedAnimations_.begin() + i);
178 unsynchronizedThreads_.erase(unsynchronizedThreads_.begin() + i);
179 break;
180 }
181 }
182 }
183 }
184
185 removeInProgress_ = false;
186}

210void AnimationManager::waitForFrame() {
211#ifdef SYNCHRONIZE_THREADS
212 #ifdef USE_SYNCHRONIZE_SPINLOCK
213 while (!hasNextFrame_.load(std::memory_order_acquire)) {
214 #ifdef SYNCHRONIZE_SPINLOCK_SLEEP
215 std::this_thread::sleep_for(std::chrono::microseconds(SYNCHRONIZE_SPINLOCK_SLEEP));
216 #endif
217 }
218 #else
219 // wait until a new frame is rendered.
220 {
221 boost::unique_lock<boost::mutex> lock(frameMut_);
222 while (!hasNextFrame_.load(std::memory_order_acquire)) {
223 frameCond_.wait(lock);
224 }
225 }
226 #endif // USE_SYNCHRONIZE_SPINLOCK
227 // toggle hasNextFrame_ to false
228 hasNextFrame_.store(false, std::memory_order_release);
229#endif
230}

254void AnimationManager::updateGraphics(RenderState *_, GLdouble dt) {
255 if (pauseFlag_) return;
256 glThreadID_ = boost::this_thread::get_id();
257
258#ifdef SYNCHRONIZE_THREADS
259 nextFrame();
260#endif
261 // wait for remove/remove to return
262 while (removeInProgress_) usleepRegen(1000);
263 while (addInProgress_) usleepRegen(1000);
264
265 // Set processing flags, so that other threads can wait
266 // for the completion of this loop
267 glInProgress_ = true;
268 std::set<Animation *> processed;
269 bool animationsRemaining = true;
270 while (animationsRemaining && !pauseFlag_) {
271 animationsRemaining = false;
272 for (auto it = gpuAnimations_.begin(); it != gpuAnimations_.end(); ++it) {
273 Animation *anim = *it;
274 processed.insert(anim);
275 if (anim->isRunning()) {
276 auto animState = anim->animationState();
277 animState->enable(RenderState::get());
278 anim->glAnimate(RenderState::get(), dt);
279 animState->disable(RenderState::get());
280 // Animation was removed in glAnimate call.
281 // We have to restart the loop because iterator is invalid.
282 if (glChangedDuringLoop_) {
283 glChangedDuringLoop_ = false;
284 animationsRemaining = true;
285 break;
286 }
287 }
288 }
289 }
290 glInProgress_ = false;
291
292 waitForStep();
293}

295void AnimationManager::runUnsynchronized(Animation *animation) const {
296 // call animate until the animation is stopped
297 // try to reach a dedicated frame rate.
298 const auto goalTime = (1.0 / animation->desiredFrameRate()) * 1000.0;
299 boost::posix_time::ptime time0, time1;
300 time0 = boost::posix_time::ptime(boost::posix_time::microsec_clock::local_time());
301 time1 = time0;
302 double dt;
303 int delta_micros;
304
305 while (animation->isRunning()) {
306 if (pauseFlag_) {
307 usleepRegen(IDLE_SLEEP);
308 time0 += boost::posix_time::microseconds(IDLE_SLEEP);
309 time1 += boost::posix_time::microseconds(IDLE_SLEEP);
310 continue;
311 }
312
313 // make a step
314 dt = static_cast<double>((time1 - time0).total_microseconds()) / 1000.0;
315 time0 = boost::posix_time::ptime(boost::posix_time::microsec_clock::local_time());
316 animation->animate(static_cast<double>(dt));
317 time1 = boost::posix_time::ptime(boost::posix_time::microsec_clock::local_time());
318
319 // synchronize to the desired frame rate
320 dt = static_cast<double>((time1 - time0).total_microseconds()) / 1000.0;
321 if (dt < goalTime) {
322 delta_micros = static_cast<int>((goalTime - dt) * 1000);
323 usleepRegen(delta_micros);
324 time1 += boost::posix_time::microseconds(delta_micros);
325 }
326 }

329void AnimationManager::run() {
330 animationThreadID_ = boost::this_thread::get_id();
331 resetTime();
332
333 while (!closeFlag_) {
334 time_ = boost::posix_time::ptime(
335 boost::posix_time::microsec_clock::local_time());
336
337 if (pauseFlag_ || synchronizedAnimations_.empty()) {
338#ifndef SYNCHRONIZE_THREADS
339 usleepRegen(IDLE_SLEEP);
340#endif // SYNCHRONIZE_THREADS
341 } else {
342 double dt = ((GLdouble) (time_ - lastTime_).total_microseconds()) / 1000.0;
343
344 // wait for remove/add to return
345 while (removeInProgress_) usleepRegen(1000);
346 while (addInProgress_) usleepRegen(1000);
347
348 animInProgress_ = true;
349 bool animsRemaining = true;
350 std::set<Animation *> processed;
351 while (animsRemaining) {
352 animsRemaining = false;
353 for (auto anim : synchronizedAnimations_) {
354 processed.insert(anim);
355 if (anim->isRunning()) {
356 anim->animate(dt);
357 // Animation was removed in animate call.
358 // We have to restart the loop because iterator is invalid.
359 if (animChangedDuringLoop_) {
360 animChangedDuringLoop_ = false;
361 animsRemaining = true;
362 break;
363 }
364 }
365 }
366 }
367 for (auto &index : spatialIndices_) {
368 index.second->update(dt);
369 }
370 animInProgress_ = false;
371#ifndef SYNCHRONIZE_THREADS
372 if(dt<10) usleepRegen((10-dt) * 1000);
373#endif // SYNCHRONIZE_THREADS
374 }
375 lastTime_ = time_;
376
377#ifdef SYNCHRONIZE_THREADS
378 nextStep();
379 waitForFrame();
380#endif // SYNCHRONIZE_THREADS
381 }
382}
