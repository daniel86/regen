16FBOState::FBOState(const ref_ptr<FBO> &fbo)
17 : State(), fbo_(fbo) {
18#ifdef USE_FBO_UBO
19 // TODO: there are some nodes in the scene graph with special handling for viewport and inverseViewport
20 // uniforms. This includes BloomPass and FilterSequence. I think the problem is that FBOState appears
21 // as a parent, and then the shader is compiled for a uniform block.
22 // However, BloomPass and FilterSequence rather use gUniform* functions to set the uniforms, which
23 // pretty sure does not work with uniform blocks.
24 joinShaderInput(fbo->uniforms());
25#else
26 joinShaderInput(fbo->viewport());
27 joinShaderInput(fbo->inverseViewport());
28#endif
29 for (auto &attachment: fbo->colorTextures()) {
30 shaderDefine(REGEN_STRING("FBO_ATTACHMENT_" << attachment->name()), "TRUE");
31 }
32}

134static std::vector<std::string> getFBOAttachments(scene::SceneInputNode &input, const std::string &key) {
135 std::vector<std::string> out;
136 auto attachments = input.getValue<std::string>(key, "");
137 if (attachments.empty()) {
138 REGEN_WARN("No attachments specified in " << input.getDescription() << ".");
139 } else {
140 boost::split(out, attachments, boost::is_any_of(","));

145ref_ptr<State> FBOState::load(LoadingContext &ctx, scene::SceneInputNode &input) {
146 auto scene = ctx.scene();
147
148 if (input.getName() == "SCREEN") {
149 GLenum drawBuffer = glenum::drawBuffer(
150 input.getValue<std::string>("draw-buffer", "FRONT"));
151 ref_ptr<ScreenState> screenState =
152 ref_ptr<ScreenState>::alloc(scene->getViewport(), drawBuffer);
153
154 return screenState;
155 } else {
156 ref_ptr<FBO> fbo = scene->getResource<FBO>(input.getName());
157 if (fbo.get() == nullptr) {
158 REGEN_WARN("Unable to find FBO for '" << input.getDescription() << "'.");
159 return {};
160 }
161 ref_ptr<FBOState> fboState = ref_ptr<FBOState>::alloc(fbo);
162
163 if (input.hasAttribute("clear-depth") &&
164 input.getValue<int>("clear-depth", 1)) {
165 fboState->setClearDepth();
166 }
167
168 if (input.hasAttribute("clear-stencil") &&
169 input.getValue<int>("clear-stencil", 1)) {
170 fboState->setClearStencil();
171 }
172
173 if (input.hasAttribute("clear-buffers")) {
174 std::vector<std::string> idVec = getFBOAttachments(input, "clear-buffers");
175 std::vector<GLenum> buffers(idVec.size());
176 for (GLuint i = 0u; i < idVec.size(); ++i) {
177 GLint v;
178 std::stringstream ss(idVec[i]);
179 ss >> v;
180 buffers[i] = GL_COLOR_ATTACHMENT0 + v;
181 }
182
183 ClearColorState::Data data;
184 data.clearColor = input.getValue<Vec4f>("clear-color", Vec4f(0.0));
185 data.colorBuffers = DrawBuffers(buffers);
186 fboState->setClearColor(data);
187 }
188
189 if (input.hasAttribute("draw-buffers")) {
190 std::vector<std::string> idVec = getFBOAttachments(input, "draw-buffers");
191 std::vector<GLenum> buffers(idVec.size());
192 for (GLuint i = 0u; i < idVec.size(); ++i) {
193 GLint v;
194 std::stringstream ss(idVec[i]);
195 ss >> v;
196 buffers[i] = GL_COLOR_ATTACHMENT0 + v;
197 }
198 fboState->setDrawBuffers(buffers);
199 } else if (input.hasAttribute("ping-pong-buffers")) {
200 std::vector<std::string> idVec = getFBOAttachments(input, "ping-pong-buffers");
201 std::vector<GLenum> buffers(idVec.size());
202 for (GLuint i = 0u; i < idVec.size(); ++i) {
203 GLint v;
204 std::stringstream ss(idVec[i]);
205 ss >> v;
206 buffers[i] = GL_COLOR_ATTACHMENT0 + v;
207 }
208 fboState->setPingPongBuffers(buffers);
209 }
210
211 return fboState;
212 }
