57 // Override
58 void processInput(
59 SceneParser *parser,
60 SceneInputNode &input,
61 const ref_ptr<StateNode> &parent) override {
62 ref_ptr<StateNode> newNode;
63 auto nodeType = input.getValue<std::string>("type", "");
64
65 if (input.hasAttribute("import")) {
66 // Handle node imports
67 const ref_ptr<SceneInputNode> &root = parser->getRoot();
68 const std::string importID = input.getValue("import");
69
70 // TODO: Use previously loaded node. It's problematic because...
71 // - Shaders below the node are configured only for the first context.
72 // - Uniforms above node are joined into shader
73 ref_ptr<SceneInputNode> imported =
74 root->getFirstChild(REGEN_NODE_CATEGORY, importID);
75 if (imported.get() == nullptr) {
76 REGEN_WARN("Unable to import node '" << importID << "'.");
77 } else {
78 newNode = createNode(parser, *imported.get(), parent);
79 handleAttributes(parser, *imported.get(), newNode);
80 handleAttributes(parser, input, newNode);
81 handleChildren(parser, input, newNode);
82 handleChildren(parser, *imported.get(), newNode);
83 }
84 }
85 if (input.hasAttribute("cull-shape")) {
86 newNode = createCullNode(parser, input, parent);
87 if (newNode.get() != nullptr) {
88 handleAttributes(parser, input, newNode);
89 handleChildren(parser, input, newNode);
90 } else {
91 REGEN_WARN("Unable to create culling node for '" << input.getDescription() << "'.");
92 }
93 } else {
94 newNode = createNode(parser, input, parent);
95 handleAttributes(parser, input, newNode);
96 handleChildren(parser, input, newNode);
97 }
98 }

100 static void handleAttributes(
101 SceneParser *parser,
102 SceneInputNode &input,
103 const ref_ptr<StateNode> &newNode) {
104 if (newNode.get() != nullptr && input.hasAttribute("name")) {
105 newNode->set_name(input.getValue("name"));
106 auto newNodeID = parser->putNamedObject(newNode);
107 auto u_objectID = ref_ptr<ShaderInput1i>::alloc("objectID");
108 u_objectID->setUniformData(newNodeID);
109 newNode->state()->joinShaderInput(u_objectID);
110 }
111 if (input.hasAttribute("sort")) {
112 // Sort node children by model view matrix.
113 auto sortMode = input.getValue<GLuint>("sort", 0);
114 ref_ptr<Camera> sortCam =
115 parser->getResources()->getCamera(parser, input.getValue<std::string>("sort-camera", ""));
116 if (sortCam.get() == nullptr) {
117 REGEN_WARN("Unable to find Camera for '" << input.getDescription() << "'.");
118 } else {
119 newNode->state()->joinStatesFront(
120 ref_ptr<SortByModelMatrix>::alloc(newNode, sortCam, (sortMode == 1)));
121 }
122 }
123 }

125 static void handleChildren(
126 SceneParser *parser,
127 SceneInputNode &input,
128 const ref_ptr<StateNode> &newNode) {
129 // Process node children
130 const std::list<ref_ptr<SceneInputNode> > &childs = input.getChildren();
131 for (auto it = childs.begin(); it != childs.end(); ++it) {
132 const ref_ptr<SceneInputNode> &x = *it;
133 // First try node processor
134 ref_ptr<NodeProcessor> nodeProcessor = parser->getNodeProcessor(x->getCategory());
135 if (nodeProcessor.get() != nullptr) {
136 nodeProcessor->processInput(parser, *x.get(), newNode);
137 continue;
138 }
139 // Second try state processor
140 ref_ptr<StateProcessor> stateProcessor = parser->getStateProcessor(x->getCategory());
141 if (stateProcessor.get() != nullptr) {
142 stateProcessor->processInput(parser, *x.get(), newNode->state());
143 continue;
144 }
145 REGEN_WARN("No processor registered for '" << x->getDescription() << "'.");
146 }
147 }

149 static ref_ptr<StateNode> createNode(
150 SceneParser *parser,
151 SceneInputNode &input,
152 const ref_ptr<StateNode> &parent) {
153 GLuint numIterations = 1;
154 if (input.hasAttribute("num-iterations")) {
155 numIterations = input.getValue<GLuint>("num-iterations", 1u);
156 }
157
158 ref_ptr<State> state = ref_ptr<State>::alloc();
159 ref_ptr<StateNode> newNode;
160 if (numIterations > 1) {
161 newNode = ref_ptr<LoopNode>::alloc(state, numIterations);
162 } else {
163 newNode = ref_ptr<StateNode>::alloc(state);
164 }
165 newNode->set_name(input.getName());
166 parent->addChild(newNode);
167 parser->putNode(input.getName(), newNode);
168
169 return newNode;
170 }

172 static ref_ptr<SpatialIndex> getSpatialIndex(SceneParser *parser, SceneInputNode &input) {
173 ref_ptr<SpatialIndex> spatialIndex;
174 if (input.hasAttribute("spatial-index")) {
175 auto spatialIndexID = input.getValue("spatial-index");
176 spatialIndex = parser->getResources()->getIndex(parser, spatialIndexID);
177 } else {
178 auto indexNode = parser->getRoot()->getFirstChild("index");
179 if (indexNode.get()) {
180 spatialIndex = parser->getResources()->getIndex(parser, indexNode->getName());
181 }
182 }
183 return spatialIndex;
184 }

186 static ref_ptr<StateNode> createCullNode(
187 SceneParser *parser,
188 SceneInputNode &input,
189 const ref_ptr<StateNode> &parent) {
190 ref_ptr<GeometricCulling> cullNode;
191 // get the parent camera. Note that this will be the light camera in case
192 // updating the shadow map.
193 auto cam = parent->getParentCamera();
194 if (cam.get() == nullptr) {
195 REGEN_WARN("No Camera can be found for '" << input.getDescription() << "'.");
196 return cullNode;
197 }
198 auto spatialIndex = getSpatialIndex(parser, input);
199 if (spatialIndex.get() == nullptr) {
200 REGEN_WARN("No SpatialIndex can be found for '" << input.getDescription() << "'.");
201 return cullNode;
202 }
203
204 auto shapeName = input.getValue<std::string>("cull-shape", "");
205 cullNode = ref_ptr<GeometricCulling>::alloc(cam, spatialIndex, shapeName);
206 cullNode->set_name(input.getName());
207 if (input.hasAttribute("sort-mode")) {
208 cullNode->setInstanceSortMode(input.getValue<SortMode>("sort-mode", SortMode::FRONT_TO_BACK));
209 }
210 parent->addChild(cullNode);
211 parser->putNode(input.getName(), cullNode);
212
213 return cullNode;
214 }
