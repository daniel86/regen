25static void transformMatrix(
26 const string &target, Mat4f &mat, const Vec3f &value) {
27 if (target == "translate") {
28 mat.x[12] += value.x;
29 mat.x[13] += value.y;
30 mat.x[14] += value.z;
31 } else if (target == "scale") {
32 mat.scale(value);
33 } else if (target == "rotate") {
34 Quaternion q(0.0, 0.0, 0.0, 1.0);
35 q.setEuler(value.x, value.y, value.z);
36 mat *= q.calculateMatrix();
37 } else {
38 REGEN_WARN("Unknown distribute target '" << target << "'.");
39 }
40}

84static void makeInstance(InstancePlaneGenerator &generator, const PlaneCell &cell) {
85 if (generator.maskData) {
86 auto density = generator.maskTexture->sampleLinear<float>(
87 cell.uv, generator.maskData);
88 if (density < 0.1f) return;
89 }
90
91 auto &instanceMat = generator.instanceData.emplace_back(Mat4f::identity());
92
93 // apply scaling, randomize between min and max scale
94 auto sizeFactor = badRandom() * cell.density;
95 auto scale = generator.objMinScale + sizeFactor * (generator.objMaxScale - generator.objMinScale);
96 instanceMat.scale(scale);
97
98 // apply random rotation around y axis
99 auto angle = badRandom() * 2.0f * M_PI;
100 Quaternion q(0.0, 0.0, 0.0, 1.0);
101 q.setEuler(angle, 0.0f, 0.0f);
102 instanceMat *= q.calculateMatrix();
103
104 // translate to cell position
105 Vec3f pos = Vec3f(cell.position.x, 0.0f, cell.position.y) + generator.cellWorldOffset;
106 if (generator.objPosVariation > 0.0f) {
107 // TODO: randomize position within cell, but need to re-compute UV then!
108 //pos.x += generator.objPosVariation * (badRandom() - 0.5f) * 2.0f;
109 //pos.z += generator.objPosVariation * (badRandom() - 0.5f) * 2.0f;
110 //cell_uv = ...
111 }
112 instanceMat.x[12] += pos.x;
113 instanceMat.x[13] += pos.y;
114 instanceMat.x[14] += pos.z;
115 // translate to height map position
116 if (generator.heightData) {
117 instanceMat.x[13] += generator.areaMaxHeight * generator.heightMap->sampleLinear<float>(
118 cell.uv, generator.heightData);
119 }
120}

122static void makeInstances(InstancePlaneGenerator &generator,
123 const PlaneCell &rootCell, const PlaneCellWeights &rootWeights) {
124 std::stack<std::pair<PlaneCell, PlaneCellWeights>> stack;
125 stack.emplace(rootCell, rootWeights);
126 auto clampedDensity = clamp(generator.objDensity, 0.1f, generator.objDensity);
127
128 while (!stack.empty()) {
129 auto &pair = stack.top();
130 auto cell = pair.first;
131 auto weights = pair.second;
132 stack.pop();
133
134 // make an instance at the cell center
135 makeInstance(generator, cell);
136
137 // subdivide the cell into four quadrants in case half of the cell is big enough
138 auto subdivideThreshold = generator.objSize * generator.objMaxScale * 2.5f / clampedDensity;
139 if (cell.size.x > subdivideThreshold.x && cell.size.y > subdivideThreshold.y) {
140 auto &bottomLeft = stack.emplace();
141 auto &bottomRight = stack.emplace();
142 auto &topRight = stack.emplace();
143 auto &topLeft = stack.emplace();
144 auto subdividedSize = cell.size * 0.5f;
145 auto halfCellDensity = cell.density * 0.5f;
146 PlaneCell *subdivideCells[4] = {
147 &bottomLeft.first, &bottomRight.first, &topRight.first, &topLeft.first
148 };
149
150 // compute center position of subdivide cells
151 bottomLeft.first.position = cell.position - cell.size * 0.25f;
152 topRight.first.position = cell.position + cell.size * 0.25f;
153 bottomRight.first.position = cell.position + Vec2f(cell.size.x * 0.25f, -cell.size.y * 0.25f);
154 topLeft.first.position = cell.position + Vec2f(-cell.size.x * 0.25f, cell.size.y * 0.25f);
155 // compute density at the center
156 bottomLeft.first.density = halfCellDensity + (weights.bottom + weights.left) * 0.25f;
157 bottomRight.first.density = halfCellDensity + (weights.bottom + weights.right) * 0.25f;
158 topRight.first.density = halfCellDensity + (weights.top + weights.right) * 0.25f;
159 topLeft.first.density = halfCellDensity + (weights.top + weights.left) * 0.25f;
160 // compute uv coordinate, and set the size to half size of parent cell
161 for (int i = 0; i < 4; i++) {
162 auto &subdivideCell = *subdivideCells[i];
163 subdivideCell.size = subdividedSize;
164 }
165 auto uvOffset = cell.size / generator.areaSize;
166 bottomLeft.first.uv = cell.uv - Vec2f(0.25f, 0.25f) * uvOffset;
167 bottomRight.first.uv = cell.uv + Vec2f(0.25f, -0.25f) * uvOffset;
168 topRight.first.uv = cell.uv + Vec2f(0.25f, 0.25f) * uvOffset;
169 topLeft.first.uv = cell.uv - Vec2f(0.25f, -0.25f) * uvOffset;
170
171 // compute weights for subdivide cells
172 bottomLeft.second.left = 0.75f * weights.left + 0.25f * weights.bottom;
173 bottomLeft.second.bottom = 0.75f * weights.bottom + 0.25f * weights.left;
174 bottomLeft.second.top = 0.5f * cell.density + 0.5f * weights.left;
175 bottomLeft.second.right = 0.5f * cell.density + 0.5f * weights.bottom;
176
177 bottomRight.second.right = 0.75f * weights.right + 0.25f * weights.bottom;
178 bottomRight.second.bottom = 0.75f * weights.bottom + 0.25f * weights.right;
179 bottomRight.second.top = 0.5f * cell.density + 0.5f * weights.right;
180 bottomRight.second.left = 0.5f * cell.density + 0.5f * weights.bottom;
181
182 topRight.second.right = 0.75f * weights.right + 0.25f * weights.top;
183 topRight.second.top = 0.75f * weights.top + 0.25f * weights.right;
184 topRight.second.left = 0.5f * cell.density + 0.5f * weights.right;
185 topRight.second.bottom = 0.5f * cell.density + 0.5f * weights.top;
186
187 topLeft.second.left = 0.75f * weights.left + 0.25f * weights.top;
188 topLeft.second.top = 0.75f * weights.top + 0.25f * weights.left;
189 topLeft.second.right = 0.5f * cell.density + 0.5f * weights.left;
190 topLeft.second.bottom = 0.5f * cell.density + 0.5f * weights.top;
191 }
192 }
193}

195static void makeInstances(InstancePlaneGenerator &generator, unsigned int i, unsigned int j) {
196 auto &cell = generator.cells[j * generator.cellCountX + i];
197 if (cell.density < 0.1f) {
198 return;
199 }
200 // read weights of adjacent cells
201 PlaneCellWeights weights;
202 if (i > 0) {
203 weights.left = generator.cells[j * generator.cellCountX + i - 1].density;
204 }
205 if (i < generator.cellCountX - 1) {
206 weights.right = generator.cells[j * generator.cellCountX + i + 1].density;
207 }
208 if (j > 0) {
209 weights.top = generator.cells[(j - 1) * generator.cellCountX + i].density;
210 }
211 if (j < generator.cellCountY - 1) {
212 weights.bottom = generator.cells[(j + 1) * generator.cellCountX + i].density;
213 }
214 makeInstances(generator, cell, weights);
215}

217static GLuint transformMatrixPlane(
218 SceneParser *parser,
219 SceneInputNode &input,
220 const ref_ptr<ShaderInputMat4> &matrixInput,
221 GLuint numInstances) {
222 auto areaSize = input.getValue<Vec2f>("area-size", Vec2f(10.0f));
223 auto areaHalfSize = areaSize * 0.5f;
224
225 InstancePlaneGenerator generator;
226 generator.areaSize = areaSize;
227 generator.areaMaxHeight = input.getValue<GLfloat>("area-max-height", 0.0f);
228 generator.objMinScale = input.getValue<GLfloat>("obj-min-scale", 0.6f);
229 generator.objMaxScale = input.getValue<GLfloat>("obj-max-scale", 1.0f);
230 generator.objPosVariation = input.getValue<GLfloat>("obj-pos-variation", 0.0f);
231 generator.objDensity = input.getValue<GLfloat>("obj-density", 1.0f);
232 generator.ws_cellSize = input.getValue<Vec2f>("cell-size", Vec2f(1.0f));
233 generator.cellHalfSize = Vec2f(generator.ws_cellSize.x, generator.ws_cellSize.y) * 0.5f;
234 generator.cellWorldOffset = input.getValue<Vec3f>("cell-offset", Vec3f(0.0f));
235 if (input.hasAttribute("area-mask-texture")) {
236 generator.maskTexture = parser->getResources()->getTexture2D(parser, input.getValue("area-mask-texture"));
237 if (generator.maskTexture.get()) {
238 generator.maskTexture->ensureTextureData();
239 generator.maskData = generator.maskTexture->textureData();
240 } else {
241 REGEN_WARN("Unable to load mask texture.");
242 }
243 }
244 if (input.hasAttribute("area-height-texture")) {
245 generator.heightMap = parser->getResources()->getTexture2D(parser, input.getValue("area-height-texture"));
246 if (generator.heightMap.get()) {
247 generator.heightMap->ensureTextureData();
248 generator.heightData = generator.heightMap->textureData();
249 } else {
250 REGEN_WARN("Unable to load height map texture.");
251 }
252 }
253 // get object bounds
254 auto meshes = parser->getResources()->getMesh(parser, input.getValue("obj-mesh"));
255 if (meshes.get() && !meshes.get()->empty()) {
256 auto &meshVec = *meshes.get();
257 auto firstMesh = meshVec[0];
258 Bounds<Vec3f> bounds(firstMesh->minPosition(), firstMesh->maxPosition());
259 for (size_t i = 1; i < meshVec.size(); i++) {
260 auto mesh = meshVec[i];
261 bounds.min.setMin(mesh->minPosition());
262 bounds.max.setMax(mesh->maxPosition());
263 }
264 generator.objSize.x = bounds.max.x - bounds.min.x;
265 generator.objSize.y = bounds.max.z - bounds.min.z;
266 }
267
268 // round up to the next cell count
269 generator.cellCountX = std::ceil(areaSize.x / generator.ws_cellSize.x);
270 generator.cellCountY = std::ceil(areaSize.y / generator.ws_cellSize.y);
271 generator.numCells = generator.cellCountX * generator.cellCountY;
272 generator.cells = new PlaneCell[generator.cellCountX * generator.cellCountY];
273 generator.ts_cellSize = Vec2f(
274 generator.ws_cellSize.x / areaSize.x,
275 generator.ws_cellSize.y / areaSize.y);
276
277 // initialize cells
278 Vec2f cellUV = generator.ts_cellSize * 0.5f;
279 for (unsigned int j = 0; j < generator.cellCountY; j++) {
280 for (unsigned int i = 0; i < generator.cellCountX; i++) {
281 auto index = j * generator.cellCountX + i;
282 auto &cell = generator.cells[index];
283 cell.position = Vec2f(static_cast<float>(i), static_cast<float>(j)) *
284 generator.ws_cellSize + generator.cellHalfSize - areaHalfSize;
285 cell.uv = cellUV;
286 cellUV.x += generator.ts_cellSize.x;
287 cell.density = 1.0f;
288 cell.size = generator.ws_cellSize;
289 }
290 cellUV.x = generator.ts_cellSize.x * 0.5f;
291 cellUV.y += generator.ts_cellSize.y;
292 }
293
294 if (generator.maskData) {
295 // compute cell density based on mask texture
296 for (unsigned int i = 0; i < generator.numCells; i++) {
297 auto &cell = generator.cells[i];
298 cell.density = generator.maskTexture->sampleMax<float>(
299 cell.uv,
300 generator.ts_cellSize,
301 generator.maskData);
302 }
303 }
304
305 for (unsigned int j = 0; j < generator.cellCountY; j++) {
306 for (unsigned int i = 0; i < generator.cellCountX; i++) {
307 makeInstances(generator, i, j);
308 }
309 }
310
311 delete[] generator.cells;
312
313 if (generator.instanceData.empty()) {
314 REGEN_WARN("No instances created.");
315 return numInstances;
316 } else {
317 numInstances = generator.instanceData.size();
318 matrixInput->setInstanceData(numInstances, 1, nullptr);
319 auto matrices = matrixInput->mapClientData<Mat4f>(ShaderData::WRITE);
320 // TODO: apply previous transform instead of overwriting
321 for (GLuint i = 0; i < numInstances; i += 1) {
322 matrices.w[i] = generator.instanceData[i];
323 }
324 }
325
326 return numInstances;
327}

329static void transformAnimation(
330 SceneParser *parser,
331 const ref_ptr<SceneInputNode> &child,
332 const ref_ptr<State> &state,
333 const ref_ptr<ModelTransformation> &tf) {
334 auto animType = child->getValue<std::string>("type", "transform");
335
336 if (animType == "boids") {
337 auto boidsAnimation = ref_ptr<BoidsSimulation_CPU>::alloc(tf);
338 boidsAnimation->loadSettings(parser, child);
339 state->attach(boidsAnimation);
340 boidsAnimation->startAnimation();
341 }
342 else {
343 auto transformAnimation = ref_ptr<TransformAnimation>::alloc(tf->get());
344
345 if (child->hasAttribute("mesh-id")) {
346 auto meshID = child->getValue("mesh-id");
347 auto meshIndex = child->getValue<GLuint>("mesh-index", 0u);
348 auto meshVec = parser->getResources()->getMesh(parser, meshID);
349 if (meshVec.get() != nullptr && meshVec->size() > meshIndex) {
350 auto mesh = (*meshVec.get())[meshIndex];
351 transformAnimation->setMesh(mesh);
352 }
353 }
354
355 for (auto &keyFrameNode: child->getChildren("key-frame")) {
356 std::optional<Vec3f> framePos = nullopt;
357 std::optional<Vec3f> frameDir = nullopt;
358 if (keyFrameNode->hasAttribute("position")) {
359 framePos = keyFrameNode->getValue<Vec3f>("position", Vec3f(0.0f));
360 }
361 if (keyFrameNode->hasAttribute("rotation")) {
362 frameDir = keyFrameNode->getValue<Vec3f>("rotation", Vec3f(0.0f));
363 }
364 auto dt = keyFrameNode->getValue<GLdouble>("dt", 1.0);
365 transformAnimation->push_back(framePos, frameDir, dt);
366 }
367
368 transformAnimation->setAnimationName(child->getName());
369 transformAnimation->startAnimation();
370 state->attach(transformAnimation);
371 }
372}

374static void transformMatrix(
375 SceneParser *parser,
376 SceneInputNode &input,
377 const ref_ptr<State> &state,
378 const ref_ptr<ModelTransformation> &tf,
379 GLuint numInstances) {
380 for (auto &child: input.getChildren()) {
381 // TODO: this seems bad, why create this every loop? why at all?
382 list<GLuint> indices = child->getIndexSequence(numInstances);
383
384 if (child->getCategory() == "set") {
385 auto mode = child->getValue("mode");
386 if (mode == "plane") {
387 numInstances = transformMatrixPlane(parser, *child.get(), tf->get(), numInstances);
388 } else {
389 auto matrices = tf->get()->mapClientData<Mat4f>(ShaderData::WRITE);
390 ValueGenerator<Vec3f> generator(child.get(), indices.size(),
391 child->getValue<Vec3f>("value", Vec3f(0.0f)));
392 const auto target = child->getValue<string>("target", "translate");
393
394 for (auto it = indices.begin(); it != indices.end(); ++it) {
395 transformMatrix(target, matrices.w[*it], generator.next());
396 }
397 }
398 } else if (child->getCategory() == "animation") {
399 transformAnimation(parser, child, state, tf);
400 } else {
401 auto matrices = tf->get()->mapClientData<Mat4f>(ShaderData::WRITE);
402 for (auto it = indices.begin(); it != indices.end(); ++it) {
403 transformMatrix(child->getCategory(), matrices.w[*it],
404 child->getValue<Vec3f>("value", Vec3f(0.0f)));
405 }
406 }
407 }

421 void processInput(
422 SceneParser *parser,
423 SceneInputNode &input,
424 const ref_ptr<State> &state) override {
425 ref_ptr<ModelTransformation> transform = parser->getResources()->getTransform(parser, input.getName());
426 if (transform.get() != nullptr) {
427 state->joinStates(transform);
428 return;
429 }
430 ref_ptr<SceneInputNode> transformNode = parser->getRoot()->getFirstChild(REGEN_TRANSFORM_STATE_CATEGORY,
431 input.getName());
432 if (transformNode.get() != nullptr && transformNode.get() != &input) {
433 processInput(parser, *transformNode.get(), state);
434 return;
435 }
436
437 bool isInstanced = input.getValue<bool>("is-instanced", false);
438 auto numInstances = input.getValue<GLuint>("num-instances", 1u);
439 transform = ref_ptr<ModelTransformation>::alloc();
440
441 // Handle instanced model matrix
442 if (isInstanced && numInstances > 1) {
443 transform->get()->setInstanceData(numInstances, 1, nullptr);
444 {
445 auto matrices = transform->get()->mapClientData<Mat4f>(ShaderData::WRITE);
446 for (GLuint i = 0; i < numInstances; i += 1) matrices.w[i] = Mat4f::identity();
447 }
448 transformMatrix(parser, input, state, transform, numInstances);
449 // add data to vbo
450 transform->setInput(transform->get());
451 } else {
452 transformMatrix(parser, input, state, transform, 1u);
453 if (transform->get()->numInstances() > 1) {
454 transform->setInput(transform->get());
455 }
456 }
457
458 state->joinStates(transform);
459 parser->getResources()->putTransform(input.getName(), transform);
460 }
461 };
