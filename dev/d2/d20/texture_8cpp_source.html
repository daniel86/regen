177Bounds<Vec2ui> Texture::getRegion(const Vec2f &texco, const Vec2f &regionTS) const {
178 auto w = static_cast<float>(width());
179 auto h = static_cast<float>(height());
180 auto startX = static_cast<unsigned int>(std::floor(texco.x * w));
181 auto startY = static_cast<unsigned int>(std::floor(texco.y * h));
182 auto endX = static_cast<unsigned int>(std::ceil((texco.x + regionTS.x) * w));
183 auto endY = static_cast<unsigned int>(std::ceil((texco.y + regionTS.y) * h));
184 if (endX >= width()) endX = width() - 1;
185 if (endY >= height()) endY = height() - 1;
186 return {
187 Vec2ui(startX, startY),
188 Vec2ui(endX, endY)};
189}

191unsigned int Texture::texelIndex(const Vec2f &texco) const {
192 auto x = static_cast<unsigned int>(std::round(texco.x * static_cast<float>(width())));
193 auto y = static_cast<unsigned int>(std::round(texco.y * static_cast<float>(height())));
194 // clamp to texture size
195 switch (wrapping_[objectIndex_]->value().x) {
196 case GL_REPEAT:
197 x = x % width();
198 y = y % height();
199 break;
200 case GL_MIRRORED_REPEAT:
201 x = x % (2 * width());
202 y = y % (2 * height());
203 if (x >= width()) x = 2 * width() - x - 1;
204 if (y >= height()) y = 2 * height() - y - 1;
205 break;
206 default: // GL_CLAMP_TO_EDGE:
207 if (x >= width()) x = width() - 1;
208 if (y >= height()) y = height() - 1;
209 break;
210 }
211 return (y * width() + x);
212}

221static std::vector<GLubyte> readTextureData_cfg(LoadingContext &ctx, scene::SceneInputNode &input, GLenum format) {
222 std::vector<GLubyte> data;
223 auto numPixelComponents = glenum::pixelComponents(format);
224 // iterate over all "texel" children
225 for (auto &child: input.getChildren("texel")) {
226 if (!child->hasAttribute("v")) {
227 REGEN_WARN("No 'v' attribute found for texel child '" << child->getName() << "'.");
228 continue;
229 }
230 int texelWidth = child->getValue<int>("width", 1);
231 for (int i = 0; i < texelWidth; ++i) {
232 if (numPixelComponents == 1) {
233 data.push_back(child->getValue<GLuint>("v", 0u));
234 } else if (numPixelComponents == 2) {
235 auto v = child->getValue<Vec2ui>("v", Vec2ui(0u));
236 data.push_back(v.x);
237 data.push_back(v.y);
238 } else if (numPixelComponents == 3) {
239 auto v = child->getValue<Vec3ui>("v", Vec3ui(0u));
240 data.push_back(v.x);
241 data.push_back(v.y);
242 data.push_back(v.z);
243 } else if (numPixelComponents == 4) {
244 auto v = child->getValue<Vec4ui>("v", Vec4ui(0u));
245 data.push_back(v.x);
246 data.push_back(v.y);
247 data.push_back(v.z);
248 data.push_back(v.w);
249 }
250 }
251 }
252 return data;
253}

300ref_ptr<Texture> Texture::load(LoadingContext &ctx, scene::SceneInputNode &input) {
301 ref_ptr<Texture> tex;
302 auto &viewport = ctx.scene()->getViewport();
303
304 if (input.hasAttribute("file")) {
305 auto mipmapFlag = GL_DONT_CARE;
306 auto forcedInternalFormat = glenum::textureInternalFormat(
307 input.getValue<std::string>("forced-internal-format", "NONE"));
308 auto forcedFormat = glenum::textureFormat(
309 input.getValue<std::string>("forced-format", "NONE"));
310 auto forcedSize =
311 input.getValue<Vec3ui>("forced-size", Vec3ui(0u));
312 auto keepData = input.getValue<bool>("keep-data", false);
313 const std::string filePath =
314 resourcePath(input.getValue("file"));
315
316 try {
317 if (input.getValue<bool>("is-cube", false)) {
318 tex = textures::loadCube(
319 filePath,
320 input.getValue<bool>("cube-flip-back", false),
321 mipmapFlag,
322 forcedInternalFormat,
323 forcedFormat,
324 forcedSize);
325 } else if (input.getValue<bool>("is-array", false)) {
326 tex = textures::loadArray(
327 filePath,
328 input.getValue<std::string>("name-pattern", ".*"),
329 mipmapFlag,
330 forcedInternalFormat,
331 forcedFormat,
332 forcedSize);
333 } else if (input.getValue<bool>("is-raw", false)) {
334 tex = textures::loadRAW(
335 filePath,
336 input.getValue<Vec3ui>("raw-size", Vec3ui(256u)),
337 input.getValue<GLuint>("raw-components", 3u),
338 input.getValue<GLuint>("raw-bytes", 4u));
339 } else {
340 tex = textures::load(
341 filePath,
342 mipmapFlag,
343 forcedInternalFormat,
344 forcedFormat,
345 forcedSize,
346 keepData);
347 }
348 }
349 catch (textures::Error &ie) {
350 REGEN_ERROR("Failed to load Texture at " << filePath << ".");
351 }
352 } else if (input.hasAttribute("video")) {
353 const std::string filePath = resourcePath(input.getValue("video"));
354 ref_ptr<VideoTexture> video = ref_ptr<VideoTexture>::alloc();
355 try {
356 video->set_file(filePath);
357 auto filename = filePath.substr(filePath.find_last_of('/') + 1);
358 video->setAnimationName(REGEN_STRING("Video" << filename));
359 video->demuxer()->set_repeat(
360 input.getValue<bool>("repeat", 1));
361 tex = video;
362 video->play();
363 video->startAnimation();
364 }
365 catch (VideoTexture::Error &ve) {
366 REGEN_ERROR("Failed to load Video at " << filePath << ".");
367 }
368 } else if (input.hasAttribute("noise")) {
369 const std::string noiseMode = input.getValue("noise");
370
371 auto sizeMode = input.getValue<std::string>("size-mode", "abs");
372 auto sizeRel = input.getValue<Vec3f>("size", Vec3f(256.0, 256.0, 1.0));
373 auto sizeAbs = getSize(viewport, sizeMode, sizeRel);
374
375 auto randomSeed = input.getValue<GLint>("random-seed", rand());
376 auto isSeamless = input.getValue<bool>("is-seamless", false);
377
378 // TODO: allow configuration of noise parameters
379 if (noiseMode == "cloud") {
380 auto noise = ref_ptr<NoiseTexture2D>::alloc(sizeAbs.x, sizeAbs.y, isSeamless);
381 noise->setNoiseGenerator(NoiseGenerator::preset_clouds(randomSeed));
382 tex = noise;
383 } else if (noiseMode == "wood") {
384 auto noise = ref_ptr<NoiseTexture2D>::alloc(sizeAbs.x, sizeAbs.y, isSeamless);
385 noise->setNoiseGenerator(NoiseGenerator::preset_wood(randomSeed));
386 tex = noise;
387 } else if (noiseMode == "granite") {
388 auto noise = ref_ptr<NoiseTexture2D>::alloc(sizeAbs.x, sizeAbs.y, isSeamless);
389 noise->setNoiseGenerator(NoiseGenerator::preset_granite(randomSeed));
390 tex = noise;
391 } else if (noiseMode == "perlin-2d" || noiseMode == "perlin") {
392 auto noise = ref_ptr<NoiseTexture2D>::alloc(sizeAbs.x, sizeAbs.y, isSeamless);
393 noise->setNoiseGenerator(NoiseGenerator::preset_perlin(randomSeed));
394 tex = noise;
395 } else if (noiseMode == "perlin-3d") {
396 auto noise = ref_ptr<NoiseTexture3D>::alloc(sizeAbs.x, sizeAbs.y, sizeAbs.z, isSeamless);
397 noise->setNoiseGenerator(NoiseGenerator::preset_perlin(randomSeed));
398 tex = noise;
399 }
400 } else if (input.hasAttribute("ramp")) {
401 auto ramp = input.getValue("ramp");
402 if (ramp == "dark-white") {
403 tex = RampTexture::darkWhite();
404 } else if (ramp == "dark-white-skewed") {
405 tex = RampTexture::darkWhiteSkewed();
406 } else if (ramp == "normal") {
407 tex = RampTexture::normal();
408 } else if (ramp == "three-step") {
409 tex = RampTexture::threeStep();
410 } else if (ramp == "four-step") {
411 tex = RampTexture::fourStep();
412 } else if (ramp == "four-step-skewed") {
413 tex = RampTexture::fourStepSkewed();
414 } else if (ramp == "black-white-black") {
415 tex = RampTexture::blackWhiteBlack();
416 } else if (ramp == "stripes") {
417 tex = RampTexture::stripes();
418 } else if (ramp == "stripe") {
419 tex = RampTexture::stripe();
420 } else if (ramp == "rgb") {
421 tex = RampTexture::rgb();
422 } else if (ramp == "inline") {
423 auto format = glenum::textureFormat(
424 input.getValue<std::string>("format", "LUMINANCE"));
425 auto internalFormat = format;
426 if (input.hasAttribute("internal-format")) {
427 internalFormat = glenum::textureInternalFormat(
428 input.getValue<std::string>("internal-format", "LUMINANCE"));
429 }
430 auto data = readTextureData_cfg(ctx, input, format);
431 tex = ref_ptr<RampTexture>::alloc(format, internalFormat, data);
432 } else {
433 REGEN_WARN("Unknown ramp type '" << ramp << "'.");
434 }
435 } else if (input.hasAttribute("spectrum")) {
436 auto spectrum = input.getValue<Vec2d>("spectrum", Vec2d(0.0, 1.0));
437 auto numTexels = input.getValue<GLint>("num-texels", 256u);
438 tex = regen::textures::loadSpectrum(spectrum.x, spectrum.y, numTexels);
439 } else if (input.hasAttribute("type")) {
440 const std::string typeName = input.getValue("type");
441 if (typeName == "bloom") {
442 auto numMips = input.getValue<GLuint>("num-mips", 5u);
443 auto bloomTexture = ref_ptr<BloomTexture>::alloc(numMips);
444 auto inputFBO = ctx.scene()->getResource<FBO>(input.getValue("input-fbo"));
445 if (inputFBO.get() == nullptr) {
446 REGEN_WARN("Unable to find FBO for '" << input.getDescription() << "'.");
447 } else {
448 auto resizer = ref_ptr<TextureResizer>::alloc(bloomTexture, viewport, 1.0, 1.0);
449 ctx.scene()->addEventHandler(Application::RESIZE_EVENT, resizer);
450 tex = bloomTexture;
451 bloomTexture->resize(inputFBO->width(), inputFBO->height());
452 }
453 } else {
454 REGEN_WARN("Unknown texture type '" << typeName << "'.");
455 }
456 } else {
457 auto sizeMode = input.getValue<std::string>("size-mode", "abs");
458 auto sizeRel = input.getValue<Vec3f>("size", Vec3f(256.0, 256.0, 1.0));
459 Vec3i sizeAbs = getSize(viewport, sizeMode, sizeRel);
460
461 auto texCount = input.getValue<GLuint>("count", 1);
462 auto pixelSize = input.getValue<GLuint>("pixel-size", 16);
463 auto pixelComponents = input.getValue<GLuint>("pixel-components", 4);
464 auto pixelType = glenum::pixelType(
465 input.getValue<std::string>("pixel-type", "UNSIGNED_BYTE"));
466 auto textureTarget = glenum::textureTarget(
467 input.getValue<std::string>("target", sizeAbs.z > 1 ? "TEXTURE_3D" : "TEXTURE_2D"));
468
469
470 tex = FBO::createTexture(
471 sizeAbs.x, sizeAbs.y, sizeAbs.z,
472 texCount,
473 textureTarget,
474 glenum::textureFormat(pixelComponents),
475 glenum::textureInternalFormat(pixelType, pixelComponents, pixelSize),
476 pixelType);
477
478 if (input.hasAttribute("size-mode") && sizeMode == "rel") {
479 auto resizer = ref_ptr<TextureResizer>::alloc(tex, viewport, sizeRel.x, sizeRel.y);
480 ctx.scene()->addEventHandler(Application::RESIZE_EVENT, resizer);
481 }
482 }
483
484 if (tex.get() == nullptr) {
485 REGEN_WARN("Failed to create Texture for '" << input.getDescription() << ".");
486 return tex;
487 }
488 tex->set_name(input.getName());
489 configure(tex, input);
490
491 return tex;
492}

494void Texture::configure(ref_ptr<Texture> &tex, scene::SceneInputNode &input) {
495 if (!input.getValue("sampler-type").empty()) {
496 tex->set_samplerType(input.getValue("sampler-type"));
497 }
498 tex->begin(RenderState::get(), 0);
499 {
500 if (!input.getValue("wrapping").empty()) {
501 tex->wrapping().push(glenum::wrappingMode(
502 input.getValue<std::string>("wrapping", "CLAMP_TO_EDGE")));
503 }
504 if (!input.getValue("aniso").empty()) {
505 tex->aniso().push(input.getValue<GLfloat>("aniso", 2.0f));
506 }
507 if (!input.getValue("lod").empty()) {
508 tex->lod().push(input.getValue<Vec2f>("lod", Vec2f(1.0f)));
509 }
510 if (!input.getValue("swizzle-r").empty() ||
511 !input.getValue("swizzle-g").empty() ||
512 !input.getValue("swizzle-b").empty() ||
513 !input.getValue("swizzle-a").empty()) {
514 auto swizzleR = static_cast<int>(glenum::textureSwizzle(
515 input.getValue<std::string>("swizzle-r", "RED")));
516 auto swizzleG = static_cast<int>(glenum::textureSwizzle(
517 input.getValue<std::string>("swizzle-g", "GREEN")));
518 auto swizzleB = static_cast<int>(glenum::textureSwizzle(
519 input.getValue<std::string>("swizzle-b", "BLUE")));
520 auto swizzleA = static_cast<int>(glenum::textureSwizzle(
521 input.getValue<std::string>("swizzle-a", "ALPHA")));
522 tex->swizzle().push(Vec4i(swizzleR, swizzleG, swizzleB, swizzleA));
523 }
524 if (!input.getValue("compare-mode").empty()) {
525 auto function = static_cast<int>(glenum::compareFunction(
526 input.getValue<std::string>("compare-function", "LEQUAL")));
527 auto mode = static_cast<int>(glenum::compareMode(
528 input.getValue<std::string>("compare-mode", "NONE")));
529 tex->compare().push(TextureCompare(mode, function));
530 }
531 if (!input.getValue("max-level").empty()) {
532 tex->maxLevel().push(input.getValue<GLint>("max-level", 1000));
533 }
534
535 if (!input.getValue("min-filter").empty() &&
536 !input.getValue("mag-filter").empty()) {
537 auto min = static_cast<int>(glenum::filterMode(input.getValue("min-filter")));
538 auto mag = static_cast<int>(glenum::filterMode(input.getValue("mag-filter")));
539 tex->filter().push(TextureFilter(min, mag));
540 } else if (!input.getValue("min-filter").empty() ||
541 !input.getValue("mag-filter").empty()) {
542 REGEN_WARN("Minification and magnification filters must be specified both." <<
543 " One missing for '"
544 << input.getDescription()
545 << "'.");
546 }
547 }
548 tex->end(RenderState::get(), 0);
549 GL_ERROR_LOG();
550}
