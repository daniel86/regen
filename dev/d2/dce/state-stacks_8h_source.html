38 template<typename StackType, typename ValueType, typename ApplyFunc>
39 class StateStack {
40 public:
45 StateStack(ApplyFunc apply, ApplyFunc lockedApply)
46 : head_(new Node(zeroValue_)),
47 isEmpty_(GL_TRUE),
48 apply_(apply),
49 applyPtr_(apply),
50 lockedApplyPtr_(lockedApply),
51 lockCounter_(0) {
52 self_ = (StackType *) this;
53 doApply_ = &applyInit;
54 }
55
56 ~StateStack() {
57 Stack<Node *> s;
58 Node *n = head_;
59 while (n->prev) n = n->prev;
60 for (; n != NULL; n = n->next) { s.push(n); }
61 while (!s.isEmpty()) {
62 Node *x = s.top();
63 delete x;
64 s.pop();
65 }
66 }
67
71 const ValueType &value() { return head_->v; }
72
77 void push(const ValueType &v) {
78 doApply_(self_, v);
79
80 if (isEmpty_) {
81 // initial push to the stack or first push after everything
82 // was popped out.
83 head_->v = v;
84 isEmpty_ = GL_FALSE;
85 } else if (head_->next) {
86 // use node created earlier
87 head_ = head_->next;
88 head_->v = v;
89 } else {
90 head_->next = new Node(v, head_);
91 head_ = head_->next;
92 }
93 }
94
98 void pop() {
99 if (head_->prev) {
100 if (head_->v != head_->prev->v) {
101 self_->apply(head_->prev->v);
102 }
103 head_ = head_->prev;
104 } else {
105 isEmpty_ = GL_TRUE;
106 }
107 }
108
112 void lock() {
113 ++lockCounter_;
114 apply_ = lockedApplyPtr_;
115 }
116
120 void unlock() {
121 --lockCounter_;
122 if (lockCounter_ < 1) {
123 apply_ = applyPtr_;
124 }
125 }
126
130 GLboolean isLocked() const {
131 return lockCounter_ > 0;
132 }

355 void push(const ValueType &v) {
356 if (counter_.y > 0) {
357 // an indexed value was pushed before
358 // if the indexed value was pushed before the last global push. only
359 // apply when the value of last global push is not equal
360 if (head_->stamp > lastStampi_) { doApply_(this, v); }
361 // else an indexed value was pushed. always call apply even if some
362 // indices may already contain the right value
363 else { apply_(v); }
364 } else {
365 // no indexed push was done before
366 doApply_(this, v);
367 }
368
369 if (isEmpty_) {
370 // initial push to the stack or first push after everything
371 // was popped out.
372 head_->v = v;
373 isEmpty_ = GL_FALSE;
374 } else if (head_->next) {
375 // use node created earlier
376 head_ = head_->next;
377 head_->v = v;
378 } else {
379 // first time someone pushed so deep
380 head_->next = new Node(v, head_);
381 head_ = head_->next;
382 }
383 head_->stamp = counter_.x;
384
385 // count number of pushes for value stamps
386 counter_.x += 1;
387 }

394 void push(GLuint index, const ValueType &v) {
395 Node *headi = headi_[index];
396
397 if (counter_.x > counter_.y) {
398 GLint lastStampi = (isEmptyi_[index] ? -1 : headi->stamp);
399 // an global value was pushed before
400 // if the global value was pushed before the last indexed push. only
401 // apply when the value of last indexed push is not equal
402 if (lastStampi > head_->stamp) { doApplyi_[index](this, index, v); }
403 // else an global value was pushed. call apply if values not equal
404 else if (v != head_->v) { applyi_(index, v); }
405 } else {
406 // no global push was done before
407 doApplyi_[index](this, index, v);
408 }
409
410 if (isEmptyi_[index]) {
411 // initial push to the stack or first push after everything
412 // was popped out.
413 headi->v = v;
414 isEmptyi_[index] = GL_FALSE;
415 } else if (headi->next) {
416 // use node created earlier
417 headi = headi->next;
418 headi->v = v;
419 headi_[index] = headi;
420 } else {
421 // first time someone pushed so deep
422 headi->next = new Node(v, headi);
423 headi = headi->next;
424 headi_[index] = headi;
425 }
426 headi->stamp = counter_.y;
427 lastStampi_ = counter_.y;
428
429 // count number of pushes for value stamps
430 counter_.x += 1;
431 counter_.y += 1;
432 }

437 void pop() {
438 // apply previously pushed global value
439 if (head_->prev) {
440 if (head_->v != head_->prev->v) {
441 apply_(head_->prev->v);
442 }
443 head_ = head_->prev;
444 } else {
445 isEmpty_ = GL_TRUE;
446 }
447
448 // if there are indexed values pushed we may
449 // have to re-enable them here...
450 if (counter_.y > 0) {
451 // Indexed states only applied with stamp>lastEqStamp
452 // Loop over all indexed stacks and compare stamps of top element.
453 for (GLuint i = 0; i < numIndices_; ++i) {
454 if (isEmptyi_[i]) { continue; }
455 Node *headi = headi_[i];
456
457 if (headi->stamp > head_->stamp && head_->v != headi->v) { applyi_(i, headi->v); }
458 }
459 }
460
461 // count number of pushes for value stamps
462 counter_.x -= 1;
463 }

469 void pop(GLuint index) {
470 Node *headi = headi_[index];
471 GLboolean valueChanged;
472
473 // apply previously pushed indexed value
474 if (headi->prev) {
475 valueChanged = (headi->v != headi->prev->v);
476 headi = headi->prev;
477 headi_[index] = headi;
478 lastStampi_ = headi->stamp;
479 } else {
480 valueChanged = GL_FALSE;
481 isEmptyi_[index] = GL_TRUE;
482 lastStampi_ = -1;
483 }
484
485 GLint lastStampi = (isEmptyi_[index] ? -1 : headi->stamp);
486 // reset to equation with latest stamp
487 if (head_->stamp > lastStampi) {
488 if (headi->v != head_->v) applyi_(index, head_->v);
489 } else if (valueChanged) { applyi_(index, headi->v); }
490
491 // count number of pushes for value stamps
492 counter_.x -= 1;
493 counter_.y -= 1;
494 }
