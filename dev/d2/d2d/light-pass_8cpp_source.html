18LightPass::LightPass(Light::Type type, const std::string &shaderKey)
19 : State(), lightType_(type), shaderKey_(shaderKey) {
20 switch (lightType_) {
21 case Light::DIRECTIONAL:
22 mesh_ = Rectangle::getUnitQuad();
23 break;
24 case Light::SPOT:
25 mesh_ = ConeClosed::getBaseCone();
26 joinStates(ref_ptr<CullFaceState>::alloc(GL_FRONT));
27 // enable depth clamping to avoid faces to be clipped away when out of depth range.
28 joinStates(ref_ptr<ToggleState>::alloc(RenderState::DEPTH_CLAMP, GL_TRUE));
29 break;
30 case Light::POINT:
31 mesh_ = Box::getUnitCube();
32 joinStates(ref_ptr<CullFaceState>::alloc(GL_FRONT));
33 // enable depth clamping to avoid faces to be clipped away when out of depth range.
34 joinStates(ref_ptr<ToggleState>::alloc(RenderState::DEPTH_CLAMP, GL_TRUE));
35 break;
36 }
37 shadowFiltering_ = SHADOW_FILTERING_NONE;
38 numShadowLayer_ = 1;
39
40 shader_ = ref_ptr<ShaderState>::alloc();
41 joinStates(shader_);
42 setShadowFiltering(SHADOW_FILTERING_NONE);
43}

62void LightPass::addLight(
63 const ref_ptr<Light> &l,
64 const ref_ptr<LightCamera> &lightCamera,
65 const ref_ptr<Texture> &shadowTexture,
66 const ref_ptr<Texture> &shadowColorTexture,
67 const std::list<ref_ptr<ShaderInput> > &inputs) {
68 LightPassLight light;
69 lights_.push_back(light);
70
71 auto it = lights_.end();
72 --it;
73 lightIterators_[l.get()] = it;
74
75 it->light = l;
76 it->camera = lightCamera;
77 it->shadow = shadowTexture;
78 it->shadowColor = shadowColorTexture;
79 it->inputs = inputs;
80 if (shadowTexture.get()) {
81 auto *tex3D = dynamic_cast<Texture3D *>(shadowTexture.get());
82 if (tex3D == nullptr) {
83 numShadowLayer_ = 1;
84 } else {
85 numShadowLayer_ = tex3D->depth();
86 }
87 shaderDefine("USE_SHADOW_MAP", "TRUE");
88 if (shadowColorTexture.get()) {
89 shaderDefine("USE_SHADOW_COLOR", "TRUE");
90 }
91 }
92
93 if (shader_->shader().get()) { addLightInput(*it); }
94}

108void LightPass::createShader(const StateConfig &cfg) {
109 StateConfigurer _cfg(cfg);
110 _cfg.addState(this);
111 _cfg.addState(mesh_.get());
112 bool hasInstancedInputs = false;
113 if (!lights_.empty()) {
114 // add first light to shader to set up shader defines and also instance count
115 auto &firstLight = lights_.front();
116 for (auto &in: firstLight.light->inputContainer()->inputs()) {
117 if (in.in_->isUniformBlock()) {
118 auto *block = dynamic_cast<UniformBlock *>(in.in_.get());
119 _cfg.addInput(in.in_->name(), in.in_);
120 for (auto &blockUniform: block->uniforms()) {
121 if (blockUniform.in_->numInstances() > 0) { hasInstancedInputs = true; }
122 }
123 } else {
124 _cfg.addInput(in.in_->name(), in.in_);
125 if (in.in_->numInstances() > 0) { hasInstancedInputs = true; }
126 }
127 }
128 }
129 if (hasInstancedInputs) {
130 _cfg.define("HAS_INSTANCES", "TRUE");
131 }
132 _cfg.define("NUM_SHADOW_LAYER", REGEN_STRING(numShadowLayer_));
133 shader_->createShader(_cfg.cfg(), shaderKey_);
134 mesh_->updateVAO(RenderState::get(), _cfg.cfg(), shader_->shader());
135
136 for (auto &light: lights_) { addLightInput(light); }
137 shadowMapLoc_ = shader_->shader()->uniformLocation("shadowTexture");
138 shadowColorLoc_ = shader_->shader()->uniformLocation("shadowColorTexture");
139}

141void LightPass::addLightInput(LightPassLight &light) {
142 // clear list of uniform loactions
143 light.inputLocations.clear();
144 // add user specified uniforms
145 for (auto jt = light.inputs.begin(); jt != light.inputs.end(); ++jt) {
146 ref_ptr<ShaderInput> &in = *jt;
147 addInputLocation(light, in, in->name());
148 }
149
150 // add shadow uniforms
151 if (light.camera.get()) {
152 addInputLocation(light, light.camera->lightCamera()->far(), "lightFar");
153 addInputLocation(light, light.camera->lightCamera()->near(), "lightNear");
154 addInputLocation(light, light.camera->lightMatrix(), "lightMatrix");
155 }
156 if (light.shadow.get()) {
157 addInputLocation(light, light.shadow->sizeInverse(), "shadowInverseSize");
158 addInputLocation(light, light.shadow->size(), "shadowSize");
159 }
160
161 // add light UBO, and special light type uniforms
162 addInputLocation(light, light.light->lightUBO(), "Light");
163 if (lightType_ == Light::SPOT) {
164 addInputLocation(light, light.light->coneMatrix(), "modelMatrix");
165 }
166}

177void LightPass::enable(RenderState *rs) {
178 State::enable(rs);
179 auto smChannel = rs->reserveTextureChannel();
180 auto smColorChannel = rs->reserveTextureChannel();
181
182 for (auto &l: lights_) {
183 ref_ptr<Texture> shadowTex;
184 // activate shadow map if specified
185 if (l.shadow.get()) {
186 l.shadow->begin(rs, smChannel);
187 glUniform1i(shadowMapLoc_, smChannel);
188 }
189 if (l.shadowColor.get()) {
190 l.shadowColor->begin(rs, smColorChannel);
191 glUniform1i(shadowColorLoc_, smColorChannel);
192 }
193 // enable light pass uniforms
194 for (auto &inputLocation: l.inputLocations) {
195 if (lights_.size() > 1 || inputLocation.uploadStamp != inputLocation.input->stamp()) {
196 inputLocation.input->enableUniform(inputLocation.location);
197 inputLocation.uploadStamp = inputLocation.input->stamp();
198 }
199 }
200
201 mesh_->enable(rs);
202 mesh_->disable(rs);
203
204 if (l.shadow.get()) { l.shadow->end(rs, smChannel); }
205 if (l.shadowColor.get()) { l.shadowColor->end(rs, smColorChannel); }
206 }
207
208 rs->releaseTextureChannel();
209 rs->releaseTextureChannel();
210}

212ref_ptr<LightPass> LightPass::load(LoadingContext &ctx, scene::SceneInputNode &input) {
213 auto scene = ctx.scene();
214
215 if (!input.hasAttribute("shader")) {
216 REGEN_WARN("Missing shader attribute for " << input.getDescription() << ".");
217 return {};
218 }
219 ref_ptr<LightPass> x = ref_ptr<LightPass>::alloc(
220 input.getValue<Light::Type>("type", Light::SPOT),
221 input.getValue("shader"));
222
223 x->setShadowFiltering(input.getValue<ShadowFilterMode>(
224 "shadow-filter", SHADOW_FILTERING_NONE));
225 bool useShadows = false, toggle = true;
226
227 for (auto &n: input.getChildren()) {
228
229 ref_ptr<Light> light = scene->getResource<Light>(n->getName());
230 if (light.get() == nullptr) {
231 continue;
232 }
233 std::list<ref_ptr<ShaderInput> > inputs;
234
235 ref_ptr<Texture> shadowMap;
236 ref_ptr<Texture> shadowColorMap;
237 ref_ptr<LightCamera> shadowCamera;
238 if (n->hasAttribute("shadow-camera")) {
239 shadowCamera = ref_ptr<LightCamera>::dynamicCast(
240 scene->getResource<Camera>(n->getValue("shadow-camera")));
241 if (shadowCamera.get() == nullptr) {
242 REGEN_WARN("Unable to find LightCamera for '" << n->getDescription() << "'.");
243 }
244 }
245 if (n->hasAttribute("shadow-buffer") || n->hasAttribute("shadow-texture")) {
246 shadowMap = TextureState::getTexture(scene, *n.get(),
247 "shadow-texture", "shadow-buffer",
248 "shadow-attachment");
249 if (shadowMap.get() == nullptr) {
250 REGEN_WARN("Unable to find ShadowMap for '" << n->getDescription() << "'.");
251 }
252 }
253 if ((n->hasAttribute("shadow-buffer") && n->hasAttribute("shadow-color-attachment")) ||
254 n->hasAttribute("shadow-color-texture")) {
255 shadowColorMap = TextureState::getTexture(scene, *n.get(),
256 "shadow-color-texture", "shadow-buffer",
257 "shadow-color-attachment");
258 }
259 if (toggle) {
260 useShadows = (shadowMap.get() != nullptr);
261 toggle = false;
262 } else if ((shadowMap.get() != nullptr) != useShadows) {
263 REGEN_WARN("Ignoring " << input.getDescription() << ".");
264 continue;
265 }
266
267 // Each light pass can have a set of ShaderInput's
268 for (auto &m: n->getChildren()) {
269 if (m->getCategory() == "input") {
270 inputs.push_back(scene::ShaderInputProcessor::createShaderInput(scene, *m.get(), x));
271 } else {
272 REGEN_WARN("Unhandled node " << m->getDescription() << ".");
273 }
274 }
275
276 x->addLight(light, shadowCamera, shadowMap, shadowColorMap, inputs);
277 }
278
279 StateConfigurer shaderConfigurer;
280 shaderConfigurer.addNode(ctx.parent().get());
281 shaderConfigurer.addState(x.get());
282 x->createShader(shaderConfigurer.cfg());
283
284 return x;
285}
