27ref_ptr<Font> Font::get(const std::string &f, GLuint size, GLuint dpi) {
28 if (!isFreetypeInitialized_) {
29 if (FT_Init_FreeType(&ftlib_)) { throw Font::Error("FT_Init_FreeType failed."); }
30 isFreetypeInitialized_ = GL_TRUE;
31 }
32 // unique font identifier
33 std::string fontKey = REGEN_STRING(f << size << "_" << dpi);
34 // check for cached font
35 auto result = fonts_.find(fontKey);
36 if (result != fonts_.end()) { return result->second; }
37
38 // create the font
39 ref_ptr<Font> font = ref_ptr<Font>::alloc(f, size, dpi);
40 fonts_[fontKey] = font;
41
42 return font;
43}

61Font::Font(const std::string &fontPath, GLuint size, GLuint dpi)
62 : fontPath_(fontPath),
63 size_(size),
64 dpi_(dpi),
65 lineHeight_(1.0f / .83f) {
66 // The object in which Freetype holds information on a given
67 // font is called a "face".
68 FT_Face face;
69
70 if (!boost::filesystem::exists(fontPath)) {
71 throw Error(REGEN_STRING(
72 "Unable to font file at '" << fontPath << "'."));
73 }
74
75 // This is where we load in the font information from the file.
76 // Of all the places where the code might die, this is the most likely,
77 // as FT_New_Face will die if the font file does not exist or is somehow broken.
78 if (FT_New_Face(ftlib_, fontPath.c_str(), 0, &face)) {
79 throw Error(REGEN_STRING(
80 "FT_New_Face failed. " <<
81 "There is probably a problem with the font file at " << fontPath
82 << "."));
83 }
84 // Freetype measures font size in terms of 1/64ths of pixels.
85 // Thus, to make a font h pixels high, we need to request a size of h*64.
86 // (h << 6 is just a prettier way of writing h*64)
87 FT_Set_Char_Size(face, size << 6, size << 6, dpi, dpi);
88
89 // find the bounding box that can hold any glyph of this font,
90 // we need to find this box because all glyph textures must have the same size
91 // for Texture2DArray.
92 int pixels_x = ::FT_MulFix((face->bbox.xMax - face->bbox.xMin), face->size->metrics.x_scale);
93 int pixels_y = ::FT_MulFix((face->bbox.yMax - face->bbox.yMin), face->size->metrics.y_scale);
94 auto textureWidth = (GLuint) (pixels_x / 64);
95 auto textureHeight = (GLuint) (pixels_y / 64);
96 // make sure texture dimensions are multiple of 2
97 if (textureWidth % 2 != 0) textureWidth += 1;
98 if (textureHeight % 2 != 0) textureHeight += 1;
99
100 // remembers some geometry infos about glyphs
101 faceData_ = new FaceData[NUMBER_OF_GLYPHS];
102
103 // create a array texture for the glyphs
104 arrayTexture_ = ref_ptr<Texture2DArray>::alloc(1);
105 arrayTexture_->set_format(GL_RED);
106 arrayTexture_->set_internalFormat(GL_R8);
107 arrayTexture_->set_pixelType(GL_UNSIGNED_BYTE);
108 arrayTexture_->set_rectangleSize(textureWidth, textureHeight);
109 arrayTexture_->set_depth(NUMBER_OF_GLYPHS);
110 arrayTexture_->begin(RenderState::get());
111 arrayTexture_->wrapping().push(GL_CLAMP_TO_BORDER);
112 arrayTexture_->filter().push(GL_LINEAR);
113 arrayTexture_->swizzle().push(GL_RED);
114 // GL expects 4byte aligned rows
115 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
116 arrayTexture_->texImage();
117 for (unsigned short i = 0; i < NUMBER_OF_GLYPHS; i++) {
118 initGlyph(face, i, textureWidth, textureHeight);
119 }
120 // reset to default
121 glPixelStorei(GL_UNPACK_ALIGNMENT, 4);
122 arrayTexture_->end(RenderState::get());
123
124 FT_Done_Face(face);
125}

136GLubyte *Font::invertPixmapWithAlpha(
137 const FT_Bitmap &bitmap, GLuint width, GLuint height) {
138 const unsigned int arraySize = width * height;
139 auto *inverse = new GLubyte[arraySize];
140 auto *inverse_ptr = inverse;
141
142 memset(inverse, 0x0, sizeof(GLubyte) * arraySize);
143
144 for (auto r = 0u; r < bitmap.rows; ++r) {
145 GLubyte *bitmap_ptr = &bitmap.buffer[bitmap.pitch * r];
146 for (auto p = 0u; p < bitmap.width; ++p) {
147 *inverse_ptr++ = *bitmap_ptr++;
148 }
149 inverse_ptr += (width - bitmap.width);
150 }
151
152 return inverse;
153}

155void Font::initGlyph(FT_Face face, GLushort ch, GLuint textureWidth, GLuint textureHeight) {
156 FaceData &glyphData = faceData_[ch];
157 FT_Glyph glyph;
158 GLubyte *inverted;
159
160 // Load the Glyph for our character.
161 if (FT_Load_Glyph(face, FT_Get_Char_Index(face, ch), FT_LOAD_DEFAULT)) {
162 throw Error("FT_Load_Glyph failed");
163 }
164 // Move the face's glyph into a Glyph object.
165 if (FT_Get_Glyph(face->glyph, &glyph)) {
166 throw Error("FT_Get_Glyph failed");
167 }
168 // Convert the glyph to a bitmap.
169 FT_Glyph_To_Bitmap(&glyph, ft_render_mode_normal, 0, 1);
170
171 auto bitmap_glyph = (FT_BitmapGlyph) glyph;
172 // This reference will make accessing the bitmap easier
173 FT_Bitmap &bitmap = bitmap_glyph->bitmap;
174
175 {
176 auto bitmapWith = (GLfloat) bitmap.width;
177 auto bitmapHeight = (GLfloat) bitmap.rows;
178 GLfloat sizeFactor = 1.0f / (GLfloat) size();
179
180 glyphData.uvX = bitmapWith / ((GLfloat) textureWidth);
181 glyphData.uvY = bitmapHeight / ((GLfloat) textureHeight);
182 glyphData.height = bitmapHeight * sizeFactor;
183 glyphData.width = bitmapWith * sizeFactor;
184 glyphData.advanceX = (face->glyph->advance.x >> 6) * sizeFactor;
185 glyphData.left = bitmap_glyph->left * sizeFactor;
186 glyphData.top = bitmap_glyph->top * sizeFactor;
187 }
188
189 {
190 inverted = invertPixmapWithAlpha(bitmap, textureWidth, textureHeight);
191 arrayTexture_->texSubImage((int) ch, inverted);
192 delete[]inverted;
193 }
194
195 FT_Done_Glyph(glyph);
196}
