51void Shader::preProcess(
52 std::map<GLenum, std::string> &ret,
53 const PreProcessorConfig &cfg,
54 const ref_ptr<PreProcessor> &preProcessor) {
55 // configure shader using macros
56 std::stringstream header;
57 header << "#version " << cfg.version << std::endl;
58 for (const auto &define: cfg.defines) {
59 const std::string &name = define.first;
60 const std::string &value = define.second;
61 if (value == "TRUE") {
62 header << "#define " << name << std::endl;
63 } else if (value == "FALSE") {
64 header << "// #undef " << name << std::endl;
65 } else {
66 header << "#define " << name << " " << value << std::endl;
67 }
68 }
69 header << "#include regen.defines.regen_InstanceID" << std::endl;
70 std::string headerStr = header.str();
71
72 // load the GLSL code.
73 PreProcessorInput preProcessorInput(
74 headerStr,
75 cfg.unprocessed,
76 cfg.externalFunctions,
77 cfg.specifiedInput);
78 ret = preProcessor->processStages(preProcessorInput);
79}

84void Shader::printLog(
85 GLuint shader,
86 GLenum shaderType,
87 const char *shaderCode,
88 GLboolean success) {
89 Logging::LogLevel logLevel;
90 if (shaderCode != nullptr) {
91 std::string shaderName = glenum::glslStageName(shaderType);
92
93 if (success) {
94 logLevel = Logging::INFO;
95 //REGEN_LOG(logLevel, shaderName << " Shader compiled successfully!");
96 } else {
97 logLevel = Logging::ERROR;
98 REGEN_LOG(logLevel, shaderName << " Shader failed to compile!");
99 }
100 } else {
101 if (success) {
102 logLevel = Logging::INFO;
103 //REGEN_LOG(logLevel, "Shader linked successfully.");
104 } else {
105 logLevel = Logging::ERROR;
106 REGEN_LOG(logLevel, "Shader failed to link.");
107 }
108 }
109
110 if (!success && shaderCode != nullptr) {
111 std::vector<std::string> codeLines;
112 boost::split(codeLines, shaderCode, boost::is_any_of("\n"));
113 for (GLuint i = 0; i < codeLines.size(); ++i) {
114 REGEN_LOG(logLevel,
115 std::setw(3) << i << std::setw(0) << " " << codeLines[i]);
116 }
117 }
118
119 int length;
120 if (shaderType == GL_NONE) {
121 glGetProgramiv(shader, GL_INFO_LOG_LENGTH, &length);
122 } else {
123 glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &length);
124 }
125 if (length > 1) {
126 char *log = new char[length];
127 std::string msgPrefix;
128 if (shaderType == GL_NONE) {
129 glGetProgramInfoLog(shader, length, nullptr, log);
130 msgPrefix = "link info: ";
131 } else {
132 glGetShaderInfoLog(shader, length, nullptr, log);
133 msgPrefix = "compile info: ";
134 }
135
136 std::vector<std::string> errorLines;
137 boost::split(errorLines, log, boost::is_any_of("\n"));
138 for (GLuint i = 0; i < errorLines.size(); ++i) {
139 if (!errorLines[i].empty()) REGEN_LOG(logLevel, msgPrefix << errorLines[i]);
140 }
141 delete[]log;
142 }
143}

267GLboolean Shader::compile() {
268 for (auto it = shaderCodes_.begin(); it != shaderCodes_.end(); ++it) {
269 const char *source = it->second.c_str();
270 ref_ptr<GLuint> shaderStage = ref_ptr<GLuint>::alloc();
271 GLuint &stage = *shaderStage.get();
272 stage = glCreateShader(it->first);
273 GLint length = -1;
274 GLint status;
275
276 glShaderSource(stage, 1, &source, &length);
277 glCompileShader(stage);
278
279 glGetShaderiv(stage, GL_COMPILE_STATUS, &status);
280 printLog(stage, it->first, source, status != 0);
281 if (!status) {
282 glDeleteShader(stage);
283 return GL_FALSE;
284 }
285
286 glAttachShader(id(), stage);
287 shaders_[it->first] = shaderStage;
288 }
289
290 return GL_TRUE;
291}

293GLboolean Shader::link() {
294 if (!transformFeedback_.empty()) {
295 std::vector<const char *> validNames(transformFeedback_.size());
296 std::set<std::string> validNames_;
297 int validCounter = 0;
298
299 GLenum stage = feedbackStage_;
300 GLuint i = 0;
301 for (; glenum::glslStages()[i] != stage; ++i) {}
302
303 // find next stage
304 std::string nextStagePrefix = "out";
305 for (GLint j = i + 1; j < glenum::glslStageCount(); ++j) {
306 GLenum nextStage = glenum::glslStages()[j];
307 if (shaders_.count(nextStage) != 0) {
308 nextStagePrefix = glenum::glslStagePrefix(nextStage);
309 break;
310 }
311 }
312
313 for (auto it = transformFeedback_.begin(); it != transformFeedback_.end(); ++it) {
314 std::string name = IOProcessor::getNameWithoutPrefix(*it);
315 if (name == "Position") {
316 name = "gl_" + name;
317 } else {
318 name = nextStagePrefix + "_" + name;
319 }
320 if (validNames_.count(name) > 0) { continue; }
321 auto needle = validNames_.insert(name);
322 validNames[validCounter++] = needle.first->c_str();
323 }
324
325 glTransformFeedbackVaryings(id(),
326 validCounter, validNames.data(), feedbackLayout_);
327 }
328
329 glLinkProgram(id());
330 GLint status;
331 glGetProgramiv(id(), GL_LINK_STATUS, &status);
332 GL_ERROR_LOG();
333 if (status == GL_FALSE) {
334 printLog(id(), GL_NONE, nullptr, false);
335 return GL_FALSE;
336 } else {
337 printLog(id(), GL_NONE, nullptr, true);
338 setupInputLocations();
339 return GL_TRUE;
340 }
341}

360void Shader::setupInputLocations() {
361 GLint count;
362 GLint arraySize;
363 GLenum type;
364 char nameC[320];
365
366 //inputs_.clear();
367 samplerLocations_.clear();
368
369 glGetProgramiv(id(), GL_ACTIVE_UNIFORMS, &count);
370 for (GLint loc_ = 0; loc_ < count; ++loc_) {
371 glGetActiveUniform(id(), loc_, 320, nullptr, &arraySize, &type, nameC);
372 std::string uniformName(nameC);
373 // for arrays..
374 GLint loc = glGetUniformLocation(id(), nameC);
375
376 // remember this uniform location
377 std::string attName(nameC);
378 if (hasPrefix(attName, "gl_")) { continue; }
379 if (boost::ends_with(uniformName, "[0]")) {
380 uniformName = uniformName.substr(0, uniformName.size() - 3);
381 }
382 if (hasPrefix(uniformName, "u_")) {
383 uniformName = truncPrefix(uniformName, "u_");
384 } else if (hasPrefix(attName, "in_")) {
385 uniformName = truncPrefix(uniformName, "in_");
386 }
387 if (hasPrefix(uniformName, "instances_")) {
388 uniformName = truncPrefix(uniformName, "instances_");
389 }
390 uniformLocations_[std::string(nameC)] = loc;
391 uniformLocations_[uniformName] = loc;
392 uniformLocations_[REGEN_STRING("u_" << uniformName)] = loc;
393 uniformLocations_[REGEN_STRING("in_" << uniformName)] = loc;
394
395 // create ShaderInput without data allocated.
396 // still setupInput must be called with this ShaderInput
397 // for the shader to enable it with applyInputs()
398 switch (type) {
399 case GL_FLOAT:
400 case GL_FLOAT_VEC2:
401 case GL_FLOAT_VEC3:
402 case GL_FLOAT_VEC4:
403 case GL_BOOL:
404 case GL_UNSIGNED_INT:
405 case GL_UNSIGNED_INT_ATOMIC_COUNTER:
406 case GL_INT:
407 case GL_BOOL_VEC2:
408 case GL_INT_VEC2:
409 case GL_BOOL_VEC3:
410 case GL_INT_VEC3:
411 case GL_BOOL_VEC4:
412 case GL_INT_VEC4:
413 case GL_FLOAT_MAT2:
414 case GL_FLOAT_MAT3:
415 case GL_FLOAT_MAT4:
416 break;
417
418 case GL_SAMPLER_BUFFER:
419 case GL_SAMPLER_1D:
420 case GL_SAMPLER_2D:
421 case GL_SAMPLER_3D:
422 case GL_SAMPLER_CUBE:
423 case GL_SAMPLER_2D_ARRAY:
424 case GL_SAMPLER_1D_SHADOW:
425 case GL_SAMPLER_2D_SHADOW:
426 case GL_SAMPLER_CUBE_SHADOW:
427 case GL_SAMPLER_2D_ARRAY_SHADOW:
428 samplerLocations_[uniformName] = loc;
429 break;
430
431 default:
432 REGEN_WARN("unknown shader type for '" << uniformName << "'");
433 break;
434
435 }
436 }
437
438 glGetProgramiv(id(), GL_ACTIVE_UNIFORM_BLOCKS, &count);
439 GLint bindingPoint = 0;
440 for (GLint blockIndex = 0; blockIndex < count; ++blockIndex) {
441 // Note: uniforms inside a uniform block do not have individual uniform locations,
442 // but each block has an index within the shader program.
443 glGetActiveUniformBlockName(id(), blockIndex, 320, &arraySize, nameC);
444 std::string blockName(truncPrefix(nameC, "in_"));
445 uniformLocations_[blockName] = bindingPoint;
446 uniformLocations_[REGEN_STRING("u_" << blockName)] = bindingPoint;
447 uniformLocations_[REGEN_STRING("in_" << blockName)] = bindingPoint;
448 // To bind data to a uniform block, we need to bind the uniform buffer object to a "binding point".
449 // We can choose which one but must make sure that the uniform buffer object is bound to the same binding point
450 // when the shader is used.
451 glUniformBlockBinding(id(), blockIndex, bindingPoint);
452 ++bindingPoint;
453 }
454
455 glGetProgramiv(id(), GL_ACTIVE_ATTRIBUTES, &count);
456 for (GLint loc_ = 0; loc_ < count; ++loc_) {
457 glGetActiveAttrib(id(), loc_, 320, nullptr, &arraySize, &type, nameC);
458 GLint loc = glGetAttribLocation(id(), nameC);
459
460 // remember this attribute location
461 std::string attName(nameC);
462 if (hasPrefix(attName, "gl_")) { continue; }
463 if (boost::ends_with(attName, "[0]")) {
464 attName = attName.substr(0, attName.size() - 3);
465 }
466 if (hasPrefix(attName, "a_")) {
467 attName = truncPrefix(attName, "a_");
468 } else if (hasPrefix(attName, "vs_")) {
469 attName = truncPrefix(attName, "vs_");
470 } else if (hasPrefix(attName, "in_")) {
471 attName = truncPrefix(attName, "in_");
472 }
473 uniformLocations_[std::string(nameC)] = loc;
474 attributeLocations_[attName] = loc;
475 attributeLocations_[REGEN_STRING("a_" << attName)] = loc;
476 attributeLocations_[REGEN_STRING("in_" << attName)] = loc;
477 attributeLocations_[REGEN_STRING("vs_" << attName)] = loc;
478 }
479}

481ref_ptr<ShaderInput> Shader::createUniform(const std::string &name) {
482 GLint loc = uniformLocation(name);
483 if (loc == -1) {
484 REGEN_WARN("Is not an active uniform '" << name << "' shader=" << id());
485 return {};
486 }
487
488 GLint arraySize;
489 GLenum type;
490 char nameC[320];
491 glGetActiveUniform(id(), loc, 320, nullptr, &arraySize, &type, nameC);
492
493 switch (type) {
494 case GL_FLOAT:
495 return ref_ptr<ShaderInput1f>::alloc(name, arraySize);
496 case GL_FLOAT_VEC2:
497 return ref_ptr<ShaderInput2f>::alloc(name, arraySize);
498 case GL_FLOAT_VEC3:
499 return ref_ptr<ShaderInput3f>::alloc(name, arraySize);
500 case GL_FLOAT_VEC4:
501 return ref_ptr<ShaderInput4f>::alloc(name, arraySize);
502 case GL_BOOL:
503 case GL_INT:
504 return ref_ptr<ShaderInput1i>::alloc(name, arraySize);
505 case GL_BOOL_VEC2:
506 case GL_INT_VEC2:
507 return ref_ptr<ShaderInput2i>::alloc(name, arraySize);
508 case GL_BOOL_VEC3:
509 case GL_INT_VEC3:
510 return ref_ptr<ShaderInput3i>::alloc(name, arraySize);
511 case GL_BOOL_VEC4:
512 case GL_INT_VEC4:
513 return ref_ptr<ShaderInput4i>::alloc(name, arraySize);
514 case GL_FLOAT_MAT2:
515 return ref_ptr<ShaderInput4f>::alloc(name, arraySize);
516 case GL_FLOAT_MAT3:
517 return ref_ptr<ShaderInputMat3>::alloc(name, arraySize);
518 case GL_FLOAT_MAT4:
519 return ref_ptr<ShaderInputMat4>::alloc(name, arraySize);
520 default:
521 REGEN_WARN("Not a known uniform type for '" << name << "' type=" << type);
522 break;
523 }
524 return {};
525}

527void Shader::setInput(const ref_ptr<ShaderInput> &in, const std::string &name) {
528 std::string inputName = (name.empty() ? in->name() : name);
529 {
530 auto needle = inputNames_.find(inputName);
531 if (needle == inputNames_.end()) {
532 inputs_.emplace_back(in, inputName);
533 auto it = inputs_.end();
534 --it;
535 inputNames_[inputName] = it;
536 } else {
537 *needle->second = NamedShaderInput(in, inputName);
538 }
539 }
540 if (in->isVertexAttribute()) {
541 auto needle = attributeLocations_.find(inputName);
542 if (!in->hasData()) { return; }
543 if (needle != attributeLocations_.end()) {
544 attributes_.emplace_back(in, needle->second);
545 }
546 } else if (!in->isConstant()) {
547 auto needle = uniformLocations_.find(inputName);
548 if (needle != uniformLocations_.end()) {
549 uniforms_.emplace_back(in, needle->second);
550 }
551 }
552}
