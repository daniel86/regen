21ref_ptr<Light> LightResource::createResource(
22 SceneParser *parser, SceneInputNode &input) {
23 auto lightType = input.getValue<Light::Type>("type", Light::SPOT);
24 ref_ptr<Light> light = ref_ptr<Light>::alloc(lightType);
25 light->set_isAttenuated(
26 input.getValue<bool>("is-attenuated", lightType != Light::DIRECTIONAL));
27
28 auto dir = input.getValue<Vec3f>("direction", Vec3f(0.0f, 0.0f, 1.0f));
29 dir.normalize();
30 light->direction()->setVertex(0, dir);
31 light->position()->setVertex(0,
32 input.getValue<Vec3f>("position", Vec3f(0.0f)));
33 light->direction()->setVertex(0,
34 input.getValue<Vec3f>("direction", Vec3f(0.0f, 0.0f, 1.0f)));
35 light->diffuse()->setVertex(0,
36 input.getValue<Vec3f>("diffuse", Vec3f(1.0f)));
37 light->specular()->setVertex(0,
38 input.getValue<Vec3f>("specular", Vec3f(1.0f)));
39 light->radius()->setVertex(0,
40 input.getValue<Vec2f>("radius", Vec2f(50.0f)));
41
42 auto angles = input.getValue<Vec2f>("cone-angles", Vec2f(50.0f, 55.0f));
43 light->set_innerConeAngle(angles.x);
44 light->set_outerConeAngle(angles.y);
45 parser->putState(input.getName(), light);
46
47 // process light node children
48 for (auto &child : input.getChildren()) {
49 if (child->getCategory() == "set") {
50 // set a given light input. The input key is given by the "target" attribute.
51 auto targetName = child->getValue("target");
52 // find the shader input in the light state
53 auto target_opt = light->findShaderInput(targetName);
54 if (!target_opt) {
55 REGEN_WARN("Cannot find light input for set in node " << child->getDescription());
56 continue;
57 }
58 auto setTarget = target_opt.value().in;
59 auto numInstances = std::max(
60 child->getValue<GLuint>("num-instances", 1u),
61 setTarget->numInstances());
62 // allocate memory for the shader input
63 setTarget->setInstanceData(numInstances, 1, nullptr);
64 InputStateProvider::setInput(*child.get(), setTarget.get(), numInstances);
65 }
66 if (child->getCategory() == "animation") {
67 auto animationType = child->getValue("type");
68 if (animationType == "boids") {
69 // let a boid simulation change the light positions
70 auto boidsAnimation = ref_ptr<BoidsSimulation_CPU>::alloc(light->position());
71 boidsAnimation->loadSettings(parser, child);
72 light->attach(boidsAnimation);
73 boidsAnimation->startAnimation();
74 } else {
75 REGEN_WARN("Unknown animation type '" << animationType << "' in node " << child->getDescription());
76 }
77 }
78 }
79
80 return light;
81}
