109const regen::Mat4f Astronomy::moonOrientation(
110 const t_aTime &aTime
111, const float latitude
112, const float longitude) const
113{
114 const t_julianDay t(jd(aTime));
115
116 t_longf l, b;
117 Moon::opticalLibrations(t, l, b);
118
119 const regen::Mat4f libLat = regen::Mat4f::rotationMatrix(-_rad(b), 0, 0);
120 const regen::Mat4f libLon = regen::Mat4f::rotationMatrix(0, _rad(l), 0);
121
122 const float a = _rad(Moon::positionAngleOfAxis(t));
123 const float p = _rad(Moon::parallacticAngle(aTime, latitude, longitude));
124
125 const regen::Mat4f zenith = regen::Mat4f::rotationMatrix(0, 0, p - a);
126
127 // finalOrientationWithLibrations
128 const regen::Mat4f R(libLat * libLon * zenith);
129
130 return R;
131}

134const float Astronomy::earthShineIntensity(
135 const t_aTime &aTime
136, const float latitude
137, const float longitude) const
138{
139 const regen::Vec3f m = moonPosition(aTime, latitude, longitude, false);
140 const regen::Vec3f s = sunPosition(aTime, latitude, longitude, false);
141
142 // ("Multiple Light Scattering" - 1980 - Van de Hulst) and
143 // ("A Physically-Based Night Sky Model" - 2001 - Wann Jensen et al.) -> the 0.19 is the earth full intensity
144
145 const float ep = (_PI - acos(s.dot(-m))) * 0.5;
146 const float Eem = 0.19 * 0.5 * (1.0 - sin(ep) * tan(ep) * log(1.0 / tan(ep * 0.5)));
147
148 return Eem;
149}

152const regen::Mat4f Astronomy::equToHorTransform(
153 const t_aTime &aTime
154, const float latitude
155, const float longitude) const
156{
157 const t_julianDay T(jCenturiesSinceSE(jd(aTime)));
158 const float s = siderealTime(aTime);
159
160 return regen::Mat4f::scaleMatrix(regen::Vec3f(-1, 1, 1))
161 * regen::Mat4f::rotationMatrix( _rad(latitude) - _PI_2, 0, 0)
162 * regen::Mat4f::rotationMatrix(0, 0, -_rad(s + longitude))
163 // precession as suggested in (Jensen et al. 2001)
164 * regen::Mat4f::rotationMatrix(0, 0, 0.01118 * T)
165 * regen::Mat4f::rotationMatrix(-0.00972 * T, 0, 0)
166 * regen::Mat4f::rotationMatrix(0, 0, 0.01118 * T);
167}
