56void Logging::log(Logging::LogLevel level,
57 const string &message,
58 const char *filePath, int line) {
59 if (message.empty()) { return; }
60
61 string fileName(filePath);
62 {
63 size_t pos = fileName.find_last_of("/");
64 if (pos != fileName.npos) {
65 fileName = fileName.substr(pos + 1);
66 }
67 }
68 for (list<Logger *>::iterator it = loggers_[level].begin();
69 it != loggers_[level].end(); ++it) {
70 (*it)->log(message, fileName, line);
71 }
72}

80void Logger::log(const string &message,
81 const string file, int line) {
82 using namespace boost::posix_time;
83 static std::locale locDate(std::cout.getloc(), new time_facet("%d.%m.%Y"));
84 static std::locale locTime(std::cout.getloc(), new time_facet("%H:%M:%S"));
85
86 ostream &os = stream();
87 os.flags(loggerFlags_); // format flags
88 os.precision(loggerPrecision_); // floating-point decimal precision
89 os.width(loggerWidth_); // field width
90
91 boost::posix_time::ptime now = second_clock::universal_time();
92
93 for (unsigned int i = 0; i < format_.size(); ++i) {
94 char &c = format_[i];
95 if (c == '%') {
96 i += 1;
97 switch (format_[i]) {
98 case '%':
99 os << '%';
100 break;
101 case MessageKey:
102 os << message;
103 break;
104 case FileKey:
105 os << file;
106 break;
107 case LineKey:
108 os << line;
109 break;
110 case DateKey: {
111 stringstream dateStream;
112 dateStream.imbue(locDate);
113 dateStream << now;
114 os << dateStream.str();
115 break;
116 }
117 case TimeKey: {
118 stringstream timeStream;
119 timeStream.imbue(locTime);
120 timeStream << now;
121 os << timeStream.str();
122 break;
123 }
124 case LevelKey:
125 switch (level_) {
126 case Logging::INFO:
127 os << "INFO";
128 break;
129 case Logging::WARN:
130 os << "WARN";
131 break;
132 case Logging::ERROR:
133 os << "ERROR";
134 break;
135 case Logging::FATAL:
136 os << "FATAL";
137 break;
138 case Logging::DEBUG:
139 os << "DEBUG";
140 break;
141 case Logging::LAST_LEVEL:
142 break;
143 }
144 break;
145 }
146 } else {
147 os << c;
148 }
149 }
150 os << endl;
151 os.flush();
152
153 os.flags(originalFlags_); // format flags
154 os.precision(originalPrecision_); // floating-point decimal precision
155 os.width(originalWidth_); // field width
156
157}
