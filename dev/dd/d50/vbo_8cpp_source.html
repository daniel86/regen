42 std::istream &operator>>(std::istream &in, VBO::Usage &mode) {
43 std::string val;
44 in >> val;
45 boost::to_upper(val);
46 if (val == "DYNAMIC") mode = VBO::USAGE_DYNAMIC;
47 else if (val == "STATIC") mode = VBO::USAGE_STATIC;
48 else if (val == "STREAM") mode = VBO::USAGE_STREAM;
49 else if (val == "FEEDBACK") mode = VBO::USAGE_FEEDBACK;
50 else if (val == "TEXTURE") mode = VBO::USAGE_TEXTURE;
51 else if (val == "UNIFORM") mode = VBO::USAGE_UNIFORM;
52 else {
53 REGEN_WARN("Unknown VBO usage '" << val << "'. Using default DYNAMIC blending.");
54 mode = VBO::USAGE_DYNAMIC;
55 }
56 return in;
57 }

100void VBO::createMemoryPools() {
101 if (dataPools_ != nullptr) return;
102
103 dataPools_ = new VBOPool[USAGE_LAST];
104 for (int i = 0; i < USAGE_LAST; ++i) { dataPools_[i].set_index(i); }
105
106#ifdef GL_ARB_texture_buffer_range
107 dataPools_[USAGE_TEXTURE].set_alignment(
108 getGLInteger(GL_TEXTURE_BUFFER_OFFSET_ALIGNMENT));
109 // XXX: hack that forces to use separate buffers with each alloc :/
110 // - nothing drawn when minSize set to a few MB
111 // - not allowed to have multiple samplerBuffer's with same buffer object ?
112 // - different format not ok ?
113 dataPools_[USAGE_TEXTURE].set_minSize(1);
114#else
115 dataPools_[USAGE_TEXTURE].set_alignment(16);
116 // no glTexBufferRange available. We have to use
117 // separate buffer objects for each texture buffer.
118 // because it is not possible to attach with offset without glTexBufferRange.
119 dataPools_[USAGE_TEXTURE].set_minSize(1);
120#endif
121}

161GLuint VBO::VBOAllocator::createAllocator(GLuint poolIndex, GLuint size) {
162 RenderState *rs = RenderState::get();
163 // create buffer
164 GLuint ref;
165 glGenBuffers(1, &ref);
166 // and allocate GPU memory
167 switch ((Usage) poolIndex) {
168 case USAGE_DYNAMIC:
169 case USAGE_FEEDBACK:
170 case USAGE_UNIFORM:
171 rs->arrayBuffer().push(ref);
172 glBufferData(GL_ARRAY_BUFFER, size, nullptr, GL_DYNAMIC_DRAW);
173 rs->arrayBuffer().pop();
174 break;
175 case USAGE_STATIC:
176 rs->arrayBuffer().push(ref);
177 glBufferData(GL_ARRAY_BUFFER, size, nullptr, GL_STATIC_DRAW);
178 rs->arrayBuffer().pop();
179 break;
180 case USAGE_STREAM:
181 rs->arrayBuffer().push(ref);
182 glBufferData(GL_ARRAY_BUFFER, size, nullptr, GL_STREAM_DRAW);
183 rs->arrayBuffer().pop();
184 break;
185 case USAGE_TEXTURE:
186 rs->textureBuffer().push(ref);
187 glBufferData(GL_TEXTURE_BUFFER, size, nullptr, GL_DYNAMIC_DRAW);
188 rs->textureBuffer().pop();
189 break;
190 case USAGE_LAST:
191 break;
192 }
193 return ref;
194}

229ref_ptr<VBO::Reference> &VBO::createReference(GLuint numBytes) {
230 VBOPool *memoryPool_ = memoryPool(usage_);
231 // get an allocator
232 VBOPool::Node *allocator = memoryPool_->chooseAllocator(numBytes);
233 if (allocator == nullptr) { allocator = memoryPool_->createAllocator(numBytes); }
234
235 ref_ptr<Reference> ref = ref_ptr<Reference>::alloc();
236 ref->poolReference_ = memoryPool_->alloc(allocator, numBytes);
237 if (ref->poolReference_.allocatorNode == nullptr) { return nullReference(); }
238
239 allocations_.push_front(ref);
240 ref->it_ = allocations_.begin();
241 ref->allocatedSize_ = numBytes;
242 ref->vbo_ = this;
243
244 allocatedSize_ += numBytes;
245 return allocations_.front();
246}

288void VBO::uploadSequential(
289 GLuint startByte,
290 GLuint endByte,
291 const std::list<ref_ptr<ShaderInput> > &attributes,
292 ref_ptr<Reference> &ref) {
293 GLuint bufferSize = endByte - startByte;
294 GLuint currOffset = 0;
295 byte *data = new byte[bufferSize];
296
297 for (auto jt = attributes.begin(); jt != attributes.end(); ++jt) {
298 ShaderInput *att = jt->get();
299 att->set_offset(currOffset + startByte);
300 att->set_stride(att->elementSize());
301 att->set_buffer(ref->bufferID(), ref);
302 // copy data
303 if (att->hasClientData()) {
304 std::memcpy(
305 data + currOffset,
306 att->mapClientDataRaw(ShaderData::READ).r,
307 att->inputSize()
308 );
309 }
310 currOffset += att->inputSize();
311 }
312
313 RenderState::get()->copyWriteBuffer().push(ref->bufferID());
314 set_bufferSubData(GL_COPY_WRITE_BUFFER, startByte, bufferSize, data);
315 RenderState::get()->copyWriteBuffer().pop();
316 delete[]data;
317}

319void VBO::uploadInterleaved(
320 GLuint startByte,
321 GLuint endByte,
322 const std::list<ref_ptr<ShaderInput> > &attributes,
323 ref_ptr<Reference> &ref) {
324 GLuint bufferSize = endByte - startByte;
325 GLuint currOffset = startByte;
326 // get the attribute struct size
327 GLuint attributeVertexSize = 0;
328 GLuint numVertices = attributes.front()->numVertices();
329 byte *data = new byte[bufferSize];
330
331 for (const auto &attribute: attributes) {
332 ShaderInput *att = attribute.get();
333
334 att->set_buffer(ref->bufferID(), ref);
335 if (att->divisor() == 0) {
336 attributeVertexSize += att->elementSize();
337
338 att->set_offset(currOffset);
339 currOffset += att->elementSize();
340 }
341 }
342
343 currOffset = (currOffset - startByte) * numVertices;
344 for (const auto &attribute: attributes) {
345 ShaderInput *att = attribute.get();
346 if (att->divisor() == 0) {
347 att->set_stride(attributeVertexSize);
348 } else {
349 // add instanced attributes to the end of the buffer
350 att->set_stride(att->elementSize());
351 att->set_offset(currOffset + startByte);
352 if (att->hasClientData()) {
353 std::memcpy(
354 data + currOffset,
355 att->mapClientDataRaw(ShaderData::READ).r,
356 att->inputSize()
357 );
358 }
359 currOffset += att->inputSize();
360 }
361 }
362
363 GLuint count = 0;
364 for (GLuint i = 0; i < numVertices; ++i) {
365 for (const auto &attribute: attributes) {
366 ShaderInput *att = attribute.get();
367 if (att->divisor() != 0) { continue; }
368
369 // size of a value for a single vertex in bytes
370 GLuint valueSize = att->valsPerElement() * att->dataTypeBytes() * att->numArrayElements();
371 // copy data
372 if (att->hasClientData()) {
373 std::memcpy(
374 data + count,
375 att->mapClientDataRaw(ShaderData::READ).r + i * valueSize,
376 valueSize
377 );
378 }
379 count += valueSize;
380 }
381 }
382
383 RenderState::get()->copyWriteBuffer().push(ref->bufferID());
384 set_bufferSubData(GL_COPY_WRITE_BUFFER, startByte, bufferSize, data);
385 RenderState::get()->copyWriteBuffer().pop();
386 delete[]data;
387}
