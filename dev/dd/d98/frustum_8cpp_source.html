23void Frustum::setPerspective(double _aspect, double _fov, double _near, double _far) {
24 fov = _fov;
25 aspect = _aspect;
26 near = _near;
27 far = _far;
28 double fovR = _fov / 57.2957795;
29 nearPlaneHalfSize.y = tan(fovR * 0.5) * near;
30 nearPlaneHalfSize.x = nearPlaneHalfSize.y * aspect;
31 farPlaneHalfSize.y = tan(fovR * 0.5) * far;
32 farPlaneHalfSize.x = farPlaneHalfSize.y * aspect;
33}

35void Frustum::setOrtho(double left, double right, double bottom, double top, double _near, double _far) {
36 fov = 0.0;
37 aspect = abs((right - left) / (top - bottom));
38 near = _near;
39 far = _far;
40 nearPlaneHalfSize.x = abs(right-left)*0.5;
41 nearPlaneHalfSize.y = abs(top-bottom)*0.5;
42 farPlaneHalfSize = nearPlaneHalfSize;
43 orthoBounds.min.x = left;
44 orthoBounds.min.y = bottom;
45 orthoBounds.max.x = right;
46 orthoBounds.max.y = top;
47}

94void Frustum::updatePointsPerspective(const Vec3f &pos, const Vec3f &dir) {
95 auto v = dir.cross(Vec3f::up());
96 v.normalize();
97 auto u = v.cross(dir);
98 u.normalize();
99 // near plane points
100 auto nc = pos + dir * near;
101 auto rw = v * nearPlaneHalfSize.x;
102 auto uh = u * nearPlaneHalfSize.y;
103 points[0] = nc - uh + rw;
104 points[1] = nc + uh - rw;
105 points[2] = nc + uh + rw;
106 points[3] = nc - uh - rw;
107 // far plane points
108 auto fc = pos + dir * far;
109 rw = v * farPlaneHalfSize.x;
110 uh = u * farPlaneHalfSize.y;
111 points[4] = fc - uh + rw;
112 points[5] = fc + uh - rw;
113 points[6] = fc + uh + rw;
114 points[7] = fc - uh - rw;
115}

117void Frustum::updatePointsOrthogonal(const Vec3f &pos, const Vec3f &dir) {
118 auto v = dir.cross(Vec3f::up());
119 v.normalize();
120 auto u = v.cross(dir);
121 u.normalize();
122 // do not assume that the frustum is centered at the far/near plane centroids!
123 // could be that left/right/top/bottom are not symmetric
124 auto vl = v * orthoBounds.min.x; // left
125 auto vr = v * orthoBounds.max.x; // right
126 auto ub = u * orthoBounds.min.y; // bottom
127 auto ut = u * orthoBounds.max.y; // top
128 // near plane points
129 auto nc = pos + dir * near;
130 points[0] = nc + vr + ub;
131 points[1] = nc + vl + ut;
132 points[2] = nc + vr + ut;
133 points[3] = nc + vl + ub;
134 // far plane points
135 auto fc = pos + dir * far;
136 points[4] = fc + vr + ub;
137 points[5] = fc + vl + ut;
138 points[6] = fc + vr + ut;
139 points[7] = fc + vl + ub;
140}

230void Frustum::split(double splitWeight, std::vector<Frustum> &frustumSplit) const {
231 const auto &n = near;
232 const auto &f = far;
233 const auto count = frustumSplit.size();
234 auto ratio = f / n;
235 double si, lastn, currf, currn;
236
237 lastn = n;
238 for (GLuint i = 1; i < count; ++i) {
239 si = i / (GLdouble) count;
240
241 // C_i = \lambda * C_i^{log} + (1-\lambda) * C_i^{uni}
242 currn = splitWeight * (n * (pow(ratio, si))) +
243 (1 - splitWeight) * (n + (f - n) * si);
244 currf = currn * 1.005;
245
246 frustumSplit[i - 1].setPerspective(aspect, fov, lastn, currf);
247
248 lastn = currn;
249 }
250 frustumSplit[count - 1].setPerspective(aspect, fov, lastn, f);
251}

253Vec3f Frustum::closestPointOnSurface(const Vec3f &point) const {
254 Vec3f closestPoint;
255 float minDistance = std::numeric_limits<float>::max();
256
257 for (const auto &plane: planes) {
258 Vec3f planePoint = plane.closestPoint(point);
259 float distance = (planePoint - point).length();
260 if (distance < minDistance) {
261 minDistance = distance;
262 closestPoint = planePoint;
263 }
264 }
265
266 return closestPoint;
267}
