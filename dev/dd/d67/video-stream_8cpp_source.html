68void VideoStream::decode(AVPacket *packet) {
69 int ret = avcodec_send_packet(codecCtx_, packet);
70 while (ret >= 0) {
71 ret = avcodec_receive_frame(codecCtx_, currFrame_);
72 if (ret == AVERROR(EAGAIN) || ret == AVERROR_EOF || ret < 0) {
73 // Need more packets or end of file or Error during decoding
74 return;
75 }
76
77 // YUV to RGB conversion
78 AVFrame *rgb = av_frame_alloc();
79 int numBytes = av_image_get_buffer_size(
80 GL_RGB_PIXEL_FORMAT,
81 codecCtx_->width,
82 codecCtx_->height,
83 1);
84 if (numBytes < 1) {
85 av_frame_free(&rgb);
86 return;
87 }
88 auto *buffer = (uint8_t *) av_malloc(numBytes * sizeof(uint8_t));
89 if (!buffer) {
90 av_frame_free(&rgb);
91 return;
92 }
93 av_image_fill_arrays(
94 rgb->data,
95 rgb->linesize,
96 buffer,
97 GL_RGB_PIXEL_FORMAT,
98 codecCtx_->width,
99 codecCtx_->height,
100 1);
101
102 sws_scale(
103 swsCtx_,
104 currFrame_->data,
105 currFrame_->linesize,
106 0,
107 codecCtx_->height,
108 rgb->data,
109 rgb->linesize);
110
111 // Remember timestamp in frame
112 auto dt = new float;
113 *dt = packet->dts * av_q2d(stream_->time_base);
114 rgb->opaque = dt;
115
116 // free package and put the frame in queue of decoded frames
117 av_frame_free(&currFrame_);
118 currFrame_ = av_frame_alloc();
119 pushFrame(rgb, numBytes);
120 }
121 av_packet_unref(packet);
122}
