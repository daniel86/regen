7static inline Vec2f findZRange(const Vec3f &lightDir, const Vec3f *frustumPoints) {
8 Vec2f range;
9 range.x = std::numeric_limits<float>::max();
10 range.y = std::numeric_limits<float>::lowest();
11
12 for (GLint i = 0; i < 8; ++i) {
13 float projection = frustumPoints[i].dot(lightDir);
14 if (projection < range.x) {
15 range.x = projection;
16 }
17 if (projection > range.y) {
18 range.y = projection;
19 }
20 }
21
22 return range;
23}

115bool LightCamera_CSM::updateFrustumSplit() {
116 bool hasChanged = false;
117 // Update near/far values when user camera projection changed
118 auto userProjStamp = userCamera_->projection()->stamp();
119 if (userProjectionStamp_ != userProjStamp) {
120 userProjectionStamp_ = userProjStamp;
121 auto proj = userCamera_->projection()->getVertex(0);
122 // update frustum splits
123 userCamera_->frustum()[0].split(splitWeight_, userCameraFrustum_);
124 // update near/far values
125 auto farValues = far_->mapClientData<GLfloat>(ShaderData::WRITE);
126 auto nearValues = near_->mapClientData<GLfloat>(ShaderData::WRITE);
127 for (unsigned int i = 0; i < numLayer_; ++i) {
128 auto &u_frustum = userCameraFrustum_[i];
129 // frustum_->far() is originally in eye space - tell's us how far we can see.
130 // Here we compute it in camera homogeneous coordinates. Basically, we calculate
131 // proj * (0, 0, far, 1)^t and then normalize to [0; 1]
132 // Note: this is used in shaders for computing z coordinate for shadow map lookup
133 farValues.w[i] = 0.5 * (-u_frustum.far * proj.r(2, 2) + proj.r(3, 2)) / u_frustum.far + 0.5;
134 nearValues.w[i] = 0.5 * (-u_frustum.near * proj.r(2, 2) + proj.r(3, 2)) / u_frustum.near + 0.5;
135 }
136 hasChanged = true;
137 }
138
139 // re-compute user frustum points if needed
140 auto userPosStamp = userCamera_->position()->stamp();
141 auto userDirStamp = userCamera_->direction()->stamp();
142 if (hasChanged || userPosStamp != userPositionStamp_ || userDirStamp != userDirectionStamp_) {
143 userPositionStamp_ = userPosStamp;
144 userDirectionStamp_ = userDirStamp;
145 auto userPos = userCamera_->position()->getVertex(0);
146 auto userDir = userCamera_->direction()->getVertex(0);
147 for (unsigned int i = 0; i < numLayer_; ++i) {
148 userCameraFrustum_[i].update(userPos.r, userDir.r);
149 userFrustumCentroids_[i] = userPos.r + userDir.r *
150 ((userCameraFrustum_[i].far - userCameraFrustum_[i].near) * 0.5f + userCameraFrustum_[i].near);
151 }
152 hasChanged = true;
153 }
154
155 return hasChanged;
156}

209bool LightCamera_CSM::updateLightProjection() {
210 // finally re-compute projection matrices, and frustum of light camera
211 auto viewStamp = view_->stamp();
212 if (viewStamp == viewStamp_) { return false; }
213 viewStamp_ = viewStamp;
214
215 for (unsigned int layerIndex = 0; layerIndex < numLayer_; ++layerIndex) {

217 Bounds<Vec3f> bounds_ls(

221#ifdef CSM_USE_SINGLE_VIEW

228 bounds_ls.min.setMin(point_ls.xyz_());
229 bounds_ls.max.setMax(point_ls.xyz_());
