113void CameraController::updateCameraPosition() {
114 if (attachedToTransform_.get()) {
115 camPos_ = meshEyeOffset_;
116 if (attachedToMesh_.get()) {
117 camPos_ += attachedToMesh_->centerPosition();
118 }
119 camPos_ = (matVal_ ^ Vec4f(camPos_, 1.0)).xyz_();
120 } else {
121 camPos_ = pos_;
122 }
123 if (isThirdPerson()) {
124 meshPos_ = camPos_;
125 rot_.setAxisAngle(Vec3f::up(), horizontalOrientation_ + meshHorizontalOrientation_);
126 Vec3f dir = rot_.rotate(Vec3f::front());
127 rot_.setAxisAngle(dir.cross(Vec3f::up()), verticalOrientation_);
128 camPos_ -= rot_.rotate(dir * meshDistance_);
129 }
130}

132void CameraController::updateCameraOrientation() {
133 if (isThirdPerson()) {
134 camDir_ = meshPos_ - camPos_;
135 camDir_.normalize();
136 } else {
137 rot_.setAxisAngle(Vec3f::up(), horizontalOrientation_ + meshHorizontalOrientation_);
138 camDir_ = rot_.rotate(Vec3f::front());
139 rot_.setAxisAngle(dirSidestep_, verticalOrientation_);
140 camDir_ = rot_.rotate(camDir_);
141 }
142}

144void CameraController::updateModel() {
145 if (attachedToTransform_.get()) {
146 // Simple rotation matrix around up vector (0,1,0)
147 GLfloat cy = cos(horizontalOrientation_), sy = sin(horizontalOrientation_);
148 matVal_.x[0] = cy;
149 matVal_.x[2] = sy;
150 matVal_.x[8] = -sy;
151 matVal_.x[10] = cy;
152 // Translate to camera position
153 matVal_.x[12] -= pos_.x;
154 matVal_.x[13] -= pos_.y;
155 matVal_.x[14] -= pos_.z;
156 pos_ = Vec3f(0.0f);
157 }

168void CameraController::animate(GLdouble dt) {
169 step_ = Vec3f(0.0f);
170 isMoving_ = moveForward_ || moveBackward_ || moveLeft_ || moveRight_;
171 auto orientation = horizontalOrientation_ + meshHorizontalOrientation_;
172 auto hasOrientationChanged = orientation != lastOrientation_;
173
174 if (hasOrientationChanged) {
175 rot_.setAxisAngle(Vec3f::up(), orientation);
176 Vec3f d = rot_.rotate(Vec3f::front());
177
178 dirXZ_ = Vec3f(d.x, 0.0f, d.z);
179 dirXZ_.normalize();
180 dirSidestep_ = dirXZ_.cross(Vec3f::up());
181 }
182
183 if (moveForward_) {
184 stepForward(moveAmount_ * dt);
185 }
186 else if (moveBackward_) {
187 stepBackward(moveAmount_ * dt);
188 }
189 if (moveLeft_) {
190 stepLeft(moveAmount_ * dt);
191 }
192 else if (moveRight_) {
193 stepRight(moveAmount_ * dt);
194 }
195 else if (moveUp_) {
196 stepUp(moveAmount_ * dt * 0.5);
197 }
198 else if (moveDown_) {
199 stepDown(moveAmount_ * dt * 0.5);
200 }
201
202 {
203 // TODO: allow setting hasUpdated_ to false to avoid unnecessary culling computations
204 // but the kinematic controller may change camera position in applyStep below though
205 // e.g. character stands on a moving platform.
206 //if (hasOrientationChanged || isMoving_)
207 applyStep(dt, step_);
208 updateModel();
209 updateCameraPosition();
210 updateCameraOrientation();
211 computeMatrices(camPos_, camDir_);
212 if (attachedToTransform_.get()) {
213 attachedToTransform_->setVertex(0, matVal_);
214 }
215 updateCamera(camPos_, camDir_, dt);
216 }
217 lastOrientation_ = orientation;
218}
