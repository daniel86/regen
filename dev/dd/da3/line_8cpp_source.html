51double Line::GetValue (double p) const
52{
53 assert (m_pModule != NULL);
54
55 double x = (m_x1 - m_x0) * p + m_x0;
56 double y = (m_y1 - m_y0) * p + m_y0;
57 double z = (m_z1 - m_z0) * p + m_z0;
58 double value = m_pModule->GetValue (x, y, z);
59
60 if (m_attenuate) {
61 return p * (1.0 - p) * 4 * value;
62 } else {
63 return value;
64 }
