39void DirectShading::updateDefine(DirectLight &l, GLuint lightIndex) {
40 shaderDefine(
41 REGEN_STRING("LIGHT" << lightIndex << "_ID"),
42 REGEN_STRING(l.id_));
43 shaderDefine(
44 REGEN_LIGHT_NAME("LIGHT_IS_ATTENUATED", l.id_),
45 l.light_->isAttenuated() ? "TRUE" : "FALSE");
46
47 std::string lightType = "UNKNOWN";
48 switch (l.light_->lightType()) {
49 case Light::DIRECTIONAL:
50 lightType = "DIRECTIONAL";
51 break;
52 case Light::POINT:
53 lightType = "POINT";
54 // define POINT_LIGHT_TYPE={CUBE, PARABOLIC} based on type of light camera
55 if (l.camera_.get()) {
56 auto *parabolic = dynamic_cast<ParabolicCamera *>(l.camera_.get());
57 if (parabolic) {
58 shaderDefine(REGEN_LIGHT_NAME("POINT_LIGHT_TYPE", l.id_), "PARABOLIC");
59 } else {
60 shaderDefine(REGEN_LIGHT_NAME("POINT_LIGHT_TYPE", l.id_), "CUBE");
61 }
62 }
63 break;
64 case Light::SPOT:
65 lightType = "SPOT";
66 break;
67 }
68 shaderDefine(REGEN_LIGHT_NAME("LIGHT_TYPE", l.id_), lightType);
69
70 // handle shadow map defined
71 if (l.shadow_.get()) {
72 shaderDefine(REGEN_LIGHT_NAME("USE_SHADOW_MAP", l.id_), "TRUE");
73 shaderDefine(REGEN_LIGHT_NAME("SHADOW_MAP_FILTER", l.id_), shadowFilterMode(l.shadowFilter_));
74
75 if (dynamic_cast<Texture3D *>(l.shadow_.get())) {
76 auto *tex3d = dynamic_cast<Texture3D *>(l.shadow_.get());
77 shaderDefine(REGEN_LIGHT_NAME("NUM_SHADOW_LAYER", l.id_), REGEN_STRING(tex3d->depth()));
78 }
79 if (l.shadowColor_.get()) {
80 shaderDefine(REGEN_LIGHT_NAME("USE_SHADOW_COLOR", l.id_), "TRUE");
81 }
82 } else {
83 shaderDefine(REGEN_LIGHT_NAME("USE_SHADOW_MAP", l.id_), "FALSE");
84 }
85}

92void DirectShading::addLight(
93 const ref_ptr<Light> &light,
94 const ref_ptr<LightCamera> &camera,
95 const ref_ptr<Texture> &shadow,
96 const ref_ptr<Texture> &shadowColor,
97 ShadowFilterMode shadowFilter) {
98 GLuint lightID = ++idCounter_;
99 GLuint lightIndex = lights_.size();
100
101 {
102 DirectLight dl;
103 dl.id_ = lightID;
104 dl.light_ = light;
105 dl.camera_ = camera;
106 dl.shadow_ = shadow;
107 dl.shadowColor_ = shadowColor;
108 dl.shadowFilter_ = shadowFilter;
109 lights_.push_back(dl);
110 }
111 DirectLight &directLight = *lights_.rbegin();
112 // remember the number of lights used
113 shaderDefine("NUM_LIGHTS", REGEN_STRING(lightIndex + 1));
114 updateDefine(directLight, lightIndex);
115
116 // join light shader inputs using a name override
117 {
118 const ShaderInputList &in = light->inputContainer()->inputs();
119 for (auto it = in.begin(); it != in.end(); ++it) {
120 if (it->in_->isUniformBlock()) {
121 // if the input is a uniform block, we add all uniforms to the shader
122 // to avoid name clash.
123 // TODO: find a way to use UBO without the name clash
124 auto *block = dynamic_cast<UniformBlock *>(it->in_.get());
125 for (auto &blockUniform: block->uniforms()) {
126 joinShaderInput(
127 blockUniform.in_,
128 REGEN_LIGHT_NAME(blockUniform.name_, lightID));
129 }
130 } else {
131 joinShaderInput(
132 it->in_,
133 REGEN_LIGHT_NAME(it->in_->name(), lightID));
134 }
135 }
136 }
137
138 if (camera.get()) {
139 joinShaderInput(camera->lightCamera()->far(), REGEN_LIGHT_NAME("lightFar", lightID));
140 joinShaderInput(camera->lightCamera()->near(), REGEN_LIGHT_NAME("lightNear", lightID));
141 joinShaderInput(camera->lightMatrix(), REGEN_LIGHT_NAME("lightMatrix", lightID));
142 }
143 if (shadow.get()) {
144 joinShaderInput(shadow->sizeInverse(), REGEN_LIGHT_NAME("shadowInverseSize", lightID));
145 joinShaderInput(shadow->size(), REGEN_LIGHT_NAME("shadowSize", lightID));
146 directLight.shadowMap_ =
147 ref_ptr<TextureState>::alloc(shadow, REGEN_LIGHT_NAME("shadowTexture", lightID));
148 joinStates(directLight.shadowMap_);
149 if (shadowColor.get()) {
150 directLight.shadowColorMap_ =
151 ref_ptr<TextureState>::alloc(shadow, REGEN_LIGHT_NAME("shadowColorTexture", lightID));
152 joinStates(directLight.shadowColorMap_);
153 }
154 }
155}

157void DirectShading::removeLight(const ref_ptr<Light> &l) {
158 std::list<DirectLight>::iterator it;
159 for (it = lights_.begin(); it != lights_.end(); ++it) {
160 ref_ptr<Light> &x = it->light_;
161 if (x.get() == l.get()) {
162 break;
163 }
164 }
165 if (it == lights_.end()) { return; }
166
167 DirectLight &directLight = *it;
168 {
169 const ShaderInputList &in = l->inputContainer()->inputs();
170 for (const auto &jt: in) { disjoinShaderInput(jt.in_); }
171 }
172 if (directLight.camera_.get()) {
173 disjoinShaderInput(directLight.camera_->lightCamera()->far());
174 disjoinShaderInput(directLight.camera_->lightCamera()->near());
175 disjoinShaderInput(directLight.camera_->lightMatrix());
176 }
177 if (directLight.shadow_.get()) {
178 disjoinShaderInput(directLight.shadow_->sizeInverse());
179 disjoinShaderInput(directLight.shadow_->size());
180 disjoinStates(directLight.shadowMap_);
181 if (directLight.shadowColor_.get()) {
182 disjoinStates(directLight.shadowColorMap_);
183 }
184 }
185 lights_.erase(it);
186
187 GLuint numLights = lights_.size(), lightIndex = 0;
188 // update shader defines
189 shaderDefine("NUM_LIGHTS", REGEN_STRING(numLights));
190 for (auto &light: lights_) {
191 updateDefine(light, lightIndex);
192 ++lightIndex;
193 }
194}

198ref_ptr<DirectShading> DirectShading::load(LoadingContext &ctx, scene::SceneInputNode &input) {
199 auto scene = ctx.scene();
200
201 ref_ptr<scene::SceneInputNode> lightNode = input.getFirstChild("direct-lights");
202 ref_ptr<scene::SceneInputNode> passNode = input.getFirstChild("direct-pass");
203 if (lightNode.get() == nullptr) {
204 REGEN_WARN("Missing direct-lights attribute for " << input.getDescription() << ".");
205 return {};
206 }
207 if (passNode.get() == nullptr) {
208 REGEN_WARN("Missing direct-pass attribute for " << input.getDescription() << ".");
209 return {};
210 }
211
212 ref_ptr<DirectShading> shadingState = ref_ptr<DirectShading>::alloc();
213 ref_ptr<StateNode> shadingNode = ref_ptr<StateNode>::alloc(shadingState);
214 ctx.parent()->addChild(shadingNode);
215
216 if (input.hasAttribute("ambient")) {
217 shadingState->ambientLight()->setVertex(0,
218 input.getValue<Vec3f>("ambient", Vec3f(0.1f)));
219 }
220
221 // load lights
222 for (auto &n: lightNode->getChildren()) {
223 if (n->getCategory() != "light") {
224 REGEN_WARN("No processor registered for '" << n->getDescription() << "'.");
225 continue;
226 }
227
228 ref_ptr<Light> light = scene->getResource<Light>(n->getName());
229 if (light.get() == nullptr) {
230 REGEN_WARN("Unable to find Light for '" << n->getDescription() << "'.");
231 continue;
232 }
233
234 auto shadowFiltering =
235 n->getValue<ShadowFilterMode>("shadow-filter", SHADOW_FILTERING_NONE);
236 ref_ptr<Texture> shadowMap;
237 ref_ptr<Texture> shadowColorMap;
238 ref_ptr<LightCamera> shadowCamera;
239 if (n->hasAttribute("shadow-camera")) {
240 shadowCamera = ref_ptr<LightCamera>::dynamicCast(
241 scene->getResource<Camera>(n->getValue("shadow-camera")));
242 if (shadowCamera.get() == nullptr) {
243 REGEN_WARN("Unable to find LightCamera for '" << n->getDescription() << "'.");
244 }
245 }
246 if (n->hasAttribute("shadow-buffer") || n->hasAttribute("shadow-texture")) {
247 shadowMap = TextureState::getTexture(scene, *n.get(),
248 "shadow-texture", "shadow-buffer",
249 "shadow-attachment");
250 if (shadowMap.get() == nullptr) {
251 REGEN_WARN("Unable to find ShadowMap for '" << n->getDescription() << "'.");
252 }
253 }
254 if (n->hasAttribute("shadow-buffer") || n->hasAttribute("shadow-color-texture")) {
255 shadowColorMap = TextureState::getTexture(scene, *n.get(),
256 "shadow-color-texture", "shadow-buffer",
257 "shadow-color-attachment");
258 }
259 shadingState->addLight(light, shadowCamera, shadowMap, shadowColorMap, shadowFiltering);
260 }
261
262 // parse passNode
263 scene::SceneNodeProcessor().processInput(scene, *passNode.get(), shadingNode);
264
265 return shadingState;
266}
