31 // Override
32 void processInput(
33 SceneParser *parser,
34 SceneInputNode &input,
35 const ref_ptr<StateNode> &parent) override {
36 if (!input.hasAttribute("texture") && !input.hasAttribute("fbo")) {
37 REGEN_WARN("Ignoring " << input.getDescription() << " without input texture.");
38 return;
39 }
40
41 ref_ptr<Texture> tex;
42 if (input.hasAttribute("texture")) {
43 tex = parser->getResources()->getTexture(parser, input.getValue("texture"));
44 if (tex.get() == nullptr) {
45 REGEN_WARN("Unable to find Texture for " << input.getDescription() << ".");
46 return;
47 }
48 } else {
49 ref_ptr<FBO> fbo = parser->getResources()->getFBO(parser, input.getValue("fbo"));
50 if (fbo.get() == nullptr) {
51 REGEN_WARN("Unable to find FBO for " << input.getDescription() << ".");
52 return;
53 }
54 auto attachment = input.getValue<GLuint>("attachment", 0);
55 vector<ref_ptr<Texture> > textures = fbo->colorTextures();
56 if (attachment >= textures.size()) {
57 REGEN_WARN("FBO " << input.getValue("fbo") <<
58 " has less then " << (attachment + 1) << " attachments.");
59 return;
60 }
61 tex = textures[attachment];
62 }
63
64 ref_ptr<FilterSequence> filterSeq = ref_ptr<FilterSequence>::alloc(
65 tex,
66 input.getValue<bool>("bind-input", true));
67 filterSeq->set_format(glenum::textureFormat(
68 input.getValue<string>("format", "NONE")));
69 filterSeq->set_internalFormat(glenum::textureInternalFormat(
70 input.getValue<string>("internal-format", "NONE")));
71 filterSeq->set_pixelType(glenum::pixelType(
72 input.getValue<string>("pixel-type", "NONE")));
73 if (input.hasAttribute("clear-color")) {
74 filterSeq->setClearColor(
75 input.getValue<Vec4f>("clear-color", Vec4f(0.0f)));
76 }
77
78 const list<ref_ptr<SceneInputNode> > &childs = input.getChildren();
79 for (auto it = childs.begin(); it != childs.end(); ++it) {
80 ref_ptr<SceneInputNode> n = *it;
81 if (!n->hasAttribute("shader")) {
82 REGEN_WARN("Ignoring filter '" << n->getDescription() << "' without shader.");
83 continue;
84 }
85 filterSeq->addFilter(ref_ptr<Filter>::alloc(
86 n->getValue("shader"),
87 n->getValue<GLfloat>("scale", 1.0f)));
88 }
89
90 parent->state()->joinStates(filterSeq);
91
92 StateConfigurer shaderConfigurer;
93 shaderConfigurer.addNode(parent.get());
94 filterSeq->createShader(shaderConfigurer.cfg());
95
96 // Make the filter output available to the Texture resource provider.
97 parser->getResources()->putTexture(input.getName(), filterSeq->output());
98 }
99 };
100 }
