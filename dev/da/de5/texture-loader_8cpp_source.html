31static void scaleImage(GLuint w, GLuint h, GLuint d) {
32 GLuint width_ = ilGetInteger(IL_IMAGE_WIDTH);
33 // scale image to desired size
34 if (w > 0 && h > 0) {
35 if (width_ > w) {
36 // use bilinear filtering for down scaling
37 iluImageParameter(ILU_FILTER, ILU_BILINEAR);
38 } else {
39 // use triangle filtering for up scaling
40 iluImageParameter(ILU_FILTER, ILU_SCALE_TRIANGLE);
41 }
42 iluScale(w, h, d);
43 }
44}

46static void convertImage(GLenum format, GLenum type) {
47 GLenum srcFormat = regenImageFormat();
48 GLenum srcType = ilGetInteger(IL_IMAGE_TYPE);
49 GLenum dstFormat = (format == GL_NONE ? srcFormat : format);
50 GLenum dstType = (type == GL_NONE ? srcType : type);
51 if (srcFormat != dstFormat || srcType != dstType) {
52 if (ilConvertImage(dstFormat, dstType) == IL_FALSE) {
53 throw Error("ilConvertImage failed");
54 }
55 }

58GLuint textures::loadImage(const std::string &file) {
59 static GLboolean devilInitialized_ = GL_FALSE;
60 if (!devilInitialized_) {
61 ilInit();
62 devilInitialized_ = GL_TRUE;
63 }
64
65 if (!boost::filesystem::exists(file)) {
66 throw Error(REGEN_STRING(
67 "Unable to open image file at '" << file << "'."));
68 }
69
70 GLuint ilID;
71 ilGenImages(1, &ilID);
72 ilBindImage(ilID);
73 if (ilLoadImage(file.c_str()) == IL_FALSE) {
74 throw Error("ilLoadImage failed");

76
77 REGEN_DEBUG("Texture '" << file << "'" <<
78 " format=" << regenImageFormat() <<
79 " type=" << ilGetInteger(IL_IMAGE_TYPE) <<

81 " channels=" << ilGetInteger(IL_IMAGE_CHANNELS) <<
82 " width=" << ilGetInteger(IL_IMAGE_WIDTH) <<
83 " height=" << ilGetInteger(IL_IMAGE_HEIGHT));

88GLenum textures::regenImageFormat() {
89 GLenum format = ilGetInteger(IL_IMAGE_FORMAT);
90 switch (format) {
91 // handle deprecated formats
92 case GL_LUMINANCE:
93 return GL_RED;
94 case GL_LUMINANCE_ALPHA:
95 return GL_RG;
96 default:
97 return format;
98 }

101void unsetData(GLuint ilID, const ref_ptr<Texture> &tex, bool keepData) {
102 if (keepData) {
103 // not sure how one could tell IL we take ownership of the data, so need to memcpy it.
104 // make sure the data tpe is unsigned byte.
105 // TODO: better to support other types than unsigned byte
106 if (ilConvertImage(ilGetInteger(IL_IMAGE_FORMAT), GL_UNSIGNED_BYTE) == IL_FALSE) {
107 throw std::runtime_error("Failed to decompress image");
108 }
109 auto numBytes =
110 ilGetInteger(IL_IMAGE_WIDTH) *
111 ilGetInteger(IL_IMAGE_HEIGHT) *
112 ilGetInteger(IL_IMAGE_BPP);
113 auto *data = new GLubyte[numBytes];
114 memcpy(data, tex->textureData(), numBytes);
115 tex->set_textureData(data, true);
116 }
117 else {
118 tex->set_textureData(nullptr);
119 }
120 ilDeleteImages(1, &ilID);
121}

123ref_ptr<Texture> textures::load(
124 const std::string &file,
125 GLenum mipmapFlag,
126 GLenum forcedInternalFormat,
127 GLenum forcedFormat,
128 const Vec3ui &forcedSize,
129 bool keepData) {
130 GLuint ilID = loadImage(file);
131 scaleImage(forcedSize.x, forcedSize.y, forcedSize.z);
132 convertImage(forcedFormat, GL_NONE);
133 GLint depth = ilGetInteger(IL_IMAGE_DEPTH);
134
135 ref_ptr<Texture> tex;
136 if (depth > 1) {
137 ref_ptr<Texture3D> tex3D = ref_ptr<Texture3D>::alloc();
138 tex3D->set_depth(depth);
139 tex = tex3D;
140 } else {
141 tex = ref_ptr<Texture2D>::alloc();
142 }
143 tex->set_rectangleSize(ilGetInteger(IL_IMAGE_WIDTH), ilGetInteger(IL_IMAGE_HEIGHT));
144 tex->set_pixelType(ilGetInteger(IL_IMAGE_TYPE));
145 tex->set_format(regenImageFormat());
146 if (forcedInternalFormat == GL_NONE) {
147 forcedInternalFormat = tex->format();
148 }
149 if (forcedInternalFormat == GL_BGRA) {
150 forcedInternalFormat = GL_RGBA;
151 } else if (forcedInternalFormat == GL_BGR) {
152 forcedInternalFormat = GL_RGB;
153 }
154 tex->set_internalFormat(forcedInternalFormat);
155 tex->set_textureData((GLubyte *) ilGetData(), false);
156 tex->begin(RenderState::get());
157 tex->texImage();
158 if (mipmapFlag != GL_NONE) {
159 tex->filter().push(TextureFilter(GL_LINEAR_MIPMAP_LINEAR, GL_LINEAR));
160 tex->setupMipmaps(mipmapFlag);
161 } else {
162 tex->filter().push(GL_LINEAR);
163 }
164 tex->wrapping().push(GL_REPEAT);
165 tex->end(RenderState::get());
166 unsetData(ilID, tex, keepData);
167 GL_ERROR_LOG();
168
169 return tex;
170}

223ref_ptr<Texture2DArray> textures::loadArray(
224 const std::string &textureDirectory,
225 const std::string &textureNamePattern,
226 GLenum mipmapFlag,
227 GLenum forcedInternalFormat,
228 GLenum forcedFormat,
229 const Vec3ui &forcedSize) {
230 GLuint numTextures = 0;
231 std::list<std::string> textureFiles;
232
233 boost::filesystem::path texturedir(textureDirectory);
234 boost::regex pattern(textureNamePattern);
235
236 std::vector<std::string> accumulator;
237 boost::filesystem::directory_iterator it(texturedir), eod;
238 BOOST_FOREACH(const boost::filesystem::path &filePath, std::make_pair(it, eod)) {
239 std::string name = filePath.filename().string();
240 if (boost::regex_match(name, pattern)) {
241 accumulator.push_back(filePath.string());
242 numTextures += 1;
243 }
244 }
245 std::sort(accumulator.begin(), accumulator.end());
246
247 ref_ptr<Texture2DArray> tex = ref_ptr<Texture2DArray>::alloc();
248 tex->set_depth(numTextures);
249 tex->begin(RenderState::get());
250
251 GLint arrayIndex = 0;
252 for (auto jt = accumulator.begin(); jt != accumulator.end(); ++jt) {
253 const std::string &textureFile = *jt;
254 GLuint ilID = loadImage(textureFile);
255 scaleImage(forcedSize.x, forcedSize.y, forcedSize.z);
256 convertImage(forcedFormat, GL_NONE);
257
258 if (arrayIndex == 0) {
259 tex->set_rectangleSize(ilGetInteger(IL_IMAGE_WIDTH), ilGetInteger(IL_IMAGE_HEIGHT));
260 tex->set_pixelType(ilGetInteger(IL_IMAGE_TYPE));
261 tex->set_format(regenImageFormat());
262 tex->set_internalFormat(
263 forcedInternalFormat == GL_NONE ? tex->format() : forcedInternalFormat);
264 tex->set_textureData(nullptr);
265 tex->texImage();
266 }
267
268 tex->texSubImage(arrayIndex, (GLubyte *) ilGetData());
269 ilDeleteImages(1, &ilID);
270 arrayIndex += 1;
271 }
272
273 if (mipmapFlag != GL_NONE) {
274 tex->filter().push(TextureFilter(GL_LINEAR_MIPMAP_LINEAR, GL_LINEAR));
275 tex->setupMipmaps(mipmapFlag);
276 } else {
277 tex->filter().push(GL_LINEAR);
278 }
279 tex->wrapping().push(GL_REPEAT);
280
281 tex->end(RenderState::get());
282 GL_ERROR_LOG();
283
284 return tex;
285}

287ref_ptr<TextureCube> textures::loadCube(
288 const std::string &file,
289 GLboolean flipBackFace,
290 GLenum mipmapFlag,
291 GLenum forcedInternalFormat,
292 GLenum forcedFormat,
293 const Vec3ui &forcedSize) {
294 GLuint ilID = loadImage(file);
295 scaleImage(forcedSize.x, forcedSize.y, forcedSize.z);
296 convertImage(forcedFormat, GL_NONE);
297
298 GLint faceWidth, faceHeight;
299 GLint width = ilGetInteger(IL_IMAGE_WIDTH);
300 GLint height = ilGetInteger(IL_IMAGE_HEIGHT);
301 GLint faces[12];
302 // guess layout
303 if (width > height) {
304 faceWidth = width / 4;
305 faceHeight = height / 3;
306 GLint faces_[12] = {
307 -1, TextureCube::TOP, -1, -1,
308 TextureCube::RIGHT, TextureCube::FRONT, TextureCube::LEFT, TextureCube::BACK,
309 -1, TextureCube::BOTTOM, -1, -1
310 };
311 for (ILint i = 0; i < 12; ++i) faces[i] = faces_[i];
312 } else {
313 faceWidth = width / 3;
314 faceHeight = height / 4;
315 GLint faces_[12] = {
316 -1, TextureCube::TOP, -1,
317 TextureCube::RIGHT, TextureCube::FRONT, TextureCube::LEFT,
318 -1, TextureCube::BOTTOM, -1,
319 -1, TextureCube::BACK, -1
320 };
321 for (ILint i = 0; i < 12; ++i) faces[i] = faces_[i];
322 }
323 const ILint numRows = height / faceHeight;
324 const ILint numCols = width / faceWidth;
325 const ILint bpp = ilGetInteger(IL_IMAGE_BPP);
326 const ILint faceBytes = bpp * faceWidth * faceHeight;
327 const ILint rowBytes = faceBytes * numCols;
328
329 ref_ptr<TextureCube> tex = ref_ptr<TextureCube>::alloc();
330 tex->begin(RenderState::get());
331 tex->set_rectangleSize(faceWidth, faceHeight);
332 tex->set_pixelType(ilGetInteger(IL_IMAGE_TYPE));
333 tex->set_format(regenImageFormat());
334 tex->set_internalFormat(
335 forcedInternalFormat == GL_NONE ? tex->format() : forcedInternalFormat);
336
337 auto *imageData = (GLbyte *) ilGetData();
338 ILint index = 0;
339 for (ILint row = 0; row < numRows; ++row) {
340 GLbyte *colData = imageData;
341 for (ILint col = 0; col < numCols; ++col) {
342 ILint mappedFace = faces[index];
343 if (mappedFace != -1) {
344 tex->set_data((TextureCube::CubeSide) mappedFace, colData);
345 }
346 index += 1;
347 colData += bpp * faceWidth;
348 }
349 imageData += rowBytes;
350 }
351 glPixelStorei(GL_UNPACK_ROW_LENGTH, faceWidth * numCols);
352 tex->cubeTexImage(TextureCube::LEFT);
353 tex->cubeTexImage(TextureCube::RIGHT);
354 tex->cubeTexImage(TextureCube::TOP);
355 tex->cubeTexImage(TextureCube::BOTTOM);
356 tex->cubeTexImage(TextureCube::FRONT);
357 if (flipBackFace) {
358 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
359 auto *flippedFace = new GLbyte[faceBytes];
360 auto *faceData = (GLbyte *) tex->cubeData()[TextureCube::BACK];
361 GLbyte *dst = flippedFace;
362
363 for (ILint row = faceWidth - 1; row >= 0; --row) {
364 GLbyte *rowData = faceData + row * faceWidth * numCols * bpp;
365 for (ILint col = faceHeight - 1; col >= 0; --col) {
366 GLbyte *pixelData = rowData + col * bpp;
367 memcpy(dst, pixelData, bpp);
368 dst += bpp;
369 }
370 }
371
372 tex->set_data(TextureCube::BACK, flippedFace);
373 tex->cubeTexImage(TextureCube::BACK);
374 delete[] flippedFace;
375 } else {
376 tex->cubeTexImage(TextureCube::BACK);
377 glPixelStorei(GL_UNPACK_ROW_LENGTH, 4);
378 }
379 if (mipmapFlag != GL_NONE) {
380 tex->filter().push(TextureFilter(GL_LINEAR_MIPMAP_LINEAR, GL_LINEAR));
381 tex->setupMipmaps(mipmapFlag);
382 } else {
383 tex->filter().push(GL_LINEAR);
384 }
385
386 tex->end(RenderState::get());
387 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
388
389 ilDeleteImages(1, &ilID);
390 GL_ERROR_LOG();
391
392 return tex;
393}

395ref_ptr<Texture> textures::loadRAW(
396 const std::string &path,
397 const Vec3ui &size,
398 GLuint numComponents,
399 GLuint bytesPerComponent) {
400 std::ifstream f(path.c_str(),
401 std::ios::in
402 | std::ios::binary
403 | std::ios::ate // start at end position
404 );
405 if (!f.is_open()) {
406 throw Error(REGEN_STRING(
407 "Unable to open data set file at '" << path << "'."));
408 }
409
410 int numBytes = size.x * size.y * size.z * numComponents;
411 char *pixels = new char[numBytes];
412 f.seekg(0, std::ios::beg);
413 f.read(pixels, numBytes);
414 f.close();
415
416 GLenum format_ = glenum::textureFormat(numComponents);
417 GLenum internalFormat_ = glenum::textureInternalFormat(GL_UNSIGNED_BYTE, numComponents, bytesPerComponent);
418
419 ref_ptr<Texture> tex;
420 if (size.z > 1) {
421 ref_ptr<Texture3D> tex3D = ref_ptr<Texture3D>::alloc();
422 tex3D->set_depth(size.z);
423 tex = tex3D;
424 } else {
425 tex = ref_ptr<Texture2D>::alloc();
426 }
427
428 tex->begin(RenderState::get());
429 tex->set_rectangleSize(size.x, size.y);
430 tex->set_pixelType(GL_UNSIGNED_BYTE);
431 tex->set_format(format_);
432 tex->set_internalFormat(internalFormat_);
433 tex->set_textureData((GLubyte *) pixels);
434 tex->filter().push(GL_LINEAR);
435 tex->wrapping().push(GL_REPEAT);
436 tex->texImage();
437 tex->end(RenderState::get());
438 tex->set_textureData(nullptr);
439
440 delete[] pixels;
441
442 return tex;
443}
