82void Disc::generateLODLevel(const Config &cfg,
83 GLuint lodLevel,
84 GLuint vertexOffset,
85 GLuint indexOffset) {
86 const float angleStep = 2.0f * M_PI / lodLevel;
87
88 auto indices = indices_->mapClientData<GLuint>(ShaderData::WRITE);
89 auto v_pos = pos_->mapClientData<Vec3f>(ShaderData::WRITE);
90 auto v_nor = (cfg.isNormalRequired ?
91 nor_->mapClientData<Vec3f>(ShaderData::WRITE) :
92 ShaderData_rw<Vec3f>::nullData());
93 auto v_tan = (cfg.isTangentRequired ?
94 tan_->mapClientData<Vec4f>(ShaderData::WRITE) :
95 ShaderData_rw<Vec4f>::nullData());
96 auto v_texco = (cfg.texcoMode == TEXCO_MODE_UV ?
97 texco_->mapClientData<Vec2f>(ShaderData::WRITE) :
98 ShaderData_rw<Vec2f>::nullData());
99
100 GLuint vertexIndex = vertexOffset;
101 for (GLuint i = 0; i <= lodLevel; ++i) {
102 float angle = i * angleStep;
103 float cosAngle = cos(angle);
104 float sinAngle = sin(angle);
105
106 Vec3f pos(cfg.discRadius * cosAngle, 0.0f, cfg.discRadius * sinAngle);
107 pos = cfg.posScale * pos;
108 v_pos.w[vertexIndex] = pos;
109
110 if (cfg.isNormalRequired) {
111 v_nor.w[vertexIndex] = Vec3f(0.0f, 1.0f, 0.0f);
112 }
113
114 if (cfg.texcoMode == TEXCO_MODE_UV) {
115 Vec2f texco(pos.x / cfg.discRadius + 0.5f, pos.z / cfg.discRadius + 0.5f);
116 v_texco.w[vertexIndex] = texco * cfg.texcoScale;
117 }
118
119 if (cfg.isTangentRequired) {
120 v_tan.w[vertexIndex] = Vec4f(-sinAngle, 0.0f, cosAngle, 1.0f);
121 }
122
123 ++vertexIndex;
124 }
125
126 // Generate indices
127 GLuint index = indexOffset;
128 for (GLuint i = 0; i < lodLevel; ++i) {
129 indices.w[index++] = vertexOffset + lodLevel;
130 indices.w[index++] = vertexOffset + i + 1;
131 indices.w[index++] = vertexOffset + i;
132 }
133}
