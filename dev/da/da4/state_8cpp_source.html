130std::optional<StateInput> State::findShaderInput(const std::string &name) {
131 StateInput ret;
132
133 auto *inState = dynamic_cast<HasInput *>(this);
134 if (inState != nullptr) {
135 ret.container = inState->inputContainer();
136 auto &l = ret.container->inputs();
137 for (const auto &inNamed : l) {
138 if (name == inNamed.name_ || name == inNamed.in_->name()) {
139 ret.in = inNamed.in_;
140 ret.block = {};
141 return ret;
142 }
143 if (inNamed.in_->isUniformBlock()) {
144 auto block = ref_ptr<UniformBlock>::dynamicCast(inNamed.in_);
145 for (auto &blockUniform : block->uniforms()) {
146 if (name == blockUniform.name_ || name == blockUniform.in_->name()) {
147 ret.block = block;
148 ret.in = blockUniform.in_;
149 return ret;
150 }
151 }
152 }
153 }
154 }
155
156 for (auto &joined : joined_) {
157 auto joinedRet = joined->findShaderInput(name);
158 if (joinedRet.has_value()) {
159 return joinedRet.value();
160 }
161 }
162
163 return std::nullopt;
164}
