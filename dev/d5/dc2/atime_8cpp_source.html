92const s_AstronomicalTime s_AstronomicalTime::fromTimeT(
93 const time_t &time
94, const time_t &utcOffset)
95{
96 // Daylight saving time should not be concidered here -> julian time functions ignore this.
97
98#ifdef __GNUC__
99 struct tm lcl(*localtime(&time));
100#else // __GNUC__
101 struct tm lcl;
102 localtime_s(&lcl, &time);
103#endif // __GNUC__
104
105 time_t mt = mktime(&lcl);
106
107 if(mt == -1)
108 return s_AstronomicalTime();
109
110 return s_AstronomicalTime(
111 static_cast<short>(lcl.tm_year + 1900)
112 , static_cast<short>(lcl.tm_mon + 1)
113 , static_cast<short>(lcl.tm_mday)
114 , static_cast<short>(lcl.tm_hour)
115 , static_cast<short>(lcl.tm_min)
116 , static_cast<short>(lcl.tm_sec)
117 , static_cast<short>(utcOffset));
118}

127const time_t s_AstronomicalTime::toTime_t() const
128{
129 time_t t = 0;
130
131#ifdef __GNUC__
132 struct tm lcl(*localtime(&t));
133#else // __GNUC__
134 struct tm lcl;
135 localtime_s(&lcl, &t);
136#endif // __GNUC__
137
138 lcl.tm_year = year - 1900;
139 lcl.tm_mon = month - 1;
140 lcl.tm_mday = day;
141
142 lcl.tm_hour = hour;
143 lcl.tm_min = minute;
144 lcl.tm_sec = second;
145
146 lcl.tm_isdst = 0;
147
148 t = mktime(&lcl);
149
150 return t;
151}
