54void BloomPass::downsample(RenderState *rs) {
55 // Downsampling of the scene color buffer into a mip chain.
56 auto nextInputTexture = inputTexture_.get();
57
58 downsampleState_->enable(rs);
59 glUniform2f(inverseInputSizeLocDS_,
60 static_cast<GLfloat>(1.0 / nextInputTexture->width()),
61 static_cast<GLfloat>(1.0 / nextInputTexture->height()));
62
63 for (auto &mip: bloomTexture_->mips()) {
64 glFramebufferTexture(GL_DRAW_FRAMEBUFFER,
65 GL_COLOR_ATTACHMENT0,
66 mip.texture->id(), 0);
67 glUniform2f(inverseViewportLocDS_, mip.sizeInverse.x, mip.sizeInverse.y);
68
69 // render a quad that fills the selected mip level
70 nextInputTexture->begin(rs, 0);
71 rs->viewport().push(mip.glViewport);
72 fullscreenMesh_->enable(rs);
73 fullscreenMesh_->disable(rs);
74 rs->viewport().pop();
75 nextInputTexture->end(rs, 0);
76
77 nextInputTexture = mip.texture;
78 glUniform2f(inverseInputSizeLocDS_, mip.sizeInverse.x, mip.sizeInverse.y);
79 }
80 downsampleState_->disable(rs);
81}

83void BloomPass::upsample(RenderState *rs) {
84 auto &mips = bloomTexture_->mips();
85 upsampleState_->enable(rs);
86 for (auto i = mips.size() - 1; i > 0; i--) {
87 // upsample mip level i into mip level i-1
88 auto &mip = mips[i];
89 auto &nextMip = mips[i - 1];
90
91 glFramebufferTexture(GL_DRAW_FRAMEBUFFER,
92 GL_COLOR_ATTACHMENT0,
93 nextMip.texture->id(), 0);
94 glUniform2f(inverseViewportLocUS_,
95 nextMip.sizeInverse.x, nextMip.sizeInverse.y);
96
97 // set next mip texture as render target, and render a quad that fills it
98 rs->viewport().push(nextMip.glViewport);
99 mip.texture->begin(rs, 0);
100 fullscreenMesh_->enable(rs);
101 fullscreenMesh_->disable(rs);
102 mip.texture->end(rs, 0);
103 rs->viewport().pop();
104 }
105 upsampleState_->disable(rs);

145 REGEN_WARN("Unable to find Texture for " << input.getDescription() << ".");

165 ctx.parent()->addChild(bloomPass);
