110void ReflectionCamera::updateReflection() {
111 if (isHidden() || !isReflectorValid_) {
112 return;
113 }
114
115 GLboolean reflectorChanged = GL_FALSE;
116 if (hasMesh_) {
117 if (transform_.get() != nullptr && transform_->stamp() != transformStamp_) {
118 reflectorChanged = GL_TRUE;
119 transformStamp_ = transform_->stamp();
120 }
121 if (nor_->stamp() != norStamp_) {
122 reflectorChanged = GL_TRUE;
123 norStamp_ = nor_->stamp();
124 }
125 if (pos_->stamp() != posStamp_) {
126 reflectorChanged = GL_TRUE;
127 posStamp_ = pos_->stamp();
128 }
129 // Compute plane parameters...
130 if (reflectorChanged) {
131 if (!pos_->hasClientData()) pos_->readServerData();
132 if (!nor_->hasClientData()) nor_->readServerData();
133 posWorld_ = pos_->mapClientData<Vec3f>(ShaderData::READ).r[vertexIndex_];
134 norWorld_ = nor_->mapClientData<Vec3f>(ShaderData::READ).r[vertexIndex_];
135
136 if (transform_.get() != nullptr) {
137 if (!transform_->hasClientData()) transform_->readServerData();
138 auto transform = transform_->mapClientData<Mat4f>(ShaderData::READ);
139 posWorld_ = (transform.r[0] ^ Vec4f(posWorld_, 1.0)).xyz_();
140 norWorld_ = (transform.r[0] ^ Vec4f(norWorld_, 0.0)).xyz_();
141 norWorld_.normalize();
142 }
143 }
144 }
145
146 // Switch normal if viewer is behind reflector.
147 GLboolean isFront = norWorld_.dot(
148 userCamera_->position()->getVertex(0).r - posWorld_) > 0.0;
149 if (isFront != isFront_) {
150 isFront_ = isFront;
151 reflectorChanged = GL_TRUE;
152 }
153 // Skip back faces
154 if (!isFront && !hasBackFace_) return;
155
156 // Compute reflection matrix...
157 if (reflectorChanged) {
158 if (isFront_) {
159 clipPlane_->setVertex(0, Vec4f(
160 norWorld_.x, norWorld_.y, norWorld_.z,
161 norWorld_.dot(posWorld_)));
162 reflectionMatrix_ = Mat4f::reflectionMatrix(posWorld_, norWorld_);
163 } else {
164 // flip reflector normal
165 Vec3f n = -norWorld_;
166 clipPlane_->setVertex(0, Vec4f(n.x, n.y, n.z, n.dot(posWorld_)));
167 reflectionMatrix_ = Mat4f::reflectionMatrix(posWorld_, n);
168 }
169 }
170
171 // Compute reflection camera direction
172 if (reflectorChanged || userCamera_->direction()->stamp() != camDirStamp_) {
173 camDirStamp_ = userCamera_->direction()->stamp();
174 Vec3f dir = reflectionMatrix_.rotateVector(userCamera_->direction()->getVertex(0).r);
175 dir.normalize();
176 direction_->setVertex(0, dir);
177
178 reflectorChanged = GL_TRUE;
179 }
180 // Compute reflection camera position
181 if (reflectorChanged || userCamera_->position()->stamp() != camPosStamp_) {
182 camPosStamp_ = userCamera_->position()->stamp();
183 Vec3f reflected = reflectionMatrix_.transformVector(
184 userCamera_->position()->getVertex(0).r);
185 position_->setVertex(0, reflected);
186
187 reflectorChanged = GL_TRUE;
188 }
189
190 // Compute view matrix
191 if (reflectorChanged) {
192 updateView();
193 cameraChanged_ = GL_TRUE;
194 }
195
196 // Compute projection matrix
197 if (userCamera_->projection()->stamp() != projStamp_) {
198 projStamp_ = userCamera_->projection()->stamp();
199 proj_->setUniformData(
200 userCamera_->projection()->getVertex(0).r);
201 projInv_->setUniformData(
202 userCamera_->projectionInverse()->getVertex(0).r);
203 cameraChanged_ = GL_TRUE;
204 }
205
206 // Compute view-projection matrix
207 if (cameraChanged_) {
208 updateViewProjection(0u,0u);
209 cameraChanged_ = GL_FALSE;
210 }
