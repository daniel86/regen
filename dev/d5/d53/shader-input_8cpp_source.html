145const byte *ShaderInput::readLock(int dataSlot) const {
146 auto &slotLock = slotLocks_[dataSlot];
147 // First get a unique lock on the slot, such that we can safely write its members.
148 std::unique_lock<std::mutex> lk(slotLock.lock);
149 while (slotLock.waitingWriters != 0) {
150 // in case the slot is write-locked, wait until all writers have finished.
151 slotLock.readerQ.wait(lk);
152 }
153 // here we know there are no waiting nor active writers, increment active readers and return the data,
154 // it can be used safely by the caller until the lock is released.
155 ++slotLock.activeReaders;
156 byte *data = dataSlots_[dataSlot];
157 lk.unlock();
158 return data;
159}

161const byte *ShaderInput::readLockTry(int dataSlot) const {
162 auto &slotLock = slotLocks_[dataSlot];
163 // First get a unique lock on the slot, such that we can safely write its members.
164 std::unique_lock<std::mutex> lk(slotLock.lock, std::try_to_lock);
165 if (!lk || slotLock.waitingWriters != 0) {
166 return nullptr;
167 }
168 // here we know there are no waiting nor active writers, increment active readers and return the data,
169 // it can be used safely by the caller until the lock is released.
170 ++slotLock.activeReaders;
171 const byte *data = dataSlots_[dataSlot];
172 lk.unlock();
173 return data;
174}

176byte *ShaderInput::writeLock(int dataSlot) const {
177 auto &slotLock = slotLocks_[dataSlot];
178 // First get a unique lock on the slot, such that we can safely write its members.
179 std::unique_lock<std::mutex> lk(slotLock.lock);
180 // increment waiting writers and wait until there are no active readers or writers.
181 ++slotLock.waitingWriters;
182 while (slotLock.activeReaders != 0 || slotLock.activeWriters != 0) {
183 slotLock.writerQ.wait(lk);
184 }
185 // also increment active writers counter
186 ++slotLock.activeWriters;
187 byte *data = dataSlots_[dataSlot];
188 lk.unlock();
189 return data;
190}

192byte *ShaderInput::writeLockTry(int dataSlot) const {
193 auto &slotLock = slotLocks_[dataSlot];
194 // First get a unique lock on the slot, such that we can safely write its members.
195 std::unique_lock<std::mutex> lk(slotLock.lock, std::try_to_lock);
196 if (!lk || slotLock.activeReaders != 0 || slotLock.activeWriters != 0) {
197 return nullptr;
198 }
199 ++slotLock.waitingWriters;
200 ++slotLock.activeWriters;
201 byte *data = dataSlots_[dataSlot];
202 lk.unlock();
203 return data;
204}

206void ShaderInput::readUnlock(int dataSlot) const {
207 auto &slotLock = slotLocks_[dataSlot];
208 // decrement active readers
209 std::unique_lock<std::mutex> lk(slotLock.lock);
210 --slotLock.activeReaders;
211 lk.unlock();
212 // notify any waiting writers
213 slotLock.writerQ.notify_one();
214}

216void ShaderInput::writeUnlock(int dataSlot, bool hasDataChanged) const {
217 auto &slotLock = slotLocks_[dataSlot];
218 std::unique_lock<std::mutex> lk(slotLock.lock);
219 --slotLock.waitingWriters;
220 --slotLock.activeWriters;
221 // TODO: if the data is small, then we could do a check here with memcmp to avoid incrementing the stamp, and
222 // switching the data slot. This could be more efficient in case the data is not changed.
223 //if (numInstances > 1 || std::memcmp(data_, data, inputSize_) != 0) { }
224 //if (std::memcmp(data_, data, inputSize_) != 0) { }
225 if (hasDataChanged) {
226 // increment the data stamp, and remember the last slot that was written to.
227 // consecutive reads will be done from this slot, next write will be done to the other slot.
228 // If the write operation did not change the data, the stamp is not incremented,
229 // and the last slot is not updated.
230 dataStamp_.fetch_add(1, std::memory_order_relaxed);
231 lastDataSlot_.exchange(dataSlot, std::memory_order_relaxed);
232 if (hasServerData()) {
233 requiresReUpload_ = true;
234 }
235 }
236 if (slotLock.waitingWriters > 0) {
237 // first notify the next writer if any
238 slotLock.writerQ.notify_one();
239 } else {
240 // else notify all waiting readers
241 slotLock.readerQ.notify_all();
242 }
243 lk.unlock();
244}

256void ShaderInput::allocateSecondSlot() const {
257 writeLock(1);
258 auto data_r = readLock(0);
259 // need to lock the slot 1 for the memcpy below, to avoid that any mapping happens in between.
260 std::unique_lock<std::mutex> lk(slotLocks_[1].lock);
261 if (dataSlots_[1] == nullptr) {
262 auto data_w = new byte[inputSize_];
263 std::memcpy(data_w, data_r, inputSize_);
264 dataSlots_[1] = data_w;
265 }
266 lk.unlock();
267 readUnlock(0);
268}

270MappedData ShaderInput::mapClientData(int mapMode) const {
271 int r_index = lastDataSlot();
272
273 if ((mapMode & ShaderData::WRITE) != 0) {
274 // NOTE: assuming we can only have two slots, which makes sense IMO, then
275 // if read slot=0, then write slot=1, else write slot=0
276 int w_index = (int) (r_index == 0);
277 byte *data_w;
278 // index mapping should e avoided! It might require to copy one slot to the other.
279 // NOTE: no index mapping needed if there is only one vertex/array element
280 bool isPartialWrite = ((mapMode & ShaderData::INDEX) != 0 && inputSize_ > (dataTypeBytes_ * valsPerElement_));
281
282 // ShaderInput initially has only one slot, the second is allocated on demand in case
283 // multiple threads are concurrently reading/writing the data.
284 // here we keep writing to the active slot as long as no one has to wait,
285 // but as soon as there is waiting time we allocate the second slot and copy the data
286 // to avoid waiting in the future.
287 if (!hasTwoSlots()) {
288 data_w = writeLockTry(r_index);
289 if (data_w) {
290 // got it!
291 w_index = r_index;
292 r_index = lastDataSlot();
293 } else {
294 // read slot is locked, and second slot was not allocated yet.
295 // create it now, and copy the data to it.
296 // note: allocateSecondSlot also acquires a lock in slot 1
297 allocateSecondSlot();
298 data_w = dataSlots_[1];
299 w_index = 1;
300 r_index = w_index;
301 }
302 } else {
303 // for now, if there are two slots, we always write to the other slot.
304 // NOTE: only in case of partial update other strategies might be better (in some cases)
305 data_w = writeLock(w_index);
306 // it could be that active slot changed in the meantime, e.g. in case writeLock above had to wait
307 // for another write operation to finish. Make sure we read from latest slot.
308 r_index = lastDataSlot();
309 }
310
311 if (isPartialWrite) {
312 // write only updates some vertices, potentially not all.
313 // this means we need to ensure the target slot for write has the current data.
314 // NOTE: as write slot must have the current data anyways, the "read" toggle can be ignored here.
315 // TODO: in some cases a better strategy could be to write into the current slot instead of copying
316 // the data to the other slot. Maybe a sensible heuristic would be the data size:
317 // for small data, especially non-array, non-vertex data, always prefer copy.
318 // for larger array and vertex data prefer write into current slot.
319 if (r_index != w_index) {
320 auto data_r = readLock(r_index);
321 {
322 // copy FULL data into write slot.
323 // NOTE: this will be inefficient if the data is large!
324 std::memcpy(data_w, data_r, inputSize_);
325 }
326 readUnlock(r_index);
327 }
328 return {data_w, -1, data_w, w_index};
329 } else { // full write.
330 const byte *data_r = nullptr;
331 if (r_index != w_index && (mapMode & ShaderData::READ) != 0) {
332 // NOTE: only need a read lock if the read slot is different from the write slot.
333 data_r = readLock(r_index);
334 } else {
335 data_r = data_w;
336 r_index = -1;
337 }
338 return {data_r, r_index, data_w, w_index};
339 }
340 } else {
341 // read only. the case of reading at index is not handled differently here.
342 if (hasTwoSlots()) {
343 auto readData = readLock(r_index);
344 return {readData, r_index};
345 } else {
346 // make an attempt to get direct access
347 auto readData = readLockTry(r_index);
348 if (readData) {
349 return {readData, r_index};
350 } else {
351 // read slot is write locked, plus the second slot was not allocated yet.
352 // allocate it now, copy the data to it, then read from it.
353 allocateSecondSlot();
354 writeUnlock(1, false);
355 r_index = lastDataSlot();
356 readData = readLock(r_index);
357 return {readData, r_index};
358 }
359 }
360 }

371void ShaderInput::writeVertex(GLuint index, const byte *data) {
372 auto mapped = mapClientDataRaw(ShaderData::WRITE | ShaderData::INDEX);
373 // NOTE: it is maybe a bit confusing, but the semantics of writeVertex is currently
374 // different for uniform array data vs vertex data.
375 // For vertex data, it is assumed that data is one vertex including all array elements.
376 // For uniform array data, it is assumed that data is one array element.
377 if (isVertexAttribute_) {
378 std::memcpy(mapped.w + index * elementSize_, data, elementSize_);
379 } else {
380 auto arrayElementSize = dataTypeBytes_ * valsPerElement_;
381 std::memcpy(mapped.w + index * arrayElementSize, data, arrayElementSize);
382 }

416void ShaderInput::setInstanceData(GLuint numInstances, GLuint divisor, const byte *data) {
417 auto dataSize_bytes = elementSize_ * numInstances / divisor;
418
419 if (dataSize_bytes != inputSize_ || isVertexAttribute_ || !hasClientData()) {
420 // size of the data has changed, need to reallocate the data buffer.
421 writeLockAll();
422 reallocateClientData(dataSize_bytes);
423 isVertexAttribute_ = false;
424 numInstances_ = std::max(1u, numInstances);
425 divisor_ = std::max(1u, divisor);
426 numVertices_ = 1u;
427 numElements_ = static_cast<int>(numArrayElements_ * numInstances_);
428 inputSize_ = dataSize_bytes;
429 writeUnlockAll(writeClientData_(data));
430 } else if (data) {
431 auto mapped = mapClientDataRaw(ShaderData::WRITE);
432 std::memcpy(mapped.w, data, dataSize_bytes);
433 }
434}

466void ShaderInput::writeServerData(GLuint index) const {
467 if (!hasClientData() || !hasServerData()) return;
468 auto mappedClientData = mapClientData(ShaderData::READ);
469 auto clientData = mappedClientData.r;
470 auto subDataStart = clientData + elementSize_ * index;
471
472 RenderState::get()->copyWriteBuffer().push(buffer_);
473 glBufferSubData(
474 GL_COPY_WRITE_BUFFER,
475 offset_ + stride_ * index,
476 elementSize_,
477 subDataStart);
478 RenderState::get()->copyWriteBuffer().pop();
479}

481void ShaderInput::writeServerData() const {
482 if (!hasClientData() || !hasServerData()) return;
483 if (bufferStamp_ == stamp()) return;
484 auto mappedClientData = mapClientData(ShaderData::READ);
485 auto clientData = mappedClientData.r;
486 auto count = std::max(numVertices_, numInstances_);
487
488 RenderState::get()->copyWriteBuffer().push(buffer_);
489 if (stride_ == elementSize_) {
490 glBufferSubData(GL_COPY_WRITE_BUFFER, offset_, inputSize_, clientData);
491 } else {
492 GLuint offset = offset_;
493 for (GLuint i = 0; i < count; ++i) {
494 glBufferSubData(GL_COPY_WRITE_BUFFER, offset, elementSize_, clientData);
495 offset += stride_;
496 clientData += elementSize_;
497 }
498 }
499 RenderState::get()->copyWriteBuffer().pop();
500
501 bufferStamp_ = stamp();
502}

504void ShaderInput::readServerData() {
505 if (!hasServerData()) return;
506 auto mappedClientData = mapClientData(ShaderData::WRITE);
507 auto clientData = mappedClientData.w;
508
509 RenderState::get()->arrayBuffer().push(buffer());
510 byte *serverData = (byte *) glMapBufferRange(
511 GL_ARRAY_BUFFER,
512 offset_,
513 numVertices_ * stride_ + elementSize_,
514 GL_MAP_READ_BIT);
515
516 if (stride_ == elementSize_) {
517 std::memcpy(clientData, serverData, inputSize_);
518 } else {
519 for (GLuint i = 0; i < numVertices_; ++i) {
520 std::memcpy(clientData, serverData, elementSize_);
521 serverData += stride_;
522 clientData += elementSize_;
523 }
524 }
525
526 glUnmapBuffer(GL_ARRAY_BUFFER);
527 RenderState::get()->arrayBuffer().pop();
528}

546ref_ptr<ShaderInput> ShaderInput::create(const ref_ptr<ShaderInput> &in) {
547 if (in->isUniformBlock()) {
548 auto newBlock = ref_ptr<UniformBlock>::alloc(in->name());
549 auto oldBlock = (UniformBlock *) (in.get());
550 for (auto &namedUniform: oldBlock->uniforms()) {
551 newBlock->addUniform(create(namedUniform.in_), namedUniform.name_);
552 }
553 return newBlock;
554 }
555
556 const std::string &name = in->name();
557 GLenum dataType = in->dataType();
558 GLuint valsPerElement = in->valsPerElement();
559
560 switch (dataType) {
561 case GL_FLOAT:
562 switch (valsPerElement) {
563 case 16:
564 return ref_ptr<ShaderInputMat4>::alloc(name);
565 case 9:
566 return ref_ptr<ShaderInputMat3>::alloc(name);
567 case 4:
568 return ref_ptr<ShaderInput4f>::alloc(name);
569 case 3:
570 return ref_ptr<ShaderInput3f>::alloc(name);
571 case 2:
572 return ref_ptr<ShaderInput2f>::alloc(name);
573 default:
574 return ref_ptr<ShaderInput1f>::alloc(name);
575 }
576 case GL_DOUBLE:
577 switch (valsPerElement) {
578 case 4:
579 return ref_ptr<ShaderInput4d>::alloc(name);
580 case 3:
581 return ref_ptr<ShaderInput3d>::alloc(name);
582 case 2:
583 return ref_ptr<ShaderInput2d>::alloc(name);
584 default:
585 return ref_ptr<ShaderInput1d>::alloc(name);
586 }
587 case GL_BOOL:
588 case GL_INT:
589 switch (valsPerElement) {
590 case 4:
591 return ref_ptr<ShaderInput4i>::alloc(name);
592 case 3:
593 return ref_ptr<ShaderInput3i>::alloc(name);
594 case 2:
595 return ref_ptr<ShaderInput2i>::alloc(name);
596 default:
597 return ref_ptr<ShaderInput1i>::alloc(name);
598 }
599 case GL_UNSIGNED_INT:
600 switch (valsPerElement) {
601 case 4:
602 return ref_ptr<ShaderInput4ui>::alloc(name);
603 case 3:
604 return ref_ptr<ShaderInput3ui>::alloc(name);
605 case 2:
606 return ref_ptr<ShaderInput2ui>::alloc(name);
607 default:
608 return ref_ptr<ShaderInput1ui>::alloc(name);
609 }
610 default:
611 return {};
612 }
613}

615ref_ptr<ShaderInput> ShaderInput::copy(const ref_ptr<ShaderInput> &in, GLboolean copyData) {
616 ref_ptr<ShaderInput> cp = create(in);
617 cp->stride_ = in->stride_;
618 cp->offset_ = in->offset_;
619 cp->inputSize_ = in->inputSize_;
620 cp->elementSize_ = in->elementSize_;
621 cp->numArrayElements_ = in->numArrayElements_;
622 cp->numVertices_ = in->numVertices_;
623 cp->numInstances_ = in->numInstances_;
624 cp->divisor_ = in->divisor_;
625 cp->buffer_ = 0;
626 cp->bufferStamp_ = 0;
627 cp->normalize_ = in->normalize_;
628 cp->isVertexAttribute_ = in->isVertexAttribute_;
629 cp->isUniformBlock_ = in->isUniformBlock_;
630 cp->isConstant_ = in->isConstant_;
631 cp->transpose_ = in->transpose_;
632 cp->forceArray_ = in->forceArray_;
633 if (in->hasClientData()) {
634 // allocate memory for one slot, copy most recent data
635 auto mapped = in->mapClientData(ShaderData::READ);
636 cp->dataSlots_[0] = new byte[cp->inputSize_];
637 std::memcpy(cp->dataSlots_[0], mapped.r, cp->inputSize_);
638 }
639 return cp;
640}
