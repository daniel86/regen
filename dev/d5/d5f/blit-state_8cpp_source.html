93ref_ptr<BlitState> BlitState::load(LoadingContext &ctx, scene::SceneInputNode &input) {
94 auto scene = ctx.scene();
95
96 if (!input.hasAttribute("src-fbo")) {
97 REGEN_WARN("Ignoring " << input.getDescription() << " without src-fbo attribute.");
98 return {};
99 }
100 if (!input.hasAttribute("dst-fbo")) {
101 REGEN_WARN("Ignoring " << input.getDescription() << " without dst-fbo attribute.");
102 return {};
103 }
104 const std::string srcID = input.getValue("src-fbo");
105 const std::string dstID = input.getValue("dst-fbo");
106 ref_ptr<FBO> src, dst;
107 if (srcID != "SCREEN") {
108 src = scene->getResource<FBO>(srcID);
109 if (src.get() == nullptr) {
110 REGEN_WARN("Unable to find FBO with name '" << srcID << "'.");
111 return {};
112 }
113 }
114 if (dstID != "SCREEN") {
115 dst = scene->getResource<FBO>(dstID);
116 if (dst.get() == nullptr) {
117 REGEN_WARN("Unable to find FBO with name '" << dstID << "'.");
118 return {};
119 }
120 }
121 bool keepAspect = input.getValue<GLuint>("keep-aspect", false);
122 if (src.get() != nullptr && dst.get() != nullptr) {
123 if (input.getValue("src-attachment") == "depth" || input.getValue("dst-attachment") == "depth") {
124 auto blit = ref_ptr<BlitToFBO>::alloc(
125 src, dst,
126 GL_DEPTH_ATTACHMENT,
127 GL_DEPTH_ATTACHMENT,
128 keepAspect);
129 blit->set_sourceBuffer(GL_DEPTH_BUFFER_BIT);
130 blit->set_filterMode(GL_NEAREST);
131 return blit;
132 } else {
133 auto srcAttachment = input.getValue<GLuint>("src-attachment", 0u);
134 auto dstAttachment = input.getValue<GLuint>("dst-attachment", 0u);
135 return ref_ptr<BlitToFBO>::alloc(
136 src, dst,
137 GL_COLOR_ATTACHMENT0 + srcAttachment,
138 GL_COLOR_ATTACHMENT0 + dstAttachment,
139 keepAspect);
140 }
141 } else if (src.get() != nullptr) {
142 // Blit Texture to Screen
143 auto srcAttachment = input.getValue<GLuint>("src-attachment", 0u);
144 return ref_ptr<BlitToScreen>::alloc(src,
145 scene->getViewport(),
146 GL_COLOR_ATTACHMENT0 + srcAttachment,
147 keepAspect);
148 } else if (dst.get() != nullptr) {
149 REGEN_WARN(input.getDescription() << ", blitting Screen to FBO not supported.");
150 } else {
151 REGEN_WARN("No src or dst FBO specified for " << input.getDescription() << ".");
152 }
153 return {};
154}
