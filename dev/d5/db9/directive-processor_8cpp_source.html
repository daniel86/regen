51{

62void DirectiveProcessor::MacroBranch::add(bool isDefined)
63{
64 MacroBranch &active = getActive();
65 MacroBranch *parent = active.parent_;
66 if(parent==NULL) { return; }
67
68 bool defined;
69 if(parent->isAnyChildDefined_) {
70 defined = false;
71 } else {
72 defined = isDefined&&parent->isDefined_;
73 }
74 if(isDefined) {
75 // remember if the argument is defined (all following #else cases can be skipped then)
76 parent->isAnyChildDefined_ = true;
77 }
78
79 parent->childs_.push_back( MacroBranch(defined, parent));
80}

112string DirectiveProcessor::MacroTree::define(const string &arg)
113{
114 static const string operatorsPattern = "(.+)[ ]*(\\+|-|\\*|/)[ ]*(.+)";
115 static boost::regex operatorsRegex(operatorsPattern);
116
117 boost::sregex_iterator it(arg.begin(), arg.end(), operatorsRegex);
118 if(it!=NO_REGEX_MATCH) {
119 string arg0_ = (*it)[1]; boost::trim(arg0_);
120 string op = (*it)[2]; boost::trim(op);
121 string arg1_ = (*it)[3]; boost::trim(arg1_);
122 const string &arg0 = define(arg0_);
123 const string &arg1 = define(arg1_);
124 if(!isNumber(arg0) || !isNumber(arg1)) return arg;
125 GLfloat number0, number1;
126 stringstream(arg0) >> number0;
127 stringstream(arg1) >> number1;
128 if(op == "+") return REGEN_STRING(number0+number1);
129 if(op == "-") return REGEN_STRING(number0-number1);
130 if(op == "/") return REGEN_STRING(number0/number1);
131 if(op == "*") return REGEN_STRING(number0*number1);
132 REGEN_WARN("Unknown operator '" << op << "'.");
133 return arg;
134 }
135 else if(isNumber(arg)) {
136 return arg;
137 }
138 else {
139 map<string,string>::iterator it = defines_.find(arg);
140 if(it==defines_.end()) {
141 return arg;
142 } else {
143 return it->second;
144 }
145 }
146}

148bool DirectiveProcessor::MacroTree::evaluateInner(const string &expression)
149{
150 static const string operatorsPattern = "(.+)[ ]*(==|!=|<=|>=|<|>)[ ]*(.+)";
151 static boost::regex operatorsRegex(operatorsPattern);
152
153 boost::sregex_iterator it(expression.begin(), expression.end(), operatorsRegex);
154 if(it!=NO_REGEX_MATCH) {
155 string arg0_ = (*it)[1]; boost::trim(arg0_);
156 string op = (*it)[2]; boost::trim(op);
157 string arg1_ = (*it)[3]; boost::trim(arg1_);
158 const string &arg0 = define(arg0_);
159 const string &arg1 = define(arg1_);
160
161 if(op == "==") return (arg0==arg1);
162 else if(op == "!=") return (arg0!=arg1);
163
164 // numeric operators left
165 if(!isNumber(arg0) || !isNumber(arg1)) { return false; }
166 float val0=0.0f, val1=0.0f;
167 stringstream(arg0) >> val0;
168 stringstream(arg1) >> val1;
169
170 if(op == "<=") return (val0<=val1);
171 else if(op == ">=") return (val0>=val1);
172 else if(op == ">") return (val0>val1);
173 else if(op == "<") return (val0<val1);
174 else return false;
175 }
176
177 // single inner argument
178 string arg = expression; boost::trim(arg);
179 if(isNumber(arg)) {
180 return arg!="0";
181 } else {
182 map<string,string>::iterator it = defines_.find(arg);
183 if(it==defines_.end() || it->second=="0") {
184 return false;
185 } else {
186 return true;
187 }
188 }

191bool DirectiveProcessor::MacroTree::evaluate(const string &expression)
192{
193 static const char* bracketsPattern = "[ ]*\\((.+)\\)[ ]*(\\|\\||\\&\\&)[ ]*([^\\)]+)";
194 static const char* pattern = "[ ]*([^\\&\\|]+)[ ]*(\\|\\||\\&\\&)[ ]*(.+)";
195 static boost::regex bracketsRegex(bracketsPattern);
196 static boost::regex expRegex(pattern);
197
198 // match leading bracket...
199 boost::sregex_iterator it(expression.begin(), expression.end(), bracketsRegex);
200 if(it!=NO_REGEX_MATCH) {
201 string exp0 = (*it)[1]; boost::trim(exp0);
202 string op = (*it)[2]; boost::trim(op);
203 string exp1 = (*it)[3]; boost::trim(exp1);
204 if(op == "&&") {
205 return evaluate(exp0) && evaluate(exp1);
206 } else {
207 return evaluate(exp0) || evaluate(exp1);
208 }
209 }
210 // match logical expression...
211 it = boost::sregex_iterator(expression.begin(), expression.end(), expRegex);
212 if(it!=NO_REGEX_MATCH) {
213 string exp0 = (*it)[1]; boost::trim(exp0);
214 string op = (*it)[2]; boost::trim(op);
215 string exp1 = (*it)[3]; boost::trim(exp1);
216 if(op == "&&") {
217 return evaluateInner(exp0) && evaluate(exp1);
218 } else {
219 return evaluateInner(exp0) || evaluate(exp1);
220 }
221 }
222 // no logical operator found...
223 return evaluateInner(expression);
224}

266void DirectiveProcessor::parseVariables(string &line)
267{
268 static const char* variablePattern = "\\$\\{[ ]*([^ \\}\\{]+)[ ]*\\}";
269 static boost::regex variableRegex(variablePattern);
270
271 set<string> variableNames;
272 boost::sregex_iterator regexIt(line.begin(), line.end(), variableRegex);
273
274 if(regexIt==NO_REGEX_MATCH) { return; }
275
276 GLboolean replacedSomething = GL_FALSE;
277
278 for(; regexIt!=NO_REGEX_MATCH; ++regexIt)
279 {
280 const string &match = (*regexIt)[1];
281 variableNames.insert(match);
282 }
283 for(set<string>::iterator it=variableNames.begin(); it!=variableNames.end(); ++it)
284 {
285 const string &define = *it;
286 string name = define.substr(0,define.find_first_of(" "));
287 const string &value = tree_.define(name);
288 if(value != name) {
289 boost::replace_all(line, "${"+name+"}", value);
290 replacedSomething = GL_TRUE;
291 }
292 }
293
294 // parse nested vars
295 if(replacedSomething==GL_TRUE) parseVariables(line);
296}

304bool DirectiveProcessor::process(PreProcessorState &state, string &line)
305{
306 if(lastStage_ != state.currStage) {
307 lastStage_ = state.currStage;
308 tree_.clear();
309 continuedLine_.clear();
310 forBranches_.clear();
311 wasEmpty_ = GL_TRUE;
312 }
313
314 if(inputs_.empty()) {
315 if(parent_.get()) {
316 inputs_.push_back(parent_);
317 } else {
318 return false;
319 }
320 }
321
322 ref_ptr<GLSLProcessor> &in = inputs_.front();
323 if(!in->getline(state, line)) {
324 inputs_.pop_front();
325 if(inputs_.empty()) return false;
326 else DirectiveProcessor::getline(state,line);
327 }
328
329 // the line stopped with '\' character
330 if(!line.empty() && *line.rbegin() == '\\') {
331 continuedLine_ += line + "\n";
332 return DirectiveProcessor::getline(state,line);
333 }
334 if(!continuedLine_.empty()) {
335 line = continuedLine_ + line;
336 continuedLine_ = "";
337 }
338
339 // evaluate ${..}
340 if(tree_.isDefined() && forBranches_.empty()) { parseVariables(line); }
341
342 string statement(line);
343 boost::trim(statement);
344
345 GLboolean isEmpty = statement.empty();
346 if(isEmpty && wasEmpty_) {
347 return DirectiveProcessor::getline(state,line);
348 }
349 wasEmpty_ = isEmpty;
350
351 if(hasPrefix(statement, "#line ")) {
352 // for now remove line directives
353 return DirectiveProcessor::getline(state,line);
354 }
355 else if(hasPrefix(statement, "#version ")) {
356 string versionStr = truncPrefix(statement, "#version ");
357 boost::trim(versionStr);
358 GLuint version = (GLuint)atoi(versionStr.c_str());
359 if(state.version<version) { state.version=version; }
360 // remove version directives, must be prepended later
361 return DirectiveProcessor::getline(state,line);
362 }
363 else if(hasPrefix(statement, "#for ")) {
364 static const char* forPattern = "#for (.+) to (.+)";
365 static boost::regex forRegex(forPattern);
366 boost::sregex_iterator regexIt(statement.begin(), statement.end(), forRegex);
367
368 if(regexIt!=NO_REGEX_MATCH) {
369 ForBranch branch;
370 branch.variableName = (*regexIt)[1]; boost::trim(branch.variableName);
371 branch.upToValue = (*regexIt)[2]; boost::trim(branch.upToValue);
372 branch.lines = "";
373 forBranches_.push_front(branch);
374 }
375 else {
376 line = "#warning Invalid Syntax: '" + statement + "'. Example: '#for INDEX to 9'.";
377 return true;
378 }
379
380 return DirectiveProcessor::getline(state,line);
381 }
382 else if(hasPrefix(statement, "#endfor")) {
383 if(forBranches_.empty()) {
384 line = "#warning Closing #endfor without opening #for.";
385 return true;
386 }
387 ForBranch &branch = forBranches_.front();
388
389 ref_ptr<StreamProcessor> forLoop = ref_ptr<StreamProcessor>::alloc();
390 stringstream &ss = forLoop->stream();
391
392 const string &def = tree_.define(branch.upToValue);
393 if(!isNumber(def)) {
394 ss << "#error " << branch.upToValue << " is not a number" << endl;
395 } else {
396 int count = boost::lexical_cast<int>(def);
397
398 for(int i=0; i<count; ++i) {
399 ss << "#define2 " << branch.variableName << " " << i << endl;
400 ss << branch.lines;
401 }
402 }
403 forBranches_.pop_front();
404 inputs_.push_front(forLoop);
405
406 return DirectiveProcessor::getline(state,line);
407 }
408 else if(!forBranches_.empty()) {
409 forBranches_.front().lines += line + "\n";
410 return DirectiveProcessor::getline(state,line);
411 }
412 else if(hasPrefix(statement, "#include ")) {
413 if(!tree_.isDefined()) {
414 return DirectiveProcessor::getline(state,line);
415 }
416 string key = truncPrefix(statement, "#include ");
417 boost::trim(key);
418 string imported;
419 map<string,string>::const_iterator needle = state.in.externFunctions.find(key);
420 if(needle != state.in.externFunctions.end()) {
421 imported = needle->second;
422 } else {
423 imported = Includer::get().include(key); boost::trim(key);
424 }
425 if(imported.empty()) {
426 line = "#warning Failed to include " + key + ". Make sure GLSW path is set up.";
427 REGEN_WARN(Includer::get().errorMessage());
428 return true;
429 } else {
430 inputs_.push_front(ref_ptr<StreamProcessor>::alloc(imported));
431 }
432 return DirectiveProcessor::getline(state,line);
433 }
434 else if(hasPrefix(statement, "#define2 ")) {
435 string v = truncPrefix(statement, "#define2 ");
436 boost::trim(v);
437 tree_._define(v);
438 }
439 else if(hasPrefix(statement, "#define ")) {
440 string v = truncPrefix(statement, "#define ");
441 boost::trim(v);
442 tree_._define(v);
443 if(tree_.isDefined()) {
444 return true;
445 }
446 }
447 else if(hasPrefix(statement, "#undef ")) {
448 string v = truncPrefix(statement, "#undef ");
449 boost::trim(v);
450 tree_._undef(v);
451 if(tree_.isDefined()) {
452 return true;
453 }
454 }
455 else if(hasPrefix(statement, "#ifdef ")) {
456 string v = truncPrefix(statement, "#ifdef ");
457 boost::trim(v);
458 tree_._ifdef(v);
459 }
460 else if(hasPrefix(statement, "#ifndef ")) {
461 string v = truncPrefix(statement, "#ifndef ");
462 boost::trim(v);
463 tree_._ifndef(v);
464 }
465 else if(hasPrefix(statement, "#if ")) {
466 string v = truncPrefix(statement, "#if ");
467 boost::trim(v);
468 tree_._ifdef(v);
469 }
470 else if(hasPrefix(statement, "#elif ")) {
471 string v = truncPrefix(statement, "#elif ");
472 boost::trim(v);
473 tree_._elif(v);
474 }
475 else if(hasPrefix(statement, "#else")) {
476 tree_._else();
477 }
478 else if(hasPrefix(statement, "#endif")) {
479 tree_._endif();
480 }
481 else if(tree_.isDefined())
482 {
483 return true;
484 }
485 return DirectiveProcessor::getline(state,line);
486}
