168GLfloat Sky::computeHorizonExtinction(const Vec3f &position, const Vec3f &dir, float radius) {
169 GLfloat u = dir.dot(-position);
170 if (u < 0.0) {
171 return 1.0;
172 }
173 Vec3f near = position + dir * u;
174 if (near.length() < radius) {
175 return 0.0;
176 } else {
177 near.normalize();
178 Vec3f v2 = near * radius - position;
179 v2.normalize();
180 GLfloat diff = acos(v2.dot(dir));
181 return math::smoothstep(0.0, 1.0, pow(diff * 2.0, 3.0));
182 }
183}

185GLfloat Sky::computeEyeExtinction(const Vec3f &eyedir) {
186 static const float surfaceHeight = 0.99f;
187 static const Vec3f eyePosition(0.0, surfaceHeight, 0.0);
188 return computeHorizonExtinction(eyePosition, eyedir, surfaceHeight - 0.15f);

199void Sky::animate(GLdouble dt) {
200 if (worldTime_) {
201 // convert to time_t
202 time_t t = boost::posix_time::to_time_t(worldTime_->p_time);
203 // get UTC offset
204 struct tm *tm = gmtime(&t);
205 time_t utcOffset = t - mktime(tm);
206 // create timef object
207 osgHimmel::TimeF time_osg(t, utcOffset);
208 astro_->update(osgHimmel::t_aTime::fromTimeF(time_osg));
209 }
210
211 // Compute sun/moon directions
212 Vec3f moon = astro_->getMoonPosition(false);
213 Vec3f sun = astro_->getSunPosition(false);
214 sun_->direction()->setVertex(0, sun);
215 moon_->direction()->setVertex(0, moon);
216 // Compute sun/moon diffuse color
217 GLfloat sunExt = computeEyeExtinction(sun);
218 Vec3f sunColor = math::mix(dawnColor_, noonColor_, abs(sunExt));
219 sun_->diffuse()->setVertex(0, computeColor(
220 sunColor,
221 sunExt)
222 );
223 moon_->diffuse()->setVertex(0, computeColor(
224 sunColor * moonSunLightReflectance_,
225 computeEyeExtinction(moon))
226 );
227
228 const float fovHalf = camera()->fov()->getVertex(0).r * 0.5f * DEGREE_TO_RAD;
229 const float height = static_cast<float>(viewport()->getVertex(0).r.y);
230 q_->setVertex(0, sqrt(2.0f) * 2.0f * tan(fovHalf) / height);
231 R_->setVertex(0, astro().getEquToHorTransform());
232 // Update random number in cmn uniform
233 updateSeed();
234}

387static ref_ptr<Atmosphere> createAtmosphereLayer(const ref_ptr<Sky> &sky,
388 scene::SceneLoader *parser, scene::SceneInputNode &input,
389 const std::string &skyName) {
390 ref_ptr<Atmosphere> atmosphere = ref_ptr<Atmosphere>::alloc(sky,
391 input.getValue<GLuint>("size", 512),
392 input.getValue<GLuint>("use-float", false),
393 input.getValue<GLuint>("lod", 0));
394
395 const std::string preset = input.getValue<std::string>("preset", "earth");
396 if (preset == "earth") atmosphere->setEarth();
397 else if (preset == "mars") atmosphere->setMars();
398 else if (preset == "venus") atmosphere->setVenus();
399 else if (preset == "uranus") atmosphere->setUranus();
400 else if (preset == "alien") atmosphere->setAlien();
401 else if (preset == "custom") {
402 auto absorption = input.getValue<Vec3f>("absorbtion", Vec3f(
403 0.18867780436772762,
404 0.4978442963618773,
405 0.6616065586417131));
406 auto rayleigh = input.getValue<Vec3f>("rayleigh", Vec3f(19.0, 359.0, 81.0));
407 auto mie = input.getValue<Vec4f>("mie", Vec4f(44.0, 308.0, 39.0, 74.0));
408 auto spot = input.getValue<GLfloat>("spot", 373.0);
409 auto strength = input.getValue<GLfloat>("strength", 54.0);
410
411 atmosphere->setRayleighBrightness(rayleigh.x);
412 atmosphere->setRayleighStrength(rayleigh.y);
413 atmosphere->setRayleighCollect(rayleigh.z);
414 atmosphere->setMieBrightness(mie.x);
415 atmosphere->setMieStrength(mie.y);
416 atmosphere->setMieCollect(mie.z);
417 atmosphere->setMieDistribution(mie.w);
418 atmosphere->setSpotBrightness(spot);
419 atmosphere->setScatterStrength(strength);
420 atmosphere->setAbsorption(absorption);
421 } else
422 REGEN_WARN("Ignoring unknown sky preset '" << preset <<
423 "' for node " << input.getDescription() << ".");
424
425 atmosphere->set_updateInterval(
426 input.getValue<GLdouble>("update-interval", 4000.0));
427
428 parser->putResource<Texture>(skyName, atmosphere->cubeMap());
429
430 sky->addLayer(atmosphere);
431
432 return atmosphere;
433}

476ref_ptr<Sky> Sky::load(LoadingContext &ctx, scene::SceneInputNode &input) {
477 auto scene = ctx.scene();
478
479 if (!input.hasAttribute("camera")) {
480 REGEN_WARN("No user-camera attribute specified for '" << input.getDescription() << "'.");
481 return {};
482 }
483 ref_ptr<Camera> cam = scene->getResource<Camera>(input.getValue("camera"));
484 if (cam.get() == nullptr) {
485 REGEN_WARN("No Camera can be found for '" << input.getDescription() << "'.");
486 return {};
487 }
488
489 ref_ptr<Sky> sky = ref_ptr<Sky>::alloc(cam, scene->getViewport());
490
491 sky->setWorldTime(&scene->application()->worldTime());
492
493 if (input.hasAttribute("noon-color"))
494 sky->set_noonColor(input.getValue<Vec3f>("noon-color", Vec3f(0.5)));
495 if (input.hasAttribute("dawn-color"))
496 sky->set_dawnColor(input.getValue<Vec3f>("dawn-color", Vec3f(0.2, 0.15, 0.15)));
497
498 if (input.hasAttribute("moon-reflectance"))
499 sky->set_moonSunLightReflectance(input.getValue<float>("moon-reflectance", 0.4));
500
501 sky->set_altitude(input.getValue<float>("altitude", 0.043));
502 sky->set_longitude(input.getValue<float>("longitude", 13.3611));
503 sky->set_latitude(input.getValue<float>("latitude", 52.5491));
504
505 for (auto &n: input.getChildren()) {
506 ref_ptr<SkyLayer> layer;
507 if (n->getCategory() == "atmosphere") {
508 layer = createAtmosphereLayer(sky, scene, *n.get(), input.getName());
509 } else if (n->getCategory() == "cloud-layer") {
510 layer = createCloudLayer(sky, scene, *n.get());
511 } else if (n->getCategory() == "moon") {
512 layer = createMoonLayer(sky, scene, *n.get());
513 } else if (n->getCategory() == "star-map") {
514 layer = createStarMapLayer(sky, scene, *n.get());
515 } else if (n->getCategory() == "stars") {
516 layer = createStarsLayer(sky, scene, *n.get());
517 }
518 if (!layer.get()) {
519 REGEN_WARN("No layer created for '" << n->getDescription() << "'.");
520 continue;
521 }
522 for (auto &layerChild: n->getChildren()) {
523 auto processor = scene->getStateProcessor(layerChild->getCategory());
524 if (processor.get() == nullptr) {
525 REGEN_WARN("No processor registered for '" << layerChild->getDescription() << "'.");
526 continue;
527 }
528 processor->processInput(scene, *layerChild.get(), sky, layer->updateState());
529 }
530 }
531 sky->createShader();
532
533 // The Sky also exposes a Light (the sun) and a Texture (the cube map)
534 scene->putResource<Light>(input.getName() + "-sun", sky->sun());
535 scene->putResource<Light>(input.getName() + "-moon", sky->moon());
536 scene->putState(input.getName() + "-sun", sky->sun());
537 scene->putState(input.getName() + "-moon", sky->moon());
538
539 sky->startAnimation();
540
541 return sky;
542}
