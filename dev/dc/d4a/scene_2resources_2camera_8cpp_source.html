30class ProjectionUpdater : public EventHandler {
31public:
37 ProjectionUpdater(const ref_ptr<Camera> &cam,
38 const ref_ptr<ShaderInput2i> &windowViewport)
39 : EventHandler(), cam_(cam), windowViewport_(windowViewport) {}
40
41 void call(EventObject *, EventData *) {
42 auto windowViewport = windowViewport_->getVertex(0);
43 auto windowAspect =
44 (GLfloat) windowViewport.r.x / (GLfloat) windowViewport.r.y;
45 if (cam_->isOrtho()) {
46 // keep the ortho width and adjust height based on aspect ratio
47 auto width = cam_->frustum()[0].nearPlaneHalfSize.x * 2.0f;
48 auto height = width / windowAspect;
49 cam_->setOrtho(
50 -width / 2.0f, width / 2.0f,
51 -height / 2.0f, height / 2.0f,
52 cam_->near()->getVertex(0).r,
53 cam_->far()->getVertex(0).r);
54 }
55 else {
56 cam_->setPerspective(
57 windowAspect,
58 cam_->fov()->getVertex(0).r,
59 cam_->near()->getVertex(0).r,
60 cam_->far()->getVertex(0).r);
61 }
62 }
63
64protected:
65 ref_ptr<Camera> cam_;
66 ref_ptr<ShaderInput2i> windowViewport_;
67};

90int getHiddenFacesMask(SceneInputNode &input) {
91 int hiddenFacesMask = 0;
92 if (input.hasAttribute("hide-faces")) {
93 auto val = input.getValue<std::string>("hide-faces", "");
94 std::vector<std::string> faces;
95 boost::split(faces, val, boost::is_any_of(","));
96 for (auto it = faces.begin(); it != faces.end(); ++it) {
97 CubeCamera::Face face;
98 std::stringstream(*it) >> face;
99 hiddenFacesMask |= face;
100 }
101 }
102 return hiddenFacesMask;
103}

105ref_ptr<Camera> createLightCamera(
106 SceneParser *parser,
107 SceneInputNode &input) {
108 ref_ptr<Light> light = parser->getResources()->getLight(parser, input.getValue("light"));
109 if (light.get() == nullptr) {
110 REGEN_WARN("Unable to find Light for '" << input.getDescription() << "'.");
111 return {};
112 }
113 auto numLayer = input.getValue<GLuint>("num-layer", 1u);
114 auto splitWeight = input.getValue<GLdouble>("split-weight", 0.9);
115 auto cameraType = input.getValue<std::string>("camera-type", "spot");
116 auto near = input.getValue<float>("near", 0.1f);
117 ref_ptr<Camera> lightCamera;
118
119 switch (light->lightType()) {
120 case Light::SPOT: {
121 auto spotCam = ref_ptr<LightCamera_Spot>::alloc(light);
122 spotCam->setLightNear(near);
123 lightCamera = spotCam;
124 break;
125 }
126 case Light::DIRECTIONAL: {
127 auto userCamera = parser->getResources()->getCamera(parser, input.getValue("camera"));
128 if (userCamera.get() == nullptr) {
129 REGEN_WARN("Unable to find user camera for '" << input.getDescription() << "'.");
130 return {};
131 }
132 auto dirCam = ref_ptr<LightCamera_CSM>::alloc(light, userCamera, numLayer);
133 dirCam->setSplitWeight(splitWeight);
134 lightCamera = dirCam;
135 break;
136 }
137 case Light::POINT: {
138 if (cameraType == "parabolic" || cameraType == "paraboloid") {
139 // parabolic camera
140 auto isDualParabolic = input.getValue<bool>("dual-paraboloid", true);
141 auto parabolic = ref_ptr<LightCamera_Parabolic>::alloc(light, isDualParabolic);
142 if (input.hasAttribute("normal")) {
143 parabolic->setNormal(input.getValue<Vec3f>("normal", Vec3f::down()));
144 }
145 parabolic->setLightNear(near);
146 lightCamera = parabolic;
147 } else {
148 // cube camera
149 auto cube = ref_ptr<LightCamera_Cube>::alloc(light, getHiddenFacesMask(input));
150 cube->setLightNear(near);
151 lightCamera = cube;
152 }
153 break;
154 }
155 }
156 if (lightCamera.get() == nullptr) {
157 REGEN_WARN("Unable to create camera for '" << input.getDescription() << "'.");
158 return {};
159 }
160 parser->putState(input.getName(), lightCamera);
161
162 return lightCamera;
163}

165ref_ptr<Camera> CameraResource::createCamera(SceneParser *parser, SceneInputNode &input) {
166 auto camType = input.getValue<string>("type", "spot");
167
168 if (input.hasAttribute("reflector") ||
169 input.hasAttribute("reflector-normal") ||
170 input.hasAttribute("reflector-point")) {
171 ref_ptr<Camera> userCamera =
172 parser->getResources()->getCamera(parser, input.getValue("camera"));
173 if (userCamera.get() == nullptr) {
174 REGEN_WARN("Unable to find Camera for '" << input.getDescription() << "'.");
175 return {};
176 }
177 ref_ptr<ReflectionCamera> cam;
178 bool hasBackFace = input.getValue<bool>("has-back-face", false);
179
180 if (input.hasAttribute("reflector")) {
181 ref_ptr<MeshVector> mesh =
182 parser->getResources()->getMesh(parser, input.getValue("reflector"));
183 if (mesh.get() == nullptr || mesh->empty()) {
184 REGEN_WARN("Unable to find Mesh for '" << input.getDescription() << "'.");
185 return {};
186 }
187 const vector<ref_ptr<Mesh> > &vec = *mesh.get();
188 cam = ref_ptr<ReflectionCamera>::alloc(
189 userCamera, vec[0], input.getValue<GLuint>("vertex-index", 0u), hasBackFace);
190 } else if (input.hasAttribute("reflector-normal")) {
191 auto normal = input.getValue<Vec3f>("reflector-normal", Vec3f(0.0f, 1.0f, 0.0f));
192 auto position = input.getValue<Vec3f>("reflector-point", Vec3f(0.0f, 0.0f, 0.0f));
193 cam = ref_ptr<ReflectionCamera>::alloc(userCamera, normal, position, hasBackFace);
194 }
195 if (cam.get()) {
196 parser->putState(input.getName(), cam);
197 }
198 return cam;
199 }
200 else if (input.hasAttribute("light")) {
201 auto cam = createLightCamera(parser, input);
202 return cam;
203 }
204 else if (camType == "cube") {
205 auto tf = parser->getResources()->getTransform(parser, input.getValue("tf"));
206 ref_ptr<CubeCamera> cam = ref_ptr<CubeCamera>::alloc(getHiddenFacesMask(input));
207 if (tf.get()) {
208 cam->attachToPosition(tf->get());
209 }
210 parser->putState(input.getName(), cam);
211 return cam;
212 }
213 else if (camType == "parabolic" || camType == "paraboloid") {
214 auto tf = parser->getResources()->getTransform(parser, input.getValue("tf"));
215 bool hasBackFace = input.getValue<bool>("dual-paraboloid", true);
216 auto cam = ref_ptr<ParabolicCamera>::alloc(hasBackFace);
217 if (input.hasAttribute("normal")) {
218 cam->setNormal(input.getValue<Vec3f>("normal", Vec3f::down()));
219 }
220
221 if (tf.get()) {
222 cam->attachToPosition(tf->get());
223 }
224 parser->putState(input.getName(), cam);
225
226 return cam;
227 }
228 else {
229 ref_ptr<Camera> cam = ref_ptr<Camera>::alloc(1);
230 cam->set_isAudioListener(
231 input.getValue<bool>("audio-listener", false));
232 cam->position()->setVertex(0,
233 input.getValue<Vec3f>("position", Vec3f(0.0f, 2.0f, -2.0f)));
234
235 auto dir = input.getValue<Vec3f>("direction", Vec3f(0.0f, 0.0f, 1.0f));
236 dir.normalize();
237 cam->direction()->setVertex(0, dir);
238
239 if (camType == "ortho" || camType == "orthographic") {
240 auto width = input.getValue<GLfloat>("width", 10.0f);
241 auto height = input.getValue<GLfloat>("height", 10.0f);
242 cam->setOrtho(
243 -width / 2.0f, width / 2.0f,
244 -height / 2.0f, height / 2.0f,
245 input.getValue<GLfloat>("near", 0.1f),
246 input.getValue<GLfloat>("far", 200.0f));
247 } else {
248 auto viewport = parser->getViewport()->getVertex(0);
249 cam->setPerspective(
250 (GLfloat) viewport.r.x / (GLfloat) viewport.r.y,
251 input.getValue<GLfloat>("fov", 45.0f),
252 input.getValue<GLfloat>("near", 0.1f),
253 input.getValue<GLfloat>("far", 200.0f));
254 }
255 cam->updateCamera();
256 // Update frustum when window size changes
257 parser->addEventHandler(Application::RESIZE_EVENT,
258 ref_ptr<ProjectionUpdater>::alloc(cam, parser->getViewport()));
259 parser->putState(input.getName(), cam);
260
261 return cam;
262 }
