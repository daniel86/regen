7template<int Count>
8std::pair<float, float> project(const std::vector<Vec2f> &points, const Vec2f &axis) {
9 std::array<float, Count> projections{};
10 std::transform(points.begin(), points.end(), projections.begin(),
11 [&axis](const Vec2f &point) {
12 return point.dot(axis);
13 });
14 auto [minIt, maxIt] = std::minmax_element(projections.begin(), projections.end());
15 return {*minIt, *maxIt};
16}

22void OrthogonalProjection::update(const BoundingShape &shape) {
23 switch (shape.shapeType()) {
24 case BoundingShapeType::SPHERE: {
25 // sphere projection is a circle
26 type = OrthogonalProjection::Type::CIRCLE;
27 auto *sphere = dynamic_cast<const BoundingSphere *>(&shape);
28 auto sphereCenter = sphere->getCenterPosition();
29 points.resize(2);
30 points[0] = Vec2f(sphereCenter.x, sphereCenter.z);
31 // note: second point stores the squared radius
32 points[1] = Vec2f(sphere->radius() * sphere->radius(), 0);
33 break;
34 }
35 case BoundingShapeType::BOX: {
36 // box projection is a rectangle
37 type = OrthogonalProjection::Type::RECTANGLE;
38 auto *box = dynamic_cast<const BoundingBox *>(&shape);
39 // generate the 4 corners of the box in the xz-plane.
40 // for this just take min/max of the box vertices as bounds,
41 // this is not a perfect fit but fast.
42 auto *vertices_3D = box->boxVertices();
43 Vec2f min(vertices_3D[0].x, vertices_3D[0].z);
44 Vec2f max(vertices_3D[0].x, vertices_3D[0].z);
45 for (int i = 1; i < 8; i++) {
46 min.x = std::min(min.x, vertices_3D[i].x);
47 min.y = std::min(min.y, vertices_3D[i].z);
48 max.x = std::max(max.x, vertices_3D[i].x);
49 max.y = std::max(max.y, vertices_3D[i].z);
50 }
51 points.resize(4);
52 // bottom-left
53 points[0] = min;
54 // bottom-right
55 points[1] = Vec2f(max.x, min.y);
56 // top-right
57 points[2] = max;
58 // top-left
59 points[3] = Vec2f(min.x, max.y);
60 // axes of the rectangle (and quad)
61 axes = {
62 Axis(Vec2f(1, 0)),
63 Axis(Vec2f(0, 1)),
64 Axis(points[1] - points[0]),
65 Axis(points[3] - points[0])
66 };
67 for (auto &axis: axes) {
68 auto [axisMin, axisMax] = project<4>(points, axis.dir);
69 axis.min = axisMin;
70 axis.max = axisMax;
71 }
72 break;
73 }
74 case BoundingShapeType::FRUSTUM: {
75 auto *frustum = dynamic_cast<const Frustum *>(&shape);
76 if (frustum->fov > 0.0) {
77 // TODO: Make a nicely fitting 2D projection of frustum if possible.
78 // But triangle is not always possible, i.e. for up and down directions.

91void OrthogonalProjection::makeParallelProjection(const Frustum &frustum) {
92 // approximate the frustum with a rectangle.
93 // To this end start with the 4 corners of the far plane,
94 // and then extend the rectangle to include the near plane.
95 // This should be a perfect fit in case of parallel projections.
96 type = OrthogonalProjection::Type::RECTANGLE;
97 auto *planePoints = frustum.points;
98 points.resize(4);
99 // the center points of near/far plane projection
100 Vec2f nearCenter_2D(0.0f), farCenter_2D(0.0f);
101 // the near plane points projected to the xz-plane
102 std::array<Vec2f, 4> nearPoints_2D;
103 for (int i = 0; i < 4; i++) {
104 // initialize with the far plane points
105 points[i] = Vec2f(planePoints[i + 4].x, planePoints[i + 4].z);
106 // project the near plane points
107 nearPoints_2D[i] = Vec2f(planePoints[i].x, planePoints[i].z);
108 farCenter_2D += points[i];
109 nearCenter_2D += nearPoints_2D[i];
110 }
111 farCenter_2D /= 4.0f;
112 nearCenter_2D /= 4.0f;
113 // distance between 2D projections of near/far plane centers
114 auto centerDistance = (farCenter_2D - nearCenter_2D).length();
115 // next find the two near plane points that are the farthest away from the far plane center
116 std::array<std::pair<float, int>, 4> distances;
117 for (int i = 0; i < 4; ++i) {
118 distances[i] = {(nearPoints_2D[i] - farCenter_2D).length(), i};
119 }
120 std::sort(distances.begin(), distances.end(), [](const auto &a, const auto &b) {
121 return a.first < b.first;
122 });
123 auto x_i1 = std::min(distances[0].second, distances[1].second);
124 auto x_i2 = std::max(distances[0].second, distances[1].second);
125 if (x_i1 == 0) {
126 auto buf = x_i1;
127 x_i1 = x_i2;
128 x_i2 = buf;
129 }
130 // x_i3/4 are the indices where the corner points of far plane must be adjusted
131 auto x_i4 = (x_i1 - 1) % 4;
132 auto x_i3 = (x_i2 + 1) % 4;
133 // edge length of near plane in 2D
134 auto near_d1 = (nearPoints_2D[x_i3] - nearPoints_2D[x_i1]).length();
135 // edge length of far plane in 2D
136 auto far_d1 = (points[x_i3] - points[x_i1]).length();
137 // compute adjusted edge length of the rectangle
138 auto edgeLength = std::max(far_d1, centerDistance + (far_d1 + near_d1) * 0.5f);
139 // finally scale the rectangle to include the near plane
140 auto dir = (points[x_i3] - points[x_i1]).normalize();
141 points[x_i4] = points[x_i1] + dir * edgeLength;
142 points[x_i3] = points[x_i2] + dir * edgeLength;
143 // axes of the rectangle
144 axes = {
145 Axis(Vec2f(1, 0)),
146 Axis(Vec2f(0, 1)),
147 Axis(points[1] - points[0]),
148 Axis(points[3] - points[0])
149 };
150 for (auto &axis: axes) {
151 auto [axisMin, axisMax] = project<4>(points, axis.dir);
152 axis.min = axisMin;
153 axis.max = axisMax;
154 }
155}

157void OrthogonalProjection::makePerspectiveProjection(const Frustum &frustum) {
158 type = OrthogonalProjection::Type::TRIANGLE;
159 points.resize(3);
160 // first point: origin of the frustum
161 auto basePoint = frustum.translation();
162 auto *farPlanePoints = frustum.points;
163 points[0] = Vec2f(basePoint.x, basePoint.z);
164 // Project far plane points onto the xz plane
165 std::array<Vec2f, 4> farPoints2D;
166 Vec2f farPlaneCenter2D;
167 for (int i = 0; i < 4; ++i) {
168 farPoints2D[i] = Vec2f(farPlanePoints[i + 4].x, farPlanePoints[i + 4].z);
169 farPlaneCenter2D += farPoints2D[i];
170 }
171 farPlaneCenter2D *= 0.25f; // Center of all far frustum points
172 auto baseToCenter = farPlaneCenter2D - points[0];
173 // Compute the angles relative to the center
174 std::array<std::pair<float, int>, 4> angles;
175 for (int i = 0; i < 4; ++i) {
176 auto baseToPt = farPoints2D[i] - points[0];
177 angles[i] = {std::acos(
178 baseToCenter.dot(baseToPt) /
179 (baseToCenter.length() * baseToPt.length())), i};
180 }
181 // Sort the angles to find the maximum
182 std::sort(angles.begin(), angles.end(), [](const auto &a, const auto &b) {
183 return a.first < b.first;
184 });
185 points[1] = farPoints2D[angles[0].second];
186 points[2] = farPoints2D[angles[1].second];
187 if ((points[1] - points[2]).length() < std::numeric_limits<float>::epsilon()) {
188 // If the two points are the same, switch the last two point
189 points[2] = farPoints2D[angles[2].second];
190 }
191
192 // axes of the triangle (and quad)
193 axes = {
194 Axis(Vec2f(1, 0)),
195 Axis(Vec2f(0, 1)),
196 Axis(points[1] - points[0]),
197 Axis(points[2] - points[1]),
198 Axis(points[0] - points[2])
199 };
200 for (auto &axis: axes) {
201 auto [axisMin, axisMax] = project<3>(points, axis.dir);
202 axis.min = axisMin;
203 axis.max = axisMax;
204 }
205}
