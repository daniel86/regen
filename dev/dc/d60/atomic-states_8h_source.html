402 explicit ClearColorState(const ref_ptr<FBO> &fbo)
403 : ServerSideState(), fbo_(fbo) {}
404
405 // override
406 void enable(RenderState *rs) override {
407 for (auto & it : data) {
408 if (!rs->drawFrameBuffer().isLocked()) {
409 fbo_->drawBuffers().push(it.colorBuffers);
410 rs->clearColor().push(it.clearColor);
411 glClear(GL_COLOR_BUFFER_BIT);
412 rs->clearColor().pop();
413 fbo_->drawBuffers().pop();
414 }
415 }
416 }
