33 // Override
34 void processInput(
35 SceneParser *parser,
36 SceneInputNode &input,
37 const ref_ptr<StateNode> &parent) {
38 if (!input.hasAttribute("shader")) {
39 REGEN_WARN("Missing shader attribute for " << input.getDescription() << ".");
40 return;
41 }
42 ref_ptr<LightPass> x = ref_ptr<LightPass>::alloc(
43 input.getValue<Light::Type>("type", Light::SPOT),
44 input.getValue("shader"));
45 parent->state()->joinStates(x);
46
47 x->setShadowFiltering(input.getValue<ShadowFilterMode>(
48 "shadow-filter", SHADOW_FILTERING_NONE));
49 bool useShadows = false, toggle = true;
50
51 const list<ref_ptr<SceneInputNode> > &childs = input.getChildren();
52 for (auto it = childs.begin(); it != childs.end(); ++it) {
53 ref_ptr<SceneInputNode> n = *it;
54
55 ref_ptr<Light> light = parser->getResources()->getLight(parser, n->getName());
56 if (light.get() == nullptr) {
57 continue;
58 }
59 list<ref_ptr<ShaderInput> > inputs;
60
61 ref_ptr<Texture> shadowMap;
62 ref_ptr<Texture> shadowColorMap;
63 ref_ptr<LightCamera> shadowCamera;
64 if (n->hasAttribute("shadow-camera")) {
65 shadowCamera = ref_ptr<LightCamera>::dynamicCast(
66 parser->getResources()->getCamera(parser, n->getValue("shadow-camera")));
67 if (shadowCamera.get() == nullptr) {
68 REGEN_WARN("Unable to find LightCamera for '" << n->getDescription() << "'.");
69 }
70 }
71 if (n->hasAttribute("shadow-buffer") || n->hasAttribute("shadow-texture")) {
72 shadowMap = TextureStateProvider::getTexture(parser, *n.get(),
73 "shadow-texture", "shadow-buffer",
74 "shadow-attachment");
75 if (shadowMap.get() == nullptr) {
76 REGEN_WARN("Unable to find ShadowMap for '" << n->getDescription() << "'.");
77 }
78 }
79 if ((n->hasAttribute("shadow-buffer") && n->hasAttribute("shadow-color-attachment")) ||
80 n->hasAttribute("shadow-color-texture")) {
81 shadowColorMap = TextureStateProvider::getTexture(parser, *n.get(),
82 "shadow-color-texture", "shadow-buffer",
83 "shadow-color-attachment");
84 }
85 if (toggle) {
86 useShadows = (shadowMap.get() != nullptr);
87 toggle = false;
88 } else if ((shadowMap.get() != nullptr) != useShadows) {
89 REGEN_WARN("Ignoring " << input.getDescription() << ".");
90 continue;
91 }
92
93 // Each light pass can have a set of ShaderInput's
94 const list<ref_ptr<SceneInputNode> > &childs = n->getChildren();
95 for (auto it = childs.begin(); it != childs.end(); ++it) {
96 ref_ptr<SceneInputNode> m = *it;
97 if (m->getCategory() == "input") {
98 inputs.push_back(InputStateProvider::createShaderInput(parser, *m.get(), x));
99 } else {
100 REGEN_WARN("Unhandled node " << m->getDescription() << ".");
101 }
102 }
103
104 x->addLight(light, shadowCamera, shadowMap, shadowColorMap, inputs);
105 }
106
107 StateConfigurer shaderConfigurer;
108 shaderConfigurer.addNode(parent.get());
109 x->createShader(shaderConfigurer.cfg());
110 }
111 };
112 }
113}
