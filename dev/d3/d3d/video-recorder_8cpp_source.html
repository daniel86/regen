65void VideoRecorder::initialize() {
66 auto &tex = fbo_->colorTextures()[attachment_ - GL_COLOR_ATTACHMENT0];
67 auto width = static_cast<int>(tex->width());
68 auto height = static_cast<int>(tex->height());
69 auto fps = static_cast<int>(encoder_->fps());
70
71 avformat_alloc_output_context2(&formatCtx_, nullptr, nullptr, filename_.c_str());
72 if (!formatCtx_) {
73 throw std::runtime_error("Could not allocate format context");
74 }
75 auto *codec = avcodec_find_encoder(AV_CODEC_ID_H264);
76 //auto *codec = avcodec_find_encoder(AV_CODEC_ID_HEVC);
77 //auto *codec = avcodec_find_encoder(AV_CODEC_ID_VP9);
78 if (!codec) {
79 throw std::runtime_error("Codec not found");
80 }
81 codecCtx_ = avcodec_alloc_context3(codec);
82 if (!codecCtx_) {
83 throw std::runtime_error("Could not allocate codec context");
84 }
85
86 // Create the codec context
87 codecCtx_->width = width;
88 codecCtx_->height = height;
89 // for a fixed frame rate, set time_base to {1, fps}
90 //codecCtx_->time_base = {1, fps};
91 codecCtx_->time_base = {1, 1000};
92 codecCtx_->framerate = {fps, 1};
93 codecCtx_->pix_fmt = AV_PIX_FMT_YUV420P;
94 if (formatCtx_->oformat->flags & AVFMT_GLOBALHEADER) {
95 codecCtx_->flags |= AV_CODEC_FLAG_GLOBAL_HEADER;
96 }
97 if (avcodec_open2(codecCtx_, codec, nullptr) < 0) {
98 throw std::runtime_error("Could not open codec");
99 }
100
101 stream_ = avformat_new_stream(formatCtx_, codec);
102 if (!stream_) {
103 throw std::runtime_error("Could not create stream");
104 }
105 avcodec_parameters_from_context(stream_->codecpar, codecCtx_);
106 if (!(formatCtx_->oformat->flags & AVFMT_NOFILE)) {
107 if (avio_open(&formatCtx_->pb, filename_.c_str(), AVIO_FLAG_WRITE) < 0) {
108 throw std::runtime_error("Could not open output file");
109 }
110 }
111 if (avformat_write_header(formatCtx_, nullptr) < 0) {
112 throw std::runtime_error("Error occurred when opening output file");
113 }
114 encoder_->setOutputStream(stream_, formatCtx_, codecCtx_);
115
116 // Create PBOs
117 pbo_ = ref_ptr<PBO>::alloc(2);
118 for (int i = 0; i < 2; ++i) {
119 pbo_->bindPackBuffer(i);
120 glBufferData(GL_PIXEL_PACK_BUFFER, frameSize_, nullptr, GL_STREAM_READ);
121 }
122 glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
123
124 // create a FBO for the encoder with the same size as the input FBO,
125 // but with a single color attachment that has the same format as used by the encoder
126 encoderFBO_ = ref_ptr<FBO>::alloc(tex->width(), tex->height());
127 encoderFBO_->addTexture(1,
128 GL_TEXTURE_2D, GL_RGB, GL_RGB, GL_UNSIGNED_BYTE);
129 GL_ERROR_LOG();
130}

137void VideoRecorder::updateFrameBuffer() {
138 auto &tex = fbo_->colorTextures()[attachment_ - GL_COLOR_ATTACHMENT0];
139 auto texWidth = static_cast<int>(tex->width());
140 auto texHeight = static_cast<int>(tex->height());
141 int nextIndex = (pboIndex_ + 1) % 2;
142 GLenum filerMode = (texWidth == codecCtx_->width && texHeight == codecCtx_->height) ? GL_NEAREST : GL_LINEAR;
143
144 // blit the input FBO to the encoder FBO
145 RenderState::get()->readFrameBuffer().push(fbo_->id());
146 fbo_->readBuffer().push(attachment_);
147 RenderState::get()->drawFrameBuffer().push(encoderFBO_->id());
148 encoderFBO_->drawBuffers().push(GL_COLOR_ATTACHMENT0);
149 glBlitFramebuffer(0, 0, texWidth, texHeight,
150 0, 0, codecCtx_->width, codecCtx_->height,
151 GL_COLOR_BUFFER_BIT, filerMode);
152 encoderFBO_->drawBuffers().pop();
153 RenderState::get()->drawFrameBuffer().pop();
154 fbo_->readBuffer().pop();
155 RenderState::get()->readFrameBuffer().pop();
156
157 // set the target framebuffer to read
158 RenderState::get()->readFrameBuffer().push(encoderFBO_->id());
159 encoderFBO_->readBuffer().push(GL_COLOR_ATTACHMENT0);
160
161 // read pixels from framebuffer to PBO glReadPixels() should return immediately.
162 pbo_->bindPackBuffer(pboIndex_);
163 glReadPixels(0, 0,
164 codecCtx_->width, codecCtx_->height,
165 GL_RGB, GL_UNSIGNED_BYTE,
166 nullptr);
167
168 // map the other PBO to process its data
169 pbo_->bindPackBuffer(nextIndex);
170 auto *ptr = (GLubyte *) glMapBuffer(GL_PIXEL_PACK_BUFFER, GL_READ_ONLY);
171 if (ptr) {
172 auto nextFrame = encoder_->reserveFrame();
173 std::memcpy(nextFrame, ptr, frameSize_);
174 encoder_->pushFrame(nextFrame, elapsedTime_);
175 glUnmapBuffer(GL_PIXEL_PACK_BUFFER);
176 }
177 glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
178
179 encoderFBO_->readBuffer().pop();
180 RenderState::get()->readFrameBuffer().pop();
181
182 pboIndex_ = nextIndex;
183}
