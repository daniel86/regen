31 void processInput(
32 SceneParser *parser,
33 SceneInputNode &input,
34 const ref_ptr<State> &state) override {
35 ref_ptr<Material> mat = ref_ptr<Material>::alloc();
36
37 if (input.hasAttribute("max-offset")) {
38 mat->set_maxOffset(input.getValue<GLfloat>("max-offset", 0.1f));
39 }
40 if (input.hasAttribute("height-map-mode")) {
41 mat->set_heightMapMode(input.getValue<Material::HeightMapMode>("height-map-mode",
42 Material::HEIGHT_MAP_VERTEX));
43 }
44 if (input.hasAttribute("color-blend-mode")) {
45 mat->set_colorBlendMode(input.getValue<BlendMode>("color-blend-mode", BLEND_MODE_SRC));
46 }
47 if (input.hasAttribute("color-blend-factor")) {
48 mat->set_colorBlendFactor(input.getValue<GLfloat>("color-blending-factor", 1.0f));
49 }
50
51
52 if (input.hasAttribute("asset")) {
53 ref_ptr<AssetImporter> assetLoader =
54 parser->getResources()->getAsset(parser, input.getValue("asset"));
55 if (assetLoader.get() == nullptr) {
56 REGEN_WARN("Skipping unknown Asset for '" << input.getDescription() << "'.");
57 } else {
58 const std::vector<ref_ptr<Material> > materials = assetLoader->materials();
59 auto materialIndex = input.getValue<GLuint>("asset-index", 0u);
60 if (materialIndex >= materials.size()) {
61 REGEN_WARN("Invalid Material index '" << materialIndex <<
62 "' for Asset '" << input.getValue("asset") << "'.");
63 } else {
64 mat = materials[materialIndex];
65 }
66 }
67 } else if (input.hasAttribute("preset")) {
68 std::string presetVal(input.getValue("preset"));
69 auto variant = input.getValue<Material::Variant>("variant", 0);
70 if (presetVal == "jade") mat->set_jade(variant);
71 else if (presetVal == "ruby") mat->set_ruby(variant);
72 else if (presetVal == "chrome") mat->set_chrome(variant);
73 else if (presetVal == "gold") mat->set_gold(variant);
74 else if (presetVal == "copper") mat->set_copper(variant);
75 else if (presetVal == "silver") mat->set_silver(variant);
76 else if (presetVal == "pewter") mat->set_pewter(variant);
77 else if (presetVal == "iron") mat->set_iron(variant);
78 else if (presetVal == "steel") mat->set_steel(variant);
79 else if (presetVal == "metal") mat->set_metal(variant);
80 else if (presetVal == "leather") mat->set_leather(variant);
81 else if (presetVal == "stone") mat->set_stone(variant);
82 else if (presetVal == "wood") mat->set_wood(variant);
83 else if (presetVal == "marble") mat->set_marble(variant);
84 else {
85 mat->set_textures(presetVal, variant);
86 }
87 }
88 if (input.hasAttribute("ambient"))
89 mat->ambient()->setVertex(0,
90 input.getValue<Vec3f>("ambient", Vec3f(0.0f)));
91 if (input.hasAttribute("diffuse"))
92 mat->diffuse()->setVertex(0,
93 input.getValue<Vec3f>("diffuse", Vec3f(1.0f)));
94 if (input.hasAttribute("specular"))
95 mat->specular()->setVertex(0,
96 input.getValue<Vec3f>("specular", Vec3f(0.0f)));
97 if (input.hasAttribute("shininess"))
98 mat->shininess()->setVertex(0,
99 input.getValue<GLfloat>("shininess", 1.0f));
100 if (input.hasAttribute("emission"))
101 mat->set_emission(input.getValue<Vec3f>("emission", Vec3f(0.0f)));
102 if (input.hasAttribute("textures")) {
103 mat->set_textures(
104 input.getValue("textures"),
105 input.getValue<Material::Variant>("variant", 0));
106 }
107
108 mat->alpha()->setVertex(0,
109 input.getValue<GLfloat>("alpha", 1.0f));
110 mat->refractionIndex()->setVertex(0,
111 input.getValue<GLfloat>("refractionIndex", 0.95f));
112 mat->set_fillMode(glenum::fillMode(
113 input.getValue<std::string>("fill-mode", "FILL")));
114 if (input.getValue<bool>("two-sided", false)) {
115 // this conflicts with shadow mapping front face culling.
116 mat->set_twoSided(true);
117 }
118
119 state->joinStates(mat);
120 }
121 };
