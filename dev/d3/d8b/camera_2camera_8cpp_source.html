124 fov_->setVertex(0, 0.0f);

145bool Camera::updateView() {
146 auto posStamp = position_->stamp();
147 auto dirStamp = direction_->stamp();
148 if (posStamp == posStamp_ && dirStamp == dirStamp_) { return false; }
149 posStamp_ = posStamp;
150 dirStamp_ = dirStamp;
151
152 auto numViewLayers = view_->numArrayElements();
153 for (unsigned int i = 0; i < numViewLayers; ++i) {
154 auto dir = direction_->getVertexClamped(i);
155 if (std::abs(dir.r.dot(Vec3f::up())) > 0.999f) {
156 auto viewMatrix = Mat4f::lookAtMatrix(
157 position_->getVertexClamped(i).r,
158 dir.r, Vec3f::right());
159 view_->setVertex(i, viewMatrix);
160 viewInv_->setVertex(i, viewMatrix.lookAtInverse());
161 } else {
162 auto viewMatrix = Mat4f::lookAtMatrix(
163 position_->getVertexClamped(i).r,
164 dir.r, Vec3f::up());
165 view_->setVertex(i, viewMatrix);
166 viewInv_->setVertex(i, viewMatrix.lookAtInverse());
167 }
168 }
169
170 return true;
171}

204void Camera::updatePose() {
205 bool updated = false;
206 if (attachedPosition_.get()) {
207 if (poseStamp_ != attachedPosition_->stamp()) {
208 poseStamp_ = attachedPosition_->stamp();
209 position_->setUniformData(attachedPosition_->getVertex(0).r);
210 updated = true;
211 }
212 }
213 else if (attachedTransform_.get()) {
214 if (poseStamp_ != attachedTransform_->stamp()) {
215 poseStamp_ = attachedTransform_->stamp();
216 auto m = attachedTransform_->getVertex(0);
217 position_->setVertex(0, m.r.position());
218 if (!isAttachedToPosition_) {
219 // TODO: change camera orientation based on transform
220 //direction_->setVertex(0, (m ^ Vec4f(Vec3f::front(),0.0)).xyz_());
221 }
222 updated = true;
223 }
224 }
225
226 if (updated) {
227 updateCamera();
228 }

281bool Camera::hasHalfSphereIntersection(const Vec3f &center, GLfloat radius) const {
282 // get the distance from the camera to the center of the sphere
283 auto d = Plane(
284 position()->getVertex(0).r,
285 direction()->getVertex(0).r).distance(center);
286 // check if the sphere is outside the far plane
287 if (d - radius > far()->getVertex(0).r) return false;
288 // check if the sphere is inside the near plane
289 if (d + radius < near()->getVertex(0).r) return false;
290 // check if the sphere is inside the half sphere
291 auto halfSphereRadius = far()->getVertex(0);
292 auto halfSphereNormal = direction()->getVertex(0);
293 auto halfSphereCenter = position()->getVertex(0).r + halfSphereNormal.r * halfSphereRadius.r;
294 return Plane(halfSphereCenter, halfSphereNormal.r).distance(center) < radius;
295}

297bool Camera::hasHalfSphereIntersection(const Vec3f &center, const Vec3f *points) const {
298 // get the distance from the camera to the center of the sphere

300 position()->getVertex(0).r,
301 direction()->getVertex(0).r).distance(center);
302 // check if the sphere is outside the far plane
303 if (d > far()->getVertex(0).r) return false;
304 // check if the sphere is inside the near plane
305 if (d < near()->getVertex(0).r) return false;
306 // check if the sphere is inside the half sphere
307 auto halfSphereRadius = far()->getVertex(0);
308 auto halfSphereNormal = direction()->getVertex(0);
309 auto halfSphereCenter = position()->getVertex(0).r + halfSphereNormal.r * halfSphereRadius.r;
310 auto halfSphere = Plane(halfSphereCenter, halfSphereNormal.r);
311 for (int i = 0; i < 8; ++i) {
312 if (halfSphere.distance(center + points[i]) < 0) {
313 return false;
314 }
315 }
316 return true;
