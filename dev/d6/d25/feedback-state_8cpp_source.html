20void FeedbackSpecification::addFeedback(const ref_ptr<ShaderInput> &in) {
21 // remove if already added
22 if (feedbackAttributeMap_.count(in->name()) > 0) { removeFeedback(in.get()); }
23
24 GLuint feedbackCount = (feedbackCount_ == 0 ? in->numVertices() : feedbackCount_);
25 feedbackCount_ = feedbackCount;
26
27 // create feedback attribute
28 ref_ptr<ShaderInput> feedback = ShaderInput::create(in);
29 feedback->set_inputSize(feedbackCount * feedback->elementSize());
30 feedback->set_numVertices(feedbackCount);
31 feedbackAttributes_.push_front(feedback);
32 feedbackAttributeMap_[in->name()] = feedbackAttributes_.begin();
33
34 requiredBufferSize_ += feedback->inputSize();
35}

37void FeedbackSpecification::removeFeedback(ShaderInput *in) {
38 auto it = feedbackAttributeMap_.find(in->name());
39 if (it == feedbackAttributeMap_.end()) { return; }
40
41 ref_ptr<ShaderInput> in_ = *(it->second);
42 requiredBufferSize_ -= in_->inputSize();
43
44 feedbackAttributes_.erase(it->second);
45 feedbackAttributeMap_.erase(it);
46}

71void FeedbackState::enable(RenderState *rs) {
72 if (requiredBufferSize_ != allocatedBufferSize_) {
73 // free previously allocated data
74 if (vboRef_.get()) { regen::VBO::free(vboRef_.get()); }
75 // allocate memory and upload to GL
76 if (feedbackMode_ == GL_INTERLEAVED_ATTRIBS) {
77 vboRef_ = feedbackBuffer_->allocInterleaved(feedbackAttributes_);
78 } else {
79 vboRef_ = feedbackBuffer_->allocSequential(feedbackAttributes_);
80 }
81 bufferRange_.buffer_ = vboRef_->bufferID();
82 bufferRange_.size_ = requiredBufferSize_;
83 allocatedBufferSize_ = requiredBufferSize_;
84 }
85
86 switch (feedbackMode_) {
87 case GL_INTERLEAVED_ATTRIBS:
88 enableInterleaved(rs);
89 break;
90 default:
91 enableSeparate(rs);
92 break;
93 }
94}
