275 template<class T>
276 T sampleAverage(const Vec2f &texco, const Vec2f &regionTS, const GLubyte *textureData) const {
277 auto bounds = getRegion(texco, regionTS);
278 T avg = T(0.0f);
279 int numSamples = 0;
280 for (unsigned int y = bounds.min.y; y <= bounds.max.y; ++y) {
281 for (unsigned int x = bounds.min.x; x <= bounds.max.x; ++x) {
282 unsigned int index = (y * width() + x);
283 avg += sample<T>(index, textureData);
284 numSamples++;
285 }
286 }
287 if (numSamples > 0) {
288 return avg / static_cast<float>(numSamples);
289 } else {
290 return avg;
291 }
292 }

302 template<class T>
303 T sampleMax(const Vec2f &texco, const Vec2f &regionTS, const GLubyte *textureData) const {
304 auto bounds = getRegion(texco, regionTS);
305 T maxVal = T(0.0f);
306 for (unsigned int y = bounds.min.y; y <= bounds.max.y; ++y) {
307 for (unsigned int x = bounds.min.x; x <= bounds.max.x; ++x) {
308 unsigned int index = (y * width() + x);
309 maxVal = std::max(maxVal, sample<T>(index, textureData));
310 }
311 }

334 template<class T>
335 T sampleLinear(const Vec2f &texco, const GLubyte *textureData) const {
336 auto w = static_cast<float>(width());
337 auto h = static_cast<float>(height());
338 auto x = texco.x * w;
339 auto y = texco.y * h;
340 auto x0 = std::floor(x);
341 auto y0 = std::floor(y);
342 auto x1 = x0 + 1;
343 auto y1 = y0 + 1;
344
345 T v00 = sampleNearest<T>(Vec2f(x0 / w, y0 / h), textureData);
346 T v01 = sampleNearest<T>(Vec2f(x0 / w, y1 / h), textureData);
347 T v10 = sampleNearest<T>(Vec2f(x1 / w, y0 / h), textureData);
348 T v11 = sampleNearest<T>(Vec2f(x1 / w, y1 / h), textureData);
349
350 auto dx = x - x0;
351 auto dy = y - y0;
352 T v0 = v00 * (1.0f - dx) + v10 * dx;
353 T v1 = v01 * (1.0f - dx) + v11 * dx;
354 return v0 * (1.0f - dy) + v1 * dy;
355 }

400 template<class T>
401 T sample(unsigned int texelIndex, const GLubyte *textureData) const {
402 auto numComponents = glenum::pixelComponents(format());
403 auto *dataOffset = textureData + texelIndex * numComponents;
404 T v(0.0f);
405 auto *typedData = (float *) &v;
406 for (unsigned int i = 0; i < numComponents; ++i) {
407 typedData[i] = static_cast<float>(dataOffset[i]) / 255.0f;
408 }
409 return v;
410 }
