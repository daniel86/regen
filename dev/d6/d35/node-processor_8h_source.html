93 static void handleAttributes(
94 scene::SceneLoader *scene,
95 SceneInputNode &input,
96 const ref_ptr<StateNode> &newNode) {
97 if (newNode.get() != nullptr && input.hasAttribute("name")) {
98 newNode->set_name(input.getValue("name"));
99 auto newNodeID = scene->putNamedObject(newNode);
100 auto u_objectID = ref_ptr<ShaderInput1i>::alloc("objectID");
101 u_objectID->setUniformData(newNodeID);
102 newNode->state()->joinShaderInput(u_objectID);
103 }
104 if (input.hasAttribute("sort")) {
105 // Sort node children by model view matrix.
106 auto sortMode = input.getValue<GLuint>("sort", 0);
107 ref_ptr<Camera> sortCam =
108 scene->getResource<Camera>(input.getValue<std::string>("sort-camera", ""));
109 if (sortCam.get() == nullptr) {
110 REGEN_WARN("Unable to find Camera for '" << input.getDescription() << "'.");
111 } else {
112 newNode->state()->joinStatesFront(
113 ref_ptr<SortByModelMatrix>::alloc(newNode, sortCam, (sortMode == 1)));
114 }

118 static void handleChildren(
119 scene::SceneLoader *scene,
120 SceneInputNode &input,
121 const ref_ptr<StateNode> &newNode) {
122 // Process node children
123 const std::list<ref_ptr<SceneInputNode> > &childs = input.getChildren();
124 for (auto it = childs.begin(); it != childs.end(); ++it) {
125 const ref_ptr<SceneInputNode> &x = *it;
126 // First try node processor
127 ref_ptr<NodeProcessor> nodeProcessor = scene->getNodeProcessor(x->getCategory());
128 if (nodeProcessor.get() != nullptr) {
129 nodeProcessor->processInput(scene, *x.get(), newNode);
130 continue;
131 }
132 // Second try state processor
133 ref_ptr<StateProcessor> stateProcessor = scene->getStateProcessor(x->getCategory());
134 if (stateProcessor.get() != nullptr) {
135 stateProcessor->processInput(scene, *x.get(), newNode, newNode->state());
136 continue;
137 }
138 REGEN_WARN("No processor registered for '" << x->getDescription() << "'.");
139 }

179 static ref_ptr<StateNode> createCullNode(
180 scene::SceneLoader *parser,
181 SceneInputNode &input,

183 ref_ptr<GeometricCulling> cullNode;
184 // get the parent camera. Note that this will be the light camera in case
185 // updating the shadow map.
186 auto cam = parent->getParentCamera();
187 if (cam.get() == nullptr) {
188 REGEN_WARN("No Camera can be found for '" << input.getDescription() << "'.");
189 return cullNode;
190 }
191 auto spatialIndex = getSpatialIndex(parser, input);
192 if (spatialIndex.get() == nullptr) {
193 REGEN_WARN("No SpatialIndex can be found for '" << input.getDescription() << "'.");
194 return cullNode;
195 }
196
197 auto shapeName = input.getValue<std::string>("cull-shape", "");
198 cullNode = ref_ptr<GeometricCulling>::alloc(cam, spatialIndex, shapeName);
199 cullNode->set_name(input.getName());
200 if (input.hasAttribute("sort-mode")) {
201 cullNode->setInstanceSortMode(input.getValue<SortMode>("sort-mode", SortMode::FRONT_TO_BACK));
202 }
203 parent->addChild(cullNode);
204 parser->putNode(input.getName(), cullNode);
205
206 return cullNode;
