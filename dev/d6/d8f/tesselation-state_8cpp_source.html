47TesselationState::TesselationState(GLuint numPatchVertices)
48 : State(),
49 lodMetric_(CAMERA_DISTANCE),

55#else
56 REGEN_WARN("GL_ARB_tessellation_shader not supported.");

63 outerLevel_->setUniformData(Vec4f(8.0f));
64
65 lodFactor_ = ref_ptr<ShaderInput1f>::alloc("lodFactor");
66 lodFactor_->setUniformData(4.0f);
67 joinShaderInput(lodFactor_);
68
69 joinStates(ref_ptr<PatchVerticesState>::alloc(numPatchVertices_));

72void TesselationState::set_lodMetric(LoDMetric metric) {
73 lodMetric_ = metric;
74
75 if (tessLevelSetter_.get()) {
76 disjoinStates(tessLevelSetter_);
77 tessLevelSetter_ = ref_ptr<State>();
78 }
79
80 shaderDefine("TESS_IS_ADAPTIVE",
81 lodMetric_ == FIXED_FUNCTION ? "FALSE" : "TRUE");
82 switch (lodMetric_) {
83 case FIXED_FUNCTION:
84 shaderDefine("TESS_LOD", "FIXED_FUNCTION");
85 tessLevelSetter_ = ref_ptr<PatchLevelState>::alloc(innerLevel_, outerLevel_);
86 joinStates(tessLevelSetter_);
87 break;
88 case EDGE_SCREEN_DISTANCE:
89 shaderDefine("TESS_LOD", "EDGE_SCREEN_DISTANCE");
90 break;
91 case EDGE_DEVICE_DISTANCE:
92 shaderDefine("TESS_LOD", "EDGE_DEVICE_DISTANCE");
93 break;
94 case CAMERA_DISTANCE:
95 shaderDefine("TESS_LOD", "CAMERA_DISTANCE");
96 break;
97 }
98}
