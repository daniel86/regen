401 inline void rotate(GLfloat angle, GLfloat x_, GLfloat y_, GLfloat z_) {
402 GLfloat c = cos(angle);
403 GLfloat s = sin(angle);
404 Vec3<GLfloat> rotated(
405 (x_ * x_ * (1 - c) + c) * x
406 + (x_ * y_ * (1 - c) - z_ * s) * y
407 + (x_ * z_ * (1 - c) + y_ * s) * z,
408 (y_ * x_ * (1 - c) + z_ * s) * x
409 + (y_ * y_ * (1 - c) + c) * y
410 + (y_ * z_ * (1 - c) - x_ * s) * z,
411 (x_ * z_ * (1 - c) - y_ * s) * x
412 + (y_ * z_ * (1 - c) + x_ * s) * y
413 + (z_ * z_ * (1 - c) + c) * z
414 );
415 x = rotated.x;
416 y = rotated.y;
417 z = rotated.z;
418 }
