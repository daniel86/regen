15Atmosphere::Atmosphere(
16 const ref_ptr<Sky> &sky,
17 unsigned int cubeMapSize,
18 bool useFloatBuffer,
19 unsigned int levelOfDetail)
20 : SkyLayer(sky) {
21 updateMesh_ = Rectangle::getUnitQuad();
22
23 state()->joinStates(ref_ptr<BlendState>::alloc(GL_SRC_ALPHA, GL_ONE));
24
25 ref_ptr<TextureCube> cubeMap = ref_ptr<TextureCube>::alloc(1);
26 cubeMap->begin(RenderState::get());
27 cubeMap->set_format(GL_RGBA);
28 if (useFloatBuffer) {
29 cubeMap->set_internalFormat(GL_RGBA16F);
30 } else {
31 cubeMap->set_internalFormat(GL_RGBA);
32 }
33 cubeMap->filter().push(GL_LINEAR);
34 cubeMap->set_rectangleSize(cubeMapSize, cubeMapSize);
35 cubeMap->wrapping().push(GL_CLAMP_TO_EDGE);
36 cubeMap->texImage();
37 cubeMap->end(RenderState::get());
38
39 // create render target for updating the sky cube map
40 fbo_ = ref_ptr<FBO>::alloc(cubeMapSize, cubeMapSize);
41 RenderState::get()->drawFrameBuffer().push(fbo_->id());
42 fbo_->drawBuffers().push(DrawBuffers::attachment0());
43 // clear negative y to black, -y cube face is not updated
44 glFramebufferTexture2D(GL_DRAW_FRAMEBUFFER, GL_COLOR_ATTACHMENT0,
45 GL_TEXTURE_CUBE_MAP_NEGATIVE_Y, cubeMap->id(), 0);
46 glClear(GL_COLOR_BUFFER_BIT);
47 // for updating bind all layers to GL_COLOR_ATTACHMENT0
48 glFramebufferTexture(GL_DRAW_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, cubeMap->id(), 0);
49 RenderState::get()->drawFrameBuffer().pop();
50
51 drawState_ = ref_ptr<SkyBox>::alloc(levelOfDetail);
52 drawState_->setCubeMap(cubeMap);
53 //state()->joinStates(drawState_);
54
58 rayleigh_ = ref_ptr<ShaderInput3f>::alloc("rayleigh");
59 rayleigh_->setUniformData(Vec3f(0.0f));
60 mie_ = ref_ptr<ShaderInput4f>::alloc("mie");
61 mie_->setUniformData(Vec4f(0.0f));
62 spotBrightness_ = ref_ptr<ShaderInput1f>::alloc("spotBrightness");
63 spotBrightness_->setUniformData(0.0f);
64 scatterStrength_ = ref_ptr<ShaderInput1f>::alloc("scatterStrength");
65 scatterStrength_->setUniformData(0.0f);
66 skyAbsorption_ = ref_ptr<ShaderInput3f>::alloc("skyAbsorption");
67 skyAbsorption_->setUniformData(Vec3f(0.0f));
71 auto atmosphereUBO = ref_ptr<UniformBlock>::alloc("Atmosphere");
72 atmosphereUBO->addUniform(sky->sun()->direction(), "sunDir");
73 atmosphereUBO->addUniform(mie_);
74 atmosphereUBO->addUniform(rayleigh_);
75 atmosphereUBO->addUniform(spotBrightness_);
76 atmosphereUBO->addUniform(skyAbsorption_);
77 atmosphereUBO->addUniform(scatterStrength_);
78 updateState_->joinShaderInput(atmosphereUBO);
79 updateShader_ = ref_ptr<ShaderState>::alloc();
80 updateState_->joinStates(updateShader_);
81 updateState_->joinShaderInput(sky_->worldTime()->in);
82}
