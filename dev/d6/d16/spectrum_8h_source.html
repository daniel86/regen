82/* UPVP_TO_XY
83
84 Given 1976 coordinates u', v', determine 1931 chromaticities x, y
85
86*/
87
88void upvp_to_xy(double up, double vp, double *xc, double *yc)
89{
90 *xc = (9 * up) / ((6 * up) - (16 * vp) + 12);
91 *yc = (4 * vp) / ((6 * up) - (16 * vp) + 12);
92}

95

106/* XYZ_TO_RGB
107
108 Given an additive tricolour system CS, defined by the CIE x
109 and y chromaticities of its three primaries (z is derived
110 trivially as 1-(x+y)), and a desired chromaticity (XC, YC,
111 ZC) in CIE space, determine the contribution of each
112 primary in a linear combination which sums to the desired
113 chromaticity. If the requested chromaticity falls outside
114 the Maxwell triangle (colour gamut) formed by the three
115 primaries, one of the r, g, or b weights will be negative.
116
117 Caller can use constrain_rgb() to desaturate an
118 outside-gamut colour to the closest representation within
119 the available gamut and/or norm_rgb to normalise the RGB
120 components so the largest nonzero component has value 1.
121
122*/
123
124void xyz_to_rgb(struct colourSystem *cs,
125 double xc, double yc, double zc,
126 double *r, double *g, double *b)
127{
128 double xr, yr, zr, xg, yg, zg, xb, yb, zb;
129 double xw, yw, zw;
130 double rx, ry, rz, gx, gy, gz, bx, by, bz;
131 double rw, gw, bw;
132
133 xr = cs->xRed; yr = cs->yRed; zr = 1 - (xr + yr);
134 xg = cs->xGreen; yg = cs->yGreen; zg = 1 - (xg + yg);
135 xb = cs->xBlue; yb = cs->yBlue; zb = 1 - (xb + yb);
136
137 xw = cs->xWhite; yw = cs->yWhite; zw = 1 - (xw + yw);
138
139 /* xyz -> rgb matrix, before scaling to white. */
140
141 rx = (yg * zb) - (yb * zg); ry = (xb * zg) - (xg * zb); rz = (xg * yb) - (xb * yg);
142 gx = (yb * zr) - (yr * zb); gy = (xr * zb) - (xb * zr); gz = (xb * yr) - (xr * yb);
143 bx = (yr * zg) - (yg * zr); by = (xg * zr) - (xr * zg); bz = (xr * yg) - (xg * yr);
144
145 /* White scaling factors.
146 Dividing by yw scales the white luminance to unity, as conventional. */
147
148 rw = ((rx * xw) + (ry * yw) + (rz * zw)) / yw;
149 gw = ((gx * xw) + (gy * yw) + (gz * zw)) / yw;
150 bw = ((bx * xw) + (by * yw) + (bz * zw)) / yw;
151
152 /* xyz -> rgb matrix, correctly scaled to white. */
153
154 rx = rx / rw; ry = ry / rw; rz = rz / rw;
155 gx = gx / gw; gy = gy / gw; gz = gz / gw;
156 bx = bx / bw; by = by / bw; bz = bz / bw;
157
158 /* rgb of the desired point */
159
160 *r = (rx * xc) + (ry * yc) + (rz * zc);
161 *g = (gx * xc) + (gy * yc) + (gz * zc);
162 *b = (bx * xc) + (by * yc) + (bz * zc);
163}

165/* INSIDE_GAMUT
166
167 Test whether a requested colour is within the gamut
168 achievable with the primaries of the current colour
169 system. This amounts simply to testing whether all the
170 primary weights are non-negative. */
171
172int inside_gamut(double r, double g, double b)
173{
174 return (r >= 0) && (g >= 0) && (b >= 0);
175}

177/* CONSTRAIN_RGB
178
179 If the requested RGB shade contains a negative weight for
180 one of the primaries, it lies outside the colour gamut
181 accessible from the given triple of primaries. Desaturate
182 it by adding white, equal quantities of R, G, and B, enough
183 to make RGB all positive. The function returns 1 if the
184 components were modified, zero otherwise.
185
186*/
187
188int constrain_rgb(double *r, double *g, double *b)
189{
190 double w;
191
192 /* Amount of white needed is w = - min(0, *r, *g, *b) */
193
194 w = (0 < *r) ? 0 : *r;
195 w = (w < *g) ? w : *g;
196 w = (w < *b) ? w : *b;
197 w = -w;
198
199 /* Add just enough white to make r, g, b all positive. */
200
201 if (w > 0) {
202 *r += w; *g += w; *b += w;
203 return 1; /* Colour modified to fit RGB gamut */
204 }
205
206 return 0; /* Colour within RGB gamut */
207}

209/* GAMMA_CORRECT_RGB
210
211 Transform linear RGB values to nonlinear RGB values. Rec.
212 709 is ITU-R Recommendation BT. 709 (1990) ``Basic
213 Parameter Values for the HDTV Standard for the Studio and
214 for International Programme Exchange'', formerly CCIR Rec.
215 709. For details see
216
217 http://www.poynton.com/ColorFAQ.html
218 http://www.poynton.com/GammaFAQ.html
219*/
220
221void gamma_correct(const struct colourSystem *cs, double *c)
222{
223 double gamma;
224
225 gamma = cs->gamma;
226
227 if (gamma == GAMMA_REC709) {
228 /* Rec. 709 gamma correction. */
229 double cc = 0.018;
230
231 if (*c < cc) {
232 *c *= ((1.099 * pow(cc, 0.45)) - 0.099) / cc;
233 } else {
234 *c = (1.099 * pow(*c, 0.45)) - 0.099;
235 }
236 } else {
237 /* Nonlinear colour = (Linear colour)^(1/gamma) */
238 *c = pow(*c, 1.0 / gamma);
239 }
240}
241
242void gamma_correct_rgb(const struct colourSystem *cs, double *r, double *g, double *b)
243{
244 gamma_correct(cs, r);
245 gamma_correct(cs, g);
246 gamma_correct(cs, b);
247}

249/* NORM_RGB
250
251 Normalise RGB components so the most intense (unless all
252 are zero) has a value of 1.
253
254*/
255
256void norm_rgb(double *r, double *g, double *b)
257{
258#define Max(a, b) (((a) > (b)) ? (a) : (b))
259 double greatest = Max(*r, Max(*g, *b));
260
261 if (greatest > 0) {
262 *r /= greatest;
263 *g /= greatest;
264 *b /= greatest;
265 }
266#undef Max
267}

348/* BB_SPECTRUM
349
350 Calculate, by Planck's radiation law, the emittance of a black body
351 of temperature bbTemp at the given wavelength (in metres). */
352
353double bbTemp = 5000; /* Hidden temperature argument
354 to BB_SPECTRUM. */
355double bb_spectrum(double wavelength)
356{
357 double wlm = wavelength * 1e-9; /* Wavelength in meters */
358
359 return (3.74183e-16 * pow(wlm, -5.0)) /
360 (exp(1.4388e-2 / (wlm * bbTemp)) - 1.0);
361}

363void xyz_to_lms(double x, double y, double z, double* l, double* m, double* s)
364{
365 *l = 0.3897*x + 0.6890*y - 0.0787*z;
366 *m = -0.2298*x + 1.1834*y + 0.0464*z;
367 *s = z;
368}

377void spectrum(double t1, double t2, int N, unsigned char* d)
378{
379 int i,j,dj;
380 double X,Y,Z,R,G,B,L,M,S, Lw=0.0, Mw=0.0, Sw=0.0;
381 struct colourSystem *cs = &CIEsystem;
382
383 j = 0; dj = 1;
384 if (t1<t2) {
385 double t = t1;
386 t1 = t2;
387 t2 = t;
388 j = N-1; dj=-1;
389 }
390
391 for (i=0; i<N; i++) {
392 bbTemp = t1 + (t2-t1)/N*i;
393
394 // integrate blackbody radiation spectrum to XYZ
395 spectrum_to_xyz(bb_spectrum, &X, &Y, &Z);
396
397 // normalize highest temperature to white (in LMS system)
398 xyz_to_lms(X,Y,Z,&L,&M,&S);
399 if (i==0) {
400 Lw=1/L; Mw=1/M; Sw=1/S;
401 }
402 L *= Lw; M *= Mw; S *= Sw;
403 lms_to_xyz(L,M,S,&X,&Y,&Z);
404
405 // convert to RGB
406 xyz_to_rgb(cs, X, Y, Z, &R, &G, &B);
407 constrain_rgb(&R, &G, &B);
408 norm_rgb(&R, &G, &B);
409 d[(j<<2)] = (unsigned char) ((double)R*255);
410 d[(j<<2)+1] = (unsigned char) ((double)G*255);
411 d[(j<<2)+2] = (unsigned char) ((double)B*255);
412 d[(j<<2)+3] = (B>0.1)? B*255 : 0;
413 j += dj;
414 }
415}
