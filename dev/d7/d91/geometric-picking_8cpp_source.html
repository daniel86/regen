58void GeomPicking::pick(RenderState *rs, GLuint feedbackCount) {
59 rs->copyReadBuffer().push(vboRef_->bufferID());
60 auto *bufferData = (PickData *) glMapBufferRange(
61 GL_COPY_READ_BUFFER,
62 vboRef_->address(),
63 bufferSize_,
64 GL_MAP_READ_BIT);
65 // find pick result with max depth (camera looks in negative z direction)
66 PickData *bestPicked = nullptr;
67 for (GLuint i = 0; i < feedbackCount; ++i) {
68 PickData &picked_x = bufferData[i];
69 if (picked_x.objectID < 0) { continue; }
70 if (!bestPicked || picked_x.depth > bestPicked->depth) {
71 bestPicked = &picked_x;
72 }
73 }
74 if (bestPicked) {
75 pickedObject_.depth = bestPicked->depth;
76 pickedObject_.instanceID = bestPicked->instanceID;
77 pickedObject_.objectID = bestPicked->objectID;
78 hasPickedObject_ = true;
79 } else {
80 hasPickedObject_ = false;
81 }
82 glUnmapBuffer(GL_COPY_READ_BUFFER);
83 rs->copyReadBuffer().pop();
84}

86void GeomPicking::updateMouse() {
87 auto inverseProjectionMatrix = camera_->projectionInverse()->getVertex(0);
88 auto mouse = mouseTexco_->getVertex(0);
89 // find view space mouse ray intersecting the frustum
90 Vec2f mouseNDC = mouse.r * 2.0 - Vec2f(1.0);
91 // in NDC space the ray starts at (mx,my,0) and ends at (mx,my,1)
92 Vec4f mouseRayNear = inverseProjectionMatrix.r ^ Vec4f(mouseNDC, 0.0, 1.0);
93 Vec4f mouseRayFar = inverseProjectionMatrix.r ^ Vec4f(mouseNDC, 1.0, 1.0);
94 mouseRayNear.xyz_() /= mouseRayNear.w;
95 mouseRayFar.xyz_() /= mouseRayFar.w;
96 mousePosVS_->setVertex(0, mouseRayNear.xyz_());
97 mouseDirVS_->setVertex(0, mouseRayNear.xyz_() - mouseRayFar.xyz_());
98}

100void GeomPicking::traverse(RenderState *rs) {
101 updateMouse();
102 state_->enable(rs);
103
104 int feedbackCount = 0;
105 GLuint feedbackQuery = 0;
106 glGenQueries(1, &feedbackQuery);
107
108 for (auto &pickableNode: childs()) {
109 auto pickableMesh = pickableNode->findStateWithType<Mesh>();
110 if (pickableMesh == nullptr) {
111 REGEN_WARN("No mesh found in pickable node.");
112 continue;
113 }
114 // update buffer range of pickable
115 bufferRange_->offset_ = feedbackCount * sizeof(PickData);
116 bufferRange_->size_ = bufferSize_ - bufferRange_->offset_;
117 bufferRange_->offset_ += vboRef_->address();
118 pickableMesh->setFeedbackRange(bufferRange_);
119 glBeginQuery(GL_PRIMITIVES_GENERATED, feedbackQuery);
120 // render pickable
121 pickableNode->traverse(rs);
122
123 glEndQuery(GL_PRIMITIVES_GENERATED);
124 feedbackCount += static_cast<int>(getGLQueryResult(feedbackQuery));
125 if (feedbackCount >= static_cast<int>(maxPickedObjects_)) {
126 break;
127 }
128 }
129
130 glDeleteQueries(1, &feedbackQuery);
131
132 if (feedbackCount > 0) {
133 auto numPicks = static_cast<GLuint>(feedbackCount);
134 pick(rs, (numPicks < maxPickedObjects_ ? numPicks : maxPickedObjects_));
135 } else {
136 hasPickedObject_ = false;
137 }
138
139 state_->disable(rs);
140 GL_ERROR_LOG();
141}
