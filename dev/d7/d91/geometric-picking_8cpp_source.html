57void GeomPicking::pick(RenderState *rs, GLuint feedbackCount) {
58 rs->copyReadBuffer().push(vboRef_->bufferID());
59 auto *bufferData = (PickData *) glMapBufferRange(
60 GL_COPY_READ_BUFFER,
61 vboRef_->address(),
62 bufferSize_,
63 GL_MAP_READ_BIT);
64 // find pick result with max depth (camera looks in negative z direction)
65 PickData *bestPicked = nullptr;
66 for (GLuint i = 0; i < feedbackCount; ++i) {
67 PickData &picked_x = bufferData[i];
68 if (picked_x.objectID<0) { continue; }
69 if (!bestPicked || picked_x.depth > bestPicked->depth) {
70 bestPicked = &picked_x;
71 }
72 }
73 if (bestPicked) {
74 pickedObject_.depth = bestPicked->depth;
75 pickedObject_.instanceID = bestPicked->instanceID;
76 pickedObject_.objectID = bestPicked->objectID;
77 hasPickedObject_ = true;
78 } else {
79 hasPickedObject_ = false;
80 }
81 glUnmapBuffer(GL_COPY_READ_BUFFER);
82 rs->copyReadBuffer().pop();
83}

85void GeomPicking::updateMouse() {
86 auto inverseProjectionMatrix = camera_->projectionInverse()->getVertex(0);
87 auto mouse = mouseTexco_->getVertex(0);
88 // find view space mouse ray intersecting the frustum
89 Vec2f mouseNDC = mouse.r * 2.0 - Vec2f(1.0);
90 // in NDC space the ray starts at (mx,my,0) and ends at (mx,my,1)
91 Vec4f mouseRayNear = inverseProjectionMatrix.r ^ Vec4f(mouseNDC, 0.0, 1.0);
92 Vec4f mouseRayFar = inverseProjectionMatrix.r ^ Vec4f(mouseNDC, 1.0, 1.0);
93 mouseRayNear.xyz_() /= mouseRayNear.w;
94 mouseRayFar.xyz_() /= mouseRayFar.w;
95 mousePosVS_->setVertex(0, mouseRayNear.xyz_());
96 mouseDirVS_->setVertex(0, mouseRayNear.xyz_() - mouseRayFar.xyz_());
97}
