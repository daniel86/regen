210 if (tex3d) {

391ref_ptr<Texture> TextureState::getTexture(
392 scene::SceneLoader *scene,
393 scene::SceneInputNode &input,
394 const std::string &idKey,
395 const std::string &bufferKey,
396 const std::string &attachmentKey) {
397 ref_ptr<Texture> tex;
398 // Find the texture resource
399 if (input.hasAttribute(idKey)) {
400 tex = scene->getResource<Texture>(input.getValue(idKey));
401 } else if (input.hasAttribute(bufferKey)) {
402 ref_ptr<FBO> fbo = scene->getResource<FBO>(input.getValue(bufferKey));
403 if (fbo.get() == nullptr) {
404 REGEN_WARN("Unable to find FBO '" << input.getValue(bufferKey) <<
405 "' for " << input.getDescription() << ".");
406 return tex;
407 }
408 const auto val = input.getValue<std::string>(attachmentKey, "0");
409 if (val == "depth") {
410 tex = fbo->depthTexture();
411 }
412 else if (val == "stencil") {
413 if (fbo->stencilTexture().get()) {
414 tex = fbo->stencilTexture();
415 } else {
416 tex = fbo->depthStencilTexture();
417 }
418 }
419 else {
420 std::vector<ref_ptr<Texture> > &textures = fbo->colorTextures();
421
422 unsigned int attachment;
423 std::stringstream ss(val);
424 ss >> attachment;
425
426 if (attachment < textures.size()) {
427 tex = textures[attachment];
428 } else {
429 REGEN_WARN("Invalid attachment '" << val <<
430 "' for " << input.getDescription() << ".");
431 }
432 }
433 }
434 if (!tex.get()) {
435 REGEN_WARN("No texture found for " << input.getDescription() << ".");
436 }
437 return tex;
438}

440ref_ptr<TextureState> TextureState::load(LoadingContext &ctx, scene::SceneInputNode &input) {
441 auto scene = ctx.scene();
442
443 ref_ptr<Texture> tex = getTexture(scene, input);
444 if (tex.get() == nullptr) {
445 REGEN_WARN("Skipping unidentified texture node for " << input.getDescription() << ".");
446 return {};
447 }
448 if (input.hasAttribute("mip-level") && input.getValue<int>("mip-level", 0) > 0) {
449 auto mipLevel = input.getValue<unsigned int>("mip-level", 1);
450 auto mipTex = dynamic_cast<TextureMips2D *>(tex.get());
451 if (mipTex) {
452 auto maxLevel = mipTex->numMips();
453 if (mipLevel < maxLevel) {
454 tex = mipTex->mipRefs()[mipLevel - 1];
455 } else {
456 REGEN_WARN("Mip level " << mipLevel << " is out of range for texture '" << tex->name() << "'.");
457 }
458 } else {
459 REGEN_WARN("Texture '" << tex->name() << "' is not a mip texture.");
460 }
461 }
462
463 // Set-Up the texture state
464 ref_ptr<TextureState> texState = ref_ptr<TextureState>::alloc(
465 tex, input.getValue("name"));
466
467 texState->set_ignoreAlpha(
468 input.getValue<bool>("ignore-alpha", false));
469 texState->set_mapTo(input.getValue<TextureState::MapTo>(
470 "map-to", TextureState::MAP_TO_CUSTOM));
471
472 // Describes how a texture will be mixed with existing pixels.
473 texState->set_blendMode(
474 input.getValue<BlendMode>("blend-mode", BLEND_MODE_SRC));
475 texState->set_blendFactor(
476 input.getValue<GLfloat>("blend-factor", 1.0f));
477
478 const std::string blendFunctionName = input.getValue("blend-function-name");
479 if (input.hasAttribute("blend-function")) {
480 texState->set_blendFunction(
481 input.getValue("blend-function"),
482 blendFunctionName);
483 }
484
485 // Defines how a texture should be mapped on geometry.
486 texState->set_mapping(input.getValue<TextureState::Mapping>(
487 "mapping", TextureState::MAPPING_TEXCO));
488
489 const std::string mappingFunctionName = input.getValue("mapping-function-name");
490 if (input.hasAttribute("mapping-function")) {
491 texState->set_mappingFunction(
492 input.getValue("mapping-function"),
493 mappingFunctionName);
494 }
495
496 // texel transfer wraps sampled texels before returning them.
497 const std::string texelTransferName = input.getValue("texel-transfer-name");
498 if (input.hasAttribute("texel-transfer-key")) {
499 texState->set_texelTransferKey(
500 input.getValue("texel-transfer-key"),
501 texelTransferName);
502 } else if (input.hasAttribute("texel-transfer-function")) {
503 texState->set_texelTransferFunction(
504 input.getValue("texel-transfer-function"),
505 texelTransferName);
506 }
507
508 // texel transfer wraps computed texture coordinates before returning them.
509 if (input.hasAttribute("texco-transfer")) {
510 texState->set_texcoTransfer(input.getValue<TextureState::TransferTexco>(
511 "texco-transfer", TextureState::TRANSFER_TEXCO_RELIEF));
512 }
513 if (input.hasAttribute("texco-flipping")) {
514 auto flipModeName = input.getValue("texco-flipping");
515 if (flipModeName == "x") {
516 texState->set_texcoFlipping(TextureState::TEXCO_FLIP_X);
517 } else if (flipModeName == "y") {
518 texState->set_texcoFlipping(TextureState::TEXCO_FLIP_Y);
519 } else {
520 texState->set_texcoFlipping(TextureState::TEXCO_FLIP_NONE);
521 }
522 }
523 if (input.hasAttribute("sampler-type")) {
524 texState->set_samplerType(input.getValue("sampler-type"));
525 }
526 const std::string texcoTransferName = input.getValue("texco-transfer-name");
527 if (input.hasAttribute("texco-transfer-key")) {
528 texState->set_texcoTransferKey(
529 input.getValue("texco-transfer-key"),
530 texcoTransferName);
531 } else if (input.hasAttribute("texco-transfer-function")) {
532 texState->set_texcoTransferFunction(
533 input.getValue("texco-transfer-function"),
534 texcoTransferName);
535 }
536
537 return texState;
538}
