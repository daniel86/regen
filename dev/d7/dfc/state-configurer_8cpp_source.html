58void StateConfigurer::addInput(const std::string &name, const ref_ptr<ShaderInput> &in, const std::string &type) {
59 auto needle = inputNames_.find(name);
60 if (needle == inputNames_.end()) {
61 cfg_.inputs_.emplace_back(in, name, type);
62 auto it = cfg_.inputs_.end();
63 --it;
64 inputNames_[name] = it;
65 } else {
66 *needle->second = NamedShaderInput(in, name, type);
67 }

70void StateConfigurer::addState(const State *s) {
71 const auto *x0 = dynamic_cast<const HasInput *>(s);
72 const auto *x1 = dynamic_cast<const FeedbackSpecification *>(s);
73 const auto *x2 = dynamic_cast<const TextureState *>(s);
74 const auto *x3 = dynamic_cast<const StateSequence *>(s);
75
76 if (x0 != nullptr) {
77 const ref_ptr<ShaderInputContainer> &container = x0->inputContainer();
78
79 // remember inputs, they will be enabled automatically
80 // when the shader is enabled.
81 for (const auto & it : container->inputs()) {
82 addInput(it.name_, it.in_);
83
84 std::queue<std::pair<const std::string&,ShaderInput*>> queue;
85 queue.emplace(it.in_->name(), it.in_.get());
86
87 while (!queue.empty()) {
88 auto [name, in] = queue.front();
89 const auto &inName = name.empty() ? in->name() : name;
90 queue.pop();
91
92 define(REGEN_STRING("HAS_" << inName), "TRUE");
93 if (in->isVertexAttribute()) {
94 define(REGEN_STRING("HAS_VERTEX_" << inName), "TRUE");
95 }
96 if (in->numInstances() > 1) {
97 define("HAS_INSTANCES", "TRUE");
98 }
99
100 if (in->isUniformBlock()) {
101 auto block = (UniformBlock *)(in);
102 for (auto& blockUniform : block->uniforms()) {
103 queue.emplace(blockUniform.name_, blockUniform.in_.get());
104 }
105 }
106 }
107 }
108 }
109 if (x1) {
110 cfg_.feedbackMode_ = x1->feedbackMode();
111 cfg_.feedbackStage_ = x1->feedbackStage();
112 for (const auto & it : x1->feedbackAttributes()) {
113 cfg_.feedbackAttributes_.push_back(it->name());
114 }
115 }
116 if (x2) {
117 // map for loop index to texture id
118 define(
119 REGEN_STRING("TEX_ID" << cfg_.textures_.size()),
120 REGEN_STRING(x2->stateID()));
121 define(
122 REGEN_STRING("TEX_ID_" << x2->name()),
123 REGEN_STRING(x2->stateID()));
124 // remember the number of textures used
125 define("NUM_TEXTURES", REGEN_STRING(cfg_.textures_.size() + 1));
126 cfg_.textures_[x2->name()] = x2->texture();
127 addInput(x2->name(), x2->texture(), x2->samplerType());
128 }
129
130 setVersion(s->shaderVersion());
131 addDefines(s->shaderDefines());
132 addFunctions(s->shaderFunctions());
133
134 if (x3) {
135 // add global sequence state
136 addState(x3->globalState().get());
137 // do not add joined states of sequences
138 return;
139 }
140 for (const auto & it : s->joined()) {
141 addState(it.get());
142 }
143}
