33static void processMeshChildren(
34 SceneParser *parser, SceneInputNode &input, MeshVector &x) {
35 const list<ref_ptr<SceneInputNode> > &childs = input.getChildren();
36 for (auto it = childs.begin(); it != childs.end(); ++it) {
37 SceneInputNode *child = it->get();
38 ref_ptr<StateProcessor> processor =
39 parser->getStateProcessor(child->getCategory());
40 if (processor.get() == nullptr) {
41 REGEN_WARN("No processor registered for '" << child->getDescription() << "'.");
42 continue;
43 }
44 if (x.size() > 1) {
45 auto state = ref_ptr<State>::alloc();
46 processor->processInput(parser, *child, state);
47 for (auto jt = x.begin(); jt != x.end(); ++jt) {
48 const ref_ptr<Mesh>& mesh = *jt;
49 mesh->joinStates(state);
50 }
51 } else {
52 processor->processInput(parser, *child, x[0]);
53 }
54 }
55}

60ref_ptr<MeshVector> MeshResource::createResource(
61 SceneParser *parser, SceneInputNode &input) {
62 const string meshType = input.getValue("type");
63 auto scaling = input.getValue<Vec3f>("scaling", Vec3f(1.0f));
64 auto texcoScaling = input.getValue<Vec2f>("texco-scaling", Vec2f(1.0f));
65 auto rotation = input.getValue<Vec3f>("rotation", Vec3f(0.0f));
66 bool useNormal = input.getValue<bool>("use-normal", true);
67 bool useTexco = input.getValue<bool>("use-texco", true);
68 bool useTangent = input.getValue<bool>("use-tangent", false);
69 auto vboUsage = input.getValue<VBO::Usage>("usage", VBO::USAGE_DYNAMIC);
70

74 std::vector<GLuint> lodLevels;
75 if (input.hasAttribute("lod-levels")) {
76 auto lodVec = input.getValue<Vec3ui>("lod-levels", Vec3ui(0));
77 lodLevels.resize(3);
78 lodLevels[0] = lodVec.x;
79 lodLevels[1] = lodVec.y;
80 lodLevels[2] = lodVec.z;
81 }
82 else {
83 lodLevels.push_back(input.getValue<GLuint>("lod", 0));
84 }
85

87 if (meshType == "sphere" || meshType == "half-sphere") {

230 if (input.hasAttribute("branch-offset")) {
231 mesh->setBranchOffset(input.getValue<GLfloat>("branch-offset", 0.5f));
232 }
233 if (input.hasAttribute("branch-length")) {
234 mesh->setBranchLength(input.getValue<GLfloat>("branch-length", 0.5f));

236 if (input.hasAttribute("branch-darkening")) {
237 mesh->setBranchDarkening(input.getValue<GLfloat>("branch-darkening", 0.5f));

240
241 (*out) = MeshVector(1);
242 (*out)[0] = mesh;
243 }
244 else if (meshType == "point") {
245 const auto numVertices = input.getValue<GLuint>("num-vertices", 1u);
246 (*out) = MeshVector(1);
247 (*out)[0] = ref_ptr<Point>::alloc(numVertices);
248 }
249 else if (meshType == "asset") {
250 ref_ptr<AssetImporter> importer = parser->getResources()->getAsset(parser, input.getValue("asset"));
251 if (importer.get() == nullptr) {
252 REGEN_WARN("Ignoring " << input.getDescription() << " with unknown Asset.");
253 } else {
254 out_ = createAssetMeshes(parser, input, importer);
255 out = out_.get();
256 }
257 for (GLuint i = 0u; i < out->size(); ++i) {
258 parser->putState(REGEN_STRING(input.getName() << i), (*out)[i]);
259 }
260 }
261 else if (meshType == "mask-patch") {
262 MaskMesh::Config meshCfg;
263 meshCfg.quad.centerAtOrigin = true;
264 meshCfg.quad.levelOfDetails = lodLevels;
265 meshCfg.quad.posScale = scaling;
266 meshCfg.quad.rotation = rotation;
267 meshCfg.quad.texcoScale = texcoScaling;
268 meshCfg.quad.isNormalRequired = input.hasAttribute("use-normal") && useNormal;
269 meshCfg.quad.isTangentRequired = input.hasAttribute("use-tangent") && useTangent;
270 meshCfg.quad.isTexcoRequired = input.hasAttribute("use-texco") && useTexco;
271 meshCfg.quad.usage = vboUsage;

273 meshCfg.heightMap = parser->getResources()->getTexture2D(parser, input.getValue("height-map"));
274 }
275 meshCfg.height = input.getValue<float>("height", 0.0f);
276 meshCfg.meshSize = input.getValue<Vec2f>("ground-size", Vec2f(10.0f));
277 auto maskTexture = parser->getResources()->getTexture2D(parser, input.getValue("mask"));
278 if (maskTexture.get() == nullptr) {
279 REGEN_WARN("Ignoring " << input.getDescription() << ", failed to load mask texture.");
280 } else {
281 (*out) = MeshVector(1);
282 (*out)[0] = ref_ptr<MaskMesh>::alloc(maskTexture, meshCfg);
283 }
284 }
285 else if (meshType == "proctree") {
286 auto procTree = ref_ptr<ProcTree>::alloc(input);
287 procTree->update();
288 (*out) = MeshVector(2);
289 (*out)[0] = procTree->trunkMesh();
290 (*out)[1] = procTree->twigMesh();
291 }
292 else if (meshType == "text") {
293 (*out) = MeshVector(1);
294 (*out)[0] = createTextMesh(parser, input);
295 }
296 else if (meshType == "mesh") {
297 GLenum primitive = glenum::primitive(input.getValue<string>("primitive", "TRIANGLES"));
298 (*out) = MeshVector(1);
299 (*out)[0] = ref_ptr<Mesh>::alloc(primitive, vboUsage);
300 } else {
301 REGEN_WARN("Ignoring " << input.getDescription() << ", unknown Mesh type.");
302 }
303
304 // configure mesh LOD
305 if (input.hasAttribute("lod-far")) {
306 auto lodFar = input.getValue<GLfloat>("lod-far", 160.0f);
307 for (GLuint i = 0u; i < out->size(); ++i) {
308 (*out)[i]->setLODFar(lodFar);
309 }
310 }
311
312 if (out->size() == 1) {
313 ref_ptr<Mesh> mesh = (*out)[0];
314 parser->putState(input.getName(), mesh);
315 }
316 parser->getResources()->putMesh(input.getName(), out_);
317
318 if (input.hasAttribute("primitive")) {
319 GLenum primitive = glenum::primitive(input.getValue("primitive"));
320 for (GLuint i = 0u; i < out->size(); ++i) {
321 (*out)[i]->set_primitive(primitive);
322 }
323 }
324
325 // Mesh resources can have State children
326 if (!out->empty()) processMeshChildren(parser, input, *out);
327
328 return out_;
329}

331ref_ptr<MeshVector> MeshResource::createAssetMeshes(
332 SceneParser *parser,
333 SceneInputNode &input,
334 const ref_ptr<AssetImporter> &importer) {
335 const auto vboUsage = input.getValue<VBO::Usage>("usage", VBO::USAGE_DYNAMIC);
336 const auto scaling = input.getValue<Vec3f>("scaling", Vec3f(1.0f));
337 const auto rotation = input.getValue<Vec3f>("rotation", Vec3f(0.0f));
338 const auto translation = input.getValue<Vec3f>("translation", Vec3f(0.0f));
339 const auto assetIndices = input.getValue<string>("asset-indices", "*");
340 bool useAnimation = input.getValue<bool>("asset-animation", false);
341
342 ref_ptr<MeshVector> out_ = ref_ptr<MeshVector>::alloc();
343 MeshVector &out = *out_.get();
344
345 const vector<ref_ptr<NodeAnimation> > &nodeAnims = importer->getNodeAnimations();
346 if (useAnimation && nodeAnims.empty()) {
347 REGEN_WARN(input.getDescription() << " has use-animation=1 but Asset '" <<
348 input.getValue("asset") << "' has not.");
349 useAnimation = false;
350 }
351
352 // Create import model transform
353 Mat4f transform = Mat4f::scaleMatrix(scaling);
354 Quaternion q(0.0, 0.0, 0.0, 1.0);
355 q.setEuler(rotation.x, rotation.y, rotation.z);
356 transform *= q.calculateMatrix();
357 transform.translate(translation);
358
359 // Asset can contain multiple meshes, they can be accessed by index.
360 // Parse user specified indices ...
361 vector<string> indicesStr;
362 boost::split(indicesStr, assetIndices, boost::is_any_of(","));
363 vector<GLuint> indices(indicesStr.size());
364 bool useAllIndices = false;
365 for (GLuint i = 0u; i < indices.size(); ++i) {
366 if (indicesStr[i] == "*") {
367 useAllIndices = true;
368 break;
369 } else {
370 stringstream ss(indicesStr[i]);
371 ss >> indices[i];
372 }
373 }
374
375 if (useAllIndices) {
376 out = importer->loadAllMeshes(transform, vboUsage);
377 } else {
378 out = importer->loadMeshes(transform, vboUsage, indices);
379 }
380 for (GLuint i = 0u; i < out.size(); ++i) {
381 ref_ptr<Mesh> mesh = out[i];
382 if (mesh.get() == nullptr) continue;
383
384 // Join in material state.
385 // Can be set to false to allow overwriting material stuff.
386 if (input.getValue<bool>("asset-material", true)) {
387 ref_ptr<Material> material =
388 importer->getMeshMaterial(mesh.get());
389 if (material.get() != nullptr) {
390 mesh->joinStates(material);
391 }
392 }
393
394 if (useAnimation) {
395 list<ref_ptr<AnimationNode> > meshBones;
396 GLuint numBoneWeights = importer->numBoneWeights(mesh.get());
397 GLuint numBones = 0u;
398
399 // Find bones influencing this mesh
400 for (auto it = nodeAnims.begin(); it != nodeAnims.end(); ++it) {
401 list<ref_ptr<AnimationNode> > ibonNodes =
402 importer->loadMeshBones(mesh.get(), it->get());
403 meshBones.insert(meshBones.end(), ibonNodes.begin(), ibonNodes.end());
404 numBones = ibonNodes.size();
405 }
406
407 // Create Bones state that is responsible for uploading
408 // animation data to GL.
409 if (!meshBones.empty()) {
410 ref_ptr<Bones> bonesState = ref_ptr<Bones>::alloc(numBoneWeights, numBones);
411 bonesState->setBones(meshBones);
412 bonesState->setAnimationName(REGEN_STRING("bones-" << input.getName()));
413 bonesState->startAnimation();
414 mesh->joinStates(bonesState);
415 }
416 }
417 }
418
419 return out_;
420}

422template <class InputType, class ValueType>
423static void configureParticleAttribute(
424 SceneParser *parser,
425 const ref_ptr<Particles> &particles,
426 SceneInputNode &input) {
427 const std::string name = input.getValue("name");
428 if (input.hasAttribute("default")) {
429 auto optional = input.getValue<ValueType>("default");
430 if (optional.has_value()) {
431 particles->setDefault<InputType,ValueType>(name, optional.value());
432 } else {
433 REGEN_WARN("Ignoring " << input.getDescription() << ", failed to parse default value.");
434 }
435 }
436 if (input.hasAttribute("variance")) {
437 auto optional = input.getValue<ValueType>("variance");
438 if (optional.has_value()) {
439 particles->setVariance<InputType,ValueType>(name, optional.value());
440 } else {
441 REGEN_WARN("Ignoring " << input.getDescription() << ", failed to parse variance value.");
442 }
443 }
444 if (input.hasAttribute("advance-mode")) {
445 auto optional = input.getValue<Particles::AdvanceMode>("advance-mode");
446 if (optional.has_value()) {
447 particles->setAdvanceMode(name, optional.value());
448 } else {
449 REGEN_WARN("Ignoring " << input.getDescription() << ", failed to parse advance mode.");
450 }
451 }
452 if (input.hasAttribute("advance-function")) {
453 auto optional = input.getValue<string>("advance-function");
454 if (optional.has_value()) {
455 particles->setAdvanceFunction(name, optional.value());
456 } else {
457 REGEN_WARN("Ignoring " << input.getDescription() << ", failed to parse advance function.");
458 }
459 }
460 if (input.hasAttribute("advance-factor")) {
461 auto optional = input.getValue<GLfloat>("advance-factor");
462 if (optional.has_value()) {
463 particles->setAdvanceFactor(name, optional.value());
464 } else {
465 REGEN_WARN("Ignoring " << input.getDescription() << ", failed to parse advance factor.");
466 }
467 }
468 if (input.hasAttribute("advance-constant")) {
469 auto optional = input.getValue<ValueType>("advance-constant");
470 if (optional.has_value()) {
471 particles->setAdvanceConstant<InputType,ValueType>(name, optional.value());
472 } else {
473 REGEN_WARN("Ignoring " << input.getDescription() << ", failed to parse advance constant.");
474 }
475 }
476 if (input.hasAttribute("advance-ramp")) {
477 auto rampMode = input.getValue<Particles::RampMode>("advance-ramp-mode", Particles::RAMP_MODE_LIFETIME);
478 auto rampTexture = parser->getResources()->getTexture(parser, input.getValue("advance-ramp"));
479 if (rampTexture.get() == nullptr) {
480 REGEN_WARN("Ignoring " << input.getDescription() << ", failed to load advance ramp texture.");
481 } else {
482 particles->setAdvanceRamp(name, rampTexture, rampMode);
483 }
484 }

487ref_ptr<Particles> MeshResource::createParticleMesh(
488 SceneParser *parser,
489 SceneInputNode &input,
490 const GLuint numParticles) {
491 ref_ptr<Particles> particles;
492 if (input.hasAttribute("update-shader")) {
493 const auto updateShader = input.getValue("update-shader");
494 particles = ref_ptr<Particles>::alloc(numParticles, updateShader);
495 } else {
496 particles = ref_ptr<Particles>::alloc(numParticles);
497 }
498 particles->setAnimationName(input.getName());
499 particles->startAnimation();
500
501 if (input.hasAttribute("max-emits")) {
502 particles->setMaxEmits(input.getValue<GLuint>("max-emits", 100u));
503 }
504 if (input.hasAttribute("animation-state")) {
505 auto animNodeName = input.getValue("animation-state");
506 auto animState = parser->getState(animNodeName);
507 if (animState.get() == nullptr) {
508 // Try to load the animation node
509 auto animNodeInput = parser->getRoot()->getFirstChild("node", animNodeName);
510 if (animNodeInput.get() != nullptr) {
511 auto animNode = ref_ptr<StateNode>::alloc();
512 animNode->set_name(animNodeName);
513 parser->processNode(animNode, animNodeName);
514 if (animNode->childs().empty()) {
515 animState = animNode->state();
516 } else {
517 animState = animNode->childs().front()->state();
518 }
519 parser->putState(animNodeName, animState);
520 }
521 }
522 if (animState.get() == nullptr) {
523 REGEN_WARN("Ignoring " << input.getDescription() << ", unknown animation-node.");
524 }
525 else {
526 particles->joinAnimationState(animState);
527 }
528 }
529
530 // the attributes of particles may have special attributes in the scene file that determine default, variance, etc.
531 // these are ignored by processMeshChildren, so we need to process them here.
532 const list<ref_ptr<SceneInputNode> > &childs = input.getChildren();
533 for (const auto & it : childs) {
534 auto &child = *it.get();
535 if (child.getCategory() == "input" && child.getValue<bool>("is-attribute", false)) {
536 const string type = child.getValue("type");
537 if (type == "float") {
538 configureParticleAttribute<ShaderInput1f, GLfloat>(parser, particles, child);
539 } else if (type == "vec2") {
540 configureParticleAttribute<ShaderInput2f, Vec2f>(parser, particles, child);
541 } else if (type == "vec3") {
542 configureParticleAttribute<ShaderInput3f, Vec3f>(parser, particles, child);
543 } else if (type == "vec4") {
544 configureParticleAttribute<ShaderInput4f, Vec4f>(parser, particles, child);
545 } else if (type == "int") {
546 configureParticleAttribute<ShaderInput1i, GLint>(parser, particles, child);
547 } else if (type == "ivec2") {
548 configureParticleAttribute<ShaderInput2i, Vec2i>(parser, particles, child);
549 } else if (type == "ivec3") {
550 configureParticleAttribute<ShaderInput3i, Vec3i>(parser, particles, child);
551 } else if (type == "ivec4") {
552 configureParticleAttribute<ShaderInput4i, Vec4i>(parser, particles, child);
553 } else {
554 REGEN_WARN("Ignoring " << child.getDescription() << ", unknown attribute type '" << type << "'.");
555 }
556 }
557 }
558
559 particles->begin();
560 // Mesh resources can have State children
561 auto x = ref_ptr<MeshVector>::alloc(1);
562 (*x.get())[0] = particles;
563 parser->getResources()->putMesh(input.getName(), x);
564 processMeshChildren(parser, input, *x.get());
565 particles->end();
566
567 return particles;
568}

570ref_ptr<TextureMappedText> MeshResource::createTextMesh(
571 SceneParser *parser, SceneInputNode &input) {
572 ref_ptr<regen::Font> font =
573 parser->getResources()->getFont(parser, input.getValue("font"));
574 if (font.get() == nullptr) {
575 REGEN_WARN("Unable to load Font for '" << input.getDescription() << "'.");
576 return {};
577 }
578
579 auto textHeight = input.getValue<GLfloat>("height", 16.0f);
580 auto textColor = input.getValue<Vec4f>("text-color", Vec4f(0.97f, 0.86f, 0.77f, 0.95f));
581
582 auto widget = ref_ptr<TextureMappedText>::alloc(font, textHeight);
583 widget->set_color(textColor);
584 widget->set_centerAtOrigin(input.getValue<bool>("center", true));
585
586 if (input.hasAttribute("text")) {
587 auto maxLineWidth = input.getValue<GLfloat>("max-line-width", 0.0f);
588 TextureMappedText::Alignment alignment = TextureMappedText::ALIGNMENT_LEFT;
