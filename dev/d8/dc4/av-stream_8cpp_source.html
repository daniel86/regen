47void AudioVideoStream::open(AVStream *stream, GLint index, GLboolean initial) {
48 if (!initial) {
49 clearQueue();
50 }
51 // Destroy the codec context if it exists
52 close();
53 // Create a new codec context for this stream
54 codecCtx_ = avcodec_alloc_context3(nullptr);
55 avcodec_parameters_to_context(codecCtx_, stream->codecpar);
56
57 // Find the decoder for the video stream
58 codec_ = avcodec_find_decoder(codecCtx_->codec_id);
59 if (!codec_) {
60 throw Error("Unsupported codec!");
61 }
62 // Open codec
63 if (avcodec_open2(codecCtx_, codec_, nullptr) < 0) {
64 throw Error("Could not open codec.");
65 }
66 stream_ = stream;
67 index_ = index;
68 cachedBytes_ = 0;
69 isActive_ = GL_TRUE;
70}

77void AudioVideoStream::pushFrame(AVFrame *frame, GLuint frameSize) {
78 {
79 boost::lock_guard<boost::mutex> lock(decodingLock_);
