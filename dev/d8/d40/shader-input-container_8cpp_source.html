78ShaderInputList::const_iterator ShaderInputContainer::setInput(
79 const ref_ptr<ShaderInput> &in, const std::string &name) {
80 const std::string &inputName = (name.empty() ? in->name() : name);
81
82 if (in->isVertexAttribute() && in->numVertices() > numVertices_) { numVertices_ = in->numVertices(); }
83 if (in->numInstances() > 1) {
84 numInstances_ = in->numInstances();
85 numVisibleInstances_ = numInstances_;
86 }
87 // check for instances of attributes within UBO
88 if (in->isUniformBlock()) {
89 auto *block = dynamic_cast<UniformBlock *>(in.get());
90 for (auto &namedInput : block->uniforms()) {
91 if (namedInput.in_->isVertexAttribute() && namedInput.in_->numVertices() > numVertices_) {
92 numVertices_ = namedInput.in_->numVertices();
93 }
94 if (namedInput.in_->numInstances() > 1) {
95 numInstances_ = namedInput.in_->numInstances();
96 numVisibleInstances_ = numInstances_;
97 }
98 }
99 }
100
101 if (inputMap_.count(inputName) > 0) {
102 removeInput(inputName);
103 } else { // insert into map of known attributes
104 inputMap_.insert(inputName);
105 }
106
107 inputs_.emplace_front(in, inputName);
108
109 if (uploadLayout_ != LAYOUT_LAST) {
110 if (in->isVertexAttribute())
111 uploadAttributes_.push_front(in);
112 uploadInputs_.push_front(*inputs_.begin());
113 }
114
115 return inputs_.begin();
116}

136void ShaderInputContainer::removeInput(const std::string &name) {
137 ShaderInputList::iterator it;
138 for (it = inputs_.begin(); it != inputs_.end(); ++it) {
139 if (it->name_ == name) { break; }
140 }
141 if (it == inputs_.end()) { return; }
142
143 if (uploadLayout_ != LAYOUT_LAST) {
144 VBOReference ref = it->in_->bufferIterator();
145 if (ref.get()) {
146 inputBuffer_->free(ref.get());
147 it->in_->set_buffer(0u, VBOReference());
148 }
149 }
150
151 inputs_.erase(it);
152}
