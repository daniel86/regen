94StateNode *StateNode::findNodeWithName(const std::string &name) {
95 std::stack<StateNode *> stack;
96 stack.push(this);
97
98 while (!stack.empty()) {
99 auto node = stack.top();
100 stack.pop();
101
102 if (node->name_ == name) {
103 return node;
104 }
105
106 for (auto &child: node->childs_) {
107 stack.push(child.get());
108 }
109 }
110
111 return nullptr;
112}
