26ModelTransformation *NodeEyeDepthComparator::findModelTransformation(StateNode *n) const {
27 State *nodeState = n->state().get();
28 auto *ret = dynamic_cast<ModelTransformation *>(nodeState);
29 if (ret != nullptr) { return ret; }
30
31 for (auto it = nodeState->joined().begin(); it != nodeState->joined().end(); ++it) {
32 auto *ret = dynamic_cast<ModelTransformation *>(it->get());
33 if (ret != nullptr) { return ret; }
34 }
35
36 for (auto it = n->childs().begin(); it != n->childs().end(); ++it) {
37 ret = findModelTransformation(it->get());
38 if (ret != nullptr) { return ret; }
39 }
40
41 return nullptr;
42}

44bool NodeEyeDepthComparator::operator()(ref_ptr<StateNode> &n0, ref_ptr<StateNode> &n1) const {
45 ModelTransformation *modelMat0 = findModelTransformation(n0.get());
46 ModelTransformation *modelMat1 = findModelTransformation(n1.get());
47 if (modelMat0 != nullptr && modelMat1 != nullptr) {
48 GLfloat diff = mode_ * (
49 getEyeDepth(modelMat0->get()->getVertex(0).r.position()) -
50 getEyeDepth(modelMat1->get()->getVertex(0).r.position()));
51 return diff < 0;
52 } else if (modelMat0 != nullptr) {
53 return true;
54 } else if (modelMat1 != nullptr) {
55 return false;
56 } else {
57 return n0 < n1;
58 }
59}

156StateNode* StateNode::findNodeWithName(const std::string &name) {
157 std::stack<StateNode *> stack;
158 stack.push(this);
159
160 while (!stack.empty()) {
161 auto node = stack.top();
162 stack.pop();
163
164 if (node->name_ == name) {
165 return node;
166 }
167
168 for (auto &child: node->childs_) {
169 stack.push(child.get());
170 }
171 }
172
173 return nullptr;
