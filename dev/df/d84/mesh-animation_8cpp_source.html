17void MeshAnimation::findFrameAfterTick(
18 GLdouble tick,
19 GLint &frame,
20 std::vector<KeyFrame> &keys) {
21 while (frame < (GLint) (keys.size() - 1)) {
22 if (tick <= keys[frame].endTick) {
23 return;
24 }
25 frame += 1;
26 }
27}

29void MeshAnimation::findFrameBeforeTick(
30 GLdouble &tick,
31 GLuint &frame,
32 std::vector<KeyFrame> &keys) {
33 for (frame = keys.size() - 1; frame > 0;) {
34 if (tick >= keys[frame].startTick) {
35 return;
36 }
37 frame -= 1;
38 }
39}

43MeshAnimation::MeshAnimation(
44 const ref_ptr<Mesh> &mesh,
45 const std::list<Interpolation> &interpolations)
46 : Animation(true, false),
47 mesh_(mesh),
48 meshBufferOffset_(-1),
49 lastFrame_(-1),
50 nextFrame_(-1),
51 pingFrame_(-1),
52 pongFrame_(-1),
53 elapsedTime_(0.0),
54 ticksPerSecond_(1.0),
55 lastTime_(0.0),
56 tickRange_(0.0, 0.0),
57 lastFramePosition_(0u),
58 startFramePosition_(0u) {
59 const ref_ptr<ShaderInputContainer> &inputContainer = mesh->inputContainer();
60 const ShaderInputList &inputs = inputContainer->inputs();
61 std::map<GLenum, std::string> shaderNames;
62 std::map<std::string, std::string> shaderConfig;
63 std::map<std::string, std::string> functions;
64 std::list<std::string> transformFeedback;
65
66 shaderNames[GL_VERTEX_SHADER] = "regen.morph.interpolate";
67
68 // find buffer size
69 bufferSize_ = 0u;
70 GLuint i = 0u;
71 for (auto it = inputs.begin(); it != inputs.end(); ++it) {
72 const ref_ptr<ShaderInput> &in = it->in_;
73 if (!in->isVertexAttribute()) continue;
74 bufferSize_ += in->inputSize();
75 transformFeedback.push_back(in->name());
76
77 if(in->stride()==0) {
78 hasMeshInterleavedAttributes_ = GL_FALSE;
79 } else {
80 hasMeshInterleavedAttributes_ = GL_TRUE;
81 }
82
83 std::string interpolationName = "interpolate_linear";
84 std::string interpolationKey;
85 for (auto jt = interpolations.begin(); jt != interpolations.end(); ++jt) {
86 if (jt->attributeName == in->name()) {
87 interpolationName = jt->interpolationName;
88 interpolationKey = jt->interpolationKey;
89 break;
90 }
91 }
92
93 shaderConfig[REGEN_STRING("ATTRIBUTE" << i << "_INTERPOLATION_NAME")] = interpolationName;
94 if (!interpolationKey.empty()) {
95 shaderConfig[REGEN_STRING("ATTRIBUTE" << i << "_INTERPOLATION_KEY")] = interpolationKey;
96 }
97 shaderConfig[REGEN_STRING("ATTRIBUTE" << i << "_NAME")] = in->name();
98 shaderConfig[REGEN_STRING("ATTRIBUTE" << i << "_TYPE")] =
99 glenum::glslDataType(in->dataType(), in->valsPerElement());
100 i += 1;
101 }
102 shaderConfig["NUM_ATTRIBUTES"] = REGEN_STRING(i);
103
104 // used to save two frames
105 animationBuffer_ = ref_ptr<VBO>::alloc(VBO::USAGE_DYNAMIC);
106 feedbackBuffer_ = ref_ptr<VBO>::alloc(VBO::USAGE_FEEDBACK);
107 feedbackRef_ = feedbackBuffer_->alloc(bufferSize_);
108
109 bufferRange_.buffer_ = feedbackRef_->bufferID();
110 bufferRange_.offset_ = 0;
111 bufferRange_.size_ = bufferSize_;
112
113 // create initial frame
114 addMeshFrame(0.0);
115
116 // init interpolation shader
117 {
118 std::map<GLenum, std::string> preProcessed;
119 Shader::preProcess(preProcessed,
120 PreProcessorConfig(330, shaderNames, shaderConfig));
121 interpolationShader_ = ref_ptr<Shader>::alloc(preProcessed);
122 }
123 if (hasMeshInterleavedAttributes_) {
124 interpolationShader_->setTransformFeedback(
125 transformFeedback, GL_INTERLEAVED_ATTRIBS, GL_VERTEX_SHADER);
126 } else {
127 interpolationShader_->setTransformFeedback(
128 transformFeedback, GL_SEPARATE_ATTRIBS, GL_VERTEX_SHADER);
129 }
130 if (interpolationShader_.get() != nullptr &&
131 interpolationShader_->compile() && interpolationShader_->link()) {
132 ref_ptr<ShaderInput> in = interpolationShader_->createUniform("frameTimeNormalized");
133 frameTimeUniform_ = (ShaderInput1f *) in.get();
134 frameTimeUniform_->setUniformData(0.0f);
135 interpolationShader_->setInput(in);
136 // join shader uniforms into animation state such that they can be
137 // configured from the outside
138 meshAnimState_ = ref_ptr<State>::alloc();
139
140 in = interpolationShader_->createUniform("friction");
141 frictionUniform_ = (ShaderInput1f *) in.get();
142 frictionUniform_->setUniformData(8.0f);
143 interpolationShader_->setInput(in);
144 meshAnimState_->joinShaderInput(in, in->name());
145
146 in = interpolationShader_->createUniform("frequency");
147 frequencyUniform_ = (ShaderInput1f *) in.get();
148 frequencyUniform_->setUniformData(5.0f);
149 interpolationShader_->setInput(in);
150 meshAnimState_->joinShaderInput(in, in->name());
151
152 joinAnimationState(meshAnimState_);
153 } else {
154 interpolationShader_ = ref_ptr<Shader>();
155 }
156}

166void MeshAnimation::setTickRange(const Vec2d &forcedTickRange) {
167 // get first and last tick of animation
168 if (forcedTickRange.x < 0.0f || forcedTickRange.y < 0.0f) {
169 tickRange_.x = 0.0;
170 tickRange_.y = 0.0;
171 for (auto it = frames_.begin(); it != frames_.end(); ++it) {
172 tickRange_.y += it->timeInTicks;
173 }
174 } else {
175 tickRange_ = forcedTickRange;
176 }
177
178 // set start frames
179 if (tickRange_.x < 0.00001) {
180 startFramePosition_ = 0u;
181 } else {
182 GLuint framePos;
183 findFrameBeforeTick(tickRange_.x, framePos, frames_);
184 startFramePosition_ = framePos;
185 }
186 // initial last frame to start frame
187 lastFramePosition_ = startFramePosition_;
188
189 // set to start pos of the new tick range
190 lastTime_ = 0.0;
191 elapsedTime_ = 0.0;
192}

194void MeshAnimation::loadFrame(GLuint frameIndex, GLboolean isPongFrame) {
195 MeshAnimation::KeyFrame &frame = frames_[frameIndex];
196
197 std::list<ref_ptr<ShaderInput> > atts;
198 for (auto & attribute : frame.attributes) {
199 atts.push_back(attribute.input);
200 }
201
202 if (isPongFrame) {
203 if (pongFrame_ != -1) { regen::VBO::free(pongIt_.get()); }
204 pongFrame_ = frameIndex;
205 if (hasMeshInterleavedAttributes_) {
206 pongIt_ = animationBuffer_->allocInterleaved(atts);
207 } else {
208 pongIt_ = animationBuffer_->allocSequential(atts);
209 }
210 frame.ref = pongIt_;
211 } else {
212 if (pingFrame_ != -1) { regen::VBO::free(pingIt_.get()); }
213 pingFrame_ = frameIndex;
214 if (hasMeshInterleavedAttributes_) {
215 pingIt_ = animationBuffer_->allocInterleaved(atts);
216 } else {
217 pingIt_ = animationBuffer_->allocSequential(atts);
218 }
219 frame.ref = pingIt_;
220 }

223void MeshAnimation::glAnimate(RenderState *rs, GLdouble dt) {
224 if (dt <= 0.00001) return;
225 if (rs->isTransformFeedbackAcive()) {
226 REGEN_WARN("Transform Feedback was active when the MeshAnimation was updated.");
227 stopAnimation();
228 return;
229 }
230
231 // find offst in the mesh vbo.
232 // in the constructor data may not be set or data moved in vbo
233 // so we lookup the offset here.
234 const ref_ptr<ShaderInputContainer> &inputContainer = mesh_->inputContainer();
235 const ShaderInputList &inputs = inputContainer->inputs();
236 std::list<ContiguousBlock> blocks;
237
238 if (hasMeshInterleavedAttributes_) {
239 meshBufferOffset_ = (inputs.empty() ? 0 : (inputs.begin()->in_)->offset());
240 for (auto it = inputs.rbegin(); it != inputs.rend(); ++it) {
241 const ref_ptr<ShaderInput> &in = it->in_;
242 if (!in->isVertexAttribute()) continue;
243 if (in->offset() < meshBufferOffset_) {
244 meshBufferOffset_ = in->offset();
245 }
246 }
247 } else {
248 // find contiguous blocks of memory in the mesh buffers.
249 auto it = inputs.begin();
250 blocks.emplace_back(it->in_);
251
252 for (++it; it != inputs.end(); ++it) {
253 const ref_ptr<ShaderInput> &in = it->in_;
254 if (!in->isVertexAttribute()) continue;
255 ContiguousBlock &activeBlock = *blocks.rbegin();
256 if (activeBlock.buffer != in->buffer()) {
257 blocks.emplace_back(in);
258 } else if (in->offset() + in->inputSize() == activeBlock.offset) {
259 // join left
260 activeBlock.offset = in->offset();
261 activeBlock.size += in->inputSize();
262 } else if (activeBlock.offset + activeBlock.size == in->offset()) {
263 // join right
264 activeBlock.size += in->inputSize();
265 } else {
266 blocks.emplace_back(in);
267 }
268 }
269 }
270
271 elapsedTime_ += dt;
272
273 // map into anim's duration
274 const GLdouble duration = tickRange_.y - tickRange_.x;
275 const GLdouble timeInTicks = elapsedTime_ * ticksPerSecond_ / 1000.0;
276 if (timeInTicks > duration) {
277 REGEN_DEBUG("Mesh animation stopped at tick " <<
278 timeInTicks << " (duration: " << duration << ", frame:" << nextFrame_ << ")");
279 elapsedTime_ = 0.0;
280 lastTime_ = 0.0;
281 tickRange_.x = 0.0;
282 tickRange_.y = 0.0;
283 stopAnimation();
284 return;
285 }
286
287 GLboolean framesChanged = GL_FALSE;
288 // Look for present frame number.
289 GLint lastFrame = lastFramePosition_;
290 GLint frame = (timeInTicks >= lastTime_ ? lastFrame : startFramePosition_);
291 findFrameAfterTick(timeInTicks, frame, frames_);
292 lastFramePosition_ = frame;
293
294 // keep two frames in animation buffer
295 lastFrame = frame - 1;
296 MeshAnimation::KeyFrame &frame0 = frames_[lastFrame];
297 if (lastFrame != pingFrame_ && lastFrame != pongFrame_) {
298 loadFrame(lastFrame, frame == pingFrame_);
299 framesChanged = GL_TRUE;
300 }
301 if (lastFrame != lastFrame_) {
302 for (auto & attribute : frame0.attributes) {
303 attribute.location = interpolationShader_->attributeLocation("next_" + attribute.input->name());
304 }
305 lastFrame_ = lastFrame;
306 framesChanged = GL_TRUE;
307 }
308 MeshAnimation::KeyFrame &frame1 = frames_[frame];
309 if (frame != pingFrame_ && frame != pongFrame_) {
310 loadFrame(frame, lastFrame == pingFrame_);
311 framesChanged = GL_TRUE;
312 }
313 if (frame != nextFrame_) {
314 for (auto it = frame1.attributes.begin(); it != frame1.attributes.end(); ++it) {
315 it->location = interpolationShader_->attributeLocation("last_" + it->input->name());
316 }
317 nextFrame_ = frame;
318 framesChanged = GL_TRUE;
319 REGEN_DEBUG("Next frame: " << nextFrame_ << " (time: " << timeInTicks << ")");
320 }
321 if (framesChanged) {
322 vao_ = ref_ptr<VAO>::alloc();
323 rs->vao().push(vao_->id());
324
325 // setup attributes
326 glBindBuffer(GL_ARRAY_BUFFER, frame0.ref->bufferID());
327 for (auto & attribute : frame0.attributes) {
328 attribute.input->enableAttribute(attribute.location);
329 }
330 glBindBuffer(GL_ARRAY_BUFFER, frame1.ref->bufferID());
331 for (auto & attribute : frame1.attributes) {
332 attribute.input->enableAttribute(attribute.location);
333 }
334
335 rs->vao().pop();
336 }
337
338 frameTimeUniform_->setVertex(0,
339 (timeInTicks - frame1.startTick) / frame1.timeInTicks);
340
341 { // Write interpolated attributes to transform feedback buffer
342 // no FS used
343 rs->toggles().push(RenderState::RASTARIZER_DISCARD, GL_TRUE);
344 rs->depthMask().push(GL_FALSE);
345 // setup the interpolation shader
346 rs->shader().push(interpolationShader_->id());
347 interpolationShader_->enable(rs);
348 rs->vao().push(vao_->id());
349
350 // setup the transform feedback
351 if (hasMeshInterleavedAttributes_) {
352 rs->feedbackBufferRange().push(0, bufferRange_);
353 } else {
354 GLint index = inputs.size() - 1;
355 bufferRange_.offset_ = 0;
356 for (auto it = inputs.rbegin(); it != inputs.rend(); ++it) {
357 const ref_ptr<ShaderInput> &in = it->in_;
358 index -= 1;
359 if (!in->isVertexAttribute()) continue;
360 bufferRange_.size_ = in->inputSize();
361 rs->feedbackBufferRange().push(index+1, bufferRange_);
362 bufferRange_.offset_ += bufferRange_.size_;
363 }
364 }
365 rs->beginTransformFeedback(GL_POINTS);
366
367 // finally the draw call
368 glDrawArrays(GL_POINTS, 0, inputContainer->numVertices());
369
370 rs->endTransformFeedback();
371 if (hasMeshInterleavedAttributes_) {
372 rs->feedbackBufferRange().pop(0);
373 } else {
374 GLint index = inputs.size() - 1;
375 for (auto it = inputs.rbegin(); it != inputs.rend(); ++it) {
376 const ref_ptr<ShaderInput> &in = it->in_;
377 index -= 1;
378 if (!in->isVertexAttribute()) continue;
379 rs->feedbackBufferRange().pop(index+1);
380 }
381 }
382 rs->vao().pop();
383 rs->shader().pop();
384 rs->depthMask().pop();
385 rs->toggles().pop(RenderState::RASTARIZER_DISCARD);
386 }
387
388 // copy transform feedback buffer content to mesh buffer
389 if (hasMeshInterleavedAttributes_) {
390 VBO::copy(
391 feedbackRef_->bufferID(),
392 inputs.begin()->in_->buffer(),
393 bufferSize_,
394 0, // feedback buffer offset
395 meshBufferOffset_);
396 } else {
397 GLuint feedbackBufferOffset = 0;
398 for (auto & block : blocks) {
399 VBO::copy(
400 feedbackRef_->bufferID(),
401 block.buffer,
402 block.size,
403 feedbackBufferOffset,
404 block.offset);
405 feedbackBufferOffset += block.size;
406 }
407 }
408
409 lastTime_ = tickRange_.x + timeInTicks;
410 GL_ERROR_LOG();
411}

415void MeshAnimation::addFrame(
416 const std::list<ref_ptr<ShaderInput> > &attributes,
417 GLdouble timeInTicks) {
418 const ref_ptr<ShaderInputContainer> &inputContainer = mesh_->inputContainer();
419 const ShaderInputList &inputs = inputContainer->inputs();
420
421 MeshAnimation::KeyFrame frame;
422
423 frame.timeInTicks = timeInTicks;
424 frame.startTick = 0.0;
425 for (auto it = frames_.rbegin(); it != frames_.rend(); ++it) {
426 MeshAnimation::KeyFrame &parentFrame = *it;
427 frame.startTick += parentFrame.timeInTicks;
428 }
429 frame.endTick = frame.startTick + frame.timeInTicks;
430
431 REGEN_DEBUG("Adding frame at tick " << frame.startTick << " (duration: " << frame.timeInTicks << ")");
432
433 // add attributes
434 for (const auto & input : inputs) {
435 const ref_ptr<ShaderInput> &in0 = input.in_;
436 if (!in0->isVertexAttribute()) continue;
437 ref_ptr<ShaderInput> att;
438 // find specified attribute
439 for (const auto & in1 : attributes) {
440 if (in0->name() == in1->name()) {
441 att = in1;
442 break;
443 }
444 }
445 if (att.get() == nullptr) {
446 // find attribute from previous frames
447 att = findLastAttribute(in0->name());
448 }
449 if (att.get() != nullptr) {
450 frame.attributes.emplace_back(att, -1);
451 } else {
452 REGEN_WARN("No attribute found for '" << in0->name() << "'");
453 }
454 }
455
456 frames_.push_back(frame);
457}

471ref_ptr<ShaderInput> MeshAnimation::findLastAttribute(const std::string &name) {
472 for (auto it = frames_.rbegin(); it != frames_.rend(); ++it) {
473 MeshAnimation::KeyFrame &f = *it;
474 for (auto jt = f.attributes.begin(); jt != f.attributes.end(); ++jt) {
475 const ref_ptr<ShaderInput> &att = jt->input;
476 if (att->name() == name) {
477 return ShaderInput::copy(att, GL_TRUE);
478 }
479 }
480 }
481 return {};

484void MeshAnimation::addSphereAttributes(
485 GLfloat horizontalRadius,
486 GLfloat verticalRadius,
487 GLdouble timeInTicks,
488 const Vec3f &offset) {
489 const ref_ptr<ShaderInputContainer> &inputContainer = mesh_->inputContainer();
490
491 if (!inputContainer->hasInput(ATTRIBUTE_NAME_POS)) {
492 REGEN_WARN("mesh has no input named '" << ATTRIBUTE_NAME_POS << "'");
493 return;
494 }
495 if (!inputContainer->hasInput(ATTRIBUTE_NAME_NOR)) {
496 REGEN_WARN("mesh has no input named '" << ATTRIBUTE_NAME_NOR << "'");
497 return;
498 }
499
500 GLfloat radiusScale = horizontalRadius / verticalRadius;
501 Vec3f scale(radiusScale, 1.0, radiusScale);
502
503 ref_ptr<ShaderInput3f> posAtt = ref_ptr<ShaderInput3f>::dynamicCast(mesh_->positions());
504 ref_ptr<ShaderInput3f> norAtt = ref_ptr<ShaderInput3f>::dynamicCast(mesh_->normals());
505 // allocate memory for the animation attributes
506 ref_ptr<ShaderInput3f> spherePos = ref_ptr<ShaderInput3f>::dynamicCast(
507 ShaderInput::copy(posAtt, GL_FALSE));
508 ref_ptr<ShaderInput3f> sphereNor = ref_ptr<ShaderInput3f>::dynamicCast(
509 ShaderInput::copy(norAtt, GL_FALSE));
510
511 // find the centroid of the mesh
512 Vec3f minPos = posAtt->getVertex(0).r;
513 Vec3f maxPos = posAtt->getVertex(0).r;
514 for (GLuint i = 1; i < posAtt->numVertices(); ++i) {
515 auto v = posAtt->getVertex(i);
516 if(minPos.x > v.r.x) minPos.x = v.r.x;
517 if(minPos.y > v.r.y) minPos.y = v.r.y;
518 if(minPos.z > v.r.z) minPos.z = v.r.z;
519 if(maxPos.x < v.r.x) maxPos.x = v.r.x;
520 if(maxPos.y < v.r.y) maxPos.y = v.r.y;
521 if(maxPos.z < v.r.z) maxPos.z = v.r.z;
522 }
523 Vec3f centroid = (minPos + maxPos) * 0.5f + offset;
524
525 // set sphere vertex data
526 // Note: this is a very simple sphere mapping, it is not perfect.
527 // There might be artifacts caused by faces that are flipped.
528 // Also make sure to use polygon offset to avoid shadow map fighting.
529 // TODO: it should be possible to do this with less artifacts by taking the faces into account.
530 for (GLuint i = 0; i < spherePos->numVertices(); ++i) {
531 Vec3f v = posAtt->getVertex(i).r;
532 Vec3f direction = v - centroid;
533 Vec3f n;
534 GLdouble l = direction.length();
535 if (l == 0) {
536 continue;
537 }
538
539 // take normalized direction vector as normal
540 n = direction;
541 n.normalize();
542 // and scaled normal as sphere position
543 // 1e-1 to avoid fighting
544 v = centroid + n * verticalRadius * (1.0f + l * 1e-6);
545
546 spherePos->setVertex(i, v);
547 sphereNor->setVertex(i, n);
548 }
549
550 std::list<ref_ptr<ShaderInput> > attributes;
551 attributes.emplace_back(spherePos);
552 attributes.emplace_back(sphereNor);
553 addFrame(attributes, timeInTicks);
554}

670void MeshAnimation::addBoxAttributes(
671 GLfloat width,
672 GLfloat height,
673 GLfloat depth,
674 GLdouble timeInTicks,
675 const Vec3f &offset) {
676 const ref_ptr<ShaderInputContainer> &inputContainer = mesh_->inputContainer();
677
678 if (!inputContainer->hasInput(ATTRIBUTE_NAME_POS)) {
679 REGEN_WARN("mesh has no input named '" << ATTRIBUTE_NAME_POS << "'");
680 return;
681 }
682 if (!inputContainer->hasInput(ATTRIBUTE_NAME_NOR)) {
683 REGEN_WARN("mesh has no input named '" << ATTRIBUTE_NAME_NOR << "'");
684 return;
685 }
686
687 Vec3f boxSize(width, height, depth);
688 GLdouble radius = sqrt(0.5f);
689
690 ref_ptr<ShaderInput3f> posAtt = ref_ptr<ShaderInput3f>::dynamicCast(mesh_->positions());
691 ref_ptr<ShaderInput3f> norAtt = ref_ptr<ShaderInput3f>::dynamicCast(mesh_->normals());
692 // allocate memory for the animation attributes
693 ref_ptr<ShaderInput3f> boxPos = ref_ptr<ShaderInput3f>::dynamicCast(
694 ShaderInput::copy(posAtt, GL_FALSE));
