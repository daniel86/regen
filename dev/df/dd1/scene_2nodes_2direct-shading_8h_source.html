32 // Override
33 void processInput(
34 SceneParser *parser,
35 SceneInputNode &input,
36 const ref_ptr<StateNode> &parent) override {
37 ref_ptr<SceneInputNode> lightNode = input.getFirstChild("direct-lights");
38 ref_ptr<SceneInputNode> passNode = input.getFirstChild("direct-pass");
39 if (lightNode.get() == nullptr) {
40 REGEN_WARN("Missing direct-lights attribute for " << input.getDescription() << ".");
41 return;
42 }
43 if (passNode.get() == nullptr) {
44 REGEN_WARN("Missing direct-pass attribute for " << input.getDescription() << ".");
45 return;
46 }
47
48 ref_ptr<DirectShading> shadingState = ref_ptr<DirectShading>::alloc();
49 ref_ptr<StateNode> shadingNode = ref_ptr<StateNode>::alloc(shadingState);
50 parent->addChild(shadingNode);
51
52 if (input.hasAttribute("ambient")) {
53 shadingState->ambientLight()->setVertex(0,
54 input.getValue<Vec3f>("ambient", Vec3f(0.1f)));
55 }
56
57 // load lights
58 const list<ref_ptr<SceneInputNode> > &childs = lightNode->getChildren();
59 for (auto it = childs.begin(); it != childs.end(); ++it) {
60 ref_ptr<SceneInputNode> n = *it;
61 if (n->getCategory() != "light") {
62 REGEN_WARN("No processor registered for '" << n->getDescription() << "'.");
63 continue;
64 }
65
66 ref_ptr<Light> light = parser->getResources()->getLight(parser, n->getName());
67 if (light.get() == nullptr) {
68 REGEN_WARN("Unable to find Light for '" << n->getDescription() << "'.");
69 continue;
70 }
71
72 auto shadowFiltering =
73 n->getValue<ShadowFilterMode>("shadow-filter", SHADOW_FILTERING_NONE);
74 ref_ptr<Texture> shadowMap;
75 ref_ptr<Texture> shadowColorMap;
76 ref_ptr<LightCamera> shadowCamera;
77 if (n->hasAttribute("shadow-camera")) {
78 shadowCamera = ref_ptr<LightCamera>::dynamicCast(
79 parser->getResources()->getCamera(parser, n->getValue("shadow-camera")));
80 if (shadowCamera.get() == nullptr) {
81 REGEN_WARN("Unable to find LightCamera for '" << n->getDescription() << "'.");
82 }
83 }
84 if (n->hasAttribute("shadow-buffer") || n->hasAttribute("shadow-texture")) {
85 shadowMap = TextureStateProvider::getTexture(parser, *n.get(),
86 "shadow-texture", "shadow-buffer",
87 "shadow-attachment");
88 if (shadowMap.get() == nullptr) {
89 REGEN_WARN("Unable to find ShadowMap for '" << n->getDescription() << "'.");
90 }
91 }
92 if (n->hasAttribute("shadow-buffer") || n->hasAttribute("shadow-color-texture")) {
93 shadowColorMap = TextureStateProvider::getTexture(parser, *n.get(),
94 "shadow-color-texture", "shadow-buffer",
95 "shadow-color-attachment");
96 }
97 shadingState->addLight(light, shadowCamera, shadowMap, shadowColorMap, shadowFiltering);
98 }
99
100 // parse passNode
101 SceneNodeProcessor().processInput(parser, *passNode.get(), shadingNode);
102 }
103 };
104 }
