41void ModelTransformation::enable(RenderState *rs) {
42 if (isAudioSource()) {
43 boost::posix_time::ptime time(
44 boost::posix_time::microsec_clock::local_time());
45 GLdouble dt = ((GLdouble) (time - lastTime_).total_microseconds()) / 1000.0;
46 lastTime_ = time;
47
48 if (dt > 1e-6) {
49 auto val = modelMat_->getVertex(0);
50 velocity_->setVertex(0, (val.r.position() - lastPosition_) / dt);
51 lastPosition_ = val.r.position();
52 if (isAudioSource()) {
53 audioSource_->set3f(AL_VELOCITY, velocity_->getVertex(0).r);
54 }
55 audioSource_->set3f(AL_POSITION, val.r.position());
56 }
57 }
58 State::enable(rs);
59}

61static void transformMatrix(
62 const std::string &target, Mat4f &mat, const Vec3f &value) {
63 if (target == "translate") {
64 mat.x[12] += value.x;
65 mat.x[13] += value.y;
66 mat.x[14] += value.z;
67 } else if (target == "scale") {
68 mat.scale(value);
69 } else if (target == "rotate") {
70 Quaternion q(0.0, 0.0, 0.0, 1.0);
71 q.setEuler(value.x, value.y, value.z);
72 mat *= q.calculateMatrix();
73 } else {
74 REGEN_WARN("Unknown distribute target '" << target << "'.");
75 }
76}

120static void makeInstance(InstancePlaneGenerator &generator, const PlaneCell &cell) {
121 if (generator.maskData) {
122 auto density = generator.maskTexture->sampleLinear<float>(
123 cell.uv, generator.maskData);
124 if (density < 0.1f) return;
125 }
126
127 auto &instanceMat = generator.instanceData.emplace_back(Mat4f::identity());
128
129 // apply scaling, randomize between min and max scale
130 auto sizeFactor = badRandom() * cell.density;
131 auto scale = generator.objMinScale + sizeFactor * (generator.objMaxScale - generator.objMinScale);
132 instanceMat.scale(scale);
133
134 // apply random rotation around y axis
135 auto angle = badRandom() * 2.0f * M_PI;
136 Quaternion q(0.0, 0.0, 0.0, 1.0);
137 q.setEuler(angle, 0.0f, 0.0f);
138 instanceMat *= q.calculateMatrix();
139
140 // translate to cell position
141 Vec3f pos = Vec3f(cell.position.x, 0.0f, cell.position.y) + generator.cellWorldOffset;
142 if (generator.objPosVariation > 0.0f) {
143 // TODO: randomize position within cell, but need to re-compute UV then!
144 //pos.x += generator.objPosVariation * (badRandom() - 0.5f) * 2.0f;
145 //pos.z += generator.objPosVariation * (badRandom() - 0.5f) * 2.0f;
146 //cell_uv = ...
147 }
148 instanceMat.x[12] += pos.x;
149 instanceMat.x[13] += pos.y;
150 instanceMat.x[14] += pos.z;
151 // translate to height map position
152 if (generator.heightData) {
153 instanceMat.x[13] += generator.areaMaxHeight * generator.heightMap->sampleLinear<float>(
154 cell.uv, generator.heightData);
155 }
156}

158static void makeInstances(InstancePlaneGenerator &generator,
159 const PlaneCell &rootCell, const PlaneCellWeights &rootWeights) {
160 std::stack<std::pair<PlaneCell, PlaneCellWeights>> stack;
161 stack.emplace(rootCell, rootWeights);
162 auto clampedDensity = math::clamp(generator.objDensity, 0.1f, generator.objDensity);
163
164 while (!stack.empty()) {
165 auto &pair = stack.top();
166 auto cell = pair.first;
167 auto weights = pair.second;
168 stack.pop();
169
170 // make an instance at the cell center
171 makeInstance(generator, cell);
172
173 // subdivide the cell into four quadrants in case half of the cell is big enough
174 auto subdivideThreshold = generator.objSize * generator.objMaxScale * 2.5f / clampedDensity;
175 if (cell.size.x > subdivideThreshold.x && cell.size.y > subdivideThreshold.y) {
176 auto &bottomLeft = stack.emplace();
177 auto &bottomRight = stack.emplace();
178 auto &topRight = stack.emplace();
179 auto &topLeft = stack.emplace();
180 auto subdividedSize = cell.size * 0.5f;
181 auto halfCellDensity = cell.density * 0.5f;
182 PlaneCell *subdivideCells[4] = {
183 &bottomLeft.first, &bottomRight.first, &topRight.first, &topLeft.first
184 };
185
186 // compute center position of subdivide cells
187 bottomLeft.first.position = cell.position - cell.size * 0.25f;
188 topRight.first.position = cell.position + cell.size * 0.25f;
189 bottomRight.first.position = cell.position + Vec2f(cell.size.x * 0.25f, -cell.size.y * 0.25f);
190 topLeft.first.position = cell.position + Vec2f(-cell.size.x * 0.25f, cell.size.y * 0.25f);
191 // compute density at the center
192 bottomLeft.first.density = halfCellDensity + (weights.bottom + weights.left) * 0.25f;
193 bottomRight.first.density = halfCellDensity + (weights.bottom + weights.right) * 0.25f;
194 topRight.first.density = halfCellDensity + (weights.top + weights.right) * 0.25f;
195 topLeft.first.density = halfCellDensity + (weights.top + weights.left) * 0.25f;
196 // compute uv coordinate, and set the size to half size of parent cell
197 for (int i = 0; i < 4; i++) {
198 auto &subdivideCell = *subdivideCells[i];
199 subdivideCell.size = subdividedSize;
200 }
201 auto uvOffset = cell.size / generator.areaSize;
202 bottomLeft.first.uv = cell.uv - Vec2f(0.25f, 0.25f) * uvOffset;
203 bottomRight.first.uv = cell.uv + Vec2f(0.25f, -0.25f) * uvOffset;
204 topRight.first.uv = cell.uv + Vec2f(0.25f, 0.25f) * uvOffset;
205 topLeft.first.uv = cell.uv - Vec2f(0.25f, -0.25f) * uvOffset;
206
207 // compute weights for subdivide cells
208 bottomLeft.second.left = 0.75f * weights.left + 0.25f * weights.bottom;
209 bottomLeft.second.bottom = 0.75f * weights.bottom + 0.25f * weights.left;
210 bottomLeft.second.top = 0.5f * cell.density + 0.5f * weights.left;
211 bottomLeft.second.right = 0.5f * cell.density + 0.5f * weights.bottom;
212
213 bottomRight.second.right = 0.75f * weights.right + 0.25f * weights.bottom;
214 bottomRight.second.bottom = 0.75f * weights.bottom + 0.25f * weights.right;
215 bottomRight.second.top = 0.5f * cell.density + 0.5f * weights.right;
216 bottomRight.second.left = 0.5f * cell.density + 0.5f * weights.bottom;
217
218 topRight.second.right = 0.75f * weights.right + 0.25f * weights.top;
219 topRight.second.top = 0.75f * weights.top + 0.25f * weights.right;
220 topRight.second.left = 0.5f * cell.density + 0.5f * weights.right;
221 topRight.second.bottom = 0.5f * cell.density + 0.5f * weights.top;
222
223 topLeft.second.left = 0.75f * weights.left + 0.25f * weights.top;
224 topLeft.second.top = 0.75f * weights.top + 0.25f * weights.left;
225 topLeft.second.right = 0.5f * cell.density + 0.5f * weights.left;
226 topLeft.second.bottom = 0.5f * cell.density + 0.5f * weights.top;
227 }
228 }

231static void makeInstances(InstancePlaneGenerator &generator, unsigned int i, unsigned int j) {
232 auto &cell = generator.cells[j * generator.cellCountX + i];
233 if (cell.density < 0.1f) {
234 return;
235 }
236 // read weights of adjacent cells
237 PlaneCellWeights weights;
238 if (i > 0) {
239 weights.left = generator.cells[j * generator.cellCountX + i - 1].density;
240 }
241 if (i < generator.cellCountX - 1) {
242 weights.right = generator.cells[j * generator.cellCountX + i + 1].density;
243 }
244 if (j > 0) {
245 weights.top = generator.cells[(j - 1) * generator.cellCountX + i].density;
246 }
247 if (j < generator.cellCountY - 1) {
248 weights.bottom = generator.cells[(j + 1) * generator.cellCountX + i].density;
249 }
250 makeInstances(generator, cell, weights);
251}

253static GLuint transformMatrixPlane(
254 scene::SceneLoader *scene,
255 scene::SceneInputNode &input,
256 const ref_ptr<ShaderInputMat4> &matrixInput,
257 GLuint numInstances) {
258 auto areaSize = input.getValue<Vec2f>("area-size", Vec2f(10.0f));
259 auto areaHalfSize = areaSize * 0.5f;
260
261 InstancePlaneGenerator generator;
262 generator.areaSize = areaSize;
263 generator.areaMaxHeight = input.getValue<GLfloat>("area-max-height", 0.0f);
264 generator.objMinScale = input.getValue<GLfloat>("obj-min-scale", 0.6f);
265 generator.objMaxScale = input.getValue<GLfloat>("obj-max-scale", 1.0f);
266 generator.objPosVariation = input.getValue<GLfloat>("obj-pos-variation", 0.0f);
267 generator.objDensity = input.getValue<GLfloat>("obj-density", 1.0f);
268 generator.ws_cellSize = input.getValue<Vec2f>("cell-size", Vec2f(1.0f));
269 generator.cellHalfSize = Vec2f(generator.ws_cellSize.x, generator.ws_cellSize.y) * 0.5f;
270 generator.cellWorldOffset = input.getValue<Vec3f>("cell-offset", Vec3f(0.0f));
271 if (input.hasAttribute("area-mask-texture")) {
272 generator.maskTexture = scene->getResource<Texture2D>(input.getValue("area-mask-texture"));
273 if (generator.maskTexture.get()) {
274 generator.maskTexture->ensureTextureData();
275 generator.maskData = generator.maskTexture->textureData();
276 } else {
277 REGEN_WARN("Unable to load mask texture.");
278 }
279 }
280 if (input.hasAttribute("area-height-texture")) {
281 generator.heightMap = scene->getResource<Texture2D>(input.getValue("area-height-texture"));
282 if (generator.heightMap.get()) {
283 generator.heightMap->ensureTextureData();
284 generator.heightData = generator.heightMap->textureData();
285 } else {
286 REGEN_WARN("Unable to load height map texture.");
287 }
288 }
289 // get object bounds
290 auto meshes = scene->getResource<MeshVector>(input.getValue("obj-mesh"));
291 if (meshes.get() && !meshes.get()->empty()) {
292 auto &meshVec = *meshes.get();
293 auto firstMesh = meshVec[0];
294 Bounds<Vec3f> bounds(firstMesh->minPosition(), firstMesh->maxPosition());
295 for (size_t i = 1; i < meshVec.size(); i++) {
296 auto mesh = meshVec[i];
297 bounds.min.setMin(mesh->minPosition());
298 bounds.max.setMax(mesh->maxPosition());
299 }
300 generator.objSize.x = bounds.max.x - bounds.min.x;
301 generator.objSize.y = bounds.max.z - bounds.min.z;
302 }
303
304 // round up to the next cell count
305 generator.cellCountX = std::ceil(areaSize.x / generator.ws_cellSize.x);
306 generator.cellCountY = std::ceil(areaSize.y / generator.ws_cellSize.y);
307 generator.numCells = generator.cellCountX * generator.cellCountY;
308 generator.cells = new PlaneCell[generator.cellCountX * generator.cellCountY];
309 generator.ts_cellSize = Vec2f(
310 generator.ws_cellSize.x / areaSize.x,
311 generator.ws_cellSize.y / areaSize.y);
312
313 // initialize cells
314 Vec2f cellUV = generator.ts_cellSize * 0.5f;
315 for (unsigned int j = 0; j < generator.cellCountY; j++) {
316 for (unsigned int i = 0; i < generator.cellCountX; i++) {
317 auto index = j * generator.cellCountX + i;
318 auto &cell = generator.cells[index];
319 cell.position = Vec2f(static_cast<float>(i), static_cast<float>(j)) *
320 generator.ws_cellSize + generator.cellHalfSize - areaHalfSize;
321 cell.uv = cellUV;
322 cellUV.x += generator.ts_cellSize.x;
323 cell.density = 1.0f;
324 cell.size = generator.ws_cellSize;
325 }
326 cellUV.x = generator.ts_cellSize.x * 0.5f;
327 cellUV.y += generator.ts_cellSize.y;
328 }
329
330 if (generator.maskData) {
331 // compute cell density based on mask texture
332 for (unsigned int i = 0; i < generator.numCells; i++) {
333 auto &cell = generator.cells[i];
334 cell.density = generator.maskTexture->sampleMax<float>(
335 cell.uv,
336 generator.ts_cellSize,
337 generator.maskData);
338 }
339 }
340
341 for (unsigned int j = 0; j < generator.cellCountY; j++) {
342 for (unsigned int i = 0; i < generator.cellCountX; i++) {
343 makeInstances(generator, i, j);
344 }
345 }
346
347 delete[] generator.cells;
348
349 if (generator.instanceData.empty()) {
350 REGEN_WARN("No instances created.");
351 return numInstances;
352 } else {
353 numInstances = generator.instanceData.size();
354 matrixInput->setInstanceData(numInstances, 1, nullptr);
355 auto matrices = matrixInput->mapClientData<Mat4f>(ShaderData::WRITE);
356 // TODO: apply previous transform instead of overwriting
357 for (GLuint i = 0; i < numInstances; i += 1) {
358 matrices.w[i] = generator.instanceData[i];
359 }
360 }
361
362 return numInstances;
363}

365static void transformAnimation(
366 scene::SceneLoader *scene,
367 const ref_ptr<scene::SceneInputNode> &child,
368 const ref_ptr<State> &state,
369 const ref_ptr<StateNode> &parent,
370 const ref_ptr<ModelTransformation> &tf) {
371 auto animType = child->getValue<std::string>("type", "transform");
372
373 if (animType == "boids") {
374 LoadingContext boidsConfig(scene, parent);
375 auto boidsAnimation = BoidsSimulation_CPU::load(boidsConfig, *child.get(), tf);
376 state->attach(boidsAnimation);
377 boidsAnimation->startAnimation();
378 } else {
379 auto transformAnimation = ref_ptr<TransformAnimation>::alloc(tf->get());
380
381 if (child->hasAttribute("mesh-id")) {
382 auto meshID = child->getValue("mesh-id");
383 auto meshIndex = child->getValue<GLuint>("mesh-index", 0u);
384 auto meshVec = scene->getResource<MeshVector>(meshID);
385 if (meshVec.get() != nullptr && meshVec->size() > meshIndex) {
386 auto mesh = (*meshVec.get())[meshIndex];
387 transformAnimation->setMesh(mesh);
388 }
389 }
390
391 for (auto &keyFrameNode: child->getChildren("key-frame")) {
392 std::optional<Vec3f> framePos = std::nullopt;
393 std::optional<Vec3f> frameDir = std::nullopt;
394 if (keyFrameNode->hasAttribute("position")) {
395 framePos = keyFrameNode->getValue<Vec3f>("position", Vec3f(0.0f));
396 }
397 if (keyFrameNode->hasAttribute("rotation")) {
398 frameDir = keyFrameNode->getValue<Vec3f>("rotation", Vec3f(0.0f));
399 }
400 auto dt = keyFrameNode->getValue<GLdouble>("dt", 1.0);
401 transformAnimation->push_back(framePos, frameDir, dt);
402 }
403
404 transformAnimation->setAnimationName(child->getName());
405 transformAnimation->startAnimation();
406 state->attach(transformAnimation);
407 }
408}

410static void transformMatrix(
411 scene::SceneLoader *scene,
412 scene::SceneInputNode &input,
413 const ref_ptr<State> &state,
414 const ref_ptr<StateNode> &parent,
415 const ref_ptr<ModelTransformation> &tf,
416 GLuint numInstances) {
417 for (auto &child: input.getChildren()) {
418 // TODO: this seems bad, why create this every loop? why at all?
419 std::list<GLuint> indices = child->getIndexSequence(numInstances);
420
421 if (child->getCategory() == "set") {
422 auto mode = child->getValue("mode");
423 if (mode == "plane") {
424 numInstances = transformMatrixPlane(scene, *child.get(), tf->get(), numInstances);
425 } else {
426 auto matrices = tf->get()->mapClientData<Mat4f>(ShaderData::WRITE);
427 scene::ValueGenerator<Vec3f> generator(child.get(), indices.size(),
428 child->getValue<Vec3f>("value", Vec3f(0.0f)));
429 const auto target = child->getValue<std::string>("target", "translate");
430
431 for (auto it = indices.begin(); it != indices.end(); ++it) {
432 transformMatrix(target, matrices.w[*it], generator.next());
433 }
434 }
435 } else if (child->getCategory() == "animation") {
436 transformAnimation(scene, child, state, parent, tf);
437 } else {
438 auto matrices = tf->get()->mapClientData<Mat4f>(ShaderData::WRITE);
439 for (auto it = indices.begin(); it != indices.end(); ++it) {
440 transformMatrix(child->getCategory(), matrices.w[*it],
441 child->getValue<Vec3f>("value", Vec3f(0.0f)));
442 }
443 }
444 }

447ref_ptr<ModelTransformation>
448ModelTransformation::load(LoadingContext &ctx, scene::SceneInputNode &input, const ref_ptr<State> &state) {
449 auto scene = ctx.scene();
450
451 ref_ptr<ModelTransformation> transform = scene->getResource<ModelTransformation>(input.getName());
452 if (transform.get() != nullptr) {
453 state->joinStates(transform);
454 return transform;
455 }
456 ref_ptr<scene::SceneInputNode> transformNode = scene->getRoot()->getFirstChild("transform", input.getName());
457 if (transformNode.get() != nullptr && transformNode.get() != &input) {
458 LoadingContext cfgTransform(scene, ctx.parent());
459 return load(cfgTransform, *transformNode.get(), state);
460 }
461
462 bool isInstanced = input.getValue<bool>("is-instanced", false);
463 auto numInstances = input.getValue<GLuint>("num-instances", 1u);
464 transform = ref_ptr<ModelTransformation>::alloc();
465
466 // Handle instanced model matrix
467 if (isInstanced && numInstances > 1) {
468 transform->get()->setInstanceData(numInstances, 1, nullptr);
469 {
470 auto matrices = transform->get()->mapClientData<Mat4f>(ShaderData::WRITE);
471 for (GLuint i = 0; i < numInstances; i += 1) matrices.w[i] = Mat4f::identity();
472 }
473 transformMatrix(scene, input, state, ctx.parent(), transform, numInstances);
474 // add data to vbo
475 transform->setInput(transform->get());
476 } else {
477 transformMatrix(scene, input, state, ctx.parent(), transform, 1u);
478 if (transform->get()->numInstances() > 1) {
479 transform->setInput(transform->get());
480 }
481 }
482
483 state->joinStates(transform);
484 scene->putResource<ModelTransformation>(input.getName(), transform);
485 return transform;
486}
