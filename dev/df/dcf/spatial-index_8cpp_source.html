149void SpatialIndex::updateVisibility() {
150 for (auto &ic: cameras_) {
151 for (auto &pair: ic.second.shapes) {
152 // keep instance IDs mapped for writing during visibility update
153 pair.second->mapInstanceIDs_internal();
154 // note: first element is the number of visible instances
155 pair.second->mappedInstanceIDs()[0] = 0;
156 pair.second->u_instanceCount_ = 0;
157 pair.second->u_visible_ = false;
158 pair.second->u_visibleSet_.clear();
159 }
160
161 if (ic.second.camera->isOmni()) {
162 // omni camera -> intersection test with bounding sphere
163 BoundingSphere sphereShape(Vec3f::zero(), ic.first->far()->getVertex(0).r);
164 sphereShape.setTransform(ic.first->position());
165 sphereShape.updateTransform(true);
166 updateVisibility(ic.second, sphereShape, false);
167 }
168 //else if (ic.second.camera->isSemiOmni()) {
169 // // TODO: Support half-spheres for culling
170 //}
171 else {
172 // spot camera -> intersection test with view frustum
173 auto &frustumShapes = ic.first->frustum();
174 for (auto &frustumShape: frustumShapes) {
175 updateVisibility(ic.second, frustumShape, frustumShapes.size() > 1);
176 }
177 }
178
179 for (auto &pair: ic.second.shapes) {
180 pair.second->unmapInstanceIDs_internal();
181 pair.second->visible_ = pair.second->u_visible_;
182 pair.second->instanceCount_ = pair.second->u_instanceCount_;
183 }
184 }
