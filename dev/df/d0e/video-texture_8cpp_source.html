134void VideoTexture::animate(GLdouble animateDT) {
135 if (!demuxer_->isPlaying()) { return; }
136 interval_ -= animateDT;
137 dt_ += animateDT;
138 if (interval_ > 0.0) { return; }
139
140 GLuint numFrames = vs_->numFrames();
141 GLboolean isIdle = (numFrames == 0);
142
143 if (isIdle) {
144 // no frames there to show
145 interval_ += idleInterval_;
146 } else {
147 boost::int64_t diff = (dt_ - intervalMili_);
148
149 // pop the first frame dropping some frames
150 // if we are not fast enough showing frames
151 AVFrame *frame = vs_->frontFrame();
152 vs_->popFrame();
153
154 {
155 // queue calling texImage
156 boost::lock_guard<boost::mutex> lock(textureUpdateLock_);
157 if (lastFrame_) {
158 av_free(lastFrame_->data[0]);
159 av_free(lastFrame_);
160 }
161 set_textureData(frame->data[0]);
162 }
163
164 // set next interval
165 GLfloat *t = (GLfloat *) frame->opaque;
166 if (!seeked_) {
167 // set timeout interval to time difference to last frame plus a correction
168 // value because the last timeout call was not exactly the wanted interval
169 GLfloat dt = (*t) - elapsedSeconds_;
170 intervalMili_ = std::max(0.0f, dt * 1000.0f - diff);
171 } else {
172 seeked_ = GL_FALSE;
173 }
174 elapsedSeconds_ = *t;
175 delete t;
176
177 if (demuxer_->audioStream().get()) {
178 // synchronize with audio
179 intervalMili_ += (elapsedSeconds_ -
180 demuxer_->audioStream()->elapsedTime()) * 2;
181 }
182
183 lastFrame_ = frame;
184 interval_ = intervalMili_;
185 }
186 dt_ = 0.0;
187}

189void VideoTexture::glAnimate(RenderState *rs, GLdouble dt) {
190 GL_ERROR_LOG();
191 auto channel = rs->reserveTextureChannel();
192 begin(rs, channel);
193 if (fileToLoaded_) { // setup the texture target
194 set_textureData(nullptr);
195 texImage();
196 filter().push(GL_LINEAR);
197 wrapping().push(GL_REPEAT);
198 fileToLoaded_ = GL_FALSE;
199 }
200 // upload texture data to GL
201 if (textureData() != NULL) {
202 boost::lock_guard<boost::mutex> lock(textureUpdateLock_);
203 texImage();
204 set_textureData(NULL);
205 }
206 end(rs, channel);
207 rs->releaseTextureChannel();
208 GL_ERROR_LOG();
209}
