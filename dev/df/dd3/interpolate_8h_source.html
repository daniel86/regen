70, IM_Squared

72, IM_Cubed

74, IM_Sin // strong in, soft out
75, IM_InvSin // soft in, strong out
76};

79template<typename T>
80inline const T interpolate(
81 const T t
82, const e_InterpolationMethod function = IM_Linear)
83{
84 switch(function)
85 {
86 case IM_SmoothStep:
87 return _smoothstep(t);
88 case IM_SmoothStep2:
89 return _smoothstep2(t);
90 case IM_SmoothStep3:
91 return _smoothstep3(t);
92 case IM_SmootherStep:
93 return _smootherstep(t);
94 case IM_Squared:
95 return _squared(t);
96 case IM_InvSquared:
97 return _invsquared(t);
98 case IM_Cubed:
99 return _cubed(t);
100 case IM_InvCubed:
101 return _invcubed(t);
102 case IM_Sin:
103 return static_cast<T>(_sin(t));
104 case IM_InvSin:
105 return static_cast<T>(_invsin(t));
106 default:
107 case IM_Linear:
108 return _linear(t);
109 }
110}
