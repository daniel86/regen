280 template<class T>
281 T sampleAverage(const Vec2f &texco, const Vec2f &regionTS, const GLubyte *textureData) const {
282 auto bounds = getRegion(texco, regionTS);
283 T avg = T(0.0f);
284 int numSamples = 0;
285 for (unsigned int y = bounds.min.y; y <= bounds.max.y; ++y) {
286 for (unsigned int x = bounds.min.x; x <= bounds.max.x; ++x) {
287 unsigned int index = (y * width() + x);
288 avg += sample<T>(index, textureData);
289 numSamples++;
290 }
291 }
292 if (numSamples > 0) {
293 return avg / static_cast<float>(numSamples);
294 } else {
295 return avg;
296 }
297 }

307 template<class T>
308 T sampleMax(const Vec2f &texco, const Vec2f &regionTS, const GLubyte *textureData) const {
309 auto bounds = getRegion(texco, regionTS);
310 T maxVal = T(0.0f);
311 for (unsigned int y = bounds.min.y; y <= bounds.max.y; ++y) {
312 for (unsigned int x = bounds.min.x; x <= bounds.max.x; ++x) {
313 unsigned int index = (y * width() + x);
314 maxVal = std::max(maxVal, sample<T>(index, textureData));

339 template<class T>
340 T sampleLinear(const Vec2f &texco, const GLubyte *textureData) const {
341 auto w = static_cast<float>(width());
342 auto h = static_cast<float>(height());
343 auto x = texco.x * w;
344 auto y = texco.y * h;
345 auto x0 = std::floor(x);
346 auto y0 = std::floor(y);
347 auto x1 = x0 + 1;
348 auto y1 = y0 + 1;
349
350 T v00 = sampleNearest<T>(Vec2f(x0 / w, y0 / h), textureData);
351 T v01 = sampleNearest<T>(Vec2f(x0 / w, y1 / h), textureData);
352 T v10 = sampleNearest<T>(Vec2f(x1 / w, y0 / h), textureData);
353 T v11 = sampleNearest<T>(Vec2f(x1 / w, y1 / h), textureData);
354
355 auto dx = x - x0;
356 auto dy = y - y0;
357 T v0 = v00 * (1.0f - dx) + v10 * dx;
358 T v1 = v01 * (1.0f - dx) + v11 * dx;
359 return v0 * (1.0f - dy) + v1 * dy;
360 }

410 template<class T>
411 T sample(unsigned int texelIndex, const GLubyte *textureData) const {
412 auto numComponents = glenum::pixelComponents(format());
413 auto *dataOffset = textureData + texelIndex * numComponents;
414 T v(0.0f);
415 auto *typedData = (float *) &v;
416 for (unsigned int i = 0; i < numComponents; ++i) {
417 typedData[i] = static_cast<float>(dataOffset[i]) / 255.0f;
418 }
419 return v;
420 }
