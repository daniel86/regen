233 void sortInForward(Node *resizedNode) {
234 unsigned int space = resizedNode->allocator.maxSpace();
235 for (Node *n = resizedNode->next;
236 n != nullptr && n->allocator.maxSpace() > space;
237 n = n->next) { swap(n->prev, n); }

246 n = n->prev) { swap(n, n->next); }
