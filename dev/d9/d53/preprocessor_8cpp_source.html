35void PreProcessor::removeProcessor(GLSLProcessor *processor) {
36 ref_ptr<GLSLProcessor> child;
37 for (ref_ptr<GLSLProcessor> p = lastProcessor_;
38 p.get() != NULL; p = p->getParent()) {
39 if (p.get() == processor) {
40 if (child.get()) {
41 child->setParent(p->getParent());

43 p->setParent(ref_ptr<GLSLProcessor>());
44 break;
45 }
46 child = p;

50std::map<GLenum, std::string> PreProcessor::processStages(const PreProcessorInput &in) {
51 static boost::regex mainRegex(mainPattern);
52 std::map<GLenum, std::string> processed;
53 const std::string *currentCode = NULL;
54
55 // find effect names
56 std::set<std::string> effectNames;
57 for (std::map<GLenum, std::string>::const_iterator
58 it = in.unprocessed.begin(); it != in.unprocessed.end(); ++it) {
59 if (it->second.empty()) { continue; }
60 if (Includer::get().isKeyValid(it->second)) {
61 std::list<std::string> path;
62 boost::split(path, it->second, boost::is_any_of("."));
63 effectNames.insert(*path.begin());
64 }
65 }
66
67 PreProcessorState state_(in);
68 state_.nextStage = GL_NONE;
69 state_.version = 150;
70
71 // reverse process stages, because stages must know inputs
72 // of next stages.
73 for (GLint i = glenum::glslStageCount() - 1; i >= 0; --i) {
74 GLenum stage = glenum::glslStages()[i];
75#ifdef DEBUG_GLSL_PREPROCESSOR
76 REGEN_DEBUG("[GLSL] Processing " << glenum::glslStageName(stage) << ".");
77#endif
78
79 std::map<GLenum, std::string>::const_iterator it = in.unprocessed.find(stage);
80 if (it != in.unprocessed.end() && !it->second.empty()) {
81 currentCode = &it->second;
82 } else {
83 if (stage == GL_VERTEX_SHADER) {
84 // no vertex shader specified. try to find one with
85 // specified effect keys.
86 for (std::set<std::string>::iterator it = effectNames.begin(); it != effectNames.end(); ++it) {
87 std::string defaultVSName = REGEN_STRING((*it) << ".vs");
88 const std::string &vsCode = Includer::get().include(defaultVSName);
89 if (!vsCode.empty()) {
90 currentCode = &vsCode;
91 break;
92 }
93 }
94 }
95 if (currentCode == NULL) continue;
96 }
97
98 state_.currStage = stage;
99
100 // fill input stream
101 state_.inStream.clear();
102 state_.inStream << "#define SHADER_STAGE " << glenum::glslStagePrefix(stage) << std::endl;
103 state_.inStream << in.header << std::endl;
104 if (Includer::get().isKeyValid(*currentCode)) { state_.inStream << "#include " << (*currentCode) << std::endl; }
105 else { state_.inStream << (*currentCode) << std::endl; }
106 currentCode = NULL;
107
108 // execute processor pipeline
109 std::stringstream out;
110 std::string line;
111 while (lastProcessor_->getline(state_, line)) {
112 out << line << std::endl;
113#ifdef DEBUG_GLSL_PREPROCESSOR
114 REGEN_DEBUG("[GLSL] -----------------------");
115#endif
116 }
117 // post process pipeline output
118 std::stringstream postProcessed;
119 // insert version statement at the top. ATI is strict about this.
120 postProcessed << "#version " << state_.version << std::endl;
121 postProcessed << out.str();
122 it = processed.insert(make_pair(stage, postProcessed.str())).first;
123
124 // check if a main function is defined
125 boost::sregex_iterator regexIt(it->second.begin(), it->second.end(), mainRegex);
126 if (regexIt == NO_REGEX_MATCH) {
127 processed.erase(stage);
128 continue;
129 }
130
131 state_.nextStage = stage;
132 }
133
134 for (ref_ptr<GLSLProcessor> p = lastProcessor_;
135 p.get() != NULL; p = p->getParent()) { p->clear(); }
136
137 return processed;
138}
