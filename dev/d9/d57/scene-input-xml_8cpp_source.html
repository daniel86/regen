16// Function to replace placeholders with actual values
17string preprocessXML(const string &xmlContent, const map<string, string> &variables) {
18 stringstream input(xmlContent);
19 stringstream output;
20 string line;
21
22 while (getline(input, line)) {
23 for (const auto &var : variables) {
24 string placeholder = "{{" + var.first + "}}";
25 if (line.find(placeholder) != string::npos) {
26 string escapedPlaceholder = regex_replace(placeholder, regex(R"([-[\]{}()*+?.,\^$|#\s])"), R"(\$&)");
27 line = regex_replace(line, regex(escapedPlaceholder), var.second);
28 }
29 }
30 output << line << '\n';
31 }
32
33 return output.str();
34}

36// Function to extract global constants from the XML content
37map<string, string> extractGlobalConstants(const string &xmlContent) {
38 map<string, string> variables;
39 regex constantRegex(R"lit(<constant\s+name="([^"]+)"\s+value="([^"]+)"\s*/>)lit");
40 smatch match;
41 string::const_iterator searchStart(xmlContent.cbegin());
42
43 while (regex_search(searchStart, xmlContent.cend(), match, constantRegex)) {
44 variables[match[1].str()] = match[2].str();
45 searchStart = match.suffix().first;
46 }
47
48 return variables;
49}
