252// This is, if required, approximatelly refraction corrected...
253
254const t_longf Earth::viewDistanceWithinAtmosphere(
255 const t_longf y
256, const bool refractionCorrected)
257{
258 const t_longf t = atmosphereThickness();
259 const t_longf r = meanRadius();
260
261 // This works, since dot product of [0, 1, 0] and
262 // eye with [x, y, z] gives y.
263
264 t_longf h = asin(y * (1.0 - 1.e-12)); // correction is required to
265 // gain t at y = 1.0 - fix for t_longf accuracy.
266
267 if(refractionCorrected)
268 h += _rad(atmosphericRefraction(_deg(asin(y))));
269
270 const t_longf cosa = cos(h);
271 const t_longf rt = r + t;
272
273 // Using law of sine for arbitrary triangle with two sides and one angle known.
274 // Since the angle is (π/2 + a), cos is used instead of sine.
275
276 const t_longf distance = cos(h + asin(cosa * r / rt)) * rt / cosa;
277
278 return distance;
279}

283// G.G. Bennet, "The Calculation of the Astronomical Refraction in marine Navigation", 1982

286const t_longf Earth::atmosphericRefraction(const t_longf altitude)
287{
288 t_longf R = 1.02 /
289 tan(_rad(altitude + 10.3 / (altitude + 5.11))) + 0.0019279;
