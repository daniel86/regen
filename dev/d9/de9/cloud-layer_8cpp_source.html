18static GLfloat *createNoiseSlice(GLuint texSize, GLuint octave) {
19 GLuint size2 = texSize * texSize;
20 GLfloat oneOverTexSize = 1.f / static_cast<GLfloat>(texSize);
21
22 osgHimmel::Noise n(1 << (octave + 2), _randf(0.f, 1.f), _randf(0.f, 1.f));
23
24 auto *noise = new float[size2];
25 GLuint o;
26 for (GLuint s = 0; s < texSize; ++s)
27 for (GLuint t = 0; t < texSize; ++t) {
28 o = t * texSize + s;
29 noise[o] = n.noise2(s * oneOverTexSize, t * oneOverTexSize, octave) * 0.5 + 0.5;
30 }
31
32 return noise;
33}

35static ref_ptr<Texture3D> createNoiseArray(GLuint texSize, GLuint octave, GLuint slices) {
36 RenderState *rs = RenderState::get();
37
38 ref_ptr<Texture3D> tex = ref_ptr<Texture3D>::alloc();
39 //ref_ptr<Texture2DArray> tex = ref_ptr<Texture2DArray>::alloc();
40 tex->begin(rs);
41 {
42 tex->set_rectangleSize(texSize, texSize);
43 tex->set_depth(slices);
44 tex->set_format(GL_RED);
45 tex->set_internalFormat(GL_R16F);
46 tex->set_pixelType(GL_FLOAT);
47
48 tex->texImage();
49 for (unsigned int s = 0; s < slices; ++s) {
50 GLfloat *data = createNoiseSlice(texSize, octave);
51 tex->texSubImage(s, (GLubyte *) data);
52 delete[]data;
53 }
54
55 tex->filter().push(GL_LINEAR);
56 tex->wrapping().push(GL_REPEAT);
57 }
58 tex->end(rs);
59
60 return tex;
61}
