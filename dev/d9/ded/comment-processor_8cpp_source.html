38 if(it==rend) {
39 if(commentActive_) {
40 // skip comment lines

42 }
43 else {
44 // no comment active

47 }

50 for ( ; it!=rend; ++it)
51 {

56 // multi line comment end
57 commentActive_ = false;
58 }
59 }
60 else {
61 lineStream << it->prefix();
62 if(comment == "/*") {
63 // multi line comment start
64 commentActive_ = true;
65 }
66 else if(comment == "//" || comment == "--") {
67 // single line comment start
68 line = lineStream.str();
69 if(line.empty()) {
70 return getline(state,line);
71 }
72 else {
73 return true;
74 }
75 }
76 }
77 last = it;
78 }
79 if(!commentActive_) {

81 }
