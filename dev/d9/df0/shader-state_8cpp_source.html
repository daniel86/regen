29void ShaderState::loadStage(
30 const std::map<std::string, std::string> &shaderConfig,
31 const std::string &effectName,
32 std::map<GLenum, std::string> &code,
33 GLenum stage) {
34 std::string stageName = glenum::glslStageName(stage);
35 std::string effectKey = REGEN_STRING(effectName << "." << glenum::glslStagePrefix(stage));
36 std::string ignoreKey = REGEN_STRING("IGNORE_" << stageName);
37
38 auto it = shaderConfig.find(ignoreKey);
39 if (it != shaderConfig.end() && it->second == "TRUE") { return; }
40
41 code[stage] = Includer::get().include(effectKey);
42 // failed to include ?
43 if (code[stage].empty()) { code.erase(stage); }
44}

46GLboolean ShaderState::createShader(const StateConfig &cfg, const std::string &shaderKey) {
47 std::map<GLenum, std::string> unprocessedCode;
48 for (GLint i = 0; i < glenum::glslStageCount(); ++i) {
49 loadStage(cfg.defines_, shaderKey, unprocessedCode, glenum::glslStages()[i]);
50 }
51 if (unprocessedCode.empty()) {
52 REGEN_ERROR("Failed to load shader with key '" << shaderKey << "'");
53 return GL_FALSE;
54 }
55 return createShader(cfg, unprocessedCode);
56}

70GLboolean ShaderState::createShader(const StateConfig &cfg, const std::map<GLenum, std::string> &unprocessedCode) {
71 const std::list<NamedShaderInput> specifiedInput = cfg.inputs_;
72 const std::map<std::string, ref_ptr<Texture> > &textures = cfg.textures_;
73 const std::map<std::string, std::string> &shaderConfig = cfg.defines_;
74 const std::map<std::string, std::string> &shaderFunctions = cfg.functions_;
75 std::map<GLenum, std::string> processedCode;
76
77 PreProcessorConfig preProcessCfg(cfg.version(),
78 unprocessedCode, shaderConfig,
79 shaderFunctions, specifiedInput);
80 Shader::preProcess(processedCode, preProcessCfg);
81 shader_ = ref_ptr<Shader>::alloc(processedCode);
82 // setup transform feedback attributes
83 shader_->setTransformFeedback(cfg.feedbackAttributes_, cfg.feedbackMode_, cfg.feedbackStage_);
84
85 if (!shader_->compile()) {
86 REGEN_ERROR("Shader failed to compiled.");
87 for (auto it = processedCode.begin(); it != processedCode.end(); ++it) {
88 REGEN_DEBUG("Shader code failed to compile:\n" << it->second);
89 }
90 return GL_FALSE;
91 }
92 if (!shader_->link()) {
93 REGEN_ERROR("Shader failed to link.");
94 for (auto it = processedCode.begin(); it != processedCode.end(); ++it) {
95 REGEN_DEBUG("Shader code failed to link:\n" << it->second);
96 }
97 } else {
98 set_isHidden(false);
99 }
100
101 shader_->setInputs(specifiedInput);
102 for (auto it = textures.begin(); it != textures.end(); ++it) {
103 shader_->setTexture(it->second, it->first);
104 }
105
106 return GL_TRUE;
107}

126ref_ptr<Shader> ShaderState::findShader(State *s) {
127 for (auto it = s->joined().rbegin(); it != s->joined().rend(); ++it) {
128 ref_ptr<Shader> out = findShader((*it).get());
129 if (out.get() != nullptr) return out;
130 }
131
132 auto *shaderState = dynamic_cast<ShaderState *>(s);
133 if (shaderState != nullptr) return shaderState->shader();
134
135 auto *hasShader = dynamic_cast<HasShader *>(s);
136 if (hasShader != nullptr) return hasShader->shaderState()->shader();
137
138 return {};
139}

150ref_ptr<ShaderState> ShaderState::load(LoadingContext &ctx, scene::SceneInputNode &input) {
151 if (!input.hasAttribute("key") && !input.hasAttribute("code")) {
152 REGEN_WARN("Ignoring " << input.getDescription() << " without shader input.");
153 return {};
154 }
155 ref_ptr<ShaderState> shaderState = ref_ptr<ShaderState>::alloc();
156
157 const std::string shaderKey = input.hasAttribute("key") ?
158 input.getValue("key") : input.getValue("code");
159 StateConfigurer stateConfigurer;
160 stateConfigurer.addNode(ctx.parent().get());
161 stateConfigurer.addState(shaderState.get());
162 shaderState->createShader(stateConfigurer.cfg(), shaderKey);
163
164 return shaderState;
165}
