30std::string regen::userDirectory() {
31#ifdef WIN32
32 char *userProfile = getenv("USERPROFILE");
33 if(userProfile!=NULL) return string(userProfile);
34
35 char *homeDrive = getenv("HOMEDRIVE");
36 char *homePath = getenv("HOMEPATH");
37 if(homeDrive!=NULL) {
38 if(homePath!=NULL) {
39 return std::string(homeDrive) + std::string(homePath);
40 } else {
41 return std::string(homeDrive);
42 }
43 } else {
44 return "C:";
45 }
46#else
47 char *home = getenv("HOME");
48 if (home != nullptr) return std::string(home);
49
50 return std::string(getpwuid(getuid())->pw_dir);
51#endif
52}

54std::string regen::filesystemPath(
55 const std::string &baseDirectory,
56 const std::string &pathString,
57 const std::string &separators) {
58 boost::filesystem::path p(baseDirectory);
59
60 std::list<std::string> pathNames;
61 boost::split(pathNames, pathString, boost::is_any_of(separators));
62 for (auto &pathName: pathNames) { p /= pathName; }
63
64 return p.string();
65}
