55Texture::Texture(GLuint numTextures)
56 : GLRectangle(glGenTextures, glDeleteTextures, numTextures),
57 ShaderInput1i(REGEN_STRING("textureChannel" << id())),
58 dim_(2),
59 format_(GL_RGBA),
60 internalFormat_(GL_RGBA8),
61 pixelType_(GL_BYTE),
62 border_(0),
63 texBind_(GL_TEXTURE_2D, 0),
64 numSamples_(1),
65 textureData_(nullptr),
66 isTextureDataOwned_(false) {
67 filter_ = new TextureParameterStack<TextureFilter> *[numObjects_];
68 lod_ = new TextureParameterStack<TextureLoD> *[numObjects_];
69 swizzle_ = new TextureParameterStack<TextureSwizzle> *[numObjects_];
70 wrapping_ = new TextureParameterStack<TextureWrapping> *[numObjects_];
71 compare_ = new TextureParameterStack<TextureCompare> *[numObjects_];
72 maxLevel_ = new TextureParameterStack<TextureMaxLevel> *[numObjects_];
73 aniso_ = new TextureParameterStack<TextureAniso> *[numObjects_];
74 for (GLuint i = 0; i < numObjects_; ++i) {
75 filter_[i] = new TextureParameterStack<TextureFilter>(texBind_, Regen_TextureFilter);
76 lod_[i] = new TextureParameterStack<TextureLoD>(texBind_, Regen_TextureLoD);
77 swizzle_[i] = new TextureParameterStack<TextureSwizzle>(texBind_, Regen_TextureSwizzle);
78 wrapping_[i] = new TextureParameterStack<TextureWrapping>(texBind_, Regen_TextureWrapping);
79 compare_[i] = new TextureParameterStack<TextureCompare>(texBind_, Regen_TextureCompare);
80 maxLevel_[i] = new TextureParameterStack<TextureMaxLevel>(texBind_, Regen_TextureMaxLevel);
81 aniso_[i] = new TextureParameterStack<TextureAniso>(texBind_, Regen_TextureAniso);
82 }
83
84 set_rectangleSize(2, 2);
85 samplerType_ = "sampler2D";
86 setUniformData(-1);
87}

168Bounds<Vec2ui> Texture::getRegion(const Vec2f &texco, const Vec2f &regionTS) const {
169 auto w = static_cast<float>(width());
170 auto h = static_cast<float>(height());
171 auto startX = static_cast<unsigned int>(std::floor(texco.x * w));
172 auto startY = static_cast<unsigned int>(std::floor(texco.y * h));
173 auto endX = static_cast<unsigned int>(std::ceil((texco.x + regionTS.x) * w));
174 auto endY = static_cast<unsigned int>(std::ceil((texco.y + regionTS.y) * h));
175 if (endX >= width()) endX = width()-1;
176 if (endY >= height()) endY = height()-1;
177 return {
178 Vec2ui(startX, startY),
179 Vec2ui(endX, endY)};
180}

182unsigned int Texture::texelIndex(const Vec2f &texco) const {
183 auto x = static_cast<unsigned int>(std::round(texco.x * static_cast<float>(width())));
184 auto y = static_cast<unsigned int>(std::round(texco.y * static_cast<float>(height())));
185 // clamp to texture size
186 switch(wrapping_[objectIndex_]->value().x) {
187 case GL_REPEAT:
188 x = x % width();
189 y = y % height();
190 break;
191 case GL_MIRRORED_REPEAT:
192 x = x % (2 * width());
193 y = y % (2 * height());
194 if (x >= width()) x = 2 * width() - x - 1;
195 if (y >= height()) y = 2 * height() - y - 1;
196 break;
197 default: // GL_CLAMP_TO_EDGE:
198 if (x >= width()) x = width()-1;
199 if (y >= height()) y = height()-1;
200 break;
201 }
202 return (y * width() + x);
203}
