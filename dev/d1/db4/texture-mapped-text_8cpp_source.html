68void TextureMappedText::updateAttributes(Alignment alignment, GLfloat maxLineWidth) {
69 Vec3f translation, glyphTranslation;
70 GLuint vertexCounter = 0u;
71
72 GLfloat actualMaxLineWidth = 0.0;
73
74 posAttribute_->setVertexData(numCharacters_ * 6);
75 texcoAttribute_->setVertexData(numCharacters_ * 6);
76 norAttribute_->setVertexData(numCharacters_ * 6);
77 inputContainer_->set_numVertices(numCharacters_ * 6);
78 // map client data for writing
79 auto v_pos = posAttribute_->mapClientData<Vec3f>(ShaderData::WRITE);
80 auto v_texco = texcoAttribute_->mapClientData<Vec3f>(ShaderData::WRITE);
81 auto v_nor = norAttribute_->mapClientData<Vec3f>(ShaderData::WRITE);
82
83 translation = Vec3f(0.0, 0.0, 0.0);
84 glyphTranslation = Vec3f(0.0, 0.0, 0.0);
85
86 for (auto it = value_.begin(); it != value_.end(); ++it) {
87 translation.y -= font_->lineHeight() * height_;
88
89 GLfloat buf;
90 // actual width for this line
91 GLfloat lineWidth = 0.0;
92 // remember space for splitting string at words
93 GLint lastSpaceIndex = 0;
94 GLfloat lastSpaceWidth = 0.0;
95
96 // get line width and split the line
97 // where it exceeds the width limit
98 for (GLuint i = 0; i < it->size(); ++i) {
99 const wchar_t &ch = (*it)[i];
100 buf = lineWidth + font_->faceData(ch).advanceX * height_;
101 if (maxLineWidth > 0.0 && buf > maxLineWidth && lastSpaceIndex != 0) {
102 // maximal line length reached
103 // split string at remembered space
104
105 // insert the rest after current iterator
106 // it cannot be placed in one line with
107 // the words before lastSpaceIndex
108 auto nextIt = it;
109 nextIt++;
110 value_.insert(nextIt, it->substr(lastSpaceIndex + 1,
111 it->size() - lastSpaceIndex - 1));
112 // set the string width validated width
113 *it = it->substr(0, lastSpaceIndex);
114
115 lineWidth = lastSpaceWidth;
116 break;
117 } else if (ch == ' ' || ch == '\t') {
118 // remember spaces
119 lastSpaceIndex = i;
120 lastSpaceWidth = lineWidth;
121 }
122 lineWidth = buf;
123 }
124 if (lineWidth > actualMaxLineWidth) {
125 actualMaxLineWidth = lineWidth;
126 }
127
128 // calculate line advance!
129 switch (alignment) {
130 case ALIGNMENT_CENTER:
131 translation.x = 0.5f * (maxLineWidth - lineWidth);
132 break;
133 case ALIGNMENT_RIGHT:
134 translation.x = maxLineWidth - lineWidth;
135 break;
136 case ALIGNMENT_LEFT:
137 translation.x = 0.0f;
138 break;
139 }
140
141 // create the geometry with appropriate
142 // translation and size for each glyph
143 for (GLuint i = 0; i < it->size(); ++i) {
144 const wchar_t &ch = (*it)[i];
145 const Font::FaceData &data = font_->faceData(ch);
146
147 glyphTranslation = Vec3f(
148 data.left * height_, (data.top - data.height) * height_, 0.001 * (i + 1)
149 );
150 makeGlyphGeometry(data, translation + glyphTranslation, (float) ch,
151 v_pos,
152 v_nor,
153 v_texco,
154 &vertexCounter);
155
156 // move cursor to next glyph
157 translation.x += data.advanceX * height_;
158 }
159
160 translation.x = 0.0;
161 }
162
163 // apply offset to each vertex
164 if (centerAtOrigin_) {
165 GLfloat centerOffset = actualMaxLineWidth * 0.5f;
166 for (GLuint i = 0; i < vertexCounter; ++i) {
167 v_pos.w[i].x -= centerOffset;
168 }
169 }
170
171 v_pos.unmap();
172 v_nor.unmap();
173 v_texco.unmap();
174
175 begin(ShaderInputContainer::INTERLEAVED);
176 setInput(posAttribute_);
177 setInput(norAttribute_);
178 setInput(texcoAttribute_);
179 end();
180 updateVAO(RenderState::get());
181
182 // set center and extends for bounding box
183 minPosition_ = v_pos.w[0];
184 maxPosition_ = v_pos.w[0];
185 for (GLuint i = 1; i < vertexCounter; ++i) {
186 minPosition_.setMin(v_pos.w[i]);
187 maxPosition_.setMax(v_pos.w[i]);
188 }
189}

191void TextureMappedText::makeGlyphGeometry(
192 const Font::FaceData &data,
193 const Vec3f &translation,
194 GLfloat layer,
195 ShaderData_rw<Vec3f> &posAttribute,
196 ShaderData_rw<Vec3f> &norAttribute,
197 ShaderData_rw<Vec3f> &texcoAttribute,
198 GLuint *vertexCounter) {
199 GLuint &i = *vertexCounter;
200 Vec3f p0 = translation + Vec3f(0.0, data.height * height_, 0.0);
201 Vec3f p1 = translation + Vec3f(0.0, 0.0, 0.0);
202 Vec3f p2 = translation + Vec3f(data.width * height_, 0.0, 0.0);
203 Vec3f p3 = translation + Vec3f(data.width * height_, data.height * height_, 0.0);
204 Vec3f n = Vec3f(0.0, 0.0, 1.0);
205 Vec3f texco0(0.0, 0.0, layer);
206 Vec3f texco1(0.0, data.uvY, layer);
207 Vec3f texco2(data.uvX, data.uvY, layer);
208 Vec3f texco3(data.uvX, 0.0, layer);
209
210 posAttribute.w[i] = p0;
211 posAttribute.w[i + 1] = p1;
212 posAttribute.w[i + 2] = p2;
213 posAttribute.w[i + 3] = p2;
214 posAttribute.w[i + 4] = p3;
215 posAttribute.w[i + 5] = p0;
216
217 norAttribute.w[i] = n;
218 norAttribute.w[i + 1] = n;
219 norAttribute.w[i + 2] = n;
220 norAttribute.w[i + 3] = n;
221 norAttribute.w[i + 4] = n;
222 norAttribute.w[i + 5] = n;
223
224 texcoAttribute.w[i] = texco0;
225 texcoAttribute.w[i + 1] = texco1;
226 texcoAttribute.w[i + 2] = texco2;
227 texcoAttribute.w[i + 3] = texco2;
228 texcoAttribute.w[i + 4] = texco3;
229 texcoAttribute.w[i + 5] = texco0;
230
231 *vertexCounter += 6;
232}
