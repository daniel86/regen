40static bool assimpLog_(std::string &msg, const std::string &prefix) {
41 if (hasPrefix(msg, prefix)) {
42 msg = truncPrefix(msg, prefix);
43 msg[msg.size() - 1] = '0';
44 return true;
45 } else {
46 return false;
47 }

50static void assimpLog(const char *msg_, char *) {
51 string msg(msg_);
52 if (assimpLog_(msg, "Info,")) { REGEN_INFO(msg); }
53 else if (assimpLog_(msg, "Warn,")) { REGEN_WARN(msg); }
54 else if (assimpLog_(msg, "Error,")) { REGEN_ERROR(msg); }
55 else if (assimpLog_(msg, "Debug,")) { REGEN_DEBUG(msg); }
56 else
57 REGEN_DEBUG(msg);
58}

60static const struct aiScene *importFile(
61 const string &assimpFile,
62 GLint userSpecifiedFlags) {
63 // get a handle to the predefined STDOUT log stream and attach
64 // it to the logging system. It remains active for all further
65 // calls to aiImportFile(Ex) and aiApplyPostProcessing.
66 static struct aiLogStream stream;
67 static GLboolean isLoggingInitialled = false;
68 if (!isLoggingInitialled) {
69 stream.callback = assimpLog;
70 stream.user = nullptr;
71 aiDetachAllLogStreams();
72 aiAttachLogStream(&stream);
73 isLoggingInitialled = true;
74 }
75
76 if (userSpecifiedFlags == -1) {
77 return aiImportFile(assimpFile.c_str(),
78 aiProcess_Triangulate
79 // Convert special texture coords to UV
80 | aiProcess_GenUVCoords
81 | aiProcess_CalcTangentSpace
82 | aiProcess_FlipUVs
83 | aiProcess_SortByPType
84 // Reorders triangles for better vertex cache locality.
85 | aiProcess_ImproveCacheLocality
86 // Searches for redundant/unreferenced materials and removes them.
87 | aiProcess_RemoveRedundantMaterials
88 // A postprocessing step to reduce the number of meshes.
89 | aiProcess_OptimizeMeshes
90 // A postprocessing step to optimize the scene hierarchy.
91 | aiProcess_OptimizeGraph
92 // If this flag is not specified,
93 // no vertices are referenced by more than one face
94 | aiProcess_JoinIdenticalVertices
95 | 0);
96 } else {
97 return aiImportFile(assimpFile.c_str(),
98 userSpecifiedFlags | 0);
99 }
100}

141static void setLightRadius(aiLight *aiLight, ref_ptr<Light> &light) {
142 GLfloat ax = aiLight->mAttenuationLinear;
143 GLfloat ay = aiLight->mAttenuationConstant;
144 GLfloat az = aiLight->mAttenuationQuadratic;
145 GLfloat z = ay / (2.0 * az);
146
147 GLfloat start = 0.01;
148 GLfloat stop = 0.99;
149
150 GLfloat inner = -z + sqrt(z * z - (ax / start - 1.0 / (start * az)));
151 GLfloat outer = -z + sqrt(z * z - (ax / stop - 1.0 / (stop * az)));
152
153 light->radius()->setVertex(0, Vec2f(inner, outer));
154}

156vector<ref_ptr<Light> > AssetImporter::loadLights() {
157 vector<ref_ptr<Light> > ret(scene_->mNumLights);
158
159 for (GLuint i = 0; i < scene_->mNumLights; ++i) {
160 aiLight *assimpLight = scene_->mLights[i];
161 // node could be animated, but for now it is ignored
162 aiNode *node = nodes_[string(assimpLight->mName.data)];
163 aiVector3D lightPos = node->mTransformation * assimpLight->mPosition;
164
165 ref_ptr<Light> light;
166 switch (assimpLight->mType) {
167 case aiLightSource_DIRECTIONAL: {
168 light = ref_ptr<Light>::alloc(Light::DIRECTIONAL);
169 light->direction()->setVertex(0, *((Vec3f *) &lightPos.x));
170 break;
171 }
172 case aiLightSource_POINT: {
173 light = ref_ptr<Light>::alloc(Light::POINT);
174 light->position()->setVertex(0, *((Vec3f *) &lightPos.x));
175 setLightRadius(assimpLight, light);
176 break;
177 }
178 case aiLightSource_SPOT: {
179 light = ref_ptr<Light>::alloc(Light::SPOT);
180 light->position()->setVertex(0, *((Vec3f *) &lightPos.x));
181 light->direction()->setVertex(0, *((Vec3f *) &assimpLight->mDirection.x));
182 light->set_outerConeAngle(
183 acos(assimpLight->mAngleOuterCone) * 360.0 / (2.0 * M_PI));
184 light->set_innerConeAngle(
185 acos(assimpLight->mAngleInnerCone) * 360.0 / (2.0 * M_PI));
186 setLightRadius(assimpLight, light);
187 break;
188 }
189 case aiLightSource_AREA:
190 REGEN_WARN("Area lights are not supported.");
191 break;
192 case aiLightSource_AMBIENT:
193 case aiLightSource_UNDEFINED:
194 case _aiLightSource_Force32Bit:
195 break;
196 }
197 if (light.get() == nullptr) { continue; }
198
199 lightToAiLight_[light.get()] = assimpLight;
200 //light->set_ambient( aiToOgle(&assimpLight->mColorAmbient) );
201 light->diffuse()->setVertex(0, aiToOgle(&assimpLight->mColorDiffuse));
202 light->specular()->setVertex(0, aiToOgle(&assimpLight->mColorSpecular));
203
204 ret[i] = light;
205 }
206
207 return ret;
208}

210ref_ptr<LightNode> AssetImporter::loadLightNode(const ref_ptr<Light> &light) {
211 aiLight *assimpLight = lightToAiLight_[light.get()];
212 if (assimpLight == nullptr) { return {}; }
213
214 aiNode *node = nodes_[string(assimpLight->mName.data)];

221}

225static void loadTexture(
226 ref_ptr<Material> &mat,
227 aiTexture *aiTexture,
228 aiMaterial *aiMat,
229 aiString &stringVal,
230 GLuint l, GLuint k,
231 const string &texturePath) {
232 ref_ptr<Texture> tex;
233 string filePath = "";
234 GLuint maxElements;
235 GLint intVal;
236 GLfloat floatVal;
237
238 if (aiTexture == nullptr) {
239 if (boost::filesystem::exists(stringVal.data)) {
240 filePath = stringVal.data;
241 } else {
242 vector<string> names;
243 filePath = stringVal.data;
244 boost::split(names, filePath, boost::is_any_of("/\\"));
245 filePath = names[names.size() - 1];
246
247 string buf = REGEN_STRING(texturePath << "/" << filePath);
248 if (boost::filesystem::exists(buf)) {
249 filePath = buf;
250 } else {
251 REGEN_ERROR("Unable to load texture '" << buf << "'.");
252 return;
253 }
254 }
255
256 try {
257 // try image texture
258 tex = textures::load(filePath);
259 }
260 catch (textures::Error &ie) {
261 // try video texture
262 ref_ptr<VideoTexture> vid = ref_ptr<VideoTexture>::alloc();
263 try {
264 vid->set_file(filePath);
265 tex = vid;
266 vid->startAnimation();
267 }
268 catch (VideoTexture::Error &ve) {
269 REGEN_ERROR("Failed to load texture '" << stringVal.data << "'.");
270 return;
271 }
272 }
273 } else if (aiTexture->mHeight == 0) {
274 // The texture is stored in a "compressed" format such as DDS or PNG
275
276 GLuint numBytes = aiTexture->mWidth;
277 string ext(aiTexture->achFormatHint);
278
279 ILenum type;
280 if (ext == "jpg" || ext == ".j")
281 type = IL_JPG;
282 else if (ext == "png")
283 type = IL_PNG;
284 else if (ext == "tga")
285 type = IL_TGA;
286 else if (ext == "bmp")
287 type = IL_BMP;
288 else if (ext == "dds")
289 type = IL_DDS;
290 else if (ext == "gif")
291 type = IL_GIF;
292 else if (ext == "tif")
293 type = IL_TIF;
294 else {
295 REGEN_ERROR("Unknown texture type '" << ext << "'.");
296 return;
297 }
298
299 try {
300 tex = textures::load(type, numBytes, aiTexture->pcData);
301 }
302 catch (textures::Error &ie) {
303 REGEN_ERROR("Failed to load texture '" << stringVal.data << "'.");
304 return;
305 }
306 } else { // The texture is NOT compressed
307 tex = ref_ptr<Texture2D>::alloc();
308 tex->begin(RenderState::get());
309 tex->set_rectangleSize(aiTexture->mWidth, aiTexture->mHeight);
310 tex->set_textureData((GLubyte *) aiTexture->pcData);
311 tex->set_pixelType(GL_UNSIGNED_BYTE);
312 tex->set_format(GL_RGBA);
313 tex->set_internalFormat(GL_RGBA8);
314 tex->filter().push(GL_LINEAR);
315 tex->wrapping().push(GL_REPEAT);
316 tex->texImage();
317 tex->end(RenderState::get());
318 tex->set_textureData(nullptr);
319 }
320
321 ref_ptr<TextureState> texState = ref_ptr<TextureState>::alloc(tex);
322 tex->begin(RenderState::get());
323
324 // Defines miscellaneous flag for the n'th texture on the stack 't'.
325 // This is a bitwise combination of the aiTextureFlags enumerated values.
326 maxElements = 1;
327 if (AI_SUCCESS == aiGetMaterialIntegerArray(aiMat,
328 AI_MATKEY_TEXFLAGS(textureTypes[l], k - 1), &intVal, &maxElements)) {
329 if (intVal & aiTextureFlags_Invert) {
330 texState->set_texelTransferKey("regen.states.textures.transfer.texel_invert");
331 }
332 if (intVal & aiTextureFlags_UseAlpha) {
333 REGEN_WARN("aiTextureFlags_UseAlpha is not supported.");
334 }
335 if (intVal & aiTextureFlags_IgnoreAlpha) {
336 texState->set_ignoreAlpha(GL_TRUE);
337 }
338 }
339
340 // Defines the height scaling of a bump map (for stuff like Parallax Occlusion Mapping)
341 maxElements = 1;
342 if (aiGetMaterialFloatArray(aiMat, AI_MATKEY_BUMPSCALING,
343 &floatVal, &maxElements) == AI_SUCCESS) {
344 texState->set_blendFactor(floatVal);
345 }
346
347 // Defines the strength the n'th texture on the stack 't'.
348 // All color components (rgb) are multipled with this factor *before* any further processing is done. -
349 maxElements = 1;
350 if (AI_SUCCESS == aiGetMaterialFloatArray(aiMat,
351 AI_MATKEY_TEXBLEND(textureTypes[l], k - 1), &floatVal, &maxElements)) {
352 texState->set_blendFactor(floatVal);
353 }
354
355 // One of the aiTextureOp enumerated values. Defines the arithmetic operation to be used
356 // to combine the n'th texture on the stack 't' with the n-1'th.
357 // TEXOP(t,0) refers to the blend operation between the base color
358 // for this stack (e.g. COLOR_DIFFUSE for the diffuse stack) and the first texture.
359 maxElements = 1;
360 if (AI_SUCCESS == aiGetMaterialIntegerArray(aiMat,
361 AI_MATKEY_TEXOP(textureTypes[l], k - 1), &intVal, &maxElements)) {
362 switch (intVal) {
363 case aiTextureOp_Multiply:
364 texState->set_blendMode(BLEND_MODE_MULTIPLY);
365 break;
366 case aiTextureOp_SignedAdd:
367 case aiTextureOp_Add:
368 texState->set_blendMode(BLEND_MODE_ADD);
369 break;
370 case aiTextureOp_Subtract:
371 texState->set_blendMode(BLEND_MODE_SUBSTRACT);
372 break;
373 case aiTextureOp_Divide:
374 texState->set_blendMode(BLEND_MODE_DIVIDE);
375 break;
376 case aiTextureOp_SmoothAdd:
377 texState->set_blendMode(BLEND_MODE_SMOOTH_ADD);
378 break;
379 }
380 }
381
382 // Defines the base axis to to compute the mapping coordinates for the n'th texture
383 // on the stack 't' from. This is not required for UV-mapped textures.
384 // For instance, if MAPPING(t,n) is aiTextureMapping_SPHERE,
385 // U and V would map to longitude and latitude of a sphere around the given axis.
386 // The axis is given in local mesh space.
387 //TEXMAP_AXIS(t,n) aiVector3D n/a
388
389 // Defines how the input mapping coordinates for sampling the n'th texture on the stack 't'
390 // are computed. Usually explicit UV coordinates are provided, but some model file formats
391 // might also be using basic shapes, such as spheres or cylinders, to project textures onto meshes.
392 maxElements = 1;
393 if (AI_SUCCESS == aiGetMaterialIntegerArray(aiMat,
394 AI_MATKEY_MAPPING(textureTypes[l], k - 1), &intVal, &maxElements)) {
395 switch (intVal) {
396 case aiTextureMapping_UV:
397 // The mapping coordinates are taken from an UV channel.
398 texState->set_mapping(TextureState::MAPPING_TEXCO);
399 break;
400 case aiTextureMapping_SPHERE:
401 texState->set_mapping(TextureState::MAPPING_SPHERE);
402 break;
403 case aiTextureMapping_CYLINDER:
404 texState->set_mapping(TextureState::MAPPING_TUBE);
405 break;
406 case aiTextureMapping_BOX:
407 texState->set_mapping(TextureState::MAPPING_CUBE);
408 break;
409 case aiTextureMapping_PLANE:
410 texState->set_mapping(TextureState::MAPPING_FLAT);
411 break;
412 case aiTextureMapping_OTHER:
413 break;
414 }
415 }
416
417 // Defines the UV channel to be used as input mapping coordinates for sampling the
418 // n'th texture on the stack 't'. All meshes assigned to this material share
419 // the same UV channel setup Presence of this key implies MAPPING(t,n) to be
420 // aiTextureMapping_UV. See How to map UV channels to textures (MATKEY_UVWSRC) for more details.
421 maxElements = 1;
422 if (AI_SUCCESS == aiGetMaterialIntegerArray(aiMat,
423 AI_MATKEY_UVWSRC(textureTypes[l], k - 1), &intVal, &maxElements)) {
424 texState->set_texcoChannel(intVal);
425 }
426
427 TextureWrapping wrapping_(GL_REPEAT);
428 // Any of the aiTextureMapMode enumerated values. Defines the texture wrapping mode on the
429 // x axis for sampling the n'th texture on the stack 't'.
430 // 'Wrapping' occurs whenever UVs lie outside the 0..1 range.
431 maxElements = 1;
432 if (AI_SUCCESS == aiGetMaterialIntegerArray(aiMat,
433 AI_MATKEY_MAPPINGMODE_U(textureTypes[l], k - 1), &intVal,
434 &maxElements)) {
435 switch (intVal) {
436 case aiTextureMapMode_Wrap:
437 wrapping_.x = GL_REPEAT;
438 break;
439 case aiTextureMapMode_Clamp:
440 wrapping_.x = GL_CLAMP;
441 break;
442 case aiTextureMapMode_Decal:
443 REGEN_WARN("ignoring texture map mode decal.");
444 break;
445 case aiTextureMapMode_Mirror:
446 wrapping_.x = GL_MIRRORED_REPEAT;
447 break;
448 }
449 }
450 // Wrap mode on the v axis. See MAPPINGMODE_U.
451 maxElements = 1;
452 if (AI_SUCCESS == aiGetMaterialIntegerArray(aiMat,
453 AI_MATKEY_MAPPINGMODE_V(textureTypes[l], k - 1), &intVal,
454 &maxElements)) {
455 switch (intVal) {
456 case aiTextureMapMode_Wrap:
457 wrapping_.y = GL_REPEAT;
458 break;
459 case aiTextureMapMode_Clamp:
460 wrapping_.y = GL_CLAMP;
461 break;
462 case aiTextureMapMode_Decal:
463 REGEN_WARN("ignoring texture map mode decal.");
464 break;
465 case aiTextureMapMode_Mirror:
466 wrapping_.y = GL_MIRRORED_REPEAT;
467 break;
468 }
469 }
470#ifdef AI_MATKEY_MAPPINGMODE_W
471 // Wrap mode on the v axis. See MAPPINGMODE_U.
472 maxElements = 1;
473 if(AI_SUCCESS == aiGetMaterialIntegerArray(aiMat,
474 AI_MATKEY_MAPPINGMODE_W(textureTypes[l],k-1), &intVal, &maxElements))
475 {
476 switch(intVal) {
477 case aiTextureMapMode_Wrap:
478 wrapping_.z = GL_REPEAT;
479 break;
480 case aiTextureMapMode_Clamp:
481 wrapping_.z = GL_CLAMP;
482 break;
483 case aiTextureMapMode_Decal:
484 REGEN_WARN("ignoring texture map mode decal.");
485 break;
486 case aiTextureMapMode_Mirror:
487 wrapping_.z = GL_MIRRORED_REPEAT;
488 break;
489 }
490 }
491#endif
492 tex->wrapping().push(wrapping_);
493
494 switch (textureTypes[l]) {
495 case aiTextureType_DIFFUSE:
496 // The texture is combined with the result of the diffuse lighting equation.
497 texState->set_mapTo(TextureState::MAP_TO_DIFFUSE);
498 break;
499 case aiTextureType_AMBIENT:
500 // The texture is combined with the result of the ambient lighting equation.
501 texState->set_mapTo(TextureState::MAP_TO_AMBIENT);
502 break;
503 case aiTextureType_SPECULAR:
504 // The texture is combined with the result of the specular lighting equation.
505 texState->set_mapTo(TextureState::MAP_TO_SPECULAR);
506 break;
507 case aiTextureType_SHININESS:
508 // The texture defines the glossiness of the material.
509 // The glossiness is in fact the exponent of the specular (phong) lighting equation.
510 // Usually there is a conversion function defined to map the linear color values
511 // in the texture to a suitable exponent. Have fun.
512 texState->set_mapTo(TextureState::MAP_TO_SHININESS);
513 break;
514 case aiTextureType_EMISSIVE:
515 // The texture is added to the result of the lighting calculation.
516 texState->set_mapTo(TextureState::MAP_TO_EMISSION);
517 // enforce multiply blending such that matEmission parameter gives easy control over
518 // amount of material emission.
519 texState->set_blendMode(BLEND_MODE_MULTIPLY);
520 break;
521 case aiTextureType_OPACITY:
522 // The texture defines per-pixel opacity.
523 // Usually 'white' means opaque and 'black' means 'transparency'.
524 // Or quite the opposite. Have fun.
525 texState->set_mapTo(TextureState::MAP_TO_ALPHA);
526 // TODO: make configurable
527 // - alpha discard threshold
528 // - texel invert
529 texState->set_discardAlpha(true, 0.25f);
530 //texState->set_texelTransferKey("regen.states.textures.transfer.texel_invert");
531 break;
532 case aiTextureType_LIGHTMAP:
533 // Lightmap texture (aka Ambient Occlusion). Both 'Lightmaps' and
534 // dedicated 'ambient occlusion maps' are covered by this material property.
535 // The texture contains a scaling value for the final color value of a pixel.
536 // Its intensity is not affected by incoming light.
537 texState->set_mapTo(TextureState::MAP_TO_LIGHT);
538 break;
539 case aiTextureType_REFLECTION:
540 // Reflection texture. Contains the color of a perfect mirror reflection.
541 // Rarely used, almost never for real-time applications.
542 //texState->setMapTo(MAP_TO_REFLECTION);
543 break;
544 case aiTextureType_DISPLACEMENT:
545 // Displacement texture. The exact purpose and format is application-dependent.
546 // Higher color values stand for higher vertex displacements.
547 texState->set_mapTo(TextureState::MAP_TO_DISPLACEMENT);
548 break;
549 case aiTextureType_HEIGHT:
550 // The texture is a height map. By convention, higher gray-scale values
551 // stand for higher elevations from the base height.
552 texState->set_mapTo(TextureState::MAP_TO_HEIGHT);
553 break;
554 case aiTextureType_NORMALS:
555 // The texture is a (tangent space) normal-map.
556 texState->set_mapTo(TextureState::MAP_TO_NORMAL);
557 break;
558 case aiTextureType_NONE:
559 // Dummy value. No texture, but the value to be used as 'texture semantic'
560 // (aiMaterialProperty::mSemantic) for all material properties *not* related to textures.
561 break;
562 case aiTextureType_UNKNOWN:
563 // Unknown texture. A texture reference that does not match any of the definitions
564 // above is considered to be 'unknown'. It is still imported, but is excluded
565 // from any further postprocessing.
566 REGEN_WARN("Unknown texture type '" << stringVal.data << "' in '" << filePath << "'.");
567 break;
568 default:
569 REGEN_WARN("Unhandled texture type '" << stringVal.data << "' in '" << filePath << "'.");
570 break;
571 }
572
573 tex->filter().push(TextureFilter(GL_LINEAR_MIPMAP_LINEAR, GL_LINEAR));
574 tex->setupMipmaps(GL_DONT_CARE);
575 mat->joinStates(texState);
576
577 tex->end(RenderState::get());
578 GL_ERROR_LOG();
579}

583vector<ref_ptr<Material> > AssetImporter::loadMaterials() {
584 vector<ref_ptr<Material> > materials(scene_->mNumMaterials);
585 aiColor4D aiCol;
586 GLfloat floatVal;
587 GLint intVal;
588 aiString stringVal;
589 GLuint maxElements;
590 GLuint l, k;
591
592 GL_ERROR_LOG();
593 for (GLuint n = 0; n < scene_->mNumMaterials; ++n) {
594 ref_ptr<Material> mat = ref_ptr<Material>::alloc();
595 materials[n] = mat;
596 aiMaterial *aiMat = scene_->mMaterials[n];
597
598 // load textures
599 for (l = 0; l < numTextureTyps; ++l) {
600 k = 0;
601 while (AI_SUCCESS == aiGetMaterialString(aiMat,
602 AI_MATKEY_TEXTURE(textureTypes[l], k), &stringVal)) {
603 k += 1;
604 if (stringVal.length < 1) continue;
605
606 aiTexture *aiTex;
607 if (stringVal.data[0] == '*') {
608 char *indexStr = stringVal.data + 1;
609 GLuint index;
610
611 stringstream ss;
612 ss << indexStr;
613 ss >> index;
614 if (index < scene_->mNumTextures) {
615 aiTex = scene_->mTextures[index];
616 } else {
617 aiTex = nullptr;
618 }
619 } else {
620 aiTex = nullptr;
621 }
622
623 loadTexture(mat, aiTex, aiMat, stringVal, l, k, texturePath_);
624 }
625 }
626
627 if (AI_SUCCESS == aiGetMaterialColor(aiMat,
628 AI_MATKEY_COLOR_DIFFUSE, &aiCol)) {
629 mat->diffuse()->setVertex(0, aiToOgle3f(&aiCol));
630 }
631 if (AI_SUCCESS == aiGetMaterialColor(aiMat,
632 AI_MATKEY_COLOR_SPECULAR, &aiCol)) {
633 mat->specular()->setVertex(0, aiToOgle3f(&aiCol));
634 }
635 if (AI_SUCCESS == aiGetMaterialColor(aiMat,
636 AI_MATKEY_COLOR_AMBIENT, &aiCol)) {
637 mat->ambient()->setVertex(0, aiToOgle3f(&aiCol));
638 }
639 if (AI_SUCCESS == aiGetMaterialColor(aiMat,
640 AI_MATKEY_COLOR_EMISSIVE, &aiCol)) {
641 mat->set_emission(aiToOgle3f(&aiCol));
642 }
643 // Defines the transparent color of the material,
644 // this is the color to be multiplied with the color of translucent light to
645 // construct the final 'destination color' for a particular position in the screen buffer.
646 if (AI_SUCCESS == aiGetMaterialColor(aiMat, AI_MATKEY_COLOR_TRANSPARENT, &aiCol)) {
647 // not supposed to be used like this but for now i think this is ok...
648 auto alpha = mat->alpha()->getVertex(0).r;
649 mat->alpha()->setVertex(0, alpha * (aiCol.r + aiCol.g + aiCol.b) / 3.0f);
650 }
651
652 maxElements = 1;
653 // Defines the base shininess of the material
654 // This is the exponent of the phong shading equation.
655 if (aiGetMaterialFloatArray(aiMat, AI_MATKEY_SHININESS,
656 &floatVal, &maxElements) == AI_SUCCESS) {
657 mat->shininess()->setVertex(0, floatVal);
658 }
659 maxElements = 1;
660 // Defines the strength of the specular highlight.
661 // This is simply a multiplier to the specular color of a material
662 if (aiGetMaterialFloatArray(aiMat, AI_MATKEY_SHININESS_STRENGTH,

773 GLuint &currentIndex,

806ref_ptr<Mesh> AssetImporter::loadMesh(const struct aiMesh &mesh, const Mat4f &transform, VBO::Usage usage) {
807 ref_ptr<Mesh> meshState = ref_ptr<Mesh>::alloc(GL_TRIANGLES, usage);
808 stringstream s;
809
810 ref_ptr<ShaderInput3f> pos = ref_ptr<ShaderInput3f>::alloc(ATTRIBUTE_NAME_POS);
811 ref_ptr<ShaderInput3f> nor = ref_ptr<ShaderInput3f>::alloc(ATTRIBUTE_NAME_NOR);
812 ref_ptr<ShaderInput4f> tan = ref_ptr<ShaderInput4f>::alloc(ATTRIBUTE_NAME_TAN);
813
814 const GLuint numFaceIndices = (mesh.mNumFaces > 0 ? mesh.mFaces[0].mNumIndices : 0);
815 GLuint numFaces = 0;
816 for (GLuint t = 0u; t < mesh.mNumFaces; ++t) {
817 const struct aiFace *face = &mesh.mFaces[t];
818 if (face->mNumIndices != numFaceIndices) { continue; }
819 numFaces += 1;
820 }
821 const GLuint numIndices = numFaceIndices * numFaces;
822
823 switch (numFaceIndices) {
824 case 1:
825 meshState->set_primitive(GL_POINTS);
826 break;
827 case 2:
828 meshState->set_primitive(GL_LINES);
829 break;
830 case 3:
831 meshState->set_primitive(GL_TRIANGLES);
832 break;
833 default:
834 meshState->set_primitive(GL_POLYGON);
835 break;
836 }
837
838 meshState->begin(ShaderInputContainer::INTERLEAVED);
839
840 {
841 ref_ptr<ShaderInput1ui> indices = ref_ptr<ShaderInput1ui>::alloc("i");
842 indices->setVertexData(numIndices);
843 auto faceIndices = indices->mapClientData<GLuint>(ShaderData::WRITE);
844 GLuint index = 0, maxIndex = 0;
845 for (GLuint t = 0u; t < mesh.mNumFaces; ++t) {
846 const struct aiFace *face = &mesh.mFaces[t];
847 if (face->mNumIndices != numFaceIndices) { continue; }
848 for (GLuint n = 0; n < face->mNumIndices; ++n) {
849 faceIndices.w[index] = face->mIndices[n];
850 if (face->mIndices[n] > maxIndex) { maxIndex = face->mIndices[n]; }
851 index += 1;
852 }
853 }
854 faceIndices.unmap();
855 meshState->setIndices(indices, maxIndex);
856 }
857
858 const auto *aiTransform = (const aiMatrix4x4 *) &transform.x;
859 // vertex positions
860 Vec3f min_(999999.9);
861 Vec3f max_(-999999.9);
862 GLuint numVertices = mesh.mNumVertices;
863 {
864 pos->setVertexData(numVertices);
865 auto v_pos = pos->mapClientData<Vec3f>(ShaderData::WRITE);
866 for (GLuint n = 0; n < numVertices; ++n) {
867 aiVector3D aiv = (*aiTransform) * mesh.mVertices[n];
868 Vec3f &v = *((Vec3f *) &aiv.x);
869 v_pos.w[n] = v;
870 min_.setMin(v);
871 max_.setMax(v);
872 }
873 v_pos.unmap();
874 meshState->setInput(pos);
875 }
876 meshState->set_bounds(min_, max_);
877
878 // per vertex normals
879 if (mesh.HasNormals()) {
880 nor->setVertexData(numVertices);
881 auto v_nor = nor->mapClientData<Vec3f>(ShaderData::WRITE);
882 for (GLuint n = 0; n < numVertices; ++n) {
883 Vec3f &v = *((Vec3f *) &mesh.mNormals[n].x);
884 v_nor.w[n] = v;
885 }
886 v_nor.unmap();
887 meshState->setInput(nor);
888 }
889
890 // per vertex colors
891 for (GLuint t = 0; t < AI_MAX_NUMBER_OF_COLOR_SETS; ++t) {
892 if (mesh.mColors[t] == nullptr) continue;
893
894 ref_ptr<ShaderInput4f> col = ref_ptr<ShaderInput4f>::alloc(REGEN_STRING("col" << t));
895 col->setVertexData(numVertices);
896 auto v_col = col->mapClientData<Vec4f>(ShaderData::WRITE);
897 for (GLuint n = 0; n < numVertices; ++n) {
898 v_col.w[n] = Vec4f(
899 mesh.mColors[t][n].r,
900 mesh.mColors[t][n].g,
901 mesh.mColors[t][n].b,
902 mesh.mColors[t][n].a);
903 }
904 v_col.unmap();
905 meshState->setInput(col);
906 }
907
908 // load texture coordinates
909 for (GLuint t = 0; t < AI_MAX_NUMBER_OF_TEXTURECOORDS; ++t) {
910 if (mesh.mTextureCoords[t] == nullptr) { continue; }
911 aiVector3D *aiTexcos = mesh.mTextureCoords[t];
912 GLuint texcoComponents = mesh.mNumUVComponents[t];
913 string texcoName = REGEN_STRING("texco" << t);
914
915 ref_ptr<ShaderInput> texco;
916 if (texcoComponents == 1) {
917 texco = ref_ptr<ShaderInput1f>::alloc(texcoName);
918 } else if (texcoComponents == 3) {
919 texco = ref_ptr<ShaderInput3f>::alloc(texcoName);
920 } else if (texcoComponents == 4) {
921 texco = ref_ptr<ShaderInput4f>::alloc(texcoName);
922 } else {
923 texco = ref_ptr<ShaderInput2f>::alloc(texcoName);
924 }
925 texco->setVertexData(numVertices);
926 auto v_texco = texco->mapClientData<float>(ShaderData::WRITE);
927
928 for (GLuint n = 0; n < numVertices; ++n) {
929 GLfloat *aiTexcoData = &(aiTexcos[n].x);
930 for (GLuint x = 0; x < texcoComponents; ++x) v_texco.w[x] = aiTexcoData[x];
931 v_texco.w += texcoComponents;
932 }
933 v_texco.unmap();
934 meshState->setInput(texco);
935 }
936
937 // load tangents
938 if (mesh.HasTangentsAndBitangents()) {
939 tan->setVertexData(numVertices);
940 auto v_tan = tan->mapClientData<Vec4f>(ShaderData::WRITE);
941 for (GLuint i = 0; i < numVertices; ++i) {
942 Vec3f &t = *((Vec3f *) &mesh.mTangents[i].x);
943 Vec3f &b = *((Vec3f *) &mesh.mBitangents[i].x);
944 Vec3f &n = *((Vec3f *) &mesh.mNormals[i].x);
945 // Calculate the handedness of the local tangent space.
946 GLfloat handeness;
947 if (n.cross(t).dot(b) < 0.0) {
948 handeness = -1.0;
949 } else {
950 handeness = 1.0;
951 }
952 v_tan.w[i] = Vec4f(t.x, t.y, t.z, handeness);
953 }
954 v_tan.unmap();
955 meshState->setInput(tan);
956 }
957 GL_ERROR_LOG();
958
959 // A mesh may have a set of bones in the form of aiBone structures..
960 // Bones are a means to deform a mesh according to the movement of a skeleton.
961 // Each bone has a name and a set of vertices on which it has influence.
962 // Its offset matrix declares the transformation needed to transform from mesh space
963 // to the local space of this bone.
964 if (mesh.HasBones()) {
965 typedef list<pair<GLfloat, GLuint> > WeightList;
966 map<GLuint, WeightList> vertexToWeights;
967 GLuint maxNumWeights = 0;
968
969 // collect weights at vertices
970 for (GLuint boneIndex = 0; boneIndex < mesh.mNumBones; ++boneIndex) {
971 aiBone *assimpBone = mesh.mBones[boneIndex];
972 for (GLuint t = 0; t < assimpBone->mNumWeights; ++t) {
973 aiVertexWeight &weight = assimpBone->mWeights[t];
974 vertexToWeights[weight.mVertexId].emplace_back(weight.mWeight, boneIndex);
975 maxNumWeights = max(maxNumWeights,
976 (GLuint) vertexToWeights[weight.mVertexId].size());
977 }
978 }
979 meshState->shaderDefine("NUM_BONE_WEIGHTS", REGEN_STRING(maxNumWeights));
980
981 if (maxNumWeights < 1) {
982 REGEN_ERROR("The model has invalid bone weights number " << maxNumWeights << ".");
983 } else {
984 // create array of bone weights and indices
985 auto boneWeights = ref_ptr<ShaderInput1f>::alloc("boneWeights", maxNumWeights);
986 auto boneIndices = ref_ptr<ShaderInput1ui>::alloc("boneIndices", maxNumWeights);
987 boneWeights->setVertexData(numVertices);
988 boneIndices->setVertexData(numVertices);
989 auto v_weights = boneWeights->mapClientData<GLfloat>(ShaderData::WRITE);
990 auto v_indices = boneIndices->mapClientData<GLuint>(ShaderData::WRITE);
991
992 for (GLuint j = 0; j < numVertices; j++) {
993 WeightList &vWeights = vertexToWeights[j];
994
995 GLuint k = 0;
996 for (auto it = vWeights.begin(); it != vWeights.end(); ++it) {
997 v_weights.w[k] = it->first;
998 v_indices.w[k] = it->second;
999 ++k;
1000 }
1001 for (; k < maxNumWeights; ++k) {
1002 v_weights.w[k] = 0.0f;
1003 v_indices.w[k] = 0u;
1004 }
1005
1006 v_weights.w += maxNumWeights;
1007 v_indices.w += maxNumWeights;
1008 }
1009
1010 v_weights.unmap();
1011 v_indices.unmap();
1012
1013 if (maxNumWeights > 1) {
1014 meshState->setInput(boneWeights);
1015 }
1016 meshState->setInput(boneIndices);
1017 }
1018 }
1019 GL_ERROR_LOG();
1020
1021 meshState->end();
1022
1023 return meshState;
1024}
