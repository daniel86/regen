19static void processMeshChildren(LoadingContext &ctx, scene::SceneInputNode &input, MeshVector &x) {
20 auto parser = ctx.scene();
21
22 for (auto &child: input.getChildren()) {
23 auto processor = parser->getStateProcessor(child->getCategory());
24 if (processor.get() == nullptr) {
25 REGEN_WARN("No processor registered for '" << child->getDescription() << "'.");
26 continue;
27 }
28 if (x.size() > 1) {
29 auto state = ref_ptr<State>::alloc();
30 processor->processInput(parser, *child.get(), ctx.parent(), state);
31 for (auto &mesh: x) {
32 mesh->joinStates(state);
33 }
34 } else {
35 processor->processInput(parser, *child.get(), ctx.parent(), x[0]);
36 }
37 }

304ref_ptr<MeshVector> MeshVector::createAssetMeshes(LoadingContext &ctx, scene::SceneInputNode &input, const ref_ptr<AssetImporter> &importer) {
305 const auto vboUsage = input.getValue<VBO::Usage>("usage", VBO::USAGE_DYNAMIC);
306 const auto scaling = input.getValue<Vec3f>("scaling", Vec3f(1.0f));
307 const auto rotation = input.getValue<Vec3f>("rotation", Vec3f(0.0f));
308 const auto translation = input.getValue<Vec3f>("translation", Vec3f(0.0f));
309 const auto assetIndices = input.getValue<std::string>("asset-indices", "*");
310 bool useAnimation = input.getValue<bool>("asset-animation", false);
311
312 ref_ptr<MeshVector> out_ = ref_ptr<MeshVector>::alloc();
313 MeshVector &out = *out_.get();
314
315 const vector <ref_ptr<NodeAnimation>> &nodeAnims = importer->getNodeAnimations();
316 if (useAnimation && nodeAnims.empty()) {
317 REGEN_WARN(input.getDescription() << " has use-animation=1 but Asset '" <<
318 input.getValue("asset") << "' has not.");
319 useAnimation = false;
320 }
321
322 // Create import model transform
323 Mat4f transform = Mat4f::scaleMatrix(scaling);
324 Quaternion q(0.0, 0.0, 0.0, 1.0);
325 q.setEuler(rotation.x, rotation.y, rotation.z);
326 transform *= q.calculateMatrix();
327 transform.translate(translation);
328
329 // Asset can contain multiple meshes, they can be accessed by index.
330 // Parse user specified indices ...
331 std::vector<std::string> indicesStr;
332 boost::split(indicesStr, assetIndices, boost::is_any_of(","));
333 vector <GLuint> indices(indicesStr.size());
334 bool useAllIndices = false;
335 for (GLuint i = 0u; i < indices.size(); ++i) {
336 if (indicesStr[i] == "*") {
337 useAllIndices = true;
338 break;
339 } else {
340 std::stringstream ss(indicesStr[i]);
341 ss >> indices[i];
342 }
343 }
344
345 if (useAllIndices) {
346 out = importer->loadAllMeshes(transform, vboUsage);
347 } else {
348 out = importer->loadMeshes(transform, vboUsage, indices);
349 }
350 for (GLuint i = 0u; i < out.size(); ++i) {
351 ref_ptr<Mesh> mesh = out[i];
352 if (mesh.get() == nullptr) continue;
353
354 // Join in material state.
355 // Can be set to false to allow overwriting material stuff.
356 if (input.getValue<bool>("asset-material", true)) {
357 ref_ptr<Material> material =
358 importer->getMeshMaterial(mesh.get());
359 if (material.get() != nullptr) {
360 mesh->joinStates(material);
361 }
362 }
363
364 if (useAnimation) {
365 std::list<ref_ptr<AnimationNode> > meshBones;
366 GLuint numBoneWeights = importer->numBoneWeights(mesh.get());
367 GLuint numBones = 0u;
368
369 // Find bones influencing this mesh
370 for (auto it = nodeAnims.begin(); it != nodeAnims.end(); ++it) {
371 std::list<ref_ptr<AnimationNode> > ibonNodes =
372 importer->loadMeshBones(mesh.get(), it->get());
373 meshBones.insert(meshBones.end(), ibonNodes.begin(), ibonNodes.end());
374 numBones = ibonNodes.size();
375 }
376
377 // Create Bones state that is responsible for uploading
378 // animation data to GL.
379 if (!meshBones.empty()) {
380 ref_ptr<Bones> bonesState = ref_ptr<Bones>::alloc(numBoneWeights, numBones);
381 bonesState->setBones(meshBones);
382 bonesState->setAnimationName(REGEN_STRING("bones-" << input.getName()));
383 bonesState->startAnimation();
384 mesh->joinStates(bonesState);
385 }
386 }
387 }
388
389 return out_;
390}

457ref_ptr<Particles> MeshVector::createParticleMesh(LoadingContext &ctx, scene::SceneInputNode &input, const GLuint numParticles) {
458 ref_ptr<Particles> particles;
459 auto parser = ctx.scene();
460 if (input.hasAttribute("update-shader")) {
461 const auto updateShader = input.getValue("update-shader");
462 particles = ref_ptr<Particles>::alloc(numParticles, updateShader);
463 } else {
464 particles = ref_ptr<Particles>::alloc(numParticles);
465 }
466 particles->setAnimationName(input.getName());
467 particles->startAnimation();
468
469 if (input.hasAttribute("max-emits")) {
470 particles->setMaxEmits(input.getValue<GLuint>("max-emits", 100u));
471 }
472 if (input.hasAttribute("animation-state")) {
473 auto animNodeName = input.getValue("animation-state");
474 auto animState = parser->getState(animNodeName);
475 if (animState.get() == nullptr) {
476 // Try to load the animation node
477 auto animNodeInput = parser->getRoot()->getFirstChild("node", animNodeName);
478 if (animNodeInput.get() != nullptr) {
479 auto animNode = ref_ptr<StateNode>::alloc();
480 animNode->set_name(animNodeName);
481 parser->processNode(animNode, animNodeName);
482 if (animNode->childs().empty()) {
483 animState = animNode->state();
484 } else {
485 animState = animNode->childs().front()->state();
486 }
487 parser->putState(animNodeName, animState);
488 }
489 }
490 if (animState.get() == nullptr) {
491 REGEN_WARN("Ignoring " << input.getDescription() << ", unknown animation-node.");
492 } else {
493 particles->joinAnimationState(animState);
494 }
495 }
496
497 // the attributes of particles may have special attributes in the scene file that determine default, variance, etc.
498 // these are ignored by processMeshChildren, so we need to process them here.
499 for (auto &child: input.getChildren()) {
500 if (child->getCategory() == "input" && child->getValue<bool>("is-attribute", false)) {
501 const std::string type = child->getValue("type");
502 if (type == "float") {
503 configureParticleAttribute<ShaderInput1f, GLfloat>(parser, particles, *child.get());
504 } else if (type == "vec2") {
505 configureParticleAttribute<ShaderInput2f, Vec2f>(parser, particles, *child.get());
506 } else if (type == "vec3") {
507 configureParticleAttribute<ShaderInput3f, Vec3f>(parser, particles, *child.get());
508 } else if (type == "vec4") {
509 configureParticleAttribute<ShaderInput4f, Vec4f>(parser, particles, *child.get());
510 } else if (type == "int") {
511 configureParticleAttribute<ShaderInput1i, GLint>(parser, particles, *child.get());
512 } else if (type == "ivec2") {
513 configureParticleAttribute<ShaderInput2i, Vec2i>(parser, particles, *child.get());
514 } else if (type == "ivec3") {
515 configureParticleAttribute<ShaderInput3i, Vec3i>(parser, particles, *child.get());
516 } else if (type == "ivec4") {
517 configureParticleAttribute<ShaderInput4i, Vec4i>(parser, particles, *child.get());
518 } else {
519 REGEN_WARN("Ignoring " << child->getDescription() << ", unknown attribute type '" << type << "'.");
520 }
521 }
522 }
523
524 particles->begin();
525 // Mesh resources can have State children
526 auto x = ref_ptr<MeshVector>::alloc(1);
527 (*x.get())[0] = particles;
528 parser->putResource<MeshVector>(input.getName(), x); // TODO check if this is needed
529 processMeshChildren(ctx, input, *x.get());
530 particles->end();
531
532 return particles;
533}
