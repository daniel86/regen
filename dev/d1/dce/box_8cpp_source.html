26 std::istream &operator>>(std::istream &in, Box::TexcoMode &mode) {
27 std::string val;
28 in >> val;
29 boost::to_upper(val);
30 if (val == "NONE") mode = Box::TEXCO_MODE_NONE;
31 else if (val == "UV") mode = Box::TEXCO_MODE_UV;
32 else if (val == "CUBE_MAP") mode = Box::TEXCO_MODE_CUBE_MAP;

93void Box::generateLODLevel(
94 const Config &cfg,
95 GLuint sideIndex,
96 GLuint lodLevel,
97 const std::vector<Tessellation> &tessellations) {
98 static const Vec3f cubeNormals[] = {
99 Vec3f(0.0f, 0.0f, 1.0f), // Front
100 Vec3f(0.0f, 0.0f, -1.0f), // Back
101 Vec3f(0.0f, 1.0f, 0.0f), // Top
102 Vec3f(0.0f, -1.0f, 0.0f), // Bottom
103 Vec3f(1.0f, 0.0f, 0.0f), // Right
104 Vec3f(-1.0f, 0.0f, 0.0f) // Left
105 };
106 static const Mat4f faceRotations[] = {
107 Mat4f::identity(), // Front
108 Mat4f::rotationMatrix(0.0f, M_PI, 0.0f), // Back
109 Mat4f::rotationMatrix(M_PI_2, 0.0f, 0.0f), // Top
110 Mat4f::rotationMatrix(-M_PI_2, 0.0f, 0.0f), // Bottom
111 Mat4f::rotationMatrix(0.0f, -M_PI_2, 0.0f), // Right
112 Mat4f::rotationMatrix(0.0f, M_PI_2, 0.0f) // Left
113 };
114
115 auto &tessellation = tessellations[lodLevel];
116 auto vertexOffset = meshLODs_[lodLevel].vertexOffset + sideIndex * tessellation.vertices.size();
117 auto indexOffset = meshLODs_[lodLevel].indexOffset + sideIndex * tessellation.outputFaces.size() * 3;
118 const Vec3f &normal = cubeNormals[sideIndex];
119 const Mat4f &faceRotMat = faceRotations[sideIndex];
120 GLuint nextIndex = indexOffset;
121
122 // map client data for writing
123 auto indices = indices_->mapClientData<GLuint>(ShaderData::WRITE);
124 auto pos = pos_->mapClientData<Vec3f>(ShaderData::WRITE);
125 auto nor = (cfg.isNormalRequired ?
126 nor_->mapClientData<Vec3f>(ShaderData::WRITE) :
127 ShaderData_rw<Vec3f>::nullData());
128 auto tan = (cfg.isTangentRequired ?
129 tan_->mapClientData<Vec4f>(ShaderData::WRITE) :
130 ShaderData_rw<Vec4f>::nullData());
131
132 for (const auto &tessFace: tessellation.outputFaces) {
133 indices.w[nextIndex++] = vertexOffset + tessFace.v1;
134 indices.w[nextIndex++] = vertexOffset + tessFace.v2;
135 indices.w[nextIndex++] = vertexOffset + tessFace.v3;
136 }
137
138 GLuint triIndices[3];
139 Vec3f triVertices[3];
140 Vec2f triTexco[3];
141
142 for (GLuint faceIndex = 0; faceIndex < tessellation.outputFaces.size(); ++faceIndex) {
143 const auto &face = tessellation.outputFaces[faceIndex];
144 GLuint faceVertIndex = 0;
145
146 for (const auto &tessIndex: {face.v1, face.v2, face.v3}) {
147 const auto &vertex = tessellation.vertices[tessIndex];
148 auto vertexIndex = vertexOffset + tessIndex;
149 triIndices[faceVertIndex] = vertexIndex;
150
151 Vec3f faceVertex = faceRotMat.transformVector(vertex) + normal;
152 Vec3f transformedVertex = cfg.posScale * modelRotation_.transformVector(faceVertex);
153 pos.w[vertexIndex] = transformedVertex;
154 minPosition_.setMin(transformedVertex);
155 maxPosition_.setMax(transformedVertex);
156 if (cfg.isNormalRequired) {
157 nor.w[vertexIndex] = normal;
158 }
159 if (texcoMode_ == TEXCO_MODE_CUBE_MAP) {
160 auto texco = texco_->mapClientData<Vec3f>(ShaderData::WRITE);
161 Vec3f v = faceVertex;
162 v.normalize();
163 texco.w[vertexIndex] = v;
164 triTexco[faceVertIndex] = Vec2f(vertex.x, vertex.y) * 0.5f + Vec2f(0.5f);
165 } else if (texcoMode_ == TEXCO_MODE_UV) {
166 Vec2f uv;
167 switch (sideIndex) {
168 case 0: // Front face
169 uv = Vec2f((faceVertex.x + 1.0f) * 0.5f, (faceVertex.y + 1.0f) * 0.5f);
170 uv.x *= cfg.posScale.x;
171 uv.y *= cfg.posScale.y;
172 break;
173 case 1: // Back face
174 uv = Vec2f((faceVertex.x + 1.0f) * 0.5f, (faceVertex.y + 1.0f) * 0.5f);
175 uv.x *= cfg.posScale.x;
176 uv.y *= cfg.posScale.y;
177 break;
178 case 2: // Top face
179 uv = Vec2f((faceVertex.x + 1.0f) * 0.5f, (faceVertex.z + 1.0f) * 0.5f);
180 uv.x *= cfg.posScale.x;
181 uv.y *= cfg.posScale.z;
182 break;
183 case 3: // Bottom face
184 uv = Vec2f((faceVertex.x + 1.0f) * 0.5f, (faceVertex.z + 1.0f) * 0.5f);
185 uv.x *= cfg.posScale.x;
186 uv.y *= cfg.posScale.z;
187 break;
188 case 4: // Right face
189 uv = Vec2f((faceVertex.z + 1.0f) * 0.5f, (faceVertex.y + 1.0f) * 0.5f);
190 uv.x *= cfg.posScale.z;
191 uv.y *= cfg.posScale.y;
192 break;
193 case 5: // Left face
194 uv = Vec2f((1.0f - faceVertex.z) * 0.5f, (faceVertex.y + 1.0f) * 0.5f);
195 uv.x *= cfg.posScale.z;
196 uv.y *= cfg.posScale.y;
197 break;
198 default:
199 uv = Vec2f(0.0f);
200 }
201 uv *= cfg.texcoScale;
202 auto texco = texco_->mapClientData<Vec2f>(ShaderData::WRITE);
203 texco.w[vertexIndex] = uv;
204 triTexco[faceVertIndex] = uv;
205 }
206 if (cfg.isTangentRequired) {
207 triVertices[faceVertIndex] = pos.w[vertexIndex];
208 }
209 faceVertIndex += 1;
210 }
211
212 if (cfg.isTangentRequired) {
213 Vec4f tangent = calculateTangent(triVertices, triTexco, normal);
214 for (GLuint i = 0; i < 3; ++i) {
215 tan.w[triIndices[i]] = tangent;
216 }
217 }
218 }

221void Box::updateAttributes(const Config &cfg) {
222 std::vector<Tessellation> tessellations;
223 GLuint numVertices = 0;
224 GLuint numIndices = 0;
225
226 // Generate base tessellation for a single face (front face)
227 Tessellation baseTess;
228 baseTess.vertices.resize(4);
229 baseTess.vertices[0] = Vec3f(-1.0, -1.0, 0.0);
230 baseTess.vertices[1] = Vec3f(1.0, -1.0, 0.0);
231 baseTess.vertices[2] = Vec3f(1.0, 1.0, 0.0);
232 baseTess.vertices[3] = Vec3f(-1.0, 1.0, 0.0);
233 baseTess.inputFaces.resize(2);
234 baseTess.inputFaces[0] = TessellationFace(0, 1, 3);
235 baseTess.inputFaces[1] = TessellationFace(1, 2, 3);
236
237 // Generate tessellations for each LOD level
238 for (GLuint lodLevel: cfg.levelOfDetails) {
239 auto &lodTess = tessellations.emplace_back();
240 lodTess.vertices = baseTess.vertices;
241 lodTess.inputFaces = baseTess.inputFaces;
242 tessellate(lodLevel, lodTess);
243
244 auto &x = meshLODs_.emplace_back();
245 x.numVertices = lodTess.vertices.size() * 6; // 6 faces
246 x.numIndices = lodTess.outputFaces.size() * 3 * 6; // 6 faces
247 x.vertexOffset = numVertices;
248 x.indexOffset = numIndices;
249 numVertices += x.numVertices;
250 numIndices += x.numIndices;
251 }
252
253 modelRotation_ = Mat4f::rotationMatrix(cfg.rotation.x, cfg.rotation.y, cfg.rotation.z);
254
255 // allocate attributes
256 pos_->setVertexData(numVertices);
257 if (cfg.isNormalRequired) {
258 nor_->setVertexData(numVertices);
259 }
260 if (cfg.isTangentRequired) {
261 tan_->setVertexData(numVertices);
262 }
263 TexcoMode texcoMode = cfg.texcoMode;
264 if (cfg.isTangentRequired && cfg.texcoMode == TEXCO_MODE_NONE) {
265 texcoMode = TEXCO_MODE_UV;
266 }
267 if (texcoMode == TEXCO_MODE_CUBE_MAP) {
268 texco_ = ref_ptr<ShaderInput3f>::alloc("texco0");
269 texco_->setVertexData(numVertices);
270 } else if (texcoMode == TEXCO_MODE_UV) {
271 texco_ = ref_ptr<ShaderInput2f>::alloc("texco0");
272 texco_->setVertexData(numVertices);
273 }
274 indices_->setVertexData(numIndices);
275
276 minPosition_ = Vec3f(999999.9);
277 maxPosition_ = Vec3f(-999999.9);
278 for (auto lodLevel = 0u; lodLevel < tessellations.size(); ++lodLevel) {
279 for (GLuint sideIndex = 0; sideIndex < 6; ++sideIndex) {
280 generateLODLevel(cfg, sideIndex, lodLevel, tessellations);
281 }
282 }
283
284 begin(ShaderInputContainer::INTERLEAVED);
285 auto indexRef = setIndices(indices_, numVertices);
286 setInput(pos_);
287 if (cfg.isNormalRequired)
288 setInput(nor_);
289 if (cfg.texcoMode != TEXCO_MODE_NONE)
290 setInput(texco_);
291 if (cfg.isTangentRequired)
292 setInput(tan_);
293 auto vertexRef = end();
294
295 for (auto &x: meshLODs_) {
296 // add the index buffer offset (in number of bytes)
297 x.indexOffset = indexRef->address() + x.indexOffset * sizeof(GLuint);
298 }
299 activateLOD(0);
