100void Camera::setPerspective(float aspect, float fov, float near, float far) {
101 bool hasLayeredProjection = proj_->numArrayElements() > 1;
102 if (hasLayeredProjection) {
103 for (unsigned int i = 0; i < numLayer_; ++i) {
104 setPerspective(aspect, fov, near, far, i);
105 }
106 } else {
107 setPerspective(aspect, fov, near, far, 0);
108 for (unsigned int i = 1; i < numLayer_; ++i) {
109 frustum_[i].setPerspective(aspect, fov, near, far);
110 }
111 }

114void Camera::setPerspective(float aspect, float fov, float near, float far, unsigned int layer) {
115 fov_->setVertexClamped(layer, fov);
116 aspect_->setVertexClamped(layer, aspect);
117 near_->setVertexClamped(layer, near);
118 far_->setVertexClamped(layer, far);
119 frustum_[layer].setPerspective(aspect, fov, near, far);
120
121 auto projectionMatrix = Mat4f::projectionMatrix(fov, aspect, near, far);
122 proj_->setVertexClamped(layer, projectionMatrix);
123 projInv_->setVertexClamped(layer, projectionMatrix.projectionInverse());
124 isOrtho_ = false;
125}

133void Camera::setOrtho(float left, float right, float bottom, float top, float near, float far, unsigned int layer) {
134 fov_->setVertex(0, 0.0f);
135 aspect_->setVertex(0, abs((right - left) / (top - bottom)));
136 near_->setVertexClamped(layer, near);
137 far_->setVertexClamped(layer, far);
138 frustum_[layer].setOrtho(left, right, bottom, top, near, far);
139 proj_->setVertex(layer, Mat4f::orthogonalMatrix(left, right, bottom, top, near, far));
140 projInv_->setVertex(layer, proj_->getVertex(layer).r.orthogonalInverse());
141 isOrtho_ = true;
142}

155bool Camera::updateView() {
156 auto posStamp = position_->stamp();
157 auto dirStamp = direction_->stamp();
158 if (posStamp == posStamp_ && dirStamp == dirStamp_) { return false; }
159 posStamp_ = posStamp;
160 dirStamp_ = dirStamp;
161
162 auto numViewLayers = view_->numArrayElements();
163 for (unsigned int i = 0; i < numViewLayers; ++i) {
164 auto dir = direction_->getVertexClamped(i);
165 if (std::abs(dir.r.dot(Vec3f::up())) > 0.999f) {
166 auto viewMatrix = Mat4f::lookAtMatrix(
167 position_->getVertexClamped(i).r,
168 dir.r, Vec3f::right());
169 view_->setVertex(i, viewMatrix);
170 viewInv_->setVertex(i, viewMatrix.lookAtInverse());
171 } else {
172 auto viewMatrix = Mat4f::lookAtMatrix(
173 position_->getVertexClamped(i).r,
174 dir.r, Vec3f::up());
175 view_->setVertex(i, viewMatrix);
176 viewInv_->setVertex(i, viewMatrix.lookAtInverse());
177 }
178 }
179
180 return true;
181}

194void Camera::updateViewProjection(unsigned int projectionIndex, unsigned int viewIndex) {
195 auto maxIndex = std::max(projectionIndex, viewIndex);

198 viewProjInv_->setVertex(maxIndex,

200 frustum_[maxIndex].update(
201 position()->getVertexClamped(maxIndex).r,
202 direction()->getVertexClamped(maxIndex).r);

215 bool updated = false;
216 if (attachedPosition_.get()) {

218 poseStamp_ = attachedPosition_->stamp();

223 if (poseStamp_ != attachedTransform_->stamp()) {
224 poseStamp_ = attachedTransform_->stamp();
225 auto m = attachedTransform_->getVertex(0);
226 position_->setVertex(0, m.r.position());
227 if (!isAttachedToPosition_) {
228 // TODO: change camera orientation based on transform
229 //direction_->setVertex(0, (m ^ Vec4f(Vec3f::front(),0.0)).xyz_());
230 }
231 updated = true;
232 }
233 }
234
235 if (updated) {
236 updateCamera();
237 }

273 auto d = Plane(
274 position()->getVertex(0).r,

276 return d - radius < far()->getVertex(0).r &&
277 d + radius > near()->getVertex(0).r;

280bool Camera::hasSphereIntersection(const Vec3f &center, const Vec3f *points) const {
281 Plane p(position()->getVertex(0).r, direction()->getVertex(0).r);
282 for (int i = 0; i < 8; ++i) {
283 auto d = p.distance(center + points[i]);
284 if (d > far()->getVertex(0).r || d < near()->getVertex(0).r)
285 return false;
286 }
287 return true;

290bool Camera::hasHalfSphereIntersection(const Vec3f &center, GLfloat radius) const {
291 // get the distance from the camera to the center of the sphere
292 auto d = Plane(
293 position()->getVertex(0).r,
294 direction()->getVertex(0).r).distance(center);
295 // check if the sphere is outside the far plane
296 if (d - radius > far()->getVertex(0).r) return false;
297 // check if the sphere is inside the near plane
298 if (d + radius < near()->getVertex(0).r) return false;
299 // check if the sphere is inside the half sphere
300 auto halfSphereRadius = far()->getVertex(0);
301 auto halfSphereNormal = direction()->getVertex(0);
302 auto halfSphereCenter = position()->getVertex(0).r + halfSphereNormal.r * halfSphereRadius.r;
303 return Plane(halfSphereCenter, halfSphereNormal.r).distance(center) < radius;
304}

306bool Camera::hasHalfSphereIntersection(const Vec3f &center, const Vec3f *points) const {
307 // get the distance from the camera to the center of the sphere
308 auto d = Plane(
309 position()->getVertex(0).r,
310 direction()->getVertex(0).r).distance(center);
311 // check if the sphere is outside the far plane
312 if (d > far()->getVertex(0).r) return false;
313 // check if the sphere is inside the near plane
314 if (d < near()->getVertex(0).r) return false;
315 // check if the sphere is inside the half sphere
316 auto halfSphereRadius = far()->getVertex(0);
317 auto halfSphereNormal = direction()->getVertex(0);
318 auto halfSphereCenter = position()->getVertex(0).r + halfSphereNormal.r * halfSphereRadius.r;
319 auto halfSphere = Plane(halfSphereCenter, halfSphereNormal.r);
320 for (int i = 0; i < 8; ++i) {
321 if (halfSphere.distance(center + points[i]) < 0) {
322 return false;
323 }
324 }
325 return true;

385int getHiddenFacesMask(scene::SceneInputNode &input) {
386 int hiddenFacesMask = 0;
387 if (input.hasAttribute("hide-faces")) {
388 auto val = input.getValue<std::string>("hide-faces", "");
389 std::vector<std::string> faces;
390 boost::split(faces, val, boost::is_any_of(","));
391 for (auto it = faces.begin(); it != faces.end(); ++it) {
392 CubeCamera::Face face;
393 std::stringstream(*it) >> face;
394 hiddenFacesMask |= face;
395 }
396 }
397 return hiddenFacesMask;

400ref_ptr<Camera> createLightCamera(LoadingContext &ctx, scene::SceneInputNode &input) {
401 ref_ptr<Light> light = ctx.scene()->getResource<Light>(input.getValue("light"));
402 if (light.get() == nullptr) {
403 REGEN_WARN("Unable to find Light for '" << input.getDescription() << "'.");
404 return {};
405 }
406 auto numLayer = input.getValue<GLuint>("num-layer", 1u);
407 auto splitWeight = input.getValue<GLdouble>("split-weight", 0.9);
408 auto cameraType = input.getValue<std::string>("camera-type", "spot");
409 auto near = input.getValue<float>("near", 0.1f);
410 ref_ptr<Camera> lightCamera;
411
412 switch (light->lightType()) {
413 case Light::SPOT: {
414 auto spotCam = ref_ptr<LightCamera_Spot>::alloc(light);
415 spotCam->setLightNear(near);
416 lightCamera = spotCam;
417 break;
418 }
419 case Light::DIRECTIONAL: {
420 auto userCamera = ctx.scene()->getResource<Camera>(input.getValue("camera"));
421 if (userCamera.get() == nullptr) {
422 REGEN_WARN("Unable to find user camera for '" << input.getDescription() << "'.");
423 return {};
424 }
425 auto dirCam = ref_ptr<LightCamera_CSM>::alloc(light, userCamera, numLayer);
426 dirCam->setSplitWeight(splitWeight);
427 lightCamera = dirCam;
428 break;
429 }
430 case Light::POINT: {
431 if (cameraType == "parabolic" || cameraType == "paraboloid") {
432 // parabolic camera
433 auto isDualParabolic = input.getValue<bool>("dual-paraboloid", true);
434 auto parabolic = ref_ptr<LightCamera_Parabolic>::alloc(light, isDualParabolic);
435 if (input.hasAttribute("normal")) {
436 parabolic->setNormal(input.getValue<Vec3f>("normal", Vec3f::down()));
437 }
438 parabolic->setLightNear(near);
439 lightCamera = parabolic;
440 } else {
441 // cube camera
442 auto cube = ref_ptr<LightCamera_Cube>::alloc(light, getHiddenFacesMask(input));
443 cube->setLightNear(near);
444 lightCamera = cube;
445 }
446 break;
447 }
448 }
449 if (lightCamera.get() == nullptr) {
450 REGEN_WARN("Unable to create camera for '" << input.getDescription() << "'.");
451 return {};
452 }
453 ctx.scene()->putState(input.getName(), lightCamera);
454
455 return lightCamera;
456}

459 class ProjectionUpdater : public EventHandler {
460 public:
461 ProjectionUpdater(const ref_ptr<Camera> &cam,
462 const ref_ptr<ShaderInput2i> &windowViewport)
463 : EventHandler(), cam_(cam), windowViewport_(windowViewport) {}
464
465 void call(EventObject *, EventData *) {
466 auto windowViewport = windowViewport_->getVertex(0);
467 auto windowAspect =
468 (GLfloat) windowViewport.r.x / (GLfloat) windowViewport.r.y;
469 if (cam_->isOrtho()) {
470 // keep the ortho width and adjust height based on aspect ratio
471 auto width = cam_->frustum()[0].nearPlaneHalfSize.x * 2.0f;
472 auto height = width / windowAspect;
473 cam_->setOrtho(
474 -width / 2.0f, width / 2.0f,
475 -height / 2.0f, height / 2.0f,
476 cam_->near()->getVertex(0).r,
477 cam_->far()->getVertex(0).r);
478 } else {
479 cam_->setPerspective(
480 windowAspect,
481 cam_->fov()->getVertex(0).r,
482 cam_->near()->getVertex(0).r,
483 cam_->far()->getVertex(0).r);
484 }
485 }
486
487 protected:
488 ref_ptr<Camera> cam_;
489 ref_ptr<ShaderInput2i> windowViewport_;
490 };

493ref_ptr<Camera> Camera::createCamera(LoadingContext &ctx, scene::SceneInputNode &input) {
494 auto camType = input.getValue<std::string>("type", "spot");
495
496 if (input.hasAttribute("reflector") ||
497 input.hasAttribute("reflector-normal") ||
498 input.hasAttribute("reflector-point")) {
499 ref_ptr<Camera> userCamera =
500 ctx.scene()->getResource<Camera>(input.getValue("camera"));
501 if (userCamera.get() == nullptr) {
502 REGEN_WARN("Unable to find Camera for '" << input.getDescription() << "'.");
503 return {};
504 }
505 ref_ptr<ReflectionCamera> cam;
506 bool hasBackFace = input.getValue<bool>("has-back-face", false);
507
508 if (input.hasAttribute("reflector")) {
509 ref_ptr<MeshVector> mesh =
510 ctx.scene()->getResource<MeshVector>(input.getValue("reflector"));
511 if (mesh.get() == nullptr || mesh->empty()) {
512 REGEN_WARN("Unable to find Mesh for '" << input.getDescription() << "'.");
513 return {};
514 }
515 const std::vector<ref_ptr<Mesh> > &vec = *mesh.get();
516 cam = ref_ptr<ReflectionCamera>::alloc(
517 userCamera, vec[0], input.getValue<GLuint>("vertex-index", 0u), hasBackFace);
518 } else if (input.hasAttribute("reflector-normal")) {
519 auto normal = input.getValue<Vec3f>("reflector-normal", Vec3f(0.0f, 1.0f, 0.0f));
520 auto position = input.getValue<Vec3f>("reflector-point", Vec3f(0.0f, 0.0f, 0.0f));
521 cam = ref_ptr<ReflectionCamera>::alloc(userCamera, normal, position, hasBackFace);
522 }
523 if (cam.get()) {
524 ctx.scene()->putState(input.getName(), cam);
525 }
526 return cam;
527 } else if (input.hasAttribute("light")) {
528 auto cam = createLightCamera(ctx, input);
529 return cam;
530 } else if (camType == "cube") {
531 auto tf = ctx.scene()->getResource<ModelTransformation>(input.getValue("tf"));
532 ref_ptr<CubeCamera> cam = ref_ptr<CubeCamera>::alloc(getHiddenFacesMask(input));
533 if (tf.get()) {
534 cam->attachToPosition(tf->get());
535 }
536 ctx.scene()->putState(input.getName(), cam);
537 return cam;
538 } else if (camType == "parabolic" || camType == "paraboloid") {
539 auto tf = ctx.scene()->getResource<ModelTransformation>(input.getValue("tf"));
540 bool hasBackFace = input.getValue<bool>("dual-paraboloid", true);
541 auto cam = ref_ptr<ParabolicCamera>::alloc(hasBackFace);
542 if (input.hasAttribute("normal")) {
543 cam->setNormal(input.getValue<Vec3f>("normal", Vec3f::down()));
544 }
545
546 if (tf.get()) {
547 cam->attachToPosition(tf->get());
548 }
549 ctx.scene()->putState(input.getName(), cam);
550
551 return cam;
552 } else {
553 ref_ptr<Camera> cam = ref_ptr<Camera>::alloc(1);
554 cam->set_isAudioListener(
555 input.getValue<bool>("audio-listener", false));
556 cam->position()->setVertex(0,
557 input.getValue<Vec3f>("position", Vec3f(0.0f, 2.0f, -2.0f)));
558
559 auto dir = input.getValue<Vec3f>("direction", Vec3f(0.0f, 0.0f, 1.0f));
560 dir.normalize();
561 cam->direction()->setVertex(0, dir);
562
563 if (camType == "ortho" || camType == "orthographic" || camType == "orthogonal") {
564 auto width = input.getValue<GLfloat>("width", 10.0f);
565 auto height = input.getValue<GLfloat>("height", 10.0f);
566 cam->setOrtho(
567 -width / 2.0f, width / 2.0f,
568 -height / 2.0f, height / 2.0f,
569 input.getValue<GLfloat>("near", 0.1f),
570 input.getValue<GLfloat>("far", 200.0f));
571 } else {
572 auto viewport = ctx.scene()->getViewport()->getVertex(0);
573 cam->setPerspective(
574 (GLfloat) viewport.r.x / (GLfloat) viewport.r.y,
575 input.getValue<GLfloat>("fov", 45.0f),
576 input.getValue<GLfloat>("near", 0.1f),
577 input.getValue<GLfloat>("far", 200.0f));
578 // Update frustum when window size changes
579 ctx.scene()->addEventHandler(Application::RESIZE_EVENT,
580 ref_ptr<ProjectionUpdater>::alloc(cam, ctx.scene()->getViewport()));
581 }
582 cam->updateCamera();
583 ctx.scene()->putState(input.getName(), cam);
584
585 return cam;
586 }
