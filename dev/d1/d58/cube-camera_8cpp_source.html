12CubeCamera::CubeCamera(int hiddenFacesMask)
13 : Camera(6),
14 hiddenFacesMask_(hiddenFacesMask) {
15 shaderDefine("RENDER_TARGET", "CUBE");

21 // Set matrix array size

23 // the matrices are still allocated for all faces. Could change the shaders to
24 // compute index based on layer and face visibility, instead of using gl_Layer directly.
25 view_->set_numArrayElements(numLayer_);

27 viewProj_->set_numArrayElements(numLayer_);
28 viewProjInv_->set_numArrayElements(numLayer_);
29
30 // Allocate matrices
31 view_->setUniformUntyped();
32 viewInv_->setUniformUntyped();
33 viewProj_->setUniformUntyped();
34 viewProjInv_->setUniformUntyped();

38 direction_->setUniformUntyped();

50bool CubeCamera::isCubeFaceVisible(int face) const {
51 switch (face) {
52 case 0: return !(hiddenFacesMask_ & POS_X);
53 case 1: return !(hiddenFacesMask_ & NEG_X);
54 case 2: return !(hiddenFacesMask_ & POS_Y);
55 case 3: return !(hiddenFacesMask_ & NEG_Y);
56 case 4: return !(hiddenFacesMask_ & POS_Z);
57 case 5: return !(hiddenFacesMask_ & NEG_Z);
58 default: return false;
59 }
60}

62bool CubeCamera::updateView() {
63 auto posStamp = position_->stamp();
64 if (posStamp == posStamp_) { return false; }
65 posStamp_ = posStamp;
66
67 const Vec3f *dir = Mat4f::cubeDirections();
68 const Vec3f *up = Mat4f::cubeUpVectors();
69 auto pos = position_->getVertex(0);
70 auto views = view_->mapClientData<Mat4f>(ShaderData::WRITE);
71 auto viewInv = viewInv_->mapClientData<Mat4f>(ShaderData::WRITE);
72 for (int i = 0; i < 6; ++i) {
73 if (isCubeFaceVisible(i)) {
74 views.w[i] = Mat4f::lookAtMatrix(pos.r, dir[i], up[i]);
75 viewInv.w[i] = views.w[i].lookAtInverse();
76 }
77 }
78 return true;
79}
