39static ALenum avToAlType(AVSampleFormat format) {
40 switch (format) {
41 case AV_SAMPLE_FMT_U8:
42 return AL_UNSIGNED_BYTE_SOFT;
43 case AV_SAMPLE_FMT_S16:
44 return AL_SHORT_SOFT;
45 case AV_SAMPLE_FMT_S32:
46 return AL_INT_SOFT;
47 case AV_SAMPLE_FMT_FLT:
48 return AL_FLOAT_SOFT;
49 case AV_SAMPLE_FMT_DBL:
50 return AL_DOUBLE_SOFT;
51 case AV_SAMPLE_FMT_U8P:
52 return AL_UNSIGNED_BYTE_SOFT;
53 case AV_SAMPLE_FMT_S16P:
54 return AL_SHORT_SOFT;
55 case AV_SAMPLE_FMT_S32P:
56 return AL_INT_SOFT;
57 case AV_SAMPLE_FMT_FLTP:
58 return AL_FLOAT_SOFT;
59 case AV_SAMPLE_FMT_DBLP:
60 return AL_DOUBLE_SOFT;
61 case AV_SAMPLE_FMT_NONE:
62 default:
63 throw AudioSource::Error(REGEN_STRING(
64 "unsupported sample format " << format));
65 }
66}

326void AudioSource::openAudioStream(AVStream *stream, GLint index, GLboolean initial) {
327 if (!initial) {
328#ifdef HAS_LIBSWRESAMPLE
329 if (resampleContext_) swr_free(&resampleContext_);
330#endif
331 clearQueue();
332 }
333
334 AudioVideoStream::open(stream, index, initial);
335 alType_ = avToAlType(codecCtx_->sample_fmt);
336 alChannelLayout_ = avToAlLayout(codecCtx_->ch_layout.u.mask);
337 alFormat_ = avFormat(alType_, alChannelLayout_);
338 rate_ = codecCtx_->sample_rate;
339
340 REGEN_DEBUG("init audio stream" <<
341 " AL format=" << alFormat_ <<
342 " sample_fmt=" << codecCtx_->sample_fmt <<
343 " channel_layout=" << codecCtx_->ch_layout.u.mask <<
344 " sample_rate=" << codecCtx_->sample_rate <<
345 " bit_rate=" << codecCtx_->bit_rate <<
346 ".");
347#ifdef HAS_LIBSWRESAMPLE
348 // create resample context for planar sample formats
349 if (av_sample_fmt_is_planar(codecCtx_->sample_fmt)) {
350 int out_sample_fmt;
351 switch (codecCtx_->sample_fmt) {
352 case AV_SAMPLE_FMT_U8P:
353 out_sample_fmt = AV_SAMPLE_FMT_U8;
354 break;
355 case AV_SAMPLE_FMT_S16P:
356 out_sample_fmt = AV_SAMPLE_FMT_S16;
357 break;
358 case AV_SAMPLE_FMT_S32P:
359 out_sample_fmt = AV_SAMPLE_FMT_S32;
360 break;
361 case AV_SAMPLE_FMT_DBLP:
362 out_sample_fmt = AV_SAMPLE_FMT_DBL;
363 break;
364 case AV_SAMPLE_FMT_FLTP:
365 default:
366 out_sample_fmt = AV_SAMPLE_FMT_FLT;
367 }
368
369 resampleContext_ = swr_alloc();
370 if (!resampleContext_) {
371 // Handle allocation error
372 }
373
374 av_opt_set_int(resampleContext_, "in_channel_layout", codecCtx_->ch_layout.u.mask, 0);
375 av_opt_set_int(resampleContext_, "in_sample_fmt", codecCtx_->sample_fmt, 0);
376 av_opt_set_int(resampleContext_, "in_sample_rate", codecCtx_->sample_rate, 0);
377 av_opt_set_int(resampleContext_, "out_channel_layout", codecCtx_->ch_layout.u.mask, 0);
378 av_opt_set_int(resampleContext_, "out_sample_fmt", out_sample_fmt, 0); // example output format
379 av_opt_set_int(resampleContext_, "out_sample_rate", codecCtx_->sample_rate, 0);
380
381 if (swr_init(resampleContext_) < 0) {
382 // Handle initialization error
383 REGEN_ERROR("failed to initialize resample context.");
384 swr_free(&resampleContext_);
385 resampleContext_ = nullptr;
386 }
387 }
388#endif
389}

409void AudioSource::decode(AVPacket *packet) {
410 // unqueue processed buffers
411 ALint processed;
412 alGetSourcei(id_, AL_BUFFERS_PROCESSED, &processed);
413 for (; processed > 0; --processed) {
414 ALuint bufid;
415 alSourceUnqueueBuffers(id_, 1, &bufid);
416 AVFrame *processedFrame = frontFrame();
417 popFrame();
418 auto af = (AudioFrame *) processedFrame->opaque;
419 elapsedTime_ = af->dts;
420 af->free();
421 delete af;
422 }
423
424 int ret = avcodec_send_packet(codecCtx_, packet);
425 while (ret >= 0) {
426 AVFrame *frame = av_frame_alloc();
427 ret = avcodec_receive_frame(codecCtx_, frame);
428 if (ret == AVERROR(EAGAIN) || ret == AVERROR_EOF || ret < 0) {
429 // Need more packets or end of file
430 av_free(frame);
431 break;
432 }
433 if (frame->data[0] == nullptr) {
434 av_free(frame);
435 break;
436 }
437
438 // Process the decoded frame
439 // Get the required buffer size for the given audio parameters.
440 int num_channels = codecCtx_->ch_layout.nb_channels;
441 int linesize;
442 int bytesDecoded = av_samples_get_buffer_size(
443 &linesize,
444 num_channels,
445 frame->nb_samples,
446 codecCtx_->sample_fmt,
447 1);
448 if (bytesDecoded <= 0) {
449 av_free(frame);
450 break;
451 }
452
453 auto frameData = (ALbyte *) av_malloc(bytesDecoded);
454
455 if (av_sample_fmt_is_planar(codecCtx_->sample_fmt)) {
456 // Planar format: copy data from each channel buffer
457 for (int ch = 0; ch < num_channels; ++ch) {
458 memcpy(frameData + ch * linesize, frame->data[ch], linesize);
459 }
460 } else {
461 // Interleaved format: copy data from the single buffer
462 memcpy(frameData, frame->data[0], bytesDecoded);
463 }
464
465 auto audioFrame = new AudioFrame;
466 audioFrame->avFrame = frame;
467 audioFrame->buffer = ref_ptr<AudioBuffer>::alloc();
468 audioFrame->dts = (packet->pts + packet->duration) * av_q2d(stream_->time_base);
469
470#ifdef HAS_LIBSWRESAMPLE
471 if (resampleContext_ != nullptr) {
472 int out_samples = swr_convert(
473 resampleContext_,
474 (uint8_t **) &frameData, // output buffer
475 frame->nb_samples, // number of samples per channel to output
476 (const uint8_t **) frame->data, // input buffer
477 frame->nb_samples // number of samples per channel to input
478 );
479 if (out_samples < 0) {
480 // Handle conversion error
481 REGEN_ERROR("failed to convert audio frame.");
482 }
483 audioFrame->convertedFrame = frameData;
484
485 } else {
486 frameData = (ALbyte *) frame->data[0];
487 audioFrame->convertedFrame = nullptr;
488 }
489#else
490 audioFrame->convertedFrame = (ALbyte *) frameData;
491#endif
492
493 // add a audio buffer to the OpenAL audio source
494 alBufferData(audioFrame->buffer->id_,
495 alFormat_, (ALbyte *) frameData,
496 bytesDecoded,
497 rate_);
498 push(audioFrame->buffer);
499 frame->opaque = audioFrame;
500
501 // (re)start playing. playback may have stop when all frames consumed.
502 if (get1i(AL_SOURCE_STATE) != AL_PLAYING) play();
503
504 pushFrame(frame, bytesDecoded);
505 }
506 av_packet_unref(packet);
507}
