90void EventObject::disconnect(unsigned int connectionID) {
91 EventHandlerIds::iterator idNeedle =
92 eventHandlerIds_.find(connectionID);
93 if (idNeedle == eventHandlerIds_.end()) {
94 REGEN_WARN("Signal with id=" << connectionID << " no known.");
95 return; // handler id not found!
96 }
97
98 unsigned int eventId = idNeedle->second;
99 EventHandlers::iterator signalHandlers =
100 eventHandlers_.find(eventId);
101 if (signalHandlers == eventHandlers_.end()) {
102 REGEN_WARN("Signal with id=" << connectionID << " has no connected handlers.");
103 return; // no handlers not found!
104 }
105
106 EventHandlerList &l = signalHandlers->second;
107 for (EventHandlerList::iterator it = l.begin(); it != l.end(); ++it) {
108 if (it->second != connectionID) continue;
109 it->first->set_handlerID(-1);
110 l.erase(it);
111 break;
112 }
113
114 eventHandlerIds_.erase(idNeedle);
115}

140void EventObject::emitQueued() {
141 // ping-pong event queue to avoid copy and long locks
142 eventLock_.lock();
143 {
144 std::list<QueuedEvent> *buf = queued_;
145 queued_ = processing_;
146 processing_ = buf;
147 }
148 eventLock_.unlock();
149
150 // process queued events
151 while (!processing_->empty()) {
152 QueuedEvent &ev = processing_->front();
153 // emit event and delete data
154 ev.emitter->emitEvent(ev.eventID, ev.data);
155 // pop out processed event
156 processing_->pop_front();
157 }
158}

162 {
163 for (std::list<QueuedEvent>::iterator it = queued_->begin(); it != queued_->end(); ++it) {
164 const QueuedEvent &ev = *it;
