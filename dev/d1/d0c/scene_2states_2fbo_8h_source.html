18static vector<string> getFBOAttachments(SceneInputNode &input, const string &key) {
19 vector<string> out;
20 auto attachments = input.getValue<string>(key, "");
21 if (attachments.empty()) {
22 REGEN_WARN("No attachments specified in " << input.getDescription() << ".");
23 } else {
24 boost::split(out, attachments, boost::is_any_of(","));
25 }
26 return out;
27}

41 // Override
42 void processInput(
43 SceneParser *parser,
44 SceneInputNode &input,
45 const ref_ptr<State> &state) override {
46 if (input.getName() == "SCREEN") {
47 GLenum drawBuffer = glenum::drawBuffer(
48 input.getValue<string>("draw-buffer", "FRONT"));
49 ref_ptr<ScreenState> screenState =
50 ref_ptr<ScreenState>::alloc(parser->getViewport(), drawBuffer);
51
52 state->joinStates(screenState);
53 } else {
54 ref_ptr<FBO> fbo = parser->getResources()->getFBO(parser, input.getName());
55 if (fbo.get() == nullptr) {
56 REGEN_WARN("Unable to find FBO for '" << input.getDescription() << "'.");
57 return;
58 }
59 ref_ptr<FBOState> fboState = ref_ptr<FBOState>::alloc(fbo);
60
61 if (input.hasAttribute("clear-depth") &&
62 input.getValue<bool>("clear-depth", true)) {
63 fboState->setClearDepth();
64 }
65
66 if (input.hasAttribute("clear-buffers")) {
67 vector<string> idVec = getFBOAttachments(input, "clear-buffers");
68 vector<GLenum> buffers(idVec.size());
69 for (GLuint i = 0u; i < idVec.size(); ++i) {
70 GLint v;
71 stringstream ss(idVec[i]);
72 ss >> v;
73 buffers[i] = GL_COLOR_ATTACHMENT0 + v;
74 }
75
76 ClearColorState::Data data;
77 data.clearColor = input.getValue<Vec4f>("clear-color", Vec4f(0.0));
78 data.colorBuffers = DrawBuffers(buffers);
79 fboState->setClearColor(data);
80 }
81
82 if (input.hasAttribute("draw-buffers")) {
83 vector<string> idVec = getFBOAttachments(input, "draw-buffers");
84 vector<GLenum> buffers(idVec.size());
85 for (GLuint i = 0u; i < idVec.size(); ++i) {
86 GLint v;
87 stringstream ss(idVec[i]);
88 ss >> v;
89 buffers[i] = GL_COLOR_ATTACHMENT0 + v;
90 }
91 fboState->setDrawBuffers(buffers);
92 } else if (input.hasAttribute("ping-pong-buffers")) {
93 vector<string> idVec = getFBOAttachments(input, "ping-pong-buffers");
94 vector<GLenum> buffers(idVec.size());
95 for (GLuint i = 0u; i < idVec.size(); ++i) {
96 GLint v;
97 stringstream ss(idVec[i]);
98 ss >> v;
99 buffers[i] = GL_COLOR_ATTACHMENT0 + v;
100 }
101 fboState->setPingPongBuffers(buffers);
102 }
103
104 state->joinStates(fboState);
105 }
106 }
107 };
108 }
