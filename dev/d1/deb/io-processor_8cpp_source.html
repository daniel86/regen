44 if (!interpolation.empty()) { ss << interpolation << " "; }
45 ss << ioType << " " << dataType << " " << name;

77void IOProcessor::defineHandleIO(PreProcessorState &state) {
78 list<InputOutput> genOut, genIn;
79 map<string, InputOutput> &nextInputs = inputs_[state.nextStage];
80 map<string, InputOutput> &inputs = inputs_[state.currStage];
81 map<string, InputOutput> &outputs = outputs_[state.currStage];
82
83 // for each input of the next stage
84 // make sure it is declared at least as output in this stage
85 for (auto & nextInput : nextInputs) {
86 const string &nameWithoutPrefix = nextInput.first;
87 const InputOutput &nextIn = nextInput.second;
88
89 if (outputs.count(nameWithoutPrefix) > 0) { continue; }
90 genOut.push_back(InputOutput(nextIn));
91 genOut.back().name = "out_" + nameWithoutPrefix;
92 genOut.back().ioType = "out";
93 genOut.back().interpolation = nextIn.interpolation;
94 if (state.currStage == GL_GEOMETRY_SHADER) {
95 genOut.back().numElements = "";
96 } else if (state.currStage == GL_VERTEX_SHADER) {
97 genOut.back().numElements = "";
98 }
99#ifdef GL_TESS_EVALUATION_SHADER
100 else if (state.currStage == GL_TESS_EVALUATION_SHADER) {
101 genOut.back().numElements = "";
102 }
103#endif
104#ifdef GL_TESS_CONTROL_SHADER
105 else if (state.currStage == GL_TESS_CONTROL_SHADER) {
106 genOut.back().numElements = " ";
107 }
108#endif
109 outputs.insert(make_pair(nameWithoutPrefix, genOut.back()));
110
111 if (inputs.count(nameWithoutPrefix) > 0) { continue; }
112 genIn.push_back(InputOutput(nextIn));
113 genIn.back().name = "in_" + nameWithoutPrefix;
114 genIn.back().ioType = "in";
115 genIn.back().interpolation = nextIn.interpolation;
116 if (state.currStage == GL_GEOMETRY_SHADER) {
117 genIn.back().numElements = " ";
118 } else if (state.currStage == GL_VERTEX_SHADER) {
119 genIn.back().numElements = "";
120 }
121#ifdef GL_TESS_EVALUATION_SHADER
122 else if (state.currStage == GL_TESS_EVALUATION_SHADER) {
123 genIn.back().numElements = " ";
124 }
125#endif
126#ifdef GL_TESS_CONTROL_SHADER
127 else if (state.currStage == GL_TESS_CONTROL_SHADER) {
128 genIn.back().numElements = " ";
129 }
130#endif
131 inputs.insert(make_pair(nameWithoutPrefix, genIn.back()));
132 }
133
134 if (genOut.empty() && genIn.empty()) {
135 lineQueue_.push_back("#define HANDLE_IO(i)");
136 return;
137 }
138
139 // declare IO:
140 // * insert a redefinition of the IO name using the stage prefix
141 // * just insert the previous declaration again
142 for (auto & it : genIn) {
143 lineQueue_.push_back("#define " + it.name + " " +
144 glenum::glslStagePrefix(state.currStage) + "_" + getNameWithoutPrefix(it.name));
145 lineQueue_.push_back(it.declaration());
146 }
147 for (auto & it : genOut) {
148 lineQueue_.push_back("#define " + it.name + " " +
149 glenum::glslStagePrefix(state.nextStage) + "_" + getNameWithoutPrefix(it.name));
150 lineQueue_.push_back(it.declaration());
151 }
152
153 // declare HANDLE_IO() function
154 lineQueue_.push_back("void HANDLE_IO(int i) {");
155 for (auto & io : genOut) {
156 const string &outName = io.name;
157 string inName = inputs[getNameWithoutPrefix(outName)].name;
158
159 switch (state.currStage) {
160 case GL_VERTEX_SHADER:
161 lineQueue_.push_back(REGEN_STRING(
162 " " << outName << " = " << inName << ";"));
163 break;
164 case GL_TESS_CONTROL_SHADER:
165 lineQueue_.push_back(REGEN_STRING(
166 " " << outName << "[ID] = " << inName << "[ID];"));
167 break;
168 case GL_TESS_EVALUATION_SHADER:
169 lineQueue_.push_back(REGEN_STRING(
170 " " << outName << " = INTERPOLATE_VALUE(" << inName << ");"));
171 break;
172 case GL_GEOMETRY_SHADER:
173 lineQueue_.push_back(REGEN_STRING(
174 " " << outName << " = " << inName << "[i];"));
175 break;
176 case GL_FRAGMENT_SHADER:
177 break;
178 }
179
180 }
181 lineQueue_.push_back("}");
182}

184IOProcessor::InputOutput IOProcessor::getUniformIO(const NamedShaderInput &uniform) {
185 string nameWithoutPrefix = getNameWithoutPrefix(uniform.name_.empty() ?
186 uniform.in_->name() : uniform.name_);
187
188 IOProcessor::InputOutput io;
189 io.layout = "";
190 io.interpolation = "";
191 io.ioType = "uniform";
192 io.value = "";
193 GLuint numElements = uniform.in_->numArrayElements() * uniform.in_->numInstances();
194 io.numElements = (numElements > 1 || uniform.in_->forceArray()) ?
195 REGEN_STRING(numElements) : "";
196 if (uniform.in_->numInstances()>1) {
197 io.name = "instances_" + nameWithoutPrefix;
198 } else {
199 io.name = "in_" + nameWithoutPrefix;
200 }
201
202 Texture *tex = dynamic_cast<Texture *>(uniform.in_.get());
203 if (tex == nullptr) {
204 io.dataType = glenum::glslDataType(uniform.in_->dataType(), uniform.in_->valsPerElement());
205 } else {
206 io.dataType = tex->samplerType();
207 }
208
209 return io;
210}

212void IOProcessor::declareSpecifiedInput(PreProcessorState &state) {
213 list<NamedShaderInput> specifiedInput = state.in.specifiedInput;
214 InputOutput io;
215 io.layout = "";
216 io.interpolation = "";
217
218 // move all uniform blocks to the begin in specifiedInput
219 std::vector<NamedShaderInput> uniformBlocks;
220 auto it = specifiedInput.begin();
221 while (it != specifiedInput.end()) {
222 if (it->in_->isUniformBlock()) {
223 uniformBlocks.push_back(*it);
224 specifiedInput.erase(it++);
225 } else {
226 ++it;
227 }
228 }
229 specifiedInput.insert(specifiedInput.begin(), uniformBlocks.begin(), uniformBlocks.end());
230
231
232 for (auto it = specifiedInput.begin(); it != specifiedInput.end(); ++it) {
233 ref_ptr<ShaderInput> in = it->in_;
234 string nameWithoutPrefix = getNameWithoutPrefix(it->name_);
235 if (inputNames_.count(nameWithoutPrefix)) continue;
236 io.block.clear();
237 io.layout.clear();
238
239 if (it->type_.empty()) {
240 Texture *tex = dynamic_cast<Texture *>(in.get());
241 if (tex == nullptr) {
242 io.dataType = glenum::glslDataType(in->dataType(), in->valsPerElement());
243 } else {
244 io.dataType = tex->samplerType();
245 }
246 } else {
247 io.dataType = it->type_;
248 }
249 GLuint numElements = in->numArrayElements() * in->numInstances();
250 io.numElements = (numElements > 1 || in->forceArray()) ?
251 REGEN_STRING(numElements) : "";
252 if (in->numInstances() > 1) {
253 io.name = "instances_" + nameWithoutPrefix;
254 lineQueue_.push_back(REGEN_STRING("#define in_" << nameWithoutPrefix <<
255 " instances_" << nameWithoutPrefix << "[regen_InstanceID]"));
256 } else {
257 io.name = "in_" + nameWithoutPrefix;
258 }
259
260 if (in->isVertexAttribute()) {
261 if (state.currStage != GL_VERTEX_SHADER) continue;
262 io.ioType = "in";
263 io.value = "";
264 inputs_[state.currStage].insert(make_pair(nameWithoutPrefix, io));
265
266 lineQueue_.push_back(REGEN_STRING("#define " << io.name << " " <<
267 glenum::glslStagePrefix(state.currStage) << "_"
268 << nameWithoutPrefix));
269 } else if (in->isConstant()) {
270 io.ioType = "const";
271
272 stringstream val;
273 val << io.dataType << "(";
274 (*in.get()).write(val);
275 val << ")";
276 io.value = val.str();
277 } else if (in->isUniformBlock()) {
278 auto *uniformBlock = dynamic_cast<UniformBlock *>(in.get());
279 io.layout = "layout(std140)";
280 io.ioType = "uniform";
281 io.value = "";
282 io.dataType = "";
283 for (auto &blockUniform: uniformBlock->uniforms()) {
284 auto memberIO = getUniformIO(blockUniform);
285 auto blockNameWithoutPrefix = getNameWithoutPrefix(blockUniform.name_.empty() ?
286 blockUniform.in_->name() : blockUniform.name_);
287 memberIO.ioType = "";
288 io.block.push_back(memberIO.declaration());
289 inputNames_.insert(blockNameWithoutPrefix);
290
291 if (blockUniform.in_->numInstances() > 1) {
292 lineQueue_.push_back(REGEN_STRING("#define in_" << blockNameWithoutPrefix <<
293 " instances_" << blockNameWithoutPrefix << "[regen_InstanceID]"));
294 }
295 }
296 uniforms_[state.currStage].insert(make_pair(nameWithoutPrefix, io));
297 } else {
298 io.ioType = "uniform";
299 io.value = "";
300 uniforms_[state.currStage].insert(make_pair(nameWithoutPrefix, io));
301 }
302
303 lineQueue_.push_back(io.declaration());
304 inputNames_.insert(nameWithoutPrefix);
305 }
306}

340bool IOProcessor::process(PreProcessorState &state, string &line) {
341 static const char *interpolationPattern_ =
342 "^[ |\t|]*((flat|noperspective|smooth|centroid)[ |\t]+(.*))$";
343 static boost::regex interpolationRegex_(interpolationPattern_);
344 static const char *pattern_ =
345 "^[ |\t|]*((in|uniform|const|out)[ |\t]+([^ ]*)[ |\t]+([^;]+);)$";
346 static boost::regex regex_(pattern_);
347 static const char *handleIOPattern_ =
348 "^[ |\t]*#define[ |\t]+HANDLE_IO[ |\t]*";
349 static boost::regex handleIORegex_(handleIOPattern_);
350 static const char *macroPattern_ = "^[ |\t]*#(.*)$";
351 static boost::regex macroRegex_(macroPattern_);
352
353 if (currStage_ != state.currStage) {
354 inputNames_.clear();
355 isInputSpecified_ = GL_FALSE;
356 currStage_ = state.currStage;
357 }
358
359 // read a line from the queue
360 if (!lineQueue_.empty()) {
361 line = lineQueue_.front();
362 lineQueue_.pop_front();
363 return true;
364 }
365 // read a line from the input stream
366 if (!getlineParent(state, line)) {
367 return false;
368 }
369 boost::sregex_iterator it;
370
371 // Insert declarations of specified inputs when the first
372 // code line occurs.
373 if (!isInputSpecified_) {
374 it = boost::sregex_iterator(line.begin(), line.end(), macroRegex_);
375 if (it == NO_REGEX_MATCH) {
376 declareSpecifiedInput(state);
377 isInputSpecified_ = GL_TRUE;
378 }
379 }
380
381 // Processing of HANLDE_IO macro.
382 it = boost::sregex_iterator(line.begin(), line.end(), handleIORegex_);
383 if (it != NO_REGEX_MATCH) {
384 if (!isInputSpecified_) {
385 declareSpecifiedInput(state);
386 isInputSpecified_ = GL_TRUE;
387 }
388 defineHandleIO(state);
389 return process(state, line);
390 }
391
392 // Parse input declaration.
393 InputOutput io;
394 it = boost::sregex_iterator(line.begin(), line.end(), interpolationRegex_);
395 if (it == NO_REGEX_MATCH) {
396 it = boost::sregex_iterator(line.begin(), line.end(), regex_);
397 } else {
398 // interpolation qualifier specified
399 io.interpolation = (*it)[2];
400 line = (*it)[3];
401 it = boost::sregex_iterator(line.begin(), line.end(), regex_);
402 }
403 // No input declaration found.
404 if (it == NO_REGEX_MATCH) {
405 lineQueue_.push_back(line);
406 return process(state, line);
407 }
408
409 io.ioType = (*it)[2];
410 io.dataType = (*it)[3];
411 io.name = (*it)[4];
412 io.numElements = "";
413 io.value = "";
414 io.layout = "";
415 io.block.clear();
416
417 parseArray(io.dataType, io.numElements);
418 parseValue(io.name, io.value);
419 parseArray(io.name, io.numElements);
420
421 string nameWithoutPrefix = getNameWithoutPrefix(io.name);
422 if (io.ioType != "out") {
423 // skip input if already defined
424 if (inputNames_.count(nameWithoutPrefix) > 0)
425 return process(state, line);
426 inputNames_.insert(nameWithoutPrefix);
427
428 // Change IO type based on specified input.
429 list<NamedShaderInput>::const_iterator needle;
430 for (needle = state.in.specifiedInput.begin();
431 needle != state.in.specifiedInput.end(); ++needle) {
432 if (needle->name_ == nameWithoutPrefix) {
433 // Change declaration based on specified input
434 const ref_ptr<ShaderInput> &in = needle->in_;
435 if (in->isVertexAttribute()) {
436 io.ioType = "in";
437 io.value = "";
438 } else if (in->isConstant()) {
439 io.ioType = "const";
440
441 stringstream val;
442 val << io.dataType << "(";
443 (*in.get()).write(val);
444 val << ")";
445 io.value = val.str();
446 } else {
447 io.ioType = "uniform";
448 io.value = "";
449 }
450 break;
451 }
452 }
453 }
454
455 if (io.ioType == "in") {
456 // define input name with matching prefix
457 lineQueue_.push_back(REGEN_STRING("#define " << io.name << " " <<
458 glenum::glslStagePrefix(state.currStage) << "_"
459 << nameWithoutPrefix));
460 } else if (io.ioType == "out" && state.nextStage != GL_NONE) {
461 // define output name with matching prefix
462 lineQueue_.push_back(REGEN_STRING("#define " << io.name << " " <<
463 glenum::glslStagePrefix(state.nextStage) << "_"
464 << nameWithoutPrefix));
465 }
466 line = io.declaration();
467
468 if (io.ioType == "out") {
469 outputs_[state.currStage].insert(make_pair(nameWithoutPrefix, io));
470 } else if (io.ioType == "in") {
471 inputs_[state.currStage].insert(make_pair(nameWithoutPrefix, io));
472 } else if (io.ioType == "uniform") {
473 uniforms_[state.currStage].insert(make_pair(nameWithoutPrefix, io));
474 }
475
476 lineQueue_.push_back(line);
477 return process(state, line);
478}
